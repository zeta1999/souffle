64  /**
65  * @brief Generate the tree based on given entry.
66  * Return a NodePtr to the root.
67  */
68  NodePtr generateTree(const RamNode& root, const RamProgram& program) {
69  this->program = const_cast<RamProgram*>(&program);
70  // Encode all relation, indexPos and viewId.
71  visitDepthFirst(root, [&](const RamNode& node) {
72  if (dynamic_cast<const RamQuery*>(&node) != nullptr) {
73  newQueryBlock();
74  }
75  if (const auto* indexSearch = dynamic_cast<const RamIndexOperation*>(&node)) {
76  encodeIndexPos(*indexSearch);
77  encodeView(indexSearch);
78  } else if (const auto* exists = dynamic_cast<const RamExistenceCheck*>(&node)) {
79  encodeIndexPos(*exists);
80  encodeView(exists);
81  } else if (const auto* provExists = dynamic_cast<const RamProvenanceExistenceCheck*>(&node)) {
82  encodeIndexPos(*provExists);
83  encodeView(provExists);
84  }
85  });
86  // Parse program
87  return visit(root);
88  }

413  NodePtr visitQuery(const RamQuery& query) override {
414  std::shared_ptr<InterpreterPreamble> preamble = std::make_shared<InterpreterPreamble>();
415  parentQueryPreamble = preamble;
416  // split terms of conditions of outer-most filter operation
417  // into terms that require a context and terms that
418  // do not require a view
419  const RamOperation* next = &query.getOperation();
420  std::vector<const RamCondition*> freeOfView;
421  if (const auto* filter = dynamic_cast<const RamFilter*>(&query.getOperation())) {
422  next = &filter->getOperation();
423  // Check terms of outer filter operation whether they can be pushed before
424  // the view-generation for speed improvements
425  auto conditions = toConjunctionList(&filter->getCondition());
426  for (auto const& cur : conditions) {
427  bool needView = false;
428  visitDepthFirst(*cur, [&](const RamNode& node) {
429  if (requireView(&node)) {
430  needView = true;
431  const auto& rel = getRelationRefForView(&node);
432  preamble->addViewInfoForFilter(
433  encodeRelation(rel), indexTable[&node], encodeView(&node));
434  }
435  });
436 
437  if (needView) {
438  preamble->addViewOperationForFilter(visit(*cur));
439  } else {
440  preamble->addViewFreeOperationForFilter(visit(*cur));
441  }
442  }
443  }
444 
445  visitDepthFirst(*next, [&](const RamNode& node) {
446  if (requireView(&node)) {
447  const auto& rel = getRelationRefForView(&node);
448  preamble->addViewInfoForNested(encodeRelation(rel), indexTable[&node], encodeView(&node));
449  };
450  });
451 
452  visitDepthFirst(*next, [&](const RamAbstractParallel&) { preamble->isParallel = true; });
453 
454  NodePtrVec children;
455  children.push_back(visit(*next));
456 
457  auto res = std::make_unique<InterpreterQuery>(I_Query, &query, visit(*next));
458  res->setPreamble(parentQueryPreamble);
459  return res;
460  }

534  /** @brief Return operation index id from the result of indexAnalysis */
535  template <class RamNode>
536  size_t encodeIndexPos(RamNode& node) {
537  const MinIndexSelection& orderSet = isa->getIndexes(node.getRelation());
538  SearchSignature signature = isa->getSearchSignature(&node);
539  // A zero signature is equivalent as a full order signature.
540  if (signature.empty()) {
541  signature = SearchSignature::getFullSearchSignature(signature.arity());
542  }
543  auto i = orderSet.getLexOrderNum(signature);
544  indexTable[&node] = i;
545  return i;
546  };

548  /** @brief Encode and return the View id of an operation. */
549  size_t encodeView(const RamNode* node) {
550  auto pos = viewTable.find(node);
551  if (pos != viewTable.end()) {
552  return pos->second;
553  }
554  size_t id = getNextViewId();
555  viewTable[node] = id;
556  return id;
557  }
558 
559  /** @brief Encode and create the relation, return the relation id */
560  size_t encodeRelation(const RamRelation& rel) {
561  auto pos = relTable.find(&rel);
562  if (pos != relTable.end()) {
563  return pos->second;
564  }
565  size_t id = getNewRelId();
566  relTable[&rel] = id;
567  createRelation(rel, isa->getIndexes(rel), id);
568  return id;
569  }

585  /**

587  */
588  bool requireView(const RamNode* node) {
589  if (dynamic_cast<const RamAbstractExistenceCheck*>(node) != nullptr) {

597  /**
598  * @brief Return the associated relation of a operation which requires a view.
599  * This function assume the operation does requires a view.

611  /**
612  * @brief Convert terms of a conjunction to a list
613  *
614  * Convert a condition of the format C1 /\ C2 /\ ... /\ Cn
615  * to a list {C1, C2, ..., Cn}.
616  */
617  inline std::vector<const RamCondition*> toConjunctionList(const RamCondition* condition) {
618  std::vector<const RamCondition*> conditionList;
619  std::queue<const RamCondition*> conditionsToProcess;
620  if (condition != nullptr) {
621  conditionsToProcess.push(condition);
622  while (!conditionsToProcess.empty()) {
623  condition = conditionsToProcess.front();
624  conditionsToProcess.pop();
625  if (const auto* ramConj = dynamic_cast<const RamConjunction*>(condition)) {
626  conditionsToProcess.push(&ramConj->getLHS());
627  conditionsToProcess.push(&ramConj->getRHS());
628  } else {
629  conditionList.emplace_back(condition);
630  }
631  }
632  }
633  return conditionList;
634  }

659  /**
660  * @brief Encode and return the super-instruction information about a index operation.
661  */
662  InterpreterSuperInstruction getIndexSuperInstInfo(const RamIndexOperation& ramIndex) {
663  size_t arity = ramIndex.getRelation().getArity();
664  InterpreterSuperInstruction indexOperation(arity);
665  const auto& first = ramIndex.getRangePattern().first;
666  for (size_t i = 0; i < arity; ++i) {
667  auto& low = first[i];
668  // Unbounded
669  if (isRamUndefValue(low)) {
670  indexOperation.first[i] = MIN_RAM_SIGNED;
671  continue;
672  }
673 
674  // Constant
675  if (dynamic_cast<RamConstant*>(low) != nullptr) {
676  indexOperation.first[i] = dynamic_cast<RamConstant*>(low)->getConstant();
677  continue;
678  }
679 
680  // TupleElement
681  if (dynamic_cast<RamTupleElement*>(low) != nullptr) {
682  auto lowTuple = dynamic_cast<RamTupleElement*>(low);
683  indexOperation.tupleFirst.push_back(
684  {i, (size_t)lowTuple->getTupleId(), lowTuple->getElement()});
685  continue;
686  }
687 
688  // Generic expression
689  indexOperation.exprFirst.push_back(
690  std::pair<size_t, std::unique_ptr<InterpreterNode>>(i, visit(low)));
691  }
692  const auto& second = ramIndex.getRangePattern().second;
693  for (size_t i = 0; i < arity; ++i) {
694  auto& hig = second[i];
695  // Unbounded
696  if (isRamUndefValue(hig)) {
697  indexOperation.second[i] = MAX_RAM_SIGNED;
698  continue;
699  }
700 
701  // Constant
702  if (dynamic_cast<RamConstant*>(hig) != nullptr) {
703  indexOperation.second[i] = dynamic_cast<RamConstant*>(hig)->getConstant();
704  continue;
705  }
706 
707  // TupleElement
708  if (dynamic_cast<RamTupleElement*>(hig) != nullptr) {
709  auto highTuple = dynamic_cast<RamTupleElement*>(hig);
710  indexOperation.tupleSecond.push_back(
711  {i, (size_t)highTuple->getTupleId(), highTuple->getElement()});
712  continue;
713  }
714 
715  // Generic expression
716  indexOperation.exprSecond.push_back(
717  std::pair<size_t, std::unique_ptr<InterpreterNode>>(i, visit(hig)));
718  }
719  return indexOperation;
720  }
