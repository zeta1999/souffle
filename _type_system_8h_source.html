208 /**
209  * A collection to represent sets of types. In addition to ordinary set capabilities
210  * it may also represent the set of all types -- without being capable of iterating over those.
211  *
212  * It is the basic entity to conduct sub- and super-type computations.
213  */
214 struct TypeSet {
215 public:
216  using const_iterator = IterDerefWrapper<typename std::set<const Type*>::const_iterator>;
217 
218  TypeSet(bool all = false) : all(all) {}
219 
220  template <typename... Types>
221  explicit TypeSet(const Types&... types) : all(false) {
222  for (const Type* cur : toVector<const Type*>(&types...)) {
223  this->types.insert(cur);
224  }
225  }
226 
227  TypeSet(const TypeSet& other) = default;
228  TypeSet(TypeSet&& other) = default;
229  TypeSet& operator=(const TypeSet& other) = default;
230  TypeSet& operator=(TypeSet&& other) = default;
231 
232  bool empty() const {
233  return !all && types.empty();
234  }
235 
236  /** Universality check */
237  bool isAll() const {
238  return all;
239  }
240 
241  /** Determines the size of this set unless it is the universal set */
242  std::size_t size() const {
243  assert(!all && "Unable to give size of universe.");
244  return types.size();
245  }
246 
247  /** Determines whether a given type is included or not */
248  bool contains(const Type& type) const {
249  return all || types.find(&type) != types.end();
250  }
251 
252  /** Adds the given type to this set */
253  void insert(const Type& type) {
254  if (!all) {
255  types.insert(&type);
256  }
257  }
258 
259  /** Calculate intersection of two TypeSet */
260  static TypeSet intersection(const TypeSet& left, const TypeSet& right) {
261  TypeSet result;
262 
263  if (left.isAll()) {
264  return right;
265  } else if (right.isAll()) {
266  return left;
267  }
268 
269  for (const auto& element : left) {
270  if (right.contains(element)) {
271  result.insert(element);
272  }
273  }
274 
275  return result;
276  }
277 
278  template <typename F>
279  TypeSet filter(TypeSet whenAll, F&& f) const {
280  if (all) return whenAll;
281 
282  TypeSet cpy;
283  for (auto&& t : *this)
284  if (f(t)) cpy.insert(t);
285  return cpy;
286  }
287 
288  /** Inserts all the types of the given set into this set */
289  void insert(const TypeSet& set) {
290  if (all) {
291  return;
292  }
293 
294  // if the other set is universal => make this one universal
295  if (set.isAll()) {
296  all = true;
297  types.clear();
298  return;
299  }
300 
301  // add types one by one
302  for (const auto& t : set) {
303  insert(t);
304  }
305  }
306 
307  /** Allows to iterate over the types contained in this set (only if not universal) */
308  const_iterator begin() const {
309  assert(!all && "Unable to enumerate universe.");
310  return derefIter(types.begin());
311  }
312 
313  /** Allows to iterate over the types contained in this set (only if not universal) */
314  const_iterator end() const {
315  assert(!all && "Unable to enumerate universe.");
316  return derefIter(types.end());
317  }
318 
319  /** Determines whether this set is a subset of the given set */
320  bool isSubsetOf(const TypeSet& b) const {
321  if (all) {
322  return b.isAll();
323  }
324  return all_of(*this, [&](const Type& cur) { return b.contains(cur); });
325  }
326 
327  /** Determines equality between type sets */
328  bool operator==(const TypeSet& other) const {
329  return all == other.all && types == other.types;
330  }
331 
332  /** Determines inequality between type sets */
333  bool operator!=(const TypeSet& other) const {
334  return !(*this == other);
335  }
336 
337  /** Adds print support for type sets */
338  void print(std::ostream& out) const {
339  if (all) {
340  out << "{ - all types - }";
341  } else {
342  out << "{"
343  << join(types, ",", [](std::ostream& out, const Type* type) { out << type->getName(); })
344  << "}";
345  }
346  }
347 
348  friend std::ostream& operator<<(std::ostream& out, const TypeSet& set) {
349  set.print(out);
350  return out;
351  }
352 
353 private:
354  /** True if it is the all-types set, false otherwise */
355  bool all;
356 
357  /** The enumeration of types in case it is not the all-types set */
358  std::set<const Type*, deref_less<Type>> types;
359 };

374  /** create type in this environment */

376  T& createType(const AstQualifiedName& name, Args&&... args) {
377  assert(types.find(name) == types.end() && "Error: registering present type!");
378  auto* newType = new T(*this, name, std::forward<Args>(args)...);
379  types[name] = std::unique_ptr<Type>(newType);
380  return *newType;
