46 public:
47  ReadStreamCSV(std::istream& file, const std::map<std::string, std::string>& rwOperation,
48  SymbolTable& symbolTable, RecordTable& recordTable)
49  : ReadStream(rwOperation, symbolTable, recordTable),
50  delimiter(getOr(rwOperation, "delimiter", "\t")), file(file), lineNumber(0),
51  inputMap(getInputColumnMap(rwOperation, arity)) {
52  while (inputMap.size() < arity) {
53  int size = static_cast<int>(inputMap.size());
54  inputMap[size] = size;
55  }
56  }
57 
58 protected:
59  /**
60  * Read and return the next tuple.
61  *
62  * Returns nullptr if no tuple was readable.
63  * @return
64  */
65  std::unique_ptr<RamDomain[]> readNextTuple() override {
66  if (file.eof()) {
67  return nullptr;
68  }
69  std::string line;
70  std::unique_ptr<RamDomain[]> tuple = std::make_unique<RamDomain[]>(typeAttributes.size());
71 
72  if (!getline(file, line)) {
73  return nullptr;
74  }
75  // Handle Windows line endings on non-Windows systems
76  if (!line.empty() && line.back() == '\r') {
77  line = line.substr(0, line.length() - 1);
78  }
79  ++lineNumber;
80 
81  size_t start = 0;
82  size_t end = 0;
83  size_t columnsFilled = 0;
84  for (uint32_t column = 0; columnsFilled < arity; column++) {
85  size_t charactersRead = 0;
86  std::string element = nextElement(line, start, end);
87  if (inputMap.count(column) == 0) {
88  continue;
89  }
90  ++columnsFilled;
91 
92  try {
93  auto&& ty = typeAttributes.at(inputMap[column]);
94  switch (ty[0]) {
95  case 's': {
96  tuple[inputMap[column]] = symbolTable.unsafeLookup(element);
97  charactersRead = element.size();
98  break;
99  }
100  case 'r': {
101  tuple[inputMap[column]] = readRecord(element, ty, 0, &charactersRead);
102  break;
103  }
104  case 'i': {
105  tuple[inputMap[column]] = RamSignedFromString(element, &charactersRead);
106  break;
107  }
108  case 'u': {
109  tuple[inputMap[column]] = ramBitCast(readRamUnsigned(element, charactersRead));
110  break;
111  }
112  case 'f': {
113  tuple[inputMap[column]] = ramBitCast(RamFloatFromString(element, &charactersRead));
114  break;
115  }
116  default: fatal("invalid type attribute: `%c`", ty[0]);
117  }
118  // Check if everything was read.
119  if (charactersRead != element.size()) {
120  throw std::invalid_argument(
121  "Expected: " + delimiter + " or \\n. Got: " + element[charactersRead]);
122  }
123  } catch (...) {
124  std::stringstream errorMessage;
125  errorMessage << "Error converting <" + element + "> in column " << column + 1 << " in line "
126  << lineNumber << "; ";
127  throw std::invalid_argument(errorMessage.str());
128  }
129  }
130 
131  return tuple;
132  }

134  /**
135  * Read an unsigned element. Possible bases are 2, 10, 16
136  * Base is indicated by the first two chars.
137  */
138  RamUnsigned readRamUnsigned(const std::string& element, size_t& charactersRead) {
139  // Sanity check
140  assert(element.size() > 0);
141 
142  RamSigned value = 0;
143 
144  // Check prefix and parse the input.
145  if (isPrefix("0b", element)) {
146  value = RamUnsignedFromString(element, &charactersRead, 2);
147  } else if (isPrefix("0x", element)) {
148  value = RamUnsignedFromString(element, &charactersRead, 16);
149  } else {
150  value = RamUnsignedFromString(element, &charactersRead);
151  }
152  return value;
153  }

155  std::string nextElement(const std::string& line, size_t& start, size_t& end) {
156  std::string element;
157 
158  // Handle record/tuple delimiter coincidence.
159  if (delimiter.find(',') != std::string::npos) {
160  int record_parens = 0;
161  size_t next_delimiter = line.find(delimiter, start);
162 
163  // Find first delimiter after the record.
164  while (end < std::min(next_delimiter, line.length()) || record_parens != 0) {
165  // Track the number of parenthesis.
166  if (line[end] == '[') {
167  ++record_parens;
168  } else if (line[end] == ']') {
169  --record_parens;
170  }
171 
172  // Check for unbalanced parenthesis.
173  if (record_parens < 0) {
174  break;
175  };
176 
177  ++end;
178 
179  // Find a next delimiter if the old one is invalid.
180  // But only if inside the unbalance parenthesis.
181  if (end == next_delimiter && record_parens != 0) {
182  next_delimiter = line.find(delimiter, end);
183  }
184  }
185 
186  // Handle the end-of-the-line case where parenthesis are unbalanced.
187  if (record_parens != 0) {
188  std::stringstream errorMessage;
189  errorMessage << "Unbalanced record parenthesis " << lineNumber << "; ";
190  throw std::invalid_argument(errorMessage.str());
191  }
192  } else {
193  end = std::min(line.find(delimiter, start), line.length());
194  }
195 
196  // Check for missing value.
197  if (start > end) {
198  std::stringstream errorMessage;
199  errorMessage << "Values missing in line " << lineNumber << "; ";
200  throw std::invalid_argument(errorMessage.str());
201  }
202 
203  element = line.substr(start, end - start);
204  start = end + delimiter.size();
205 
206  return element;
207  }

209  std::map<int, int> getInputColumnMap(
210  const std::map<std::string, std::string>& rwOperation, const unsigned arity_) const {
211  std::string columnString = getOr(rwOperation, "columns", "");
212  std::map<int, int> inputColumnMap;
213 
214  if (!columnString.empty()) {
215  std::istringstream iss(columnString);
216  std::string mapping;
217  int index = 0;
218  while (std::getline(iss, mapping, ':')) {
219  inputColumnMap[stoi(mapping)] = index++;
220  }
221  if (inputColumnMap.size() < arity_) {
222  throw std::invalid_argument("Invalid column set was given: <" + columnString + ">");
223  }
224  } else {
225  while (inputColumnMap.size() < arity_) {
226  int size = static_cast<int>(inputColumnMap.size());
227  inputColumnMap[size] = size;
228  }
229  }
230  return inputColumnMap;
231  }

236  std::map<int, int> inputMap;
