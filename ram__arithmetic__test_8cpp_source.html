46 /** Function to evaluate a single RamExpression. */
47 RamDomain evalExpression(std::unique_ptr<RamExpression> expression, SymbolTable& symTab) {
48  // Set up RamProgram and translation unit
49  std::vector<std::unique_ptr<RamExpression>> returnValues;
50  returnValues.emplace_back(std::move(expression));
51 
52  Global::config().set("jobs", "1");
53  std::unique_ptr<RamStatement> query =
54  std::make_unique<RamQuery>(std::make_unique<RamSubroutineReturn>(std::move(returnValues)));
55  std::map<std::string, std::unique_ptr<RamStatement>> subs;
56  subs.insert(std::make_pair("test", std::move(query)));
57  std::vector<std::unique_ptr<RamRelation>> rels;
58 
59  std::unique_ptr<RamProgram> prog =
60  std::make_unique<RamProgram>(std::move(rels), std::make_unique<RamSequence>(), std::move(subs));
61 
62  ErrorReport errReport;
63  DebugReport debugReport;
64 
65  RamTranslationUnit translationUnit(std::move(prog), symTab, errReport, debugReport);
66 
67  // configure and execute interpreter
68  std::unique_ptr<InterpreterEngine> interpreter = std::make_unique<InterpreterEngine>(translationUnit);
69 
70  std::string name("test");
71  std::vector<RamDomain> ret;
72 
73  interpreter->executeSubroutine(name, {}, ret);
74 
75  return ret.at(0);
76 }

96  std::vector<std::unique_ptr<RamExpression>> args;

102 /** Evaluate a binary operator */
103 RamDomain evalBinary(FunctorOp functor, RamDomain arg1, RamDomain arg2) {
104  std::vector<std::unique_ptr<RamExpression>> args;
105  args.push_back(std::make_unique<RamSignedConstant>(arg1));
106  args.push_back(std::make_unique<RamSignedConstant>(arg2));
107 
108  return evalMultiArg(functor, std::move(args));
109 }
