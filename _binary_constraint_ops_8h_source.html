35 enum class BinaryConstraintOp {
36  EQ, // equivalence of two values
37  FEQ, // float equiv; b/c NaNs are never equiv
38  NE, // whether two values are different
39  FNE, // float diff; b/c NaNs are always different
40  LT, // signed <
41  ULT, // Unsigned <
42  FLT, // Float <
43  SLT, // Symbol <
44  LE, // signed ≤
45  ULE, // Unsigned ≤
46  FLE, // Float ≤
47  SLE, // Symbol ≤
48  GT, // signed >
49  UGT, // unsigned >
50  FGT, // float >
51  SGT, // Symbol >
52  GE, // signed ≥
53  UGE, // Unsigned ≥
54  FGE, // Float ≥
55  SGE, // Symbol ≥
56  MATCH, // matching string
57  CONTAINS, // whether a sub-string is contained in a string
58  NOT_MATCH, // not matching string
59  NOT_CONTAINS // whether a sub-string is not contained in a string
60 };

244 /**
245  * Converts symbolic representation of an operator to the operator.
246  * Note that this won't tell you which polymorphic overload is actually used.
247  */
248 inline BinaryConstraintOp toBinaryConstraintOp(const std::string& symbol) {
249  if (symbol == "=") return BinaryConstraintOp::EQ;
250  if (symbol == "!=") return BinaryConstraintOp::NE;
251  if (symbol == "<") return BinaryConstraintOp::LT;
252  if (symbol == "<=") return BinaryConstraintOp::LE;
253  if (symbol == ">=") return BinaryConstraintOp::GE;
254  if (symbol == ">") return BinaryConstraintOp::GT;
255  if (symbol == "match") return BinaryConstraintOp::MATCH;
256  if (symbol == "contains") return BinaryConstraintOp::CONTAINS;
257  if (symbol == "not_match") return BinaryConstraintOp::NOT_MATCH;
258  if (symbol == "not_contains") return BinaryConstraintOp::NOT_CONTAINS;
259 
260  fatal("unrecognised binary operator: symbol = `%s`", symbol);
261 }
