49 bool RecursiveClausesAnalysis::computeIsRecursive(
50  const AstClause& clause, const AstTranslationUnit& translationUnit) const {
51  const auto& relationDetail = *translationUnit.getAnalysis<RelationDetailCacheAnalysis>();
52  const AstProgram& program = *translationUnit.getProgram();
53 
54  // we want to reach the atom of the head through the body
55  const AstRelation* trg = getHeadRelation(&clause, &program);
56 
57  std::set<const AstRelation*> reached;
58  std::vector<const AstRelation*> worklist;
59 
60  // set up start list
61  for (const auto* cur : getBodyLiterals<AstAtom>(clause)) {
62  auto rel = relationDetail.getRelation(cur->getQualifiedName());
63  if (rel == trg) {
64  return true;
65  }
66  worklist.push_back(rel);
67  }
68 
69  // process remaining elements
70  while (!worklist.empty()) {
71  // get next to process
72  const AstRelation* cur = worklist.back();
73  worklist.pop_back();
74 
75  // skip null pointers (errors in the input code)
76  if (cur == nullptr) {
77  continue;
78  }
79 
80  // check whether this one has been checked before
81  if (!reached.insert(cur).second) {
82  continue;
83  }
84 
85  // check all atoms in the relations
86  for (const AstClause* cl : relationDetail.getClauses(cur)) {
87  for (const AstAtom* at : getBodyLiterals<AstAtom>(*cl)) {
88  auto rel = relationDetail.getRelation(at->getQualifiedName());
89  if (rel == trg) {
90  return true;
91  }
92  worklist.push_back(rel);
93  }
94  }
95  }
96 
97  // no cycles found
98  return false;
99 }
