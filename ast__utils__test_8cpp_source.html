49 TEST(AstUtils, Grounded) {
50  // create an example clause:
51  auto clause = std::make_unique<AstClause>();
52 
53  // something like:
54  // r(X,Y,Z) :- a(X), X = Y, !b(Z).
55 
56  // r(X,Y,Z)
57  auto* head = new AstAtom("r");
58  head->addArgument(std::unique_ptr<AstArgument>(new AstVariable("X")));
59  head->addArgument(std::unique_ptr<AstArgument>(new AstVariable("Y")));
60  head->addArgument(std::unique_ptr<AstArgument>(new AstVariable("Z")));
61  clause->setHead(std::unique_ptr<AstAtom>(head));
62 
63  // a(X)
64  auto* a = new AstAtom("a");
65  a->addArgument(std::unique_ptr<AstArgument>(new AstVariable("X")));
66  clause->addToBody(std::unique_ptr<AstLiteral>(a));
67 
68  // X = Y
69  AstLiteral* e1 = new AstBinaryConstraint(BinaryConstraintOp::EQ,
70  std::unique_ptr<AstArgument>(new AstVariable("X")),
71  std::unique_ptr<AstArgument>(new AstVariable("Y")));
72  clause->addToBody(std::unique_ptr<AstLiteral>(e1));
73 
74  // !b(Z)
75  auto* b = new AstAtom("b");
76  b->addArgument(std::unique_ptr<AstArgument>(new AstVariable("Z")));
77  auto* neg = new AstNegation(std::unique_ptr<AstAtom>(b));
78  clause->addToBody(std::unique_ptr<AstLiteral>(neg));
79 
80  // check construction
81  EXPECT_EQ("r(X,Y,Z) :- \n a(X),\n X = Y,\n !b(Z).", toString(*clause));
82 
83  auto program = std::make_unique<AstProgram>();
84  program->addClause(std::move(clause));
85  DebugReport dbgReport;
86  ErrorReport errReport;
87  AstTranslationUnit tu{std::move(program), errReport, dbgReport};
88 
89  // obtain groundness
90  auto isGrounded = getGroundedTerms(tu, *tu.getProgram()->getClauses()[0]);
91 
92  auto args = head->getArguments();
93  // check selected sub-terms
94  EXPECT_TRUE(isGrounded[args[0]]); // X
95  EXPECT_TRUE(isGrounded[args[1]]); // Y
96  EXPECT_FALSE(isGrounded[args[2]]); // Z
97 }
