243  .decl A(x:number, y:number)

247  A(0,0).
248  A(0,0).
249  A(0,1).
250 
251  B(1).
252 
253  C(z) :- A(z,y), A(z,x), x != 3, x < y, !B(x), y > 3, B(y).
254  C(r) :- A(r,y), A(r,x), x != 3, x < y, !B(y), y > 3, B(y), B(x), x < y.
255  C(x) :- A(x,a), a != 3, !B(a), A(x,b), b > 3, B(c), a < b, c = b.
256  )",
257  errorReport, debugReport);
258

267  EXPECT_EQ(3, aClauses.size());

275  EXPECT_EQ(3, cClauses.size());

289  EXPECT_FALSE(MinimiseProgramTransformer::areBijectivelyEquivalent(aClauses[1], aClauses[2]));

291 
292  // -- C

294  EXPECT_FALSE(MinimiseProgramTransformer::areBijectivelyEquivalent(cClauses[0], cClauses[1]));
295  EXPECT_FALSE(MinimiseProgramTransformer::areBijectivelyEquivalent(cClauses[2], cClauses[1]));
296

319 /**
320  * Test the removal of redundancies within clauses using the MinimiseProgramTransformer.
321  *
322  * In particular, the removal of:
323  * - intraclausal literals equivalent to another literal in the body
324  * e.g. a(x) :- b(x), b(x), c(x). --> a(x) :- b(x), c(x).
325  * - clauses that are only trivially satisfiable
326  * e.g. a(x) :- a(x), x != 0. is only true if a(x) is already true
327  */
328 TEST(AstTransformers, RemoveClauseRedundancies) {
329  ErrorReport errorReport;
330  DebugReport debugReport;
331 
332  std::unique_ptr<AstTranslationUnit> tu = ParserDriver::parseTranslationUnit(
333  R"(
334  .decl a,b,c(X:number)
335  a(0).
336  b(1).
337  c(X) :- b(X).
338 
339  a(X) :- b(X), c(X).
340  a(X) :- a(X).
341  a(X) :- a(X), X != 1.
342 
343  q(X) :- a(X).
344 
345  .decl q(X:number)
346  .output q()
347  )",
348  errorReport, debugReport);
349 
350  const auto& program = *tu->getProgram();
351 
352  // Invoking the `RemoveRelationCopiesTransformer` to create some extra redundancy
353  // In particular: The relation `c` will be replaced with `b` throughout, creating
354  // the clause b(x) :- b(x).
355  std::make_unique<RemoveRelationCopiesTransformer>()->apply(*tu);
356  EXPECT_EQ(nullptr, getRelation(program, "c"));
357  auto bIntermediateClauses = getClauses(program, "b");
358  EXPECT_EQ(2, bIntermediateClauses.size());
359  EXPECT_EQ("b(1).", toString(*bIntermediateClauses[0]));
360  EXPECT_EQ("b(X) :- \n b(X).", toString(*bIntermediateClauses[1]));
361 
362  // Attempt to minimise the program
363  std::make_unique<MinimiseProgramTransformer>()->apply(*tu);
364  EXPECT_EQ(3, program.getRelations().size());
365 
366  auto aClauses = getClauses(program, "a");
367  EXPECT_EQ(2, aClauses.size());
368  EXPECT_EQ("a(0).", toString(*aClauses[0]));
369  EXPECT_EQ("a(X) :- \n b(X).", toString(*aClauses[1]));
370 
371  auto bClauses = getClauses(program, "b");
372  EXPECT_EQ(1, bClauses.size());
373  EXPECT_EQ("b(1).", toString(*bClauses[0]));
374 
375  auto qClauses = getClauses(program, "q");
376  EXPECT_EQ(1, qClauses.size());
377  EXPECT_EQ("q(X) :- \n a(X).", toString(*qClauses[0]));
378 }
