78  void outputRecord(std::ostream& destination, const RamDomain value, const std::string& name) {
79  auto&& recordInfo = types["records"][name];

81  // Check if record type information are present
82  assert(!recordInfo.is_null() && "Missing record type information");
83 
84  // Check for nil

89

91  const size_t recordArity = recordInfo["arity"].long_value();
92 
93  const RamDomain* tuplePtr = recordTable.unpack(value, recordArity);
94 
95  destination << "[";
96 
97  // print record's elements
98  for (size_t i = 0; i < recordArity; ++i) {
99  if (i > 0) {
100  destination << ", ";
101  }
102 
103  const std::string& recordType = recordTypes[i].string_value();
104  const RamDomain recordValue = tuplePtr[i];
105 
106  switch (recordType[0]) {
107  case 'i': destination << recordValue; break;
108  case 'f': destination << ramBitCast<RamFloat>(recordValue); break;
109  case 'u': destination << ramBitCast<RamUnsigned>(recordValue); break;
110  case 's': destination << symbolTable.unsafeResolve(recordValue); break;
111  case 'r': outputRecord(destination, recordValue, recordType); break;
112  default: fatal("Unsupported type attribute: `%c`", recordType[0]);
113  }
114  }
115  destination << "]";
116  }
