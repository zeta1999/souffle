109 /**
110  * Executes a binary file.
111  */
112 void executeBinary(const std::string& binaryFilename) {
113  assert(!binaryFilename.empty() && "binary filename cannot be blank");
114 
115  // check whether the executable exists
116  if (!isExecutable(binaryFilename)) {
117  throw std::invalid_argument("Generated executable <" + binaryFilename + "> could not be found");
118  }
119 
120  // run the executable
121  if (Global::config().has("library-dir")) {
122  std::string ldPath;
123  for (const std::string& library : splitString(Global::config().get("library-dir"), ' ')) {
124  ldPath += library + ':';
125  }
126  ldPath.pop_back();
127  setenv("LD_LIBRARY_PATH", ldPath.c_str(), 1);
128  setenv("DYLD_LIBRARY_PATH", ldPath.c_str(), 1);
129  }
130 
131  int exitCode = system(binaryFilename.c_str());
132 
133  if (Global::config().get("dl-program").empty()) {
134  remove(binaryFilename.c_str());
135  remove((binaryFilename + ".cpp").c_str());
136  }
137 
138  // exit with same code as executable
139  if (exitCode != EXIT_SUCCESS) {
140  exit(exitCode);
141  }
142 }

144 /**
145  * Compiles the given source file to a binary file.
146  */
147 void compileToBinary(std::string compileCmd, const std::string& sourceFilename) {
148  // add source code
149  compileCmd += ' ';
150  for (const std::string& path : splitString(Global::config().get("library-dir"), ' ')) {
151  // The first entry may be blank
152  if (path.empty()) {
153  continue;
154  }
155  compileCmd += "-L" + path + ' ';
156  }
157  for (const std::string& library : splitString(Global::config().get("libraries"), ' ')) {
158  // The first entry may be blank
159  if (library.empty()) {
160  continue;
161  }
162  compileCmd += "-l" + library + ' ';
163  }
164 
165  compileCmd += sourceFilename;
166 
167  // run executable
168  if (system(compileCmd.c_str()) != 0) {
169  throw std::invalid_argument("failed to compile C++ source <" + sourceFilename + ">");
170  }
171 }
