54 struct bool_or {

84 /**
85  * A constraint factory for a constraint ensuring that the value assigned to the
86  * given variable is (at least) {true}
87  */
88 BoolDisjunctConstraint isTrue(const BoolDisjunctVar& var) {
89  struct C : public Constraint<BoolDisjunctVar> {
90  BoolDisjunctVar var;
91  C(BoolDisjunctVar var) : var(std::move(var)) {}
92  bool update(Assignment<BoolDisjunctVar>& ass) const override {
93  auto res = !ass[var];
94  ass[var] = true;
95  return res;
96  }
97  void print(std::ostream& out) const override {
98  out << var << " is true";
99  }
100  };
101  return std::make_shared<C>(var);
102 }

104 /**
105  * A constraint factory for a constraint ensuring the constraint
106  *
107  * a ⇒ b
108  *
109  * Hence, whenever a is mapped to {true}, so is b.
110  */
111 BoolDisjunctConstraint imply(const BoolDisjunctVar& a, const BoolDisjunctVar& b) {
112  return sub(a, b, "⇒");
113 }

115 /**
116  * A constraint factory for a constraint ensuring the constraint
117  *
118  * vars[0] ∧ vars[1] ∧ ... ∧ vars[n] ⇒ res
119  *
120  * Hence, whenever all variables vars[i] are mapped to true, so is res.
121  */
122 BoolDisjunctConstraint imply(const std::vector<BoolDisjunctVar>& vars, const BoolDisjunctVar& res) {
123  struct C : public Constraint<BoolDisjunctVar> {
124  BoolDisjunctVar res;
125  std::vector<BoolDisjunctVar> vars;
126 
127  C(BoolDisjunctVar res, std::vector<BoolDisjunctVar> vars)
128  : res(std::move(res)), vars(std::move(vars)) {}
129 
130  bool update(Assignment<BoolDisjunctVar>& ass) const override {
131  bool r = ass[res];
132  if (r) {
133  return false;
134  }
135 
136  for (const auto& cur : vars) {
137  if (!ass[cur]) {
138  return false;
139  }
140  }
141 
142  ass[res] = true;
143  return true;
144  }
145 
146  void print(std::ostream& out) const override {
147  out << join(vars, " ∧ ") << " ⇒ " << res;
148  }
149  };
150 
151  return std::make_shared<C>(res, vars);
152 }

159 std::map<const AstArgument*, bool> getGroundedTerms(const AstTranslationUnit& tu, const AstClause& clause) {
160  struct Analysis : public AstConstraintAnalysis<BoolDisjunctVar> {
161  const RelationDetailCacheAnalysis& relCache;
162  std::set<const AstAtom*> ignore;
163 
164  Analysis(const AstTranslationUnit& tu) : relCache(*tu.getAnalysis<RelationDetailCacheAnalysis>()) {}
165 
166  // atoms are producing grounded variables
167  void visitAtom(const AstAtom& cur) override {
168  // some atoms need to be skipped (head or negation)
169  if (ignore.find(&cur) != ignore.end()) {
170  return;
171  }
172 
173  // all arguments are grounded
174  for (const auto& arg : cur.getArguments()) {
175  addConstraint(isTrue(getVar(arg)));
176  }
177  }
178 
179  // negations need to be skipped
180  void visitNegation(const AstNegation& cur) override {
181  // add nested atom to black-list
182  ignore.insert(cur.getAtom());
183  }
184 
185  // also skip head if we don't have an inline qualifier
186  void visitClause(const AstClause& clause) override {
187  if (auto clauseHead = clause.getHead()) {
188  auto relation = relCache.getRelation(clauseHead->getQualifiedName());
189  // Only skip the head if the relation ISN'T inline. Keeping the head will ground
190  // any mentioned variables, allowing us to pretend they're grounded.
191  if (!(relation && relation->hasQualifier(RelationQualifier::INLINE))) {
192  ignore.insert(clauseHead);
193  }
194  }
195  }
196 
197  // binary equality relations propagates groundness
198  void visitBinaryConstraint(const AstBinaryConstraint& cur) override {
199  // only target equality
200  if (!isEqConstraint(cur.getOperator())) {
201  return;
202  }
203 
204  // if equal, link right and left side
205  auto lhs = getVar(cur.getLHS());
206  auto rhs = getVar(cur.getRHS());
207 
208  addConstraint(imply(lhs, rhs));
209  addConstraint(imply(rhs, lhs));
210  }
211 
212  // record init nodes
213  void visitRecordInit(const AstRecordInit& init) override {
214  auto cur = getVar(init);
215 
216  std::vector<BoolDisjunctVar> vars;
217 
218  // if record is grounded, so are all its arguments
219  for (const auto& arg : init.getArguments()) {
220  auto arg_var = getVar(arg);
221  addConstraint(imply(cur, arg_var));
222  vars.push_back(arg_var);
223  }
224 
225  // if all arguments are grounded, so is the record
226  addConstraint(imply(vars, cur));
227  }
228 
229  // constants are also sources of grounded values
230  void visitConstant(const AstConstant& c) override {
231  addConstraint(isTrue(getVar(c)));
232  }
233 
234  // aggregators are grounding values
235  void visitAggregator(const AstAggregator& c) override {
236  addConstraint(isTrue(getVar(c)));
237  }
238 
239  // functors with grounded values are grounded values
240  void visitFunctor(const AstFunctor& cur) override {
241  auto fun = getVar(cur);
242  std::vector<BoolDisjunctVar> varArgs;
243  for (const auto& arg : cur.getArguments()) {
244  varArgs.push_back(getVar(arg));
245  }
246  addConstraint(imply(varArgs, fun));
247  }
248 
249  // casts propogate groundedness in and out
250  void visitTypeCast(const AstTypeCast& cast) override {
251  addConstraint(imply(getVar(cast.getValue()), getVar(cast)));
252  }
253  };
254 
255  // run analysis on given clause
256  return Analysis(tu).analyse(clause);
257 }
