25 template <typename A, typename F /* Tuple<RamDomain,1> -> void */>
26 void runRange(A from, A to, A step, F&& go) {
27 #define GO(x) go(Tuple<RamDomain, 1>{ramBitCast(x)})
28  if (0 < step) {
29  for (auto x = from; x < to; x += step) {
30  GO(x);
31  }
32  } else if (step < 0) {
33  for (auto x = from; to < x; x += step) {
34  GO(x);
35  }
36  } else if (from != to) {
37  // `step = 0` edge case, only if non-empty range
38  GO(from);
39  }
40 #undef GO
41 }
