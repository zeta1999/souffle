44 std::string MaterializeSingletonAggregationTransformer::findUniqueVariableName(const AstClause& clause) {
45  static int counter = 0;
46  std::set<std::string> variableNames;
47  visitDepthFirst(clause, [&](const AstVariable& variable) { variableNames.insert(variable.getName()); });
48  std::string candidateVariableName = "z"; // completely arbitrary
49  while (variableNames.find(candidateVariableName) != variableNames.end()) {
50  candidateVariableName = "z" + toString(counter++);
51  }
52  return candidateVariableName;
53 }

65 bool MaterializeSingletonAggregationTransformer::transform(AstTranslationUnit& translationUnit) {
66  AstProgram& program = *translationUnit.getProgram();
67  std::set<std::pair<AstAggregator*, AstClause*>> pairs;
68  // collect references to clause / aggregate pairs
69  visitDepthFirst(program, [&](const AstClause& clause) {
70  visitDepthFirst(clause, [&](const AstAggregator& agg) {
71  // if the aggregate isn't single valued
72  // (ie it relies on a grounding from the outer scope)
73  // or it's the only atom in the clause, then there's no point materialising it!
74  if (!isSingleValued(agg, clause) || clause.getBodyLiterals().size() == 1) {
75  return;
76  }
77  auto* foundAggregate = const_cast<AstAggregator*>(&agg);
78  auto* foundClause = const_cast<AstClause*>(&clause);
79  pairs.insert(std::make_pair(foundAggregate, foundClause));
80  });
81  });
82  for (auto pair : pairs) {
83  // Clone the aggregate that we're going to be deleting.
84  auto aggregate = souffle::clone(pair.first);
85  AstClause* clause = pair.second;
86  // synthesise an aggregate relation
87  // __agg_rel_0()
88  auto aggRel = std::make_unique<AstRelation>();
89  auto aggHead = std::make_unique<AstAtom>();
90  auto aggClause = std::make_unique<AstClause>();
91 
92  std::string aggRelName = findUniqueAggregateRelationName(program);
93  aggRel->setQualifiedName(aggRelName);
94  aggHead->setQualifiedName(aggRelName);
95 
96  // create a synthesised variable to replace the aggregate term!
97  std::string variableName = findUniqueVariableName(*clause);
98  auto variable = std::make_unique<AstVariable>(variableName);
99 
100  // __agg_rel_0(z) :- ...
101  aggHead->addArgument(souffle::clone(variable));
102  aggRel->addAttribute(std::make_unique<AstAttribute>(variableName, "number"));
103  aggClause->setHead(souffle::clone(aggHead));
104 
105  // A(x) :- x = sum .., B(x).
106  // -> A(x) :- x = z, B(x), __agg_rel_0(z).
107  auto equalityLiteral = std::make_unique<AstBinaryConstraint>(
108  BinaryConstraintOp::EQ, souffle::clone(variable), souffle::clone(aggregate));
109  // __agg_rel_0(z) :- z = sum ...
110  aggClause->addToBody(std::move(equalityLiteral));
111  program.addRelation(std::move(aggRel));
112  program.addClause(std::move(aggClause));
113 
114  // the only thing left to do is just replace the aggregate terms in the original
115  // clause with the synthesised variable
116  struct replaceAggregate : public AstNodeMapper {
117  const AstAggregator& aggregate;
118  const std::unique_ptr<AstVariable> variable;
119  replaceAggregate(const AstAggregator& aggregate, std::unique_ptr<AstVariable> variable)
120  : aggregate(aggregate), variable(std::move(variable)) {}
121  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
122  assert(node != nullptr);
123  if (auto* current = dynamic_cast<AstAggregator*>(node.get())) {
124  if (*current == aggregate) {
125  auto replacement = souffle::clone(variable);
126  assert(replacement != nullptr);
127  return replacement;
128  }
129  }
130  node->apply(*this);
131  assert(node != nullptr);
132  return node;
133  }
134  };
135  replaceAggregate update(*aggregate, std::move(variable));
136  clause->apply(update);
137  clause->addToBody(std::move(aggHead));
138  }
139  return pairs.size() > 0;
140 }

142 bool MaterializeSingletonAggregationTransformer::isSingleValued(
143  const AstAggregator& agg, const AstClause& clause) {
144  std::map<std::string, int> occurrences;
145  visitDepthFirst(clause, [&](const AstVariable& v) {
146  if (occurrences.find(v.getName()) == occurrences.end()) {
147  occurrences[v.getName()] = 0;
148  }
149  occurrences[v.getName()] = occurrences[v.getName()] + 1;
150  });
151  std::set<std::string> aggVariables;
152  visitDepthFirst(agg, [&](const AstVariable& v) {
153  aggVariables.insert(v.getName());
154  occurrences[v.getName()] = occurrences[v.getName()] - 1;
155  });
156  for (std::string variableName : aggVariables) {
157  if (occurrences[variableName] != 0) {
158  return false;
159  }
160  }
161  return true;
162 }
