32 /**

34  * associated with an AstArgument which's property it is describing.
35  *
36  * @tparam PropertySpace the property space associated to the analysis

39 struct AstConstraintAnalysisVar : public Variable<const AstArgument*, PropertySpace> {
40  explicit AstConstraintAnalysisVar(const AstArgument* arg)

43  : Variable<const AstArgument*, PropertySpace>(&arg) {}
44 
45  /** adds print support */

47  out << "var(" << *(this->id) << ")";
48  }
49 };

51 /**
52  * A base class for AstConstraintAnalysis collecting constraints for an analysis
53  * by visiting every node of a given AST. The collected constraints are
54  * then utilized to obtain the desired analysis result.
55  *
56  * @tparam AnalysisVar the type of variable (and included property space)
57  * to be utilized by this analysis.
58  */
59 template <typename AnalysisVar>
60 class AstConstraintAnalysis : public AstVisitor<void> {
61 public:
62  using value_type = typename AnalysisVar::property_space::value_type;
63  using constraint_type = std::shared_ptr<Constraint<AnalysisVar>>;
64  using solution_type = std::map<const AstArgument*, value_type>;
65 
66  virtual void collectConstraints(const AstClause& clause) {
67  visitDepthFirstPreOrder(clause, *this);
68  }
69 
70  /**
71  * Runs this constraint analysis on the given clause.
72  *
73  * @param clause the close to be analysed
74  * @param debug a flag enabling the printing of debug information
75  * @return an assignment mapping a property to each argument in the given clause
76  */
77  solution_type analyse(const AstClause& clause, std::ostream* debugOutput = nullptr) {
78  collectConstraints(clause);
79 
80  assignment = constraints.solve();
81 
82  // print debug information if desired
83  if (debugOutput != nullptr) {
84  *debugOutput << "Clause: " << clause << "\n";
85  *debugOutput << "Problem:\n" << constraints << "\n";
86  *debugOutput << "Solution:\n" << assignment << "\n";
87  }
88 
89  // convert assignment to result
90  solution_type solution;
91  visitDepthFirst(clause, [&](const AstArgument& arg) { solution[&arg] = assignment[getVar(arg)]; });
92  return solution;
93  }
94 
95 protected:
96  /**
97  * A utility function mapping an AstArgument to its associated analysis variable.
98  *
99  * @param arg the AST argument to be mapped
100  * @return the analysis variable representing its associated value
101  */
102  AnalysisVar getVar(const AstArgument& arg) {
103  const auto* var = dynamic_cast<const AstVariable*>(&arg);
104  if (var == nullptr) {
105  // no mapping required
106  return AnalysisVar(arg);
107  }
108 
109  // filter through map => always take the same variable
110  auto res = variables.insert({var->getName(), AnalysisVar(var)}).first;
111  return res->second;
112  }
113 
114  /**
115  * A utility function mapping an AstArgument to its associated analysis variable.
116  *
117  * @param arg the AST argument to be mapped
118  * @return the analysis variable representing its associated value
119  */
120  AnalysisVar getVar(const AstArgument* arg) {
121  return getVar(*arg);
122  }
123 
124  /** Adds another constraint to the internally maintained list of constraints */
125  void addConstraint(const constraint_type& constraint) {
126  constraints.add(constraint);
127  }
128 
129  Assignment<AnalysisVar> assignment;
130 
131  /** The list of constraints making underlying this analysis */
132  Problem<AnalysisVar> constraints;
133 
134  /** A map mapping variables to unique instances to facilitate the unification of variables */
135  std::map<std::string, AnalysisVar> variables;
136 };
