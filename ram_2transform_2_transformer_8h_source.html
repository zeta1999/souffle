30 /**
31  * @Class RamTransformer
32  * @Brief abstract transformer class for a translation unit
33  *
34  * This is an abstract class to implement transformers. A
35  * transformer takes a translation unit and changes its
36  * state.
37  *
38  * Transformers can be composed using other transformers.
39  *
40  * For debugging purposes, a transformer has a name
41  * (this will show up in the debug report) and a
42  * protected method transform(), that performs the
43  * actual transformation.
44  *
45  * The method apply is used to call transform() and
46  * does the reporting of the debug information.
47  *
48  */
49 
50 class RamTransformer {
51 public:
52  virtual ~RamTransformer() = default;
53 
54  /**
55  * @Brief apply the transformer to a translation unit
56  * @Param translationUnit that will be transformed.
57  * @Return flag reporting whether the RAM program has changed
58  */
59  bool apply(RamTranslationUnit& translationUnit);
60 
61  /**
62  * @Brief get name of the transformer
63  */
64  virtual std::string getName() const = 0;
65 
66 protected:
67  /**
68  * @Brief transform the translation unit / used by apply
69  * @Param translationUnit that will be transformed.
70  * @Return flag reporting whether the RAM program has changed
71  */
72  virtual bool transform(RamTranslationUnit& translationUnit) = 0;
73 };

81 /**
82  * @Class RamTransformerSequence
83  * @Brief Composite sequence transformer
84  *
85  * A sequence of transformations is applied to a translation unit
86  * sequentially. The last transformation decides the outcome whether
87  * the code has been changed.
88  *
89  */
90 class RamTransformerSequence : public RamMetaTransformer {
91 public:
92  template <typename... Tfs>
93  RamTransformerSequence(std::unique_ptr<Tfs>&&... tf) : RamTransformerSequence() {
94  std::unique_ptr<RamTransformer> tmp[] = {std::move(tf)...};
95  for (auto& cur : tmp) {
96  transformers.emplace_back(std::move(cur));
97  }
98  for (const auto& cur : transformers) {
99  (void)cur;
100  assert(cur);
101  }
102  }
103  RamTransformerSequence() = default;
104  std::string getName() const override {
105  return "RamTransformerSequence";
106  }
107  bool transform(RamTranslationUnit& tU) override {
108  bool changed = false;
109  // The last transformer decides the status
110  // of the change flag.
111  // Note that for other semantics, new transformer
112  // sequence class needs to be introduced.
113  for (auto const& cur : transformers) {
114  changed = cur->apply(tU);
115  }
116  return changed;
117  }
118 
119 protected:
120  /** sequence of transformers */
121  std::vector<std::unique_ptr<RamTransformer>> transformers;
122 };
