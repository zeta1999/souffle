36  * Subclass of Literal that represents a negated atom, * e.g., !parent(x,y).
37  * A Negated atom occurs in a body of clause and cannot occur in a head of a clause.
38  */

41  AstNegation(Own<AstAtom> atom, SrcLocation loc = {})
42  : AstLiteral(std::move(loc)), atom(std::move(atom)) {}
43

47  }

51  }

56

59  }
