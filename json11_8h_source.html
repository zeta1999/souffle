1 /* json11
2  *
3  * json11 is a tiny JSON library for C++11, providing JSON parsing and serialization.
4  *
5  * The core object provided by the library is json11::Json. A Json object represents any JSON
6  * value: null, bool, number (int or double), string (std::string), array (std::vector), or
7  * object (std::map).
8  *
9  * Json objects act like values: they can be assigned, copied, moved, compared for equality or
10  * order, etc. There are also helper methods Json::dump, to serialize a Json to a string, and
11  * Json::parse (static) to parse a std::string as a Json object.
12  *
13  * Internally, the various types of Json object are represented by the JsonValue class
14  * hierarchy.
15  *
16  * A note on numbers - JSON specifies the syntax of number formatting but not its semantics,
17  * so some JSON implementations distinguish between integers and floating-point numbers, while
18  * some don't. In json11, we choose the latter. Because some JSON implementations (namely
19  * Javascript itself) treat all numbers as the same type, distinguishing the two leads
20  * to JSON that will be *silently* changed by a round-trip through those implementations.
21  * Dangerous! To avoid that risk, json11 stores all numbers as double internally, but also
22  * provides integer helpers.
23  *
24  * Fortunately, double-precision IEEE754 ('double') can precisely store any integer in the
25  * range +/-2^53, which includes every 'int' on most systems. (Timestamps often use int64
26  * or long long to avoid the Y2038K problem; a double storing microseconds since some epoch
27  * will be exact for +/- 275 years.)
28  */

308 static void dump(const std::string& value, std::string& out) {
309  out += '"';
310  for (size_t i = 0; i < value.length(); i++) {
311  const char ch = value[i];
312  if (ch == '\\') {
313  out += "\\\\";
314  } else if (ch == '"') {
315  out += "\\\"";
316  } else if (ch == '\b') {
317  out += "\\b";
318  } else if (ch == '\f') {
319  out += "\\f";
320  } else if (ch == '\n') {
321  out += "\\n";
322  } else if (ch == '\r') {
323  out += "\\r";
324  } else if (ch == '\t') {
325  out += "\\t";
326  } else if (static_cast<uint8_t>(ch) <= 0x1f) {
327  char buf[8];
328  snprintf(buf, sizeof buf, "\\u%04x", ch);
329  out += buf;
330  } else if (static_cast<uint8_t>(ch) == 0xe2 && static_cast<uint8_t>(value[i + 1]) == 0x80 &&
331  static_cast<uint8_t>(value[i + 2]) == 0xa8) {
332  out += "\\u2028";
333  i += 2;
334  } else if (static_cast<uint8_t>(ch) == 0xe2 && static_cast<uint8_t>(value[i + 1]) == 0x80 &&
335  static_cast<uint8_t>(value[i + 2]) == 0xa9) {
336  out += "\\u2029";
337  i += 2;
338  } else {
339  out += ch;
340  }
341  }
342  out += '"';
343 }

766  /* encode_utf8(pt, out)
767  *
768  * Encode pt as UTF-8 and add it to out.
769  */
770  void encode_utf8(long pt, std::string& out) {
771  if (pt < 0) {
772  return;
773  }
774 
775  if (pt < 0x80) {
776  out += static_cast<char>(pt);
777  } else if (pt < 0x800) {
778  out += static_cast<char>((pt >> 6) | 0xC0);
779  out += static_cast<char>((pt & 0x3F) | 0x80);
780  } else if (pt < 0x10000) {
781  out += static_cast<char>((pt >> 12) | 0xE0);
782  out += static_cast<char>(((pt >> 6) & 0x3F) | 0x80);
783  out += static_cast<char>((pt & 0x3F) | 0x80);
784  } else {
785  out += static_cast<char>((pt >> 18) | 0xF0);
786  out += static_cast<char>(((pt >> 12) & 0x3F) | 0x80);
787  out += static_cast<char>(((pt >> 6) & 0x3F) | 0x80);
788  out += static_cast<char>((pt & 0x3F) | 0x80);
789  }

792  /* parse_string()
793  *
794  * Parse a std::string, starting at the current position.
795  */
796  std::string parse_string() {
797  std::string out;
798  long last_escaped_codepoint = -1;
799  while (true) {
800  if (i == str.size()) return fail("unexpected end of input in std::string", "");
801 
802  char ch = str[i++];
803 
804  if (ch == '"') {
805  encode_utf8(last_escaped_codepoint, out);
806  return out;
807  }
808 
809  if (in_range(ch, 0, 0x1f)) {
810  return fail("unescaped " + esc(ch) + " in std::string", "");
811  }
812 
813  // The usual case: non-escaped characters
814  if (ch != '\\') {
815  encode_utf8(last_escaped_codepoint, out);
816  last_escaped_codepoint = -1;
817  out += ch;
818  continue;
819  }
820 
821  // Handle escapes
822  if (i == str.size()) {
823  return fail("unexpected end of input in std::string", "");
824  }
825 
826  ch = str[i++];
827 
828  if (ch == 'u') {
829  // Extract 4-byte escape sequence
830  std::string esc = str.substr(i, 4);
831  // Explicitly check length of the substring. The following loop
832  // relies on std::string returning the terminating NUL when
833  // accessing str[length]. Checking here reduces brittleness.
834  if (esc.length() < 4) {
835  return fail("bad \\u escape: " + esc, "");
836  }
837  for (size_t j = 0; j < 4; j++) {
838  if (!in_range(esc[j], 'a', 'f') && !in_range(esc[j], 'A', 'F') &&
839  !in_range(esc[j], '0', '9'))
840  return fail("bad \\u escape: " + esc, "");
841  }
842 
843  long codepoint = strtol(esc.data(), nullptr, 16);
844 
845  // JSON specifies that characters outside the BMP shall be encoded as a pair
846  // of 4-hex-digit \u escapes encoding their surrogate pair components. Check
847  // whether we're in the middle of such a beast: the previous codepoint was an
848  // escaped lead (high) surrogate, and this is a trail (low) surrogate.
849  if (in_range(last_escaped_codepoint, 0xD800, 0xDBFF) && in_range(codepoint, 0xDC00, 0xDFFF)) {
850  // Reassemble the two surrogate pairs into one astral-plane character, per
851  // the UTF-16 algorithm.
852  encode_utf8((((last_escaped_codepoint - 0xD800) << 10) | (codepoint - 0xDC00)) + 0x10000,
853  out);
854  last_escaped_codepoint = -1;
855  } else {
856  encode_utf8(last_escaped_codepoint, out);
857  last_escaped_codepoint = codepoint;
858  }
859 
860  i += 4;
861  continue;
862  }
863 
864  encode_utf8(last_escaped_codepoint, out);
865  last_escaped_codepoint = -1;
866 
867  if (ch == 'b') {
868  out += '\b';
869  } else if (ch == 'f') {
870  out += '\f';
871  } else if (ch == 'n') {
872  out += '\n';
873  } else if (ch == 'r') {
874  out += '\r';
875  } else if (ch == 't') {
876  out += '\t';
877  } else if (ch == '"' || ch == '\\' || ch == '/') {
878  out += ch;
879  } else {
880  return fail("invalid escape character " + esc(ch), "");
881  }
882  }
883  }

885  /* parse_number()
886  *
887  * Parse a double.
888  */
889  Json parse_number() {
890  size_t start_pos = i;
891 
892  if (str[i] == '-') {
893  i++;
894  }
895 
896  // Integer part
897  if (str[i] == '0') {
898  i++;
899  if (in_range(str[i], '0', '9')) {
900  return fail("leading 0s not permitted in numbers");
901  }
902  } else if (in_range(str[i], '1', '9')) {
903  i++;
904  while (in_range(str[i], '0', '9')) {
905  i++;
906  }
907  } else {
908  return fail("invalid " + esc(str[i]) + " in number");
909  }
910 
911  if (str[i] != '.' && str[i] != 'e' && str[i] != 'E' &&
912  (i - start_pos) <= static_cast<size_t>(std::numeric_limits<int>::digits10)) {
913  return std::atoll(str.c_str() + start_pos);
914  }
915 
916  // Decimal part
917  if (str[i] == '.') {
918  i++;
919  if (!in_range(str[i], '0', '9')) {
920  return fail("at least one digit required in fractional part");
921  }
922 
923  while (in_range(str[i], '0', '9')) {
924  i++;
925  }
926  }
927 
928  // Exponent part
929  if (str[i] == 'e' || str[i] == 'E') {
930  i++;
931 
932  if (str[i] == '+' || str[i] == '-') {
933  i++;
934  }
935 
936  if (!in_range(str[i], '0', '9')) {
937  return fail("at least one digit required in exponent");
938  }
939 
940  while (in_range(str[i], '0', '9')) {
941  i++;
942  }
943  }
944 
945  return std::strtod(str.c_str() + start_pos, nullptr);
946  }

964  /* parse_json()
965  *
966  * Parse a JSON object.
967  */
968  Json parse_json(int depth) {
969  if (depth > max_depth) {
970  return fail("exceeded maximum nesting depth");
971  }
972 
973  char ch = get_next_token();
974  if (failed) {
975  return Json();
976  }
977 
978  if (ch == '-' || (ch >= '0' && ch <= '9')) {

985  }
986 
987  if (ch == 'f') {

989  }
990 
991  if (ch == 'n') {
992  return expect("null", Json());
993  }
994 
995  if (ch == '"') {
996  return parse_string();
997  }
998 
999  if (ch == '{') {

1001  ch = get_next_token();
1002  if (ch == '}') {
1003  return data;
1004  }
1005

1008

1010  if (failed) {
1011  return Json();
1012  }
1013

1017  }

1020  if (failed) {
1021  return Json();
1022  }
1023 
1024  ch = get_next_token();
1025  if (ch == '}') {
1026  break;
1027  }
1028  if (ch != ',') {
1029  return fail("expected ',' in object, got " + esc(ch));
1030  }
1031 
1032  ch = get_next_token();
1033  }
1034  return data;
1035  }
1036 
1037  if (ch == '[') {
1038  std::vector<Json> data;
1039  ch = get_next_token();
1040  if (ch == ']') {
1041  return data;
1042  }
1043 
1044  while (true) {
1045  i--;
1046  data.push_back(parse_json(depth + 1));
1047  if (failed) {

1049  }
1050 
1051  ch = get_next_token();
1052  if (ch == ']') {
1053  break;
1054  }
1055  if (ch != ',') {
1056  return fail("expected ',' in list, got " + esc(ch));
1057  }

1061  }

1065  return fail("expected value, got " + esc(ch));
1066  }
