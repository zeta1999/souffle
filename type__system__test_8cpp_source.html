153 TEST(TypeSystem, GreatestCommonSubtype) {
154  TypeEnvironment env;
155 
156  const Type& N = env.getType("number");
157 
158  const Type& A = env.createType<SubsetType>("A", env.getType("number"));
159  const Type& B = env.createType<SubsetType>("B", env.getType("number"));
160  const Type& C = env.createType<SubsetType>("C", env.getType("symbol"));
161 
162  EXPECT_EQ("{number}", toString(getGreatestCommonSubtypes(N, N)));
163 
164  EXPECT_EQ("{A}", toString(getGreatestCommonSubtypes(A, A)));
165  EXPECT_EQ("{B}", toString(getGreatestCommonSubtypes(B, B)));
166  EXPECT_EQ("{C}", toString(getGreatestCommonSubtypes(C, C)));
167 
168  EXPECT_EQ("{}", toString(getGreatestCommonSubtypes(A, B)));
169  EXPECT_EQ("{}", toString(getGreatestCommonSubtypes(A, C)));
170  EXPECT_EQ("{}", toString(getGreatestCommonSubtypes(B, C)));
171 
172  EXPECT_EQ("{}", toString(getGreatestCommonSubtypes(A, B, C)));
173 
174  EXPECT_EQ("{A}", toString(getGreatestCommonSubtypes(A, N)));
175  EXPECT_EQ("{A}", toString(getGreatestCommonSubtypes(N, A)));
176 
177  EXPECT_EQ("{B}", toString(getGreatestCommonSubtypes(B, N)));
178  EXPECT_EQ("{B}", toString(getGreatestCommonSubtypes(N, B)));
179 
180  EXPECT_EQ("{}", toString(getGreatestCommonSubtypes(C, N)));
181  EXPECT_EQ("{}", toString(getGreatestCommonSubtypes(N, C)));
182 
183  // // bring in unions
184 
185  auto& U = env.createType<UnionType>("U");
186  auto& S = env.createType<UnionType>("S");
187 
188  U.setElements(toVector(&A));
189  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(U, S)));
190 
191  S.setElements(toVector(&A));
192  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, S)));
193 
194  U.setElements(toVector(&A, &B));
195  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(U, S)));
196  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(U, S, N)));
197 
198  S.setElements(toVector(&A, &B));
199  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, S)));
200  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, S, N)));
201 
202  // bring in a union of unions
203  auto& R = env.createType<UnionType>("R");
204 
205  EXPECT_EQ("{R}", toString(getGreatestCommonSubtypes(U, R)));
206  EXPECT_EQ("{R}", toString(getGreatestCommonSubtypes(S, R)));
207 
208  EXPECT_EQ("{R}", toString(getGreatestCommonSubtypes(U, R, N)));
209  EXPECT_EQ("{R}", toString(getGreatestCommonSubtypes(S, R, N)));
210 
211  R.setElements(toVector(static_cast<const Type*>(&U))); // R = U = S
212 
213  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, R)));
214  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(S, R)));
215 
216  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, R, N)));
217  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(S, R, N)));
218 
219  R.setElements(toVector(static_cast<const Type*>(&U), static_cast<const Type*>(&S)));
220 
221  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, R)));
222  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(S, R)));
223 
224  EXPECT_EQ("{U}", toString(getGreatestCommonSubtypes(U, R, N)));
225  EXPECT_EQ("{S}", toString(getGreatestCommonSubtypes(S, R, N)));
226 }
