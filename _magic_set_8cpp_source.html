300 // Given a set of relations R, add in all relations that use one of these
301 // relations in their clauses. Repeat until a fixed point is reached.
302 std::set<AstQualifiedName> addBackwardDependencies(
303  const AstProgram* program, std::set<AstQualifiedName> relations) {
304  bool relationsAdded = false;
305  std::set<AstQualifiedName> result;
306 
307  for (AstQualifiedName relName : relations) {
308  // Add the relation itself
309  result.insert(relName);
310  }
311 
312  // Add in all relations that need to use an ignored relation
313  for (AstRelation* rel : program->getRelations()) {
314  for (AstClause* clause : getClauses(*program, *rel)) {
315  AstQualifiedName clauseHeadName = clause->getHead()->getQualifiedName();
316  if (!contains(relations, clauseHeadName)) {
317  // Clause hasn't been added yet, so check if it needs to be added
318  visitDepthFirst(*clause, [&](const AstAtom& subatom) {
319  AstQualifiedName atomName = subatom.getQualifiedName();
320  if (contains(relations, atomName)) {
321  // Clause uses one of the given relations
322  result.insert(clauseHeadName);
323 
324  // Clause name hasn't been seen yet, so fixed point not reached
325  relationsAdded = true;
326  }
327  });
328  }
329  }
330  }
331 
332  if (relationsAdded) {
333  // Keep going until we reach a fixed point
334  return addBackwardDependencies(program, result);
335  } else {
336  return result;
337  }
338 }

375 std::set<AstQualifiedName> addIgnoredRelations(
376  const AstProgram* program, std::set<AstQualifiedName> relations) {

378  std::vector<std::string> specifiedRelations = splitString(Global::config().get("magic-transform"), ',');
379

381  if (contains(specifiedRelations, "*")) {

386  std::set<AstQualifiedName> targetRelations;
387  for (AstRelation* rel : program->getRelations()) {
388  std::string mainName = rel->getQualifiedName().getQualifiers()[0];
389  if (contains(specifiedRelations, mainName)) {
390  targetRelations.insert(rel->getQualifiedName());

392  }

395  // if we want to magic transform 'a', then we also have to magic transform

403  retVal.insert(rel->getQualifiedName());

405  }

414 // reorders a vector of integers to fit the clause atom-reordering function
415 std::vector<unsigned int> reorderOrdering(std::vector<unsigned int> order) {
416  // when the adornment is computed, the atoms are numbered based on
417  // which was chosen by the SIPS first - this is the 'order' vector.
418  // want to reorder clause atoms so that the atom labelled 0 is first, and so on.
419  // i.e. order[i] denotes where labels[i] should move
420  // e.g.: [a, b, c] with label [1, 2, 0] should become [c, a, b]
421 
422  // the atom reordering function for clauses, however, moves it as follows:
423  // [a, b, c] with label [1, 2, 0] becomes [b, c, a]
424  // i.e. labels[i] goes to the position of i in the order vector
425 
426  // this function reorders the ordering scheme to match the second type
427  std::vector<unsigned int> neworder(order.size());
428  for (size_t i = 0; i < order.size(); i++) {
429  // this took embarrassingly long to figure out
430  neworder[order[i]] = i;
431  }
432  return neworder;
433 }

663  for (AstClause* clause : getClauses(*program, *rel)) {
664  std::set<AstBinaryConstraint*> constraints;
665  M update(compositeBindings, constraints, changeCount);
666  clause->apply(update);
667 
668  changeCount = update.getChangeCount();
669 
670  for (AstBinaryConstraint* constraint : constraints) {
671  clause->addToBody(std::unique_ptr<AstBinaryConstraint>(constraint));
672  }
673  }
674  }
675 
676  return compositeBindings;

679 // runs the adornment algorithm on an input program
680 // Adornment algorithm:
681 
682 // Let P be the set of all adorned predicates (initially empty)
683 // Let D' be the set of all adorned clauses (initially empty)
684 // Let S be the set of all seen predicate adornments
685 
686 // Get the program
687 // Get the query
688 // Adorn the query based on boundness, and add it to P and S
689 // While P is not empty
690 // -- Pop the first atom out, call it R^c, where c is the adornment
691 // -- For every clause Q defining R:
692 // -- -- Adorn Q using R^c based on the SIPS chosen
693 // -- -- Add the adorned clause to D'
694 // -- -- If the body of the adorned clause contains an
695 // unseen predicate adornment, add it to S and P
696 
697 // Output: D' [the set of all adorned clauses]
698 void Adornment::run(const AstTranslationUnit& translationUnit) {
699  // -------------
700  // --- Setup ---
701  // -------------
702  const AstProgram* program = translationUnit.getProgram();
703  auto* ioTypes = translationUnit.getAnalysis<IOType>();
704 
705  // normalises and tracks bindings of composite arguments (namely records and functors)
706  BindingStore compositeBindings = bindComposites(program);
707 
708  // set up IDB/EDB and the output queries
709  std::vector<AstQualifiedName> outputQueries;
710  std::vector<std::vector<AdornedClause>> adornedProgram;
711 
712  // sort out the relations in the program into EDB/IDB and find computed relations
713  for (AstRelation* rel : program->getRelations()) {
714  AstQualifiedName relName = rel->getQualifiedName();
715 
716  // find computed relations for the topdown part
717  if (ioTypes->isOutput(rel)) {
718  outputQueries.push_back(rel->getQualifiedName());
719  // add relation to adornment
720  adornmentRelations.push_back(rel->getQualifiedName());
721  }
722 
723  // check whether edb or idb
724  bool is_edb = true;
725  for (AstClause* clause : getClauses(*program, *rel)) {
726  if (!isFact(*clause)) {
727  is_edb = false;
728  break;
729  }
730  }
731 
732  if (is_edb) {
733  adornmentEdb.insert(relName);
734  } else {
735  adornmentIdb.insert(relName);
736  }
737  }
738 
739  // find all negated literals
740  visitDepthFirst(*program, [&](const AstNegation& negation) {
741  negatedAtoms.insert(negation.getAtom()->getQualifiedName());
742  });
743 
744  // add the relations needed for negated relations to be computed
745  negatedAtoms = addForwardDependencies(program, negatedAtoms);
746 
747  // find atoms that should be ignored
748  for (AstRelation* rel : program->getRelations()) {
749  for (AstClause* clause : getClauses(*program, *rel)) {
750  // ignore atoms that have rules containing aggregators
751  if (containsAggregators(clause)) {
752  ignoredAtoms.insert(clause->getHead()->getQualifiedName());
753  }
754 
755  // ignore all atoms used inside an aggregator within the clause
756  ignoredAtoms = addAggregators(clause, ignoredAtoms);
757  }
758  }
759 
760  // find atoms that should be ignored based on magic-transform option
761  ignoredAtoms = addIgnoredRelations(program, ignoredAtoms);
762 
763  // if a relation is ignored, then all the atoms in its bodies need to be ignored
764  ignoredAtoms = addForwardDependencies(program, ignoredAtoms);
765 
766  // -----------------
767  // --- Adornment ---
768  // -----------------
769  // begin adornment algorithm
770  // adornment is performed for each output query separately
771  for (auto outputQuery : outputQueries) {
772  std::vector<AdornedPredicate> currentPredicates;
773  std::set<AdornedPredicate> seenPredicates;
774  std::vector<AdornedClause> adornedClauses;
775 
776  // create an adorned predicate of the form outputName_ff..f
777  size_t arity = getRelation(*program, outputQuery)->getArity();
778  std::string frepeat = std::string(arity, 'f'); // #fs = #args
779  AdornedPredicate outputPredicate(outputQuery, frepeat);
780  currentPredicates.push_back(outputPredicate);
781  seenPredicates.insert(outputPredicate);
782 
783  // keep going through the remaining predicates that need to be adorned
784  while (!currentPredicates.empty()) {
785  // pop out the first element
786  AdornedPredicate currPredicate = currentPredicates[0];
787  currentPredicates.erase(currentPredicates.begin());
788 
789  // don't bother adorning ignored predicates
790  if (contains(ignoredAtoms, currPredicate.getQualifiedName())) {
791  continue;
792  }
793 
794  // go through and adorn all IDB clauses defining the relation
795  AstRelation* rel = getRelation(*program, currPredicate.getQualifiedName());
796  for (AstClause* clause : getClauses(*program, *rel)) {
797  if (isFact(*clause)) {
798  continue;
799  }
800 
801  size_t numAtoms = getBodyLiterals<AstAtom>(*clause).size();
802  std::vector<std::string> clauseAtomAdornments(numAtoms);
803  std::vector<unsigned int> ordering(numAtoms);
804  std::set<std::string> boundArgs;
805 
806  // mark all bound arguments in the head as bound
807  AstAtom* clauseHead = clause->getHead();
808  std::string headAdornment = currPredicate.getAdornment();
809  std::vector<AstArgument*> headArguments = clauseHead->getArguments();
810 
811  for (size_t argnum = 0; argnum < headArguments.size(); argnum++) {
812  if (headAdornment[argnum] == 'b') {
813  std::string name = getString(headArguments[argnum]);
814  boundArgs.insert(name);
815  }
816  }
817 
818  // mark all bound arguments from the body
819  for (const auto* bc : getBodyLiterals<AstBinaryConstraint>(*clause)) {
820  BinaryConstraintOp op = bc->getOperator();
821  if (!isEqConstraint(op)) {
822  continue;
823  }
824 
825  // have an equality constraint
826  AstArgument* lhs = bc->getLHS();
827  AstArgument* rhs = bc->getRHS();
828  if (isBindingConstraint(lhs, rhs, boundArgs)) {
829  boundArgs.insert(getString(lhs));
830  }
831  if (isBindingConstraint(rhs, lhs, boundArgs)) {
832  boundArgs.insert(getString(rhs));
833  }
834  }
835 
836  std::vector<AstAtom*> atoms = getBodyLiterals<AstAtom>(*clause);
837  int atomsAdorned = 0;
838  int atomsTotal = atoms.size();
839 
840  while (atomsAdorned < atomsTotal) {
841  // get the next body atom to adorn based on our SIPS
842  int currIndex = getNextAtomSIPS(atoms, boundArgs, adornmentEdb, compositeBindings);
843  AstAtom* currAtom = atoms[currIndex];
844  AstQualifiedName atomName = currAtom->getQualifiedName();
845 
846  // compute the adornment pattern of this atom, and
847  // add all its arguments to the list of bound args
848  std::pair<std::string, std::set<std::string>> result =
849  bindArguments(currAtom, boundArgs, compositeBindings);
850  std::string atomAdornment = result.first;
851  boundArgs = result.second;
852 
853  // check if we've already dealt with this adornment before
854  if (!contains(seenPredicates, atomName, atomAdornment)) {
855  // not seen before, so push it onto the computation list
856  // and mark it as seen
857  currentPredicates.push_back(AdornedPredicate(atomName, atomAdornment));
858  seenPredicates.insert(AdornedPredicate(atomName, atomAdornment));
859  }
860 
861  clauseAtomAdornments[currIndex] = atomAdornment; // store the adornment
862  ordering[currIndex] = atomsAdorned; // mark what atom number this is
863  atoms[currIndex] = nullptr; // mark as done
864 
865  atomsAdorned++;
866  }
867 
868  // adornment of this clause is complete - add it to the list of
869  // adorned clauses
870  adornedClauses.push_back(
871  AdornedClause(clause, headAdornment, clauseAtomAdornments, ordering));
872  }
873  }
874 
875  // add the list of adorned clauses matching the current output relation
876  adornmentClauses.push_back(adornedClauses);
877  }
878 
879  this->bindings = std::move(compositeBindings);
880 }

900 // transforms the program so that a relation is either purely made up of
901 // facts or has no facts at all
902 void separateDBs(AstProgram* program) {
903  for (AstRelation* relation : program->getRelations()) {
904  AstQualifiedName relName = relation->getQualifiedName();
905 
906  // determine whether the relation fits into the EDB, IDB, or both
907  bool is_edb = false;
908  bool is_idb = false;
909 
910  for (AstClause* clause : getClauses(*program, *relation)) {
911  if (isFact(*clause)) {
912  is_edb = true;
913  } else {
914  is_idb = true;
915  }
916  if (is_edb && is_idb) {
917  break;
918  }
919  }
920 
921  if (is_edb && is_idb) {
922  // relation is part of EDB and IDB
923 
924  // move all the relation's facts to a new relation with a unique name
925  std::string newEdbName = getNextEdbName(program);
926  AstRelation* newEdbRel = createNewRelation(relation, newEdbName);
927  program->addRelation(std::unique_ptr<AstRelation>(newEdbRel));
928 
929  // find all facts for the relation
930  for (AstClause* clause : getClauses(*program, *relation)) {
931  if (isFact(*clause)) {
932  // clause is fact - add it to the new EDB relation
933  AstClause* newEdbClause = clause->clone();
934  newEdbClause->getHead()->setQualifiedName(newEdbName);
935  program->addClause(std::unique_ptr<AstClause>(newEdbClause));
936  }
937  }
938 
939  // add a rule to the old relation that relates it to the new relation
940  auto* newIdbClause = new AstClause();
941  newIdbClause->setSrcLoc(nextSrcLoc(relation->getSrcLoc()));
942 
943  // oldname(arg1...argn) :- newname(arg1...argn)
944  auto* headAtom = new AstAtom(relName);
945  auto* bodyAtom = new AstAtom(newEdbName);
946 
947  size_t numargs = relation->getArity();
948  for (size_t j = 0; j < numargs; j++) {
949  std::stringstream argName;
950  argName.str("");
951  argName << "arg" << j;
952  headAtom->addArgument(std::make_unique<AstVariable>(argName.str()));
953  bodyAtom->addArgument(std::make_unique<AstVariable>(argName.str()));
954  }
955 
956  newIdbClause->setHead(std::unique_ptr<AstAtom>(headAtom));
957  newIdbClause->addToBody(std::unique_ptr<AstAtom>(bodyAtom));
958 
959  program->addClause(std::unique_ptr<AstClause>(newIdbClause));
960  }
961  }
962 }

1041 // Magic Set Transformation
1042 // STEPS:
1043 // For all output relations G:
1044 // -- Get the adornment S for this clause
1045 // -- Add to S the set of magic rules for all clauses in S:
1046 // -- -- For each clause C = A^a :- A1^a1, A2^a2, ..., An^an
1047 // -- -- -- For each IDB literal A_i in the body of C
1048 // -- -- -- -- Add mag(Ai^ai) :- mag(A^a), A1^a1, ..., Ai-1^ai-1 to the program
1049 // -- For all clauses H :- T in S:
1050 // -- -- Replace the clause with H :- mag(H), T.
1051 // -- Add the fact m_G_f...f to S
1052 // Remove all old idb rules
1053 bool MagicSetTransformer::transform(AstTranslationUnit& translationUnit) {
1054  AstProgram* program = translationUnit.getProgram();
1055  auto* ioTypes = translationUnit.getAnalysis<IOType>();
1056 
1057  separateDBs(program); // make EDB int IDB = empty
1058 
1059  auto* adornment = translationUnit.getAnalysis<Adornment>(); // perform adornment
1060  const BindingStore& compositeBindings = adornment->getBindings();
1061 
1062  // edb/idb handling
1063  std::vector<std::vector<AdornedClause>> allAdornedClauses = adornment->getAdornedClauses();
1064  std::set<AstQualifiedName> negatedAtoms = adornment->getNegatedAtoms();
1065  std::set<AstQualifiedName> ignoredAtoms = adornment->getIgnoredAtoms();
1066  std::set<AstQualifiedName> oldIdb = adornment->getIDB();
1067  std::set<AstQualifiedName> newIdb;
1068 
1069  // additions
1070  std::vector<AstQualifiedName> newQueryNames;
1071  std::vector<AstClause*> newClauses;
1072 
1073  // output handling
1074  std::vector<AstQualifiedName> outputQueries = adornment->getRelations();
1075 
1076  // ignore negated atoms
1077  for (AstQualifiedName relation : negatedAtoms) {
1078  ignoredAtoms.insert(relation);
1079  }
1080 
1081  // perform magic set algorithm for each output
1082  for (size_t querynum = 0; querynum < outputQueries.size(); querynum++) {
1083  AstQualifiedName outputQuery = outputQueries[querynum];
1084  std::vector<AdornedClause> adornedClauses = allAdornedClauses[querynum];
1085  AstRelation* originalOutputRelation = getRelation(*program, outputQuery);
1086 
1087  // add a relation for the output query
1088  // mN_outputname_ff...f()
1089  auto* magicOutputRelation = new AstRelation();
1090  std::string frepeat = std::string(originalOutputRelation->getArity(), 'f');
1091  AstQualifiedName magicOutputName =
1092  createMagicIdentifier(createAdornedIdentifier(outputQuery, frepeat), querynum);
1093  magicOutputRelation->setQualifiedName(magicOutputName);
1094  newQueryNames.push_back(magicOutputName);
1095 
1096  // add the new relation to the program
1097  program->addRelation(std::unique_ptr<AstRelation>(magicOutputRelation));
1098 
1099  // add an empty fact to the program
1100  // i.e. mN_outputname_ff...f().
1101  auto* outputFact = new AstClause();
1102  outputFact->setSrcLoc(nextSrcLoc(originalOutputRelation->getSrcLoc()));
1103  outputFact->setHead(std::make_unique<AstAtom>(magicOutputName));
1104  program->addClause(std::unique_ptr<AstClause>(outputFact));
1105 
1106  // perform the magic transformation based on the adornment for this output query
1107  for (AdornedClause adornedClause : adornedClauses) {
1108  AstClause* clause = adornedClause.getClause();
1109  AstQualifiedName originalName = clause->getHead()->getQualifiedName();
1110 
1111  // dont perform the magic transformation on ignored relations
1112  if (contains(ignoredAtoms, originalName)) {
1113  continue;
1114  }
1115 
1116  // find the adorned version of this relation
1117  std::string headAdornment = adornedClause.getHeadAdornment();
1118  AstQualifiedName newRelName = createAdornedIdentifier(originalName, headAdornment);
1119  AstRelation* adornedRelation = getRelation(*program, newRelName);
1120 
1121  // check if adorned relation already created previously
1122  if (adornedRelation == nullptr) {
1123  // adorned relation not created yet, so
1124  // create the relation with the new adornment
1125  AstRelation* originalRelation = getRelation(*program, originalName);
1126  AstRelation* newRelation = createNewRelation(originalRelation, newRelName);
1127 
1128  // add the created adorned relation to the program
1129  program->addRelation(std::unique_ptr<AstRelation>(newRelation));
1130 
1131  // copy over input directives to new adorned relation
1132  // also - update input directives to correctly use default fact file names
1133  if (ioTypes->isInput(originalRelation)) {
1134  for (AstIO* io : program->getIOs()) {
1135  if (io->getQualifiedName() != originalName || io->getType() != AstIoType::input) {
1136  continue;
1137  }
1138  io->setQualifiedName(newRelName);
1139  if (!io->hasDirective("IO")) {
1140  io->addDirective("IO", "file");
1141  }
1142  if (io->getDirective("IO") == "file" && !io->hasDirective("filename")) {
1143  io->addDirective("filename", originalName.getQualifiers()[0] + ".facts");
1144  }
1145  }
1146  }
1147  adornedRelation = newRelation;
1148  }
1149 
1150  // create the adorned version of this clause
1151  AstClause* newClause = clause->clone();
1152  newClause->getHead()->setQualifiedName(newRelName);
1153  // reorder atoms based on SIPS ordering
1154  AstClause* tmp = reorderAtoms(newClause, reorderOrdering(adornedClause.getOrdering()));
1155  delete newClause;
1156  newClause = tmp;
1157 
1158  // get corresponding adornments for each body atom
1159  std::vector<std::string> bodyAdornment =
1160  reorderAdornment(adornedClause.getBodyAdornment(), adornedClause.getOrdering());
1161 
1162  // set the name of each IDB pred in the clause to be the adorned version
1163  int atomsSeen = 0;
1164  for (AstLiteral* lit : newClause->getBodyLiterals()) {
1165  if (dynamic_cast<AstAtom*>(lit) != nullptr) {
1166  auto* bodyAtom = dynamic_cast<AstAtom*>(lit);
1167  AstQualifiedName atomName = bodyAtom->getQualifiedName();
1168  // note that all atoms in the original clause were adorned,
1169  // but only the IDB atom adornments should be added here
1170  if (contains(oldIdb, atomName)) {
1171  if (!contains(ignoredAtoms, atomName)) {
1172  // ignored atoms should not be changed
1173  AstQualifiedName newAtomName =
1174  createAdornedIdentifier(atomName, bodyAdornment[atomsSeen]);
1175  bodyAtom->setQualifiedName(newAtomName);
1176  newIdb.insert(newAtomName);
1177  } else {
1178  newIdb.insert(atomName);
1179  }
1180  }
1181  atomsSeen++;
1182  }
1183  }
1184 
1185  // Add the set of magic rules for this clause C = A^a :- A1^a1, A2^a2, ..., An^an
1186  // -- For each clause C = A^a :- A1^a1, A2^a2, ..., An^an
1187  // -- -- For each IDB literal A_i in the body of C
1188  // -- -- -- Add mag(Ai^ai) :- mag(A^a), A1^a1, ..., Ai-1^ai-1 to the program
1189  std::vector<AstAtom*> body = getBodyLiterals<AstAtom>(*newClause);
1190  for (size_t i = 0; i < body.size(); i++) {
1191  AstAtom* currentLiteral = body[i];
1192 
1193  // only care about atoms in the body
1194  if (dynamic_cast<AstAtom*>(currentLiteral) != nullptr) {
1195  auto* atom = dynamic_cast<AstAtom*>(currentLiteral);
1196  AstQualifiedName atomName = atom->getQualifiedName();
1197 
1198  // only IDB atoms that are not being ignored matter
1199  if (contains(newIdb, atomName) && !contains(ignoredAtoms, atomName)) {
1200  std::string currAdornment = bodyAdornment[i];
1201 
1202  // generate the name of the magic version of this adorned literal
1203  AstQualifiedName newAtomName = createMagicIdentifier(atomName, querynum);
1204 
1205  // if the magic version does not exist, create it
1206  if (getRelation(*program, newAtomName) == nullptr) {
1207  auto* magicRelation = new AstRelation();
1208  magicRelation->setQualifiedName(newAtomName);
1209 
1210  // find out the original name of the relation (pre-adornment)
1211  std::string baseAtomName = atomName.getQualifiers()[0];
1212  int endpt = getEndpoint(baseAtomName);
1213  AstQualifiedName originalRelationName = createSubIdentifier(
1214  atomName, 0, endpt - 1); // get rid of the extra + at the end
1215  AstRelation* originalRelation = getRelation(*program, originalRelationName);
1216 
1217  // copy over the (bound) attributes from the original relation
1218  int argcount = 0;
1219  for (AstAttribute* attr : originalRelation->getAttributes()) {
1220  if (currAdornment[argcount] == 'b') {
1221  magicRelation->addAttribute(souffle::clone(attr));
1222  }
1223  argcount++;
1224  }
1225 
1226  // copy over internal representation
1227  magicRelation->setRepresentation(originalRelation->getRepresentation());
1228 
1229  // add the new magic relation to the program
1230  program->addRelation(std::unique_ptr<AstRelation>(magicRelation));
1231  }
1232 
1233  // start setting up the magic rule
1234  auto* magicClause = new AstClause();
1235  magicClause->setSrcLoc(nextSrcLoc(atom->getSrcLoc()));
