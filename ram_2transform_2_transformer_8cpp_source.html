32 bool RamTransformer::apply(RamTranslationUnit& translationUnit) {
33  const bool debug = Global::config().has("debug-report");

35  std::string ramProgStrOld = debug ? toString(translationUnit.getProgram()) : "";

40  auto end = std::chrono::high_resolution_clock::now();
41

44  translationUnit.invalidateAnalyses();
45  }
46 
47  // print runtime & change info for transformer in verbose mode
48  if (verbose && (nullptr == dynamic_cast<RamMetaTransformer*>(this))) {
49  std::string changedString = changed ? "changed" : "unchanged";
50  std::cout << getName() << " time: " << std::chrono::duration<double>(end - start).count() << "sec ["
51  << changedString << "]" << std::endl;
52  }
53 
54  // print program after transformation in debug report
55  if (debug) {
56  translationUnit.getDebugReport().startSection();
57  if (changed) {
58  translationUnit.getDebugReport().addCodeSection(getName(), "RAM Program after " + getName(),
59  "ram", ramProgStrOld, toString(translationUnit.getProgram()));
60 
61  translationUnit.getDebugReport().endSection(getName(), getName());
62  } else {
63  translationUnit.getDebugReport().endSection(getName(), getName() + " " + " (unchanged)");
64  }
65  }
66 
67  // abort evaluation of the program if errors were encountered
68  if (translationUnit.getErrorReport().getNumErrors() != 0) {
69  std::cerr << translationUnit.getErrorReport();

73  }
74

76 }
