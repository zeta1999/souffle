197 TEST(AstProgram, RemoveClause) {
198  auto atom = std::make_unique<AstAtom>("B");
199  atom->addArgument(std::make_unique<AstVariable>("x"));
200  auto sum = std::make_unique<AstAggregator>(AggregateOp::SUM, std::make_unique<AstVariable>("x"));
201  std::vector<std::unique_ptr<AstLiteral>> body;
202  body.push_back(std::move(atom));
203  sum->setBody(std::move(body));
204 
205  auto tu1 = makeATU(".decl A,B(x:number) \n A(sum x : B(x)).");
206  auto clause = makeClause("A", std::move(sum));
207 
208  tu1->getProgram()->removeClause(clause.get());
209  auto tu2 = makeATU(".decl A,B(x:number)");
210  EXPECT_EQ(*tu1->getProgram(), *tu2->getProgram());
211 }
