31 void IOType::run(const AstTranslationUnit& translationUnit) {
32  const AstProgram& program = *translationUnit.getProgram();
33  visitDepthFirst(program, [&](const AstIO& io) {
34  auto* relation = getRelation(program, io.getQualifiedName());
35  if (relation == nullptr) {
36  return;
37  }
38  switch (io.getType()) {
39  case AstIoType::input: inputRelations.insert(relation); break;
40  case AstIoType::output: outputRelations.insert(relation); break;
41  case AstIoType::printsize:
42  printSizeRelations.insert(relation);
43  outputRelations.insert(relation);
44  break;
45  }
46  });
47 }
