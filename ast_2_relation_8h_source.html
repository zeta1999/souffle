40 /*!
41  * @class Relation
42  * @brief Intermediate representation of a datalog relation
43  *
44  * A relation has a name, types of its arguments, qualifier type,
45  * and dependencies to other relations.
46  *
47  */
48 class AstRelation : public AstNode {
49 public:
50  AstRelation() = default;
51  AstRelation(AstQualifiedName name, SrcLocation loc = {}) : name(std::move(name)) {
52  setSrcLoc(std::move(loc));
53  }
54 
55  /** get qualified relation name */
56  const AstQualifiedName& getQualifiedName() const {
57  return name;
58  }
59 
60  /** Set name for this relation */
61  void setQualifiedName(AstQualifiedName n) {
62  name = std::move(n);
63  }
64 
65  /** Add a new used type to this relation */
66  void addAttribute(Own<AstAttribute> attr) {
67  assert(attr && "Undefined attribute");
68  attributes.push_back(std::move(attr));
69  }
70 
71  /** Return the arity of this relation */
72  size_t getArity() const {
73  return attributes.size();
74  }
75 
76  /** Set relation attributes */
77  void setAttributes(VecOwn<AstAttribute> attrs) {
78  attributes = std::move(attrs);
79  }
80 
81  /** Get relation attributes */
82  std::vector<AstAttribute*> getAttributes() const {
83  return toPtrVector(attributes);
84  }
85 
86  /** Get relation qualifiers */
87  const std::set<RelationQualifier>& getQualifiers() const {
88  return qualifiers;
89  }
90 
91  /** Add qualifier to this relation */
92  void addQualifier(RelationQualifier q) {
93  qualifiers.insert(q);
94  }
95 
96  /** Remove qualifier from this relation */
97  void removeQualifier(RelationQualifier q) {
98  qualifiers.erase(q);
99  }
100 
101  /** Get relation representation */
102  RelationRepresentation getRepresentation() const {
103  return representation;
104  }
105 
106  /** Set relation representation */
107  void setRepresentation(RelationRepresentation representation) {
108  this->representation = representation;
109  }
110 
111  /** check for a relation qualifier */
112  bool hasQualifier(RelationQualifier q) const {
113  return qualifiers.find(q) != qualifiers.end();
114  }
115 
116  AstRelation* clone() const override {
117  auto res = new AstRelation(name, getSrcLoc());
118  res->attributes = souffle::clone(attributes);
119  res->qualifiers = qualifiers;
120  res->representation = representation;
121  return res;
122  }
123 
124  void apply(const AstNodeMapper& map) override {
125  for (auto& cur : attributes) {
126  cur = map(std::move(cur));
127  }
128  }
129 
130  std::vector<const AstNode*> getChildNodes() const override {
131  std::vector<const AstNode*> res;
132  for (const auto& cur : attributes) {
133  res.push_back(cur.get());
134  }
135  return res;
136  }
137 
138 protected:
139  void print(std::ostream& os) const override {
140  os << ".decl " << getQualifiedName() << "(" << join(attributes, ", ") << ")" << join(qualifiers, " ")
141  << " " << representation;
142  }
143 
144  bool equal(const AstNode& node) const override {
145  const auto& other = static_cast<const AstRelation&>(node);
146  return name == other.name && equal_targets(attributes, other.attributes);
147  }
148 
149  /** Name of relation */
150  AstQualifiedName name;
151 
152  /** Attributes of the relation */
153  VecOwn<AstAttribute> attributes;
154 
155  /** Qualifiers of relation */
156  std::set<RelationQualifier> qualifiers;
157 
158  /** Datastructure to use for this relation */
159  RelationRepresentation representation{RelationRepresentation::DEFAULT};
160 };

162 /**
163  * Lexicographical order for AstRelation
164  * using the qualified name as an ordering criteria.
165  */
166 struct AstNameComparison {
167  bool operator()(const AstRelation* x, const AstRelation* y) const {
168  if (x != nullptr && y != nullptr) {
169  return x->getQualifiedName() < y->getQualifiedName();
170  }
171  return y != nullptr;
172  }
173 };
