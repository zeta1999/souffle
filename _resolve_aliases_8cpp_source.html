55 /**
56  * A utility class for the unification process required to eliminate aliases.
57  * A substitution maps variables to terms and can be applied as a transformation
58  * to AstArguments.
59  */
60 class Substitution {
61  // map type used for internally storing var->term mappings
62  // - note: variables are identified by their names
63  using map_t = std::map<std::string, std::unique_ptr<AstArgument>>;
64 
65  // the mapping of variables to terms
66  map_t varToTerm;
67 
68 public:
69  // -- Constructors/Destructors --
70 
71  Substitution() = default;
72 
73  Substitution(const std::string& var, const AstArgument* arg) {
74  varToTerm.insert(std::make_pair(var, souffle::clone(arg)));
75  }
76 
77  ~Substitution() = default;
78 
79  /**
80  * Applies this substitution to the given argument and returns a pointer
81  * to the modified argument.
82  *
83  * @param node the node to be transformed
84  * @return a pointer to the modified or replaced node
85  */
86  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const {
87  // create a substitution mapper
88  struct M : public AstNodeMapper {
89  const map_t& map;
90 
91  M(const map_t& map) : map(map) {}
92 
93  using AstNodeMapper::operator();
94 
95  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
96  // see whether it is a variable to be substituted
97  if (auto var = dynamic_cast<AstVariable*>(node.get())) {
98  auto pos = map.find(var->getName());
99  if (pos != map.end()) {
100  return souffle::clone(pos->second);
101  }
102  }
103 
104  // otherwise, apply the mapper recursively
105  node->apply(*this);
106  return node;
107  }
108  };
109 
110  // apply the mapper
111  return M(varToTerm)(std::move(node));
112  }
113 
114  /**
115  * A generic, type consistent wrapper of the transformation operation above.
116  */
117  template <typename T>
118  std::unique_ptr<T> operator()(std::unique_ptr<T> node) const {
119  std::unique_ptr<AstNode> resPtr = (*this)(std::unique_ptr<AstNode>(node.release()));
120  assert(nullptr != dynamic_cast<T*>(resPtr.get()) && "Invalid node type mapping.");
121  return std::unique_ptr<T>(dynamic_cast<T*>(resPtr.release()));
122  }
123 
124  /**
125  * Appends the given substitution s to this substitution t such that the
126  * result t' is s composed with t (s o t).
127  * i.e.,
128  * - if t(x) = y, then t'(x) = s(y)
129  * - if s(x) = y, and x is not mapped by t, then t'(x) = y
130  */
131  void append(const Substitution& sub) {
132  // apply substitution on the rhs of all current mappings
133  for (auto& pair : varToTerm) {
134  pair.second = sub(std::move(pair.second));
135  }
136 
137  // append unseen variables to the end
138  for (const auto& pair : sub.varToTerm) {
139  if (varToTerm.find(pair.first) == varToTerm.end()) {
140  // not seen yet, add it in
141  varToTerm.insert(std::make_pair(pair.first, souffle::clone(pair.second)));
142  }
143  }
144  }
145 
146  /** A print function (for debugging) */
147  void print(std::ostream& out) const {
148  out << "{"
149  << join(varToTerm, ",",
150  [](std::ostream& out,
151  const std::pair<const std::string, std::unique_ptr<AstArgument>>& cur) {
152  out << cur.first << " -> " << *cur.second;
153  })
154  << "}";
155  }
156 
157  friend std::ostream& operator<<(std::ostream& out, const Substitution& s) __attribute__((unused)) {
158  s.print(out);
159  return out;
160  }
161 };

208 std::unique_ptr<AstClause> ResolveAliasesTransformer::resolveAliases(const AstClause& clause) {
209  // -- utilities --
210 
211  // tests whether something is a variable
212  auto isVar = [&](const AstArgument& arg) { return dynamic_cast<const AstVariable*>(&arg) != nullptr; };
213 
214  // tests whether something is a record
215  auto isRec = [&](const AstArgument& arg) { return dynamic_cast<const AstRecordInit*>(&arg) != nullptr; };
216 
217  // tests whether a value `a` occurs in a term `b`
218  auto occurs = [](const AstArgument& a, const AstArgument& b) {
219  bool res = false;
220  visitDepthFirst(b, [&](const AstArgument& arg) { res = (res || (arg == a)); });
221  return res;
222  };
223 
224  // variables appearing as functorless arguments in atoms or records should not
225  // be resolved
226  std::set<std::string> baseGroundedVariables;
227  for (const auto* atom : getBodyLiterals<AstAtom>(clause)) {
228  for (const AstArgument* arg : atom->getArguments()) {
229  if (const auto* var = dynamic_cast<const AstVariable*>(arg)) {
230  baseGroundedVariables.insert(var->getName());
231  }
232  }
233  visitDepthFirst(*atom, [&](const AstRecordInit& rec) {
234  for (const AstArgument* arg : rec.getArguments()) {
235  if (const auto* var = dynamic_cast<const AstVariable*>(arg)) {
236  baseGroundedVariables.insert(var->getName());
237  }
238  }
239  });
240  }
241 
242  // I) extract equations
243  std::vector<Equation> equations;
244  visitDepthFirst(clause, [&](const AstBinaryConstraint& constraint) {
245  if (isEqConstraint(constraint.getOperator())) {
246  equations.push_back(Equation(constraint.getLHS(), constraint.getRHS()));
247  }
248  });
249 
250  // II) compute unifying substitution
251  Substitution substitution;
252 
253  // a utility for processing newly identified mappings
254  auto newMapping = [&](const std::string& var, const AstArgument* term) {
255  // found a new substitution
256  Substitution newMapping(var, term);
257 
258  // apply substitution to all remaining equations
259  for (auto& equation : equations) {
260  equation.apply(newMapping);
261  }
262 
263  // add mapping v -> t to substitution
264  substitution.append(newMapping);
265  };
266 
267  while (!equations.empty()) {
268  // get next equation to compute
269  Equation equation = equations.back();
270  equations.pop_back();
271 
272  // shortcuts for left/right
273  const AstArgument& lhs = *equation.lhs;
274  const AstArgument& rhs = *equation.rhs;
275 
276  // #1: t = t => skip
277  if (lhs == rhs) {
278  continue;
279  }
280 
281  // #2: [..] = [..] => decompose
282  if (isRec(lhs) && isRec(rhs)) {
283  // get arguments
284  const auto& lhs_args = static_cast<const AstRecordInit&>(lhs).getArguments();
285  const auto& rhs_args = static_cast<const AstRecordInit&>(rhs).getArguments();
286 
287  // make sure sizes are identical
288  assert(lhs_args.size() == rhs_args.size() && "Record lengths not equal");
289 
290  // create new equalities
291  for (size_t i = 0; i < lhs_args.size(); i++) {
292  equations.push_back(Equation(lhs_args[i], rhs_args[i]));
293  }
294 
295  continue;
296  }
297 
298  // #3: neither is a variable => skip
299  if (!isVar(lhs) && !isVar(rhs)) {
300  continue;
301  }
302 
303  // #4: v = w => add mapping
304  if (isVar(lhs) && isVar(rhs)) {
305  auto& var = static_cast<const AstVariable&>(lhs);
306  newMapping(var.getName(), &rhs);
307  continue;
308  }
309 
310  // #5: t = v => swap
311  if (!isVar(lhs)) {
312  equations.push_back(Equation(rhs, lhs));
313  continue;
314  }
315 
316  // now we know lhs is a variable
317  assert(isVar(lhs));
318 
319  // therefore, we have v = t
320  const auto& v = static_cast<const AstVariable&>(lhs);
321  const AstArgument& t = rhs;
322 
323  // #6: v occurs in t => skip
324  if (occurs(v, t)) {
325  continue;
326  }
327 
328  assert(!occurs(v, t));
329 
330  // #7: t is a record => add mapping
331  if (isRec(t)) {
332  newMapping(v.getName(), &t);
333  continue;
334  }
335 
336  // #8: v is already grounded => skip
337  auto pos = baseGroundedVariables.find(v.getName());
338  if (pos != baseGroundedVariables.end()) {
339  continue;
340  }
341 
342  // add new mapping
343  newMapping(v.getName(), &t);
344  }
345 
346  // III) compute resulting clause
347  return substitution(souffle::clone(&clause));
348 }

350 std::unique_ptr<AstClause> ResolveAliasesTransformer::removeTrivialEquality(const AstClause& clause) {
351  std::unique_ptr<AstClause> res(cloneHead(&clause));
352 
353  // add all literals, except filtering out t = t constraints
354  for (AstLiteral* literal : clause.getBodyLiterals()) {
355  if (auto* constraint = dynamic_cast<AstBinaryConstraint*>(literal)) {
356  // TODO: don't filter out `FEQ` constraints, since `x = x` can fail when `x` is a NaN
357  if (isEqConstraint(constraint->getOperator())) {
358  if (*constraint->getLHS() == *constraint->getRHS()) {
359  continue; // skip this one
360  }
361  }
362  }
363 
364  res->addToBody(souffle::clone(literal));
365  }
366 
367  // done
368  return res;
369 }

371 std::unique_ptr<AstClause> ResolveAliasesTransformer::removeComplexTermsInAtoms(const AstClause& clause) {
372  std::unique_ptr<AstClause> res(clause.clone());
373 
374  // get list of atoms
375  std::vector<AstAtom*> atoms = getBodyLiterals<AstAtom>(*res);
376 
377  // find all functors in atoms
378  std::vector<const AstArgument*> terms;
379  for (const AstAtom* atom : atoms) {
380  for (const AstArgument* arg : atom->getArguments()) {
381  // ignore if not a functor
382  if (dynamic_cast<const AstFunctor*>(arg) == nullptr) {
383  continue;
384  }
385 
386  // add this functor if not seen yet
387  if (!any_of(terms, [&](const AstArgument* cur) { return *cur == *arg; })) {
388  terms.push_back(arg);
389  }
390  }
391  }
392 
393  // find all functors in records too
394  visitDepthFirst(atoms, [&](const AstRecordInit& rec) {
395  for (const AstArgument* arg : rec.getArguments()) {
396  // ignore if not a functor
397  if (dynamic_cast<const AstFunctor*>(arg) == nullptr) {
398  continue;
399  }
400 
401  // add this functor if not seen yet
402  if (!any_of(terms, [&](const AstArgument* cur) { return *cur == *arg; })) {
403  terms.push_back(arg);
404  }
405  }
406  });
407 
408  // substitute them with new variables (a real map would compare pointers)
409  using substitution_map =
410  std::vector<std::pair<std::unique_ptr<AstArgument>, std::unique_ptr<AstVariable>>>;
411  substitution_map termToVar;
412 
413  static int varCounter = 0;
414  for (const AstArgument* arg : terms) {
415  // create a new mapping for this term
416  auto term = souffle::clone(arg);
417  auto newVariable = std::make_unique<AstVariable>(" _tmp_" + toString(varCounter++));
418  termToVar.push_back(std::make_pair(std::move(term), std::move(newVariable)));
419  }
420 
421  // apply mapping to replace the terms with the variables
422  struct Update : public AstNodeMapper {
423  const substitution_map& map;
424 
425  Update(const substitution_map& map) : map(map) {}
426 
427  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
428  // check whether node needs to be replaced
429  for (const auto& pair : map) {
430  auto& term = pair.first;
431  auto& variable = pair.second;
432 
433  if (*term == *node) {
434  return souffle::clone(variable);
435  }
436  }
437 
438  // continue recursively
439  node->apply(*this);
440  return node;
441  }
442  };
443 
444  // update atoms
445  Update update(termToVar);
446  for (AstAtom* atom : atoms) {
447  atom->apply(update);
448  }
449 
450  // add the necessary variable constraints to the clause
451  for (const auto& pair : termToVar) {
452  auto& term = pair.first;
453  auto& variable = pair.second;
454 
455  res->addToBody(std::make_unique<AstBinaryConstraint>(
456  BinaryConstraintOp::EQ, souffle::clone(variable), souffle::clone(term)));
457  }
458 
459  return res;
460 }

462 bool ResolveAliasesTransformer::transform(AstTranslationUnit& translationUnit) {
463  bool changed = false;
464  AstProgram& program = *translationUnit.getProgram();
465 
466  // get all clauses
467  std::vector<const AstClause*> clauses;
468  visitDepthFirst(program, [&](const AstRelation& rel) {
469  for (const auto& clause : getClauses(program, rel)) {
470  clauses.push_back(clause);
471  }
472  });
473 
474  // clean all clauses
475  for (const AstClause* clause : clauses) {
476  // -- Step 1 --
477  // get rid of aliases
478  std::unique_ptr<AstClause> noAlias = resolveAliases(*clause);
479 
480  // clean up equalities
481  std::unique_ptr<AstClause> cleaned = removeTrivialEquality(*noAlias);
482 
483  // -- Step 2 --
484  // restore simple terms in atoms
485  std::unique_ptr<AstClause> normalised = removeComplexTermsInAtoms(*cleaned);
486 
487  // swap if changed
488  if (*normalised != *clause) {
489  changed = true;
490  program.removeClause(clause);
491  program.addClause(std::move(normalised));
492  }
493  }
494 
495  return changed;
496 }
