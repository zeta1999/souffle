168 AstSemanticCheckerImpl::AstSemanticCheckerImpl(AstTranslationUnit& tu) : tu(tu) {
169  // suppress warnings for given relations
170  if (Global::config().has("suppress-warnings")) {
171  std::vector<std::string> suppressedRelations =
172  splitString(Global::config().get("suppress-warnings"), ',');
173 
174  if (std::find(suppressedRelations.begin(), suppressedRelations.end(), "*") !=
175  suppressedRelations.end()) {
176  // mute all relations
177  for (AstRelation* rel : program.getRelations()) {
178  rel->addQualifier(RelationQualifier::SUPPRESSED);
179  }
180  } else {
181  // mute only the given relations (if they exist)
182  for (auto& relname : suppressedRelations) {
183  const std::vector<std::string> comps = splitString(relname, '.');
184  if (!comps.empty()) {
185  // generate the relation identifier
186  AstQualifiedName relid(comps[0]);
187  for (size_t i = 1; i < comps.size(); i++) {
188  relid.append(comps[i]);
189  }
190 
191  // update suppressed qualifier if the relation is found
192  if (AstRelation* rel = getRelation(program, relid)) {
193  rel->addQualifier(RelationQualifier::SUPPRESSED);
194  }
195  }
196  }
197  }
198  }
199 
200  // Check types in AST.
201  for (const auto* astType : program.getTypes()) {
202  checkType(*astType);
203  }
204 
205  // check rules
206  for (auto* rel : program.getRelations()) {
207  checkRelation(*rel);
208  }
209  for (auto* clause : program.getClauses()) {
210  checkClause(*clause);
211  }
212 
213  // Group clauses that stem from a single complex rule
214  // with multiple headers/disjunction etc. The grouping
215  // is performed via their source-location.
216  std::map<SrcLocation, std::set<const AstClause*>> multiRuleMap;
217  for (auto* clause : program.getClauses()) {
218  // collect clauses of a multi rule, i.e., they have the same source locator
219  multiRuleMap[clause->getSrcLoc()].insert(clause);
220  }
221 
222  // check complex rule
223  for (const auto& multiRule : multiRuleMap) {
224  checkComplexRule(multiRule.second);
225  }
226 
227  checkNamespaces();
228  checkIO();
229  checkWitnessProblem();
230  checkInlining();
231 
232  // Run grounded terms checker
233  GroundedTermsChecker().verify(tu);
234 
235  // Check types
236  TypeChecker{tu}.run();
237 
238  // - stratification --
239  // check for cyclic dependencies
240  for (AstRelation* cur : program.getRelations()) {
241  size_t scc = sccGraph.getSCC(cur);
242  if (sccGraph.isRecursive(scc)) {
243  for (const AstRelation* cyclicRelation : sccGraph.getInternalRelations(scc)) {
244  // Negations and aggregations need to be stratified
245  const AstLiteral* foundLiteral = nullptr;
246  bool hasNegation = hasClauseWithNegatedRelation(cyclicRelation, cur, &program, foundLiteral);
247  if (hasNegation ||
248  hasClauseWithAggregatedRelation(cyclicRelation, cur, &program, foundLiteral)) {
249  auto const& relSet = sccGraph.getInternalRelations(scc);
250  std::set<const AstRelation*, AstNameComparison> sortedRelSet(
251  relSet.begin(), relSet.end());
252  // Negations and aggregations need to be stratified
253  std::string relationsListStr = toString(join(sortedRelSet, ",",
254  [](std::ostream& out, const AstRelation* r) { out << r->getQualifiedName(); }));
255  std::vector<DiagnosticMessage> messages;
256  messages.push_back(DiagnosticMessage(
257  "Relation " + toString(cur->getQualifiedName()), cur->getSrcLoc()));
258  std::string negOrAgg = hasNegation ? "negation" : "aggregation";
259  messages.push_back(
260  DiagnosticMessage("has cyclic " + negOrAgg, foundLiteral->getSrcLoc()));
261  report.addDiagnostic(Diagnostic(Diagnostic::Type::ERROR,
262  DiagnosticMessage("Unable to stratify relation(s) {" + relationsListStr + "}"),
263  messages));
264  break;
265  }
266  }
267  }
268  }

292  * Get unnamed variables except those that appear inside aggregates.

294 std::set<const AstUnnamedVariable*> getUnnamedVariables(const AstNode& node) {
295  std::set<const AstUnnamedVariable*> unnamedInAggregates;
296  visitDepthFirst(node, [&](const AstAggregator& agg) {
297  visitDepthFirst(agg, [&](const AstUnnamedVariable& var) { unnamedInAggregates.insert(&var); });
298  });
299 
300  std::set<const AstUnnamedVariable*> unnamed;
301  visitDepthFirst(node, [&](const AstUnnamedVariable& var) {
302  if (!contains(unnamedInAggregates, &var)) {
303  unnamed.insert(&var);
304  }
305  });
306 
307  return unnamed;
308 }

312 void AstSemanticCheckerImpl::checkLiteral(const AstLiteral& literal) {
313  // check potential nested atom
314  if (const auto* atom = as<AstAtom>(literal)) {
315  checkAtom(*atom);
316  }
317 
318  if (const auto* neg = as<AstNegation>(literal)) {
319  checkAtom(*neg->getAtom());
320  }
321 
322  if (const auto* constraint = as<AstBinaryConstraint>(literal)) {
323  checkArgument(*constraint->getLHS());
324  checkArgument(*constraint->getRHS());
325 
326  std::set<const AstUnnamedVariable*> unnamedInRecord;
327  visitDepthFirst(*constraint, [&](const AstRecordInit& record) {
328  for (auto* arg : record.getArguments()) {
329  if (auto* unnamed = as<AstUnnamedVariable>(arg)) {
330  unnamedInRecord.insert(unnamed);
331  }
332  }
333  });
334 
335  // Check if constraint contains unnamed variables.
336  for (auto* unnamed : getUnnamedVariables(*constraint)) {
337  if (!contains(unnamedInRecord, unnamed)) {
338  report.addError("Underscore in binary relation", unnamed->getSrcLoc());
339  }
340  }
341  }

344 /**
345  * agg1, agg2 are clauses which contain no head, and consist of a single literal
346  * that contains an aggregate.
347  * agg1 is dependent on agg2 if agg1 contains a variable which is grounded by agg2, and not by agg1.
348  */
349 bool AstSemanticCheckerImpl::isDependent(const AstClause& agg1, const AstClause& agg2) {
350  auto groundedInAgg1 = getGroundedTerms(tu, agg1);
351  auto groundedInAgg2 = getGroundedTerms(tu, agg2);
352  bool dependent = false;
353  // For each variable X in the first aggregate
354  visitDepthFirst(agg1, [&](const AstVariable& searchVar) {
355  // Try to find the corresponding variable X in the second aggregate
356  // by string comparison
357  const AstVariable* matchingVarPtr = nullptr;
358  visitDepthFirst(agg2, [&](const AstVariable& var) {
359  if (var == searchVar) {
360  matchingVarPtr = &var;
361  return;
362  }
363  });
364  // If the variable occurs in both clauses (a match was found)
365  if (matchingVarPtr != nullptr) {
366  if (!groundedInAgg1[&searchVar] && groundedInAgg2[matchingVarPtr]) {
367  dependent = true;
368  }
369  }
370  });
371  return dependent;
372 }

374 void AstSemanticCheckerImpl::checkAggregator(const AstAggregator& aggregator) {
375  auto& report = tu.getErrorReport();
376  auto& program = *tu.getProgram();
377  const AstAggregator* inner = nullptr;
378 
379  // check for disallowed nested aggregates
380  visitDepthFirst(aggregator, [&](const AstAggregator& innerAgg) {
381  if (aggregator != innerAgg) {
382  inner = &innerAgg;
383  }
384  });
385 
386  if (inner != nullptr) {
387  report.addError("Unsupported nested aggregate", inner->getSrcLoc());
388  }
389 
390  AstClause dummyClauseAggregator;
391 
392  visitDepthFirst(program, [&](const AstLiteral& parentLiteral) {
393  visitDepthFirst(parentLiteral, [&](const AstAggregator& candidateAggregate) {
394  if (candidateAggregate != aggregator) {
395  return;
396  }
397  // Get the literal containing the aggregator and put it into a dummy clause
398  // so we can get information about groundedness
399  dummyClauseAggregator.addToBody(souffle::clone(&parentLiteral));
400  });
401  });
402 
403  visitDepthFirst(program, [&](const AstLiteral& parentLiteral) {
404  visitDepthFirst(parentLiteral, [&](const AstAggregator& /* otherAggregate */) {
405  // Create the other aggregate's dummy clause
406  AstClause dummyClauseOther;
407  dummyClauseOther.addToBody(souffle::clone(&parentLiteral));
408  // Check dependency between the aggregator and this one
409  if (isDependent(dummyClauseAggregator, dummyClauseOther) &&
410  isDependent(dummyClauseOther, dummyClauseAggregator)) {
411  report.addError("Mutually dependent aggregate", aggregator.getSrcLoc());
412  }
413  });
414  });
415 
416  for (AstLiteral* literal : aggregator.getBodyLiterals()) {
417  checkLiteral(*literal);
418  }
419 }

433 /**
434  * Check if the argument can be statically evaluated
435  * and thus in particular, if it should be allowed to appear as argument in facts.
436  **/
437 bool isConstantArgument(const AstArgument* arg) {
438  assert(arg != nullptr);
439 
440  if (isA<AstVariable>(arg) || isA<AstUnnamedVariable>(arg)) {
441  return false;
442  } else if (isA<AstUserDefinedFunctor>(arg)) {
443  return false;
444  } else if (isA<AstCounter>(arg)) {
445  return false;
446  } else if (auto* typeCast = as<AstTypeCast>(arg)) {
447  return isConstantArgument(typeCast->getValue());
448  } else if (auto* term = as<AstTerm>(arg)) {
449  // Term covers intrinsic functor and records. User-functors are handled earlier.
450  return all_of(term->getArguments(), isConstantArgument);
451  } else if (isA<AstConstant>(arg)) {
452  return true;
453  } else {
454  fatal("unsupported argument type: %s", typeid(arg).name());
455  }
456 }

482 void AstSemanticCheckerImpl::checkClause(const AstClause& clause) {
483  // check head atom
484  checkAtom(*clause.getHead());
485 
486  // Check for absence of underscores in head
487  for (auto* unnamed : getUnnamedVariables(*clause.getHead())) {
488  report.addError("Underscore in head of rule", unnamed->getSrcLoc());
489  }
490 
491  // check body literals
492  for (AstLiteral* lit : clause.getBodyLiterals()) {
493  checkLiteral(*lit);
494  }
495 
496  // check facts
497  if (isFact(clause)) {
498  checkFact(clause);
499  }
500 
501  // check whether named unnamed variables of the form _<ident>
502  // are only used once in a clause; if not, warnings will be
503  // issued.
504  std::map<std::string, int> var_count;
505  std::map<std::string, const AstVariable*> var_pos;
506  visitDepthFirst(clause, [&](const AstVariable& var) {
507  var_count[var.getName()]++;
508  var_pos[var.getName()] = &var;
509  });
510  for (const auto& cur : var_count) {
511  int numAppearances = cur.second;
512  const auto& varName = cur.first;
513  const auto& varLocation = var_pos[varName]->getSrcLoc();
514  if (varName[0] == '_') {
515  assert(varName.size() > 1 && "named variable should not be a single underscore");
516  if (numAppearances > 1) {
517  report.addWarning("Variable " + varName + " marked as singleton but occurs more than once",
518  varLocation);
519  }
520  }
521  }
522 
523  // check execution plan
524  if (clause.getExecutionPlan() != nullptr) {
525  auto numAtoms = getBodyLiterals<AstAtom>(clause).size();
526  for (const auto& cur : clause.getExecutionPlan()->getOrders()) {
527  bool isComplete = true;
528  auto order = cur.second->getOrder();
529  for (unsigned i = 1; i <= order.size(); i++) {
530  if (!contains(order, i)) {
531  isComplete = false;
532  break;
533  }
534  }
535  if (order.size() != numAtoms || !isComplete) {
536  report.addError("Invalid execution order in plan", cur.second->getSrcLoc());
537  }
538  }
539  }
540 
541  // check auto-increment
542  if (recursiveClauses.recursive(&clause)) {
543  visitDepthFirst(clause, [&](const AstCounter& ctr) {
544  report.addError("Auto-increment functor in a recursive rule", ctr.getSrcLoc());
545  });
546  }
547 }

549 void AstSemanticCheckerImpl::checkComplexRule(std::set<const AstClause*> multiRule) {
550  std::map<std::string, int> var_count;
551  std::map<std::string, const AstVariable*> var_pos;
552 
553  // Count the variable occurrence for the body of a
554  // complex rule only once.
555  // TODO (b-scholz): for negation / disjunction this is not quite
556  // right; we would need more semantic information here.
557  for (auto literal : (*multiRule.begin())->getBodyLiterals()) {
558  visitDepthFirst(*literal, [&](const AstVariable& var) {
559  var_count[var.getName()]++;
560  var_pos[var.getName()] = &var;
561  });
562  }
563 
564  // Count variable occurrence for each head separately
565  for (auto clause : multiRule) {
566  visitDepthFirst(*(clause->getHead()), [&](const AstVariable& var) {
567  var_count[var.getName()]++;
568  var_pos[var.getName()] = &var;
569  });
570  }
571 
572  // Check that a variables occurs more than once
573  for (const auto& cur : var_count) {
574  int numAppearances = cur.second;
575  const auto& varName = cur.first;
576  const auto& varLocation = var_pos[varName]->getSrcLoc();
577  if (varName[0] != '_' && numAppearances == 1) {
578  report.addWarning("Variable " + varName + " only occurs once", varLocation);
579  }
580  }

583 void AstSemanticCheckerImpl::checkRelationDeclaration(const AstRelation& relation) {
584  const auto& attributes = relation.getAttributes();
585  assert(attributes.size() == relation.getArity() && "mismatching attribute size and arity");
586 
587  for (size_t i = 0; i < relation.getArity(); i++) {
588  AstAttribute* attr = attributes[i];
589  auto&& typeName = attr->getTypeName();
590 
591  /* check whether type exists */
592  if (!typeEnv.isPrimitiveType(typeName) && !getType(program, typeName)) {
593  report.addError(tfm::format("Undefined type in attribute %s", *attr), attr->getSrcLoc());
594  }
595

597  for (size_t j = 0; j < i; j++) {
598  if (attr->getName() == attributes[j]->getName()) {

600  }

605 void AstSemanticCheckerImpl::checkRelation(const AstRelation& relation) {
606  if (relation.getRepresentation() == RelationRepresentation::EQREL) {
607  if (relation.getArity() == 2) {
608  const auto& attributes = relation.getAttributes();
609  assert(attributes.size() == 2 && "mismatching attribute size and arity");
610  if (attributes[0]->getTypeName() != attributes[1]->getTypeName()) {
611  report.addError("Domains of equivalence relation " + toString(relation.getQualifiedName()) +
612  " are different",
613  relation.getSrcLoc());
614  }
615  } else {
616  report.addError(
617  "Equivalence relation " + toString(relation.getQualifiedName()) + " is not binary",
618  relation.getSrcLoc());
619  }
620  }
621 
622  // start with declaration
623  checkRelationDeclaration(relation);
624 
625  // check whether this relation is empty
626  if (getClauses(program, relation).empty() && !ioTypes.isInput(&relation) &&
627  !relation.hasQualifier(RelationQualifier::SUPPRESSED)) {
628  report.addWarning("No rules/facts defined for relation " + toString(relation.getQualifiedName()),
629  relation.getSrcLoc());

636  // check presence of all the element types and that all element types are based off a primitive
637  for (const AstQualifiedName& sub : type.getTypes()) {

758 static const std::vector<SrcLocation> usesInvalidWitness(AstTranslationUnit& tu,
759  const std::vector<AstLiteral*>& literals,
760  const std::set<std::unique_ptr<AstArgument>>& groundedArguments) {
761  // Node-mapper that replaces aggregators with new (unique) variables
762  struct M : public AstNodeMapper {
763  // Variables introduced to replace aggregators
764  mutable std::set<std::string> aggregatorVariables;
765 
766  const std::set<std::string>& getAggregatorVariables() {
767  return aggregatorVariables;
768  }
769 
770  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
771  static int numReplaced = 0;
772  if (dynamic_cast<AstAggregator*>(node.get()) != nullptr) {
773  // Replace the aggregator with a variable
774  std::stringstream newVariableName;
775  newVariableName << "+aggr_var_" << numReplaced++;
776 
777  // Keep track of which variables are bound to aggregators
778  aggregatorVariables.insert(newVariableName.str());
779 
780  return std::make_unique<AstVariable>(newVariableName.str());
781  }
782  node->apply(*this);
783  return node;
784  }
785  };
786 
787  std::vector<SrcLocation> result;
788 
789  // Create two versions of the original clause
790 
791  // Clause 1 - will remain equivalent to the original clause in terms of variable groundedness
792  auto originalClause = std::make_unique<AstClause>();
793  originalClause->setHead(std::make_unique<AstAtom>("*"));
794 
795  // Clause 2 - will have aggregators replaced with intrinsically grounded variables
796  auto aggregatorlessClause = std::make_unique<AstClause>();
797  aggregatorlessClause->setHead(std::make_unique<AstAtom>("*"));
798 
799  // Construct both clauses in the same manner to match the original clause
800  // Must keep track of the subnode in Clause 1 that each subnode in Clause 2 matches to
801  std::map<const AstArgument*, const AstArgument*> identicalSubnodeMap;
802  for (const AstLiteral* lit : literals) {
803  auto firstClone = souffle::clone(lit);
804  auto secondClone = souffle::clone(lit);
805 
806  // Construct the mapping between equivalent literal subnodes
807  std::vector<const AstArgument*> firstCloneArguments;
808  visitDepthFirst(*firstClone, [&](const AstArgument& arg) { firstCloneArguments.push_back(&arg); });
809 
810  std::vector<const AstArgument*> secondCloneArguments;
811  visitDepthFirst(*secondClone, [&](const AstArgument& arg) { secondCloneArguments.push_back(&arg); });
812 
813  for (size_t i = 0; i < firstCloneArguments.size(); i++) {
814  identicalSubnodeMap[secondCloneArguments[i]] = firstCloneArguments[i];
815  }
816 
817  // Actually add the literal clones to each clause
818  originalClause->addToBody(std::move(firstClone));
819  aggregatorlessClause->addToBody(std::move(secondClone));
820  }
821 
822  // Replace the aggregators in Clause 2 with variables
823  M update;
824  aggregatorlessClause->apply(update);
825 
826  // Create a dummy atom to force certain arguments to be grounded in the aggregatorlessClause
827  auto groundingAtomAggregatorless = std::make_unique<AstAtom>("grounding_atom");
828  auto groundingAtomOriginal = std::make_unique<AstAtom>("grounding_atom");
829 
830  // Force the new aggregator variables to be grounded in the aggregatorless clause
831  const std::set<std::string>& aggregatorVariables = update.getAggregatorVariables();
832  for (const std::string& str : aggregatorVariables) {
833  groundingAtomAggregatorless->addArgument(std::make_unique<AstVariable>(str));
834  }
835 
836  // Force the given grounded arguments to be grounded in both clauses
837  for (const std::unique_ptr<AstArgument>& arg : groundedArguments) {
838  groundingAtomAggregatorless->addArgument(souffle::clone(arg));
839  groundingAtomOriginal->addArgument(souffle::clone(arg));
840  }
841 
842  aggregatorlessClause->addToBody(std::move(groundingAtomAggregatorless));
843  originalClause->addToBody(std::move(groundingAtomOriginal));
844 
845  // Compare the grounded analysis of both generated clauses
846  // All added arguments in Clause 2 were forced to be grounded, so if an ungrounded argument
847  // appears in Clause 2, it must also appear in Clause 1. Consequently, have two cases:
848  // - The argument is also ungrounded in Clause 1 - handled by another check
849  // - The argument is grounded in Clause 1 => the argument was grounded in the
850  // first clause somewhere along the line by an aggregator-body - not allowed!
851  std::set<std::unique_ptr<AstArgument>> newlyGroundedArguments;
852  auto originalGrounded = getGroundedTerms(tu, *originalClause);
853  for (auto&& pair : getGroundedTerms(tu, *aggregatorlessClause)) {
854  if (!pair.second && originalGrounded[identicalSubnodeMap[pair.first]]) {
855  result.push_back(pair.first->getSrcLoc());
856  }
857 
858  // Otherwise, it can now be considered grounded
859  newlyGroundedArguments.insert(souffle::clone(pair.first));
860  }
861 
862  // All previously grounded are still grounded
863  for (const std::unique_ptr<AstArgument>& arg : groundedArguments) {
864  newlyGroundedArguments.insert(souffle::clone(arg));
865  }
866 
867  // Everything on this level is fine, check subaggregators of each literal
868  for (const AstLiteral* lit : literals) {
869  visitDepthFirst(*lit, [&](const AstAggregator& aggr) {
870  // Check recursively if an invalid witness is used
871  for (auto&& argloc : usesInvalidWitness(tu, aggr.getBodyLiterals(), newlyGroundedArguments)) {
872  result.push_back(argloc);
873  }
874  });
875  }
876 
877  return result;
878 }

904 /**
905  * Find a cycle consisting entirely of inlined relations.
906  * If no cycle exists, then an empty vector is returned.
907  */
908 std::vector<AstQualifiedName> findInlineCycle(const PrecedenceGraphAnalysis& precedenceGraph,
909  std::map<const AstRelation*, const AstRelation*>& origins, const AstRelation* current,
910  AstRelationSet& unvisited, AstRelationSet& visiting, AstRelationSet& visited) {
911  std::vector<AstQualifiedName> result;
912 
913  if (current == nullptr) {
914  // Not looking at any nodes at the moment, so choose any node from the unvisited list
915 
916  if (unvisited.empty()) {
917  // Nothing left to visit - so no cycles exist!
918  return result;
919  }
920 
921  // Choose any element from the unvisited set
922  current = *unvisited.begin();
923  origins[current] = nullptr;
924 
925  // Move it to "currently visiting"
926  unvisited.erase(current);
927  visiting.insert(current);
928 
929  // Check if we can find a cycle beginning from this node
930  std::vector<AstQualifiedName> subresult =
931  findInlineCycle(precedenceGraph, origins, current, unvisited, visiting, visited);
932 
933  if (subresult.empty()) {
934  // No cycle found, try again from another node
935  return findInlineCycle(precedenceGraph, origins, nullptr, unvisited, visiting, visited);
936  } else {
937  // Cycle found! Return it
938  return subresult;
939  }
940  }
941 
942  // Check neighbours
943  const AstRelationSet& successors = precedenceGraph.graph().successors(current);
944  for (const AstRelation* successor : successors) {
945  // Only care about inlined neighbours in the graph
946  if (successor->hasQualifier(RelationQualifier::INLINE)) {
947  if (visited.find(successor) != visited.end()) {
948  // The neighbour has already been visited, so move on
949  continue;
950  }
951 
952  if (visiting.find(successor) != visiting.end()) {
953  // Found a cycle!!
954  // Construct the cycle in reverse
955  while (current != nullptr) {
956  result.push_back(current->getQualifiedName());
957  current = origins[current];
958  }
959  return result;
960  }
961 
962  // Node has not been visited yet
963  origins[successor] = current;
964 
965  // Move from unvisited to visiting
966  unvisited.erase(successor);
967  visiting.insert(successor);
968 
969  // Visit recursively and check if a cycle is formed
970  std::vector<AstQualifiedName> subgraphCycle =
971  findInlineCycle(precedenceGraph, origins, successor, unvisited, visiting, visited);
972 
973  if (!subgraphCycle.empty()) {
974  // Found a cycle!
975  return subgraphCycle;
976  }
977  }
978  }
979 
980  // Visited all neighbours with no cycle found, so done visiting this node.
981  visiting.erase(current);
982  visited.insert(current);
983  return result;
984 }

986 void AstSemanticCheckerImpl::checkInlining() {
987  auto isInline = [&](const AstRelation* rel) { return rel->hasQualifier(RelationQualifier::INLINE); };
988 
989  // Find all inlined relations
990  AstRelationSet inlinedRelations;
991  for (const auto& relation : program.getRelations()) {
992  if (isInline(relation)) {
993  inlinedRelations.insert(relation);
994  if (ioTypes.isIO(relation)) {
995  report.addError(
996  "IO relation " + toString(relation->getQualifiedName()) + " cannot be inlined",
997  relation->getSrcLoc());
998  }
999  }
1000  }
1001 
1002  // Check 1:
1003  // Let G' be the subgraph of the precedence graph G containing only those nodes
1004  // which are marked with the inline directive.
1005  // If G' contains a cycle, then inlining cannot be performed.
1006 
1007  AstRelationSet unvisited; // nodes that have not been visited yet
1008  AstRelationSet visiting; // nodes that we are currently visiting
1009  AstRelationSet visited; // nodes that have been completely explored
1010 
1011  // All nodes are initially unvisited
1012  for (const AstRelation* rel : inlinedRelations) {
1013  unvisited.insert(rel);
1014  }
1015 
1016  // Remember the parent node of each visited node to construct the found cycle
1017  std::map<const AstRelation*, const AstRelation*> origins;
1018 
1019  std::vector<AstQualifiedName> result =
1020  findInlineCycle(precedenceGraph, origins, nullptr, unvisited, visiting, visited);
1021 
1022  // If the result contains anything, then a cycle was found
1023  if (!result.empty()) {
1024  AstRelation* cycleOrigin = getRelation(program, result[result.size() - 1]);
1025 
1026  // Construct the string representation of the cycle
1027  std::stringstream cycle;
1028  cycle << "{" << cycleOrigin->getQualifiedName();
1029 
1030  // Print it backwards to preserve the initial cycle order
1031  for (int i = result.size() - 2; i >= 0; i--) {
1032  cycle << ", " << result[i];
1033  }
1034 
1035  cycle << "}";
1036 
1037  report.addError(
1038  "Cannot inline cyclically dependent relations " + cycle.str(), cycleOrigin->getSrcLoc());
1039  }
1040 
1041  // Check 2:
1042  // Cannot use the counter argument ('$') in inlined relations
1043 
1044  // Check if an inlined literal ever takes in a $
1045  visitDepthFirst(program, [&](const AstAtom& atom) {
1046  AstRelation* associatedRelation = getRelation(program, atom.getQualifiedName());
1047  if (associatedRelation != nullptr && isInline(associatedRelation)) {
1048  visitDepthFirst(atom, [&](const AstArgument& arg) {
1049  if (dynamic_cast<const AstCounter*>(&arg) != nullptr) {
1050  report.addError(
1051  "Cannot inline literal containing a counter argument '$'", arg.getSrcLoc());
1052  }
1053  });
1054  }
1055  });
1056 
1057  // Check if an inlined clause ever contains a $
1058  for (const AstRelation* rel : inlinedRelations) {
1059  for (AstClause* clause : getClauses(program, *rel)) {
1060  visitDepthFirst(*clause, [&](const AstArgument& arg) {
1061  if (dynamic_cast<const AstCounter*>(&arg) != nullptr) {
1062  report.addError(
1063  "Cannot inline clause containing a counter argument '$'", arg.getSrcLoc());
1064  }
1065  });
1066  }
1067  }
1068 
1069  // Check 3:
1070  // Suppose the relation b is marked with the inline directive, but appears negated
1071  // in a clause. Then, if b introduces a new variable in its body, we cannot inline
1072  // the relation b.
1073 
1074  // Find all relations with the inline declarative that introduce new variables in their bodies
1075  AstRelationSet nonNegatableRelations;
1076  for (const AstRelation* rel : inlinedRelations) {
1077  bool foundNonNegatable = false;
1078  for (const AstClause* clause : getClauses(program, *rel)) {
1079  // Get the variables in the head
1080  std::set<std::string> headVariables;
1081  visitDepthFirst(
1082  *clause->getHead(), [&](const AstVariable& var) { headVariables.insert(var.getName()); });
1083 
1084  // Get the variables in the body
1085  std::set<std::string> bodyVariables;
1086  visitDepthFirst(clause->getBodyLiterals(),
1087  [&](const AstVariable& var) { bodyVariables.insert(var.getName()); });
1088 
1089  // Check if all body variables are in the head
1090  // Do this separately to the above so only one error is printed per variable
1091  for (const std::string& var : bodyVariables) {
1092  if (headVariables.find(var) == headVariables.end()) {
1093  nonNegatableRelations.insert(rel);
1094  foundNonNegatable = true;
1095  break;
1096  }
1097  }
1098 
1099  if (foundNonNegatable) {
1100  break;
1101  }
1102  }
1103  }
1104 
1105  // Check that these relations never appear negated
1106  visitDepthFirst(program, [&](const AstNegation& neg) {
1107  AstRelation* associatedRelation = getRelation(program, neg.getAtom()->getQualifiedName());
1108  if (associatedRelation != nullptr &&
1109  nonNegatableRelations.find(associatedRelation) != nonNegatableRelations.end()) {
1110  report.addError(
1111  "Cannot inline negated relation which may introduce new variables", neg.getSrcLoc());
1112  }
1113  });
1114 
1115  // Check 4:
1116  // Don't support inlining atoms within aggregators at this point.
1117 
1118  // Reasoning: Suppose we have an aggregator like `max X: a(X)`, where `a` is inlined to `a1` and `a2`.
1119  // Then, `max X: a(X)` will become `max( max X: a1(X), max X: a2(X) )`. Suppose further that a(X) has
1120  // values X where it is true, while a2(X) does not. Then, the produced argument
1121  // `max( max X: a1(X), max X: a2(X) )` will not return anything (as one of its arguments fails), while
1122  // `max X: a(X)` will.
1123  // Can work around this with emptiness checks (e.g. `!a1(_), ... ; !a2(_), ... ; ...`)
1124 
1125  // This corner case prevents generalising aggregator inlining with the current set up.
1126 
1127  visitDepthFirst(program, [&](const AstAggregator& aggr) {
1128  visitDepthFirst(aggr, [&](const AstAtom& subatom) {
1129  const AstRelation* rel = getRelation(program, subatom.getQualifiedName());
1130  if (rel != nullptr && isInline(rel)) {
1131  report.addError("Cannot inline relations that appear in aggregator", subatom.getSrcLoc());
1132  }
1133  });
1134  });
1135 
1136  // Check 5:
1137  // Suppose a relation `a` is inlined, appears negated in a clause, and contains a
1138  // (possibly nested) unnamed variable in its arguments. Then, the atom can't be
1139  // inlined, as unnamed variables are named during inlining (since they may appear
1140  // multiple times in an inlined-clause's body) => ungroundedness!
1141 
1142  // Exception: It's fine if the unnamed variable appears in a nested aggregator, as
1143  // the entire aggregator will automatically be grounded.
1144 
1145  // TODO (azreika): special case where all rules defined for `a` use the
1146  // underscored-argument exactly once: can workaround by remapping the variable
1147  // back to an underscore - involves changes to the actual inlining algo, though
1148 
1149  // Returns the pair (isValid, lastSrcLoc) where:
1150  // - isValid is true if and only if the node contains an invalid underscore, and
1151  // - lastSrcLoc is the source location of the last visited node
1152  std::function<std::pair<bool, SrcLocation>(const AstNode*)> checkInvalidUnderscore =
1153  [&](const AstNode* node) {
1154  if (dynamic_cast<const AstUnnamedVariable*>(node) != nullptr) {
1155  // Found an invalid underscore
1156  return std::make_pair(true, node->getSrcLoc());
1157  } else if (dynamic_cast<const AstAggregator*>(node) != nullptr) {
1158  // Don't care about underscores within aggregators
1159  return std::make_pair(false, node->getSrcLoc());
1160  }
1161 
1162  // Check if any children nodes use invalid underscores
1163  for (const AstNode* child : node->getChildNodes()) {
1164  std::pair<bool, SrcLocation> childStatus = checkInvalidUnderscore(child);
1165  if (childStatus.first) {
1166  // Found an invalid underscore
1167  return childStatus;
1168  }
1169  }
1170 
1171  return std::make_pair(false, node->getSrcLoc());
1172  };
1173 
1174  // Perform the check
1175  visitDepthFirst(program, [&](const AstNegation& negation) {
1176  const AstAtom* associatedAtom = negation.getAtom();
1177  const AstRelation* associatedRelation = getRelation(program, associatedAtom->getQualifiedName());
1178  if (associatedRelation != nullptr && isInline(associatedRelation)) {
1179  std::pair<bool, SrcLocation> atomStatus = checkInvalidUnderscore(associatedAtom);
1180  if (atomStatus.first) {
1181  report.addError(
1182  "Cannot inline negated atom containing an unnamed variable unless the variable is "
1183  "within an aggregator",
1184  atomStatus.second);
1185  }
1186  }
1187  });
1188 }

1214 void TypeChecker::visitAtom(const AstAtom& atom) {
1215  auto relation = getAtomRelation(&atom, &program);
1216  if (relation == nullptr) {
1217  return; // error unrelated to types.
1218  }
1219 
1220  auto attributes = relation->getAttributes();
1221  auto arguments = atom.getArguments();
1222  if (attributes.size() != arguments.size()) {
1223  return; // error in input program
1224  }
1225 
1226  for (size_t i = 0; i < attributes.size(); ++i) {
1227  auto& typeName = attributes[i]->getTypeName();
1228  if (typeEnv.isType(typeName)) {
1229  auto argTypes = typeAnalysis.getTypes(arguments[i]);
1230  auto& attributeType = typeEnv.getType(typeName);
1231 
1232  if (argTypes.isAll() || argTypes.empty()) {
1233  continue; // This will be reported later.
1234  }
1235 
1236  // Attribute and argument type agree if, argument type is a subtype of declared type
