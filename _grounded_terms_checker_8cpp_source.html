34 void GroundedTermsChecker::verify(AstTranslationUnit& translationUnit) {
35  auto&& program = *translationUnit.getProgram();
36  auto&& report = translationUnit.getErrorReport();
37 
38  // -- check grounded variables and records --
39  visitDepthFirst(program.getClauses(), [&](const AstClause& clause) {
40  if (isFact(clause)) return; // only interested in rules
41 
42  auto isGrounded = getGroundedTerms(translationUnit, clause);
43 
44  std::set<std::string> reportedVars;
45  // all terms in head need to be grounded
46  for (auto&& cur : getVariables(clause)) {
47  if (!isGrounded[cur] && reportedVars.insert(cur->getName()).second) {
48  report.addError("Ungrounded variable " + cur->getName(), cur->getSrcLoc());
49  }
50  }
51 
52  // all records need to be grounded
53  for (auto&& cur : getRecords(clause)) {
54  if (!isGrounded[cur]) {
55  report.addError("Ungrounded record", cur->getSrcLoc());
56  }
57  }
58  });
59 }
