37 bool IndexedInequalityTransformer::transformIndexToFilter(RamProgram& program) {
38  bool changed = false;
39 
40  // helper for collecting conditions from filter operations
41  auto addCondition = [](std::unique_ptr<RamCondition> condition,
42  std::unique_ptr<RamCondition> c) -> std::unique_ptr<RamCondition> {
43  if (condition == nullptr) {
44  return c;
45  } else {
46  return std::make_unique<RamConjunction>(std::move(condition), std::move(c));
47  }
48  };

50  visitDepthFirst(program, [&](const RamQuery& query) {
51  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> indexToFilterRewriter =
52  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
53  // find a RamIndexOperation
54  if (const RamIndexOperation* indexOperation = dynamic_cast<RamIndexOperation*>(node.get())) {
55  auto attributesToDischarge =
56  idxAnalysis->getIndexes(indexOperation->getRelation()).getAttributesToDischarge();
57 
58  auto pattern = indexOperation->getRangePattern();
59  std::unique_ptr<RamCondition> condition;
60  RamPattern updatedPattern;
61 
62  for (RamExpression* p : indexOperation->getRangePattern().first) {
63  updatedPattern.first.emplace_back(p->clone());
64  }
65  for (RamExpression* p : indexOperation->getRangePattern().second) {
66  updatedPattern.second.emplace_back(p->clone());
67  }
68 
69  for (auto i : attributesToDischarge) {
70  // move constraints out of the indexed inequality and into a conjuction
71  std::unique_ptr<RamConstraint> lowerBound;
72  std::unique_ptr<RamConstraint> upperBound;
73  changed = true;
74 
75  if (!isRamUndefValue(pattern.first[i])) {
76  lowerBound = std::make_unique<RamConstraint>(BinaryConstraintOp::GE,
77  std::make_unique<RamTupleElement>(indexOperation->getTupleId(), i),
78  souffle::clone(pattern.first[i]));
79  condition = addCondition(std::move(condition), souffle::clone(lowerBound));
80  }
81 
82  if (!isRamUndefValue(pattern.second[i])) {
83  upperBound = std::make_unique<RamConstraint>(BinaryConstraintOp::LE,
84  std::make_unique<RamTupleElement>(indexOperation->getTupleId(), i),
85  souffle::clone(pattern.second[i]));
86  condition = addCondition(std::move(condition), souffle::clone(upperBound));
87  }
88 
89  // reset the bounds
90  updatedPattern.first[i] = std::make_unique<RamUndefValue>();
91  updatedPattern.second[i] = std::make_unique<RamUndefValue>();
92  }
93 
94  if (condition) {
95  auto nestedOp = souffle::clone(&indexOperation->getOperation());
96  auto filter = std::make_unique<RamFilter>(std::move(condition), std::move(nestedOp));
97 
98  // need to rewrite the node with the same index operation
99  if (const RamIndexScan* iscan = dynamic_cast<RamIndexScan*>(node.get())) {
100  node = std::make_unique<RamIndexScan>(
101  std::make_unique<RamRelationReference>(&iscan->getRelation()),
102  iscan->getTupleId(), std::move(updatedPattern), std::move(filter),
103  iscan->getProfileText());
104  } else if (const RamParallelIndexScan* pscan =
105  dynamic_cast<RamParallelIndexScan*>(node.get())) {
106  node = std::make_unique<RamParallelIndexScan>(
107  std::make_unique<RamRelationReference>(&pscan->getRelation()),
108  pscan->getTupleId(), std::move(updatedPattern), std::move(filter),
109  pscan->getProfileText());
110  } else if (const RamIndexChoice* ichoice = dynamic_cast<RamIndexChoice*>(node.get())) {
111  node = std::make_unique<RamIndexChoice>(
112  std::make_unique<RamRelationReference>(&ichoice->getRelation()),
113  ichoice->getTupleId(), souffle::clone(&ichoice->getCondition()),
114  std::move(updatedPattern), std::move(filter), ichoice->getProfileText());
115  } else if (const RamIndexAggregate* iagg = dynamic_cast<RamIndexAggregate*>(node.get())) {
116  node = std::make_unique<RamIndexAggregate>(std::move(filter), iagg->getFunction(),
117  std::make_unique<RamRelationReference>(&iagg->getRelation()),
118  souffle::clone(&iagg->getExpression()), souffle::clone(&iagg->getCondition()),
119  std::move(updatedPattern), iagg->getTupleId());
120  } else {
121  fatal("New RamIndexOperation subclass found but not supported while making index.");
122  }
123  }
124  }
125  node->apply(makeLambdaRamMapper(indexToFilterRewriter));
126  return node;
127  };
128  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(indexToFilterRewriter));
129  });

131  visitDepthFirst(program, [&](const RamQuery& query) {
132  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> removeEmptyIndexRewriter =
133  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
134  // find an IndexOperation
135  if (const RamIndexOperation* indexOperation = dynamic_cast<RamIndexOperation*>(node.get())) {
136  auto pattern = indexOperation->getRangePattern();
137  size_t length = pattern.first.size();
138  bool foundRealIndexableOperation = false;
139 
140  for (size_t i = 0; i < length; ++i) {
141  // if both bounds are undefined we don't have a box query
142  if (isRamUndefValue(pattern.first[i]) && isRamUndefValue(pattern.second[i])) {
143  continue;
144  }
145  // if lower and upper bounds are equal its also not a box query
146  if (*(pattern.first[i]) == *(pattern.second[i])) {
147  foundRealIndexableOperation = true;
148  break;
149  }
150  }
151  if (!foundRealIndexableOperation) {
152  // need to rewrite the node with a semantically equivalent operation to get rid of the
153  // index operation i.e. RamIndexScan with no indexable attributes -> RamScan
154  if (const RamIndexScan* iscan = dynamic_cast<RamIndexScan*>(node.get())) {
155  node = std::make_unique<RamScan>(
156  std::make_unique<RamRelationReference>(&iscan->getRelation()),
157  iscan->getTupleId(), souffle::clone(&iscan->getOperation()),
158  iscan->getProfileText());
159  } else if (const RamParallelIndexScan* pscan =
160  dynamic_cast<RamParallelIndexScan*>(node.get())) {
161  node = std::make_unique<RamParallelScan>(
162  std::make_unique<RamRelationReference>(&pscan->getRelation()),
163  pscan->getTupleId(), souffle::clone(&pscan->getOperation()),
164  pscan->getProfileText());
165  } else if (const RamIndexChoice* ichoice = dynamic_cast<RamIndexChoice*>(node.get())) {
166  node = std::make_unique<RamChoice>(
167  std::make_unique<RamRelationReference>(&ichoice->getRelation()),
168  ichoice->getTupleId(), souffle::clone(&ichoice->getCondition()),
169  souffle::clone(&ichoice->getOperation()), ichoice->getProfileText());
170  } else if (const RamIndexAggregate* iagg = dynamic_cast<RamIndexAggregate*>(node.get())) {
171  node = std::make_unique<RamAggregate>(souffle::clone(&iagg->getOperation()),
172  iagg->getFunction(),
173  std::make_unique<RamRelationReference>(&iagg->getRelation()),
174  souffle::clone(&iagg->getExpression()), souffle::clone(&iagg->getCondition()),
175  iagg->getTupleId());
176  } else {
177  fatal("New RamIndexOperation subclass found but not supported while transforming "
178  "index.");
179  }
180  }
181  }
182  node->apply(makeLambdaRamMapper(removeEmptyIndexRewriter));
183  return node;
184  };
185  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(removeEmptyIndexRewriter));
186  });
187  return changed;
188 }
