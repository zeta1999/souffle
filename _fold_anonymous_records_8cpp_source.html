37 bool FoldAnonymousRecords::isValidRecordConstraint(const AstLiteral* literal) {
38  auto constraint = dynamic_cast<const AstBinaryConstraint*>(literal);
39 
40  if (constraint == nullptr) {
41  return false;
42  }
43 
44  const auto* left = constraint->getLHS();
45  const auto* right = constraint->getRHS();
46 
47  const auto* leftRecord = dynamic_cast<const AstRecordInit*>(left);
48  const auto* rightRecord = dynamic_cast<const AstRecordInit*>(right);
49 
50  // Check if arguments are records records.
51  if ((leftRecord == nullptr) || (rightRecord == nullptr)) {
52  return false;
53  }
54 
55  // Check if records are of the same size.
56  if (leftRecord->getChildNodes().size() != rightRecord->getChildNodes().size()) {
57  return false;
58  }
59 
60  // Check if operator is "=" or "!="
61  auto op = constraint->getOperator();
62 
63  return isEqConstraint(op) || isEqConstraint(negatedConstraintOp(op));
64 }

74 std::vector<std::unique_ptr<AstLiteral>> FoldAnonymousRecords::expandRecordBinaryConstraint(
75  const AstBinaryConstraint& constraint) {
76  std::vector<std::unique_ptr<AstLiteral>> replacedContraint;
77 
78  const auto* left = dynamic_cast<AstRecordInit*>(constraint.getLHS());
79  const auto* right = dynamic_cast<AstRecordInit*>(constraint.getRHS());
80  assert(left != nullptr && "Non-record passed to record method");
81  assert(right != nullptr && "Non-record passed to record method");
82 
83  auto leftChildren = left->getArguments();
84  auto rightChildren = right->getArguments();
85 
86  assert(leftChildren.size() == rightChildren.size());
87 
88  // [a, b..] = [c, d...] â†’ a = c, b = d ...
89  for (size_t i = 0; i < leftChildren.size(); ++i) {
90  auto newConstraint = std::make_unique<AstBinaryConstraint>(
91  constraint.getOperator(), souffle::clone(leftChildren[i]), souffle::clone(rightChildren[i]));
92  replacedContraint.push_back(std::move(newConstraint));
93  }
94 
95  // Handle edge case. Empty records.
96  if (leftChildren.size() == 0) {
97  if (isEqConstraint(constraint.getOperator())) {
98  replacedContraint.emplace_back(new AstBooleanConstraint(true));
99  } else {
100  replacedContraint.emplace_back(new AstBooleanConstraint(false));
101  }
102  }
103 
104  return replacedContraint;
105 }

107 void FoldAnonymousRecords::transformClause(
108  const AstClause& clause, std::vector<std::unique_ptr<AstClause>>& newClauses) {
109  // If we have an inequality constraint, we need to create new clauses
110  // At most one inequality constraint will be expanded in a single pass.
111  AstBinaryConstraint* neqConstraint = nullptr;
112 
113  std::vector<std::unique_ptr<AstLiteral>> newBody;
114  for (auto* literal : clause.getBodyLiterals()) {
115  if (isValidRecordConstraint(literal)) {
116  const AstBinaryConstraint& constraint = dynamic_cast<AstBinaryConstraint&>(*literal);
117 
118  // Simple case, [a_0, ..., a_n] = [b_0, ..., b_n]
119  if (isEqConstraint(constraint.getOperator())) {
120  auto transformedLiterals = expandRecordBinaryConstraint(constraint);
121  std::move(std::begin(transformedLiterals), std::end(transformedLiterals),
122  std::back_inserter(newBody));
123 
124  // else if: Case [a_0, ..., a_n] != [b_0, ..., b_n].
125  // track single such case, it will be expanded in the end.
126  } else if (neqConstraint == nullptr) {
127  neqConstraint = dynamic_cast<AstBinaryConstraint*>(literal);
128 
129  // Else: repeated inequality.
130  } else {
131  newBody.push_back(souffle::clone(literal));
132  }
133 
134  // else, we simply copy the literal.
135  } else {
136  newBody.push_back(souffle::clone(literal));
137  }
138  }
139 
140  // If no inequality: create a single modified clause.
141  if (neqConstraint == nullptr) {
142  auto newClause = souffle::clone(&clause);
143  newClause->setBodyLiterals(std::move(newBody));
144  newClauses.emplace_back(std::move(newClause));
145 
146  // Else: For each pair in negation, we need an extra clause.
147  } else {
148  auto transformedLiterals = expandRecordBinaryConstraint(*neqConstraint);
149 
150  for (auto it = begin(transformedLiterals); it != end(transformedLiterals); ++it) {
151  auto newClause = souffle::clone(&clause);
152  auto copyBody = souffle::clone(newBody);
153  copyBody.push_back(std::move(*it));
154 
155  newClause->setBodyLiterals(std::move(copyBody));
156 
157  newClauses.push_back(std::move(newClause));
158  }
159  }
160 }

162 bool FoldAnonymousRecords::transform(AstTranslationUnit& translationUnit) {
163  bool changed = false;
164  AstProgram& program = *translationUnit.getProgram();
165 
166  std::vector<std::unique_ptr<AstClause>> newClauses;
167 
168  for (const auto* clause : program.getClauses()) {
169  if (containsValidRecordConstraint(*clause)) {
170  changed = true;
171  transformClause(*clause, newClauses);
172  } else {
173  newClauses.emplace_back(clause->clone());
174  }
175  }
176 
177  // Update AstProgram.
178  if (changed) {
179  program.setClauses(std::move(newClauses));
180  }
181  return changed;
182 }
