26 InterpreterRelation::InterpreterRelation(std::size_t arity, std::size_t auxiliaryArity, std::string name,
27  std::vector<std::string> attributeTypes, const MinIndexSelection& orderSet, IndexFactory factory)
28  : relName(std::move(name)), arity(arity), auxiliaryArity(auxiliaryArity),
29  attributeTypes(std::move(attributeTypes)) {
30  for (auto order : orderSet.getAllOrders()) {
31  // Expand the order to a total order
32  AttributeSet set{order.begin(), order.end()};
33 
34  for (std::size_t i = 0; i < arity; ++i) {
35  if (set.find(i) == set.end()) {
36  order.push_back(i);
37  }
38  }
39  indexes.push_back(factory(Order(order)));
40  }
41 
42  // Use the first index as default main index
43  main = indexes[0].get();
44 }

57 bool InterpreterRelation::insert(const TupleRef& tuple) {
58  if (!main->insert(tuple)) {
59  return false;

162 bool InterpreterIndirectRelation::insert(const TupleRef& tuple) {
163  if (main->contains(tuple)) {
164  return false;
165  }
166 
167  int blockIndex = numTuples / (BLOCK_SIZE / arity);
168  int tupleIndex = (numTuples % (BLOCK_SIZE / arity)) * arity;
169 
170  if (tupleIndex == 0) {
171  blockList.push_back(std::make_unique<RamDomain[]>(BLOCK_SIZE));
172  }
173 
174  RamDomain* newTuple = &blockList[blockIndex][tupleIndex];
175  for (size_t i = 0; i < arity; ++i) {
176  newTuple[i] = tuple[i];
177  }
178 
179  // update all indexes with new tuple
180  for (auto& cur : indexes) {
181  cur->insert(TupleRef(newTuple, arity));
182  }
183 
184  // increment relation size
185  numTuples++;
186 
187  return true;
188 }
