33 void RedundantRelationsAnalysis::run(const AstTranslationUnit& translationUnit) {
34  precedenceGraph = translationUnit.getAnalysis<PrecedenceGraphAnalysis>();
35 
36  std::set<const AstRelation*> work;
37  std::set<const AstRelation*> notRedundant;
38  auto* ioType = translationUnit.getAnalysis<IOType>();
39 
40  const std::vector<AstRelation*>& relations = translationUnit.getProgram()->getRelations();
41  /* Add all output relations to the work set */
42  for (const AstRelation* r : relations) {
43  if (ioType->isOutput(r)) {
44  work.insert(r);
45  }
46  }
47 
48  /* Find all relations which are not redundant for the computations of the
49  output relations. */
50  while (!work.empty()) {
51  /* Chose one element in the work set and add it to notRedundant */
52  const AstRelation* u = *(work.begin());
53  work.erase(work.begin());
54  notRedundant.insert(u);
55 
56  /* Find all predecessors of u and add them to the worklist
57  if they are not in the set notRedundant */
58  for (const AstRelation* predecessor : precedenceGraph->graph().predecessors(u)) {
59  if (notRedundant.count(predecessor) == 0u) {
60  work.insert(predecessor);
61  }
62  }
63  }
64 
65  /* All remaining relations are redundant. */
66  redundantRelations.clear();
67  for (const AstRelation* r : relations) {
68  if (notRedundant.count(r) == 0u) {
69  redundantRelations.insert(r);
70  }
71  }
72 }
