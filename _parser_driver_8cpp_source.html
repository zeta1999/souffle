51 std::unique_ptr<AstTranslationUnit> ParserDriver::parse(
52  const std::string& filename, FILE* in, ErrorReport& errorReport, DebugReport& debugReport) {
53  translationUnit =
54  std::make_unique<AstTranslationUnit>(std::make_unique<AstProgram>(), errorReport, debugReport);
55  yyscan_t scanner;
56  scanner_data data;
57  data.yyfilename = filename;
58  yylex_init_extra(&data, &scanner);
59  yyset_in(in, scanner);
60 
61  yy::parser parser(*this, scanner);
62  parser.parse();
63 
64  yylex_destroy(scanner);
65

199  SrcLocation tagLoc, std::set<RelationTag> tags) {
200  if (any_of(incompatible, [&](auto&& x) { return contains(tags, x); })) {
201  error(tagLoc, tfm::format("%s qualifier already set", join(incompatible, "/")));
202  }

204  tags.insert(tag);
