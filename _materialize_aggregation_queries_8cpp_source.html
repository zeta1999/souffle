49 bool MaterializeAggregationQueriesTransformer::materializeAggregationQueries(
50  AstTranslationUnit& translationUnit) {
51  bool changed = false;
52 
53  AstProgram& program = *translationUnit.getProgram();
54  const TypeEnvironment& env = translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
55 
56  // if an aggregator has a body consisting of more than an atom => create new relation
57  int counter = 0;
58  visitDepthFirst(program, [&](const AstClause& clause) {
59  visitDepthFirst(clause, [&](const AstAggregator& agg) {
60  // check whether a materialization is required
61  if (!needsMaterializedRelation(const_cast<AstAggregator&>(agg), program)) {
62  return;
63  }
64  changed = true;
65 
66  // -- create a new clause --
67 
68  auto relName = "__agg_body_rel_" + toString(counter++);
69  while (getRelation(program, relName) != nullptr) {
70  relName = "__agg_body_rel_" + toString(counter++);
71  }
72  // create the new clause for the materialised rule
73  auto* aggClause = new AstClause();
74  // create the body of the new materialised rule
75  for (const auto& cur : agg.getBodyLiterals()) {
76  aggClause->addToBody(souffle::clone(cur));
77  }
78  // find stuff for which we need a grounding
79  for (const auto& argPair : getGroundedTerms(translationUnit, *aggClause)) {
80  const auto* variable = dynamic_cast<const AstVariable*>(argPair.first);
81  bool variableIsGrounded = argPair.second;
82  // if it's not even a variable type or the term is grounded
83  // then skip it
84  if (variable == nullptr || variableIsGrounded) {
85  continue;
86  }
87 
88  for (const auto& lit : clause.getBodyLiterals()) {
89  const auto* atom = dynamic_cast<const AstAtom*>(lit);
90  if (atom == nullptr) {
91  continue; // it's not an atom so it can't help ground anything
92  }
93  for (const auto& arg : atom->getArguments()) {
94  const auto* atomVariable = dynamic_cast<const AstVariable*>(arg);
95  // if this atom contains the variable I need to ground, add it
96  if ((atomVariable != nullptr) && variable->getName() == atomVariable->getName()) {
97  // expand the body with this one so that it will ground this variable
98  aggClause->addToBody(souffle::clone(atom));
99  break;
100  }
101  }
102  }
103  }
104 
105  auto* head = new AstAtom();
106  head->setQualifiedName(relName);
107  std::vector<bool> symbolArguments;
108 
109  // Ensure each variable is only added once
110  std::set<std::string> variables;
111  visitDepthFirst(*aggClause, [&](const AstVariable& var) { variables.insert(var.getName()); });
112 
113  // Insert all variables occurring in the body of the aggregate into the head
114  for (const auto& var : variables) {
115  head->addArgument(std::make_unique<AstVariable>(var));
116  }
117 
118  aggClause->setHead(std::unique_ptr<AstAtom>(head));
119 
120  // instantiate unnamed variables in count operations
121  if (agg.getOperator() == AggregateOp::COUNT) {
122  int count = 0;
123  for (const auto& cur : aggClause->getBodyLiterals()) {
124  cur->apply(makeLambdaAstMapper(
125  [&](std::unique_ptr<AstNode> node) -> std::unique_ptr<AstNode> {
126  // check whether it is a unnamed variable
127  auto* var = dynamic_cast<AstUnnamedVariable*>(node.get());
128  if (var == nullptr) {
129  return node;
130  }
131 
132  // replace by variable
133  auto name = " _" + toString(count++);
134  auto res = new AstVariable(name);
135 
136  // extend head
137  head->addArgument(souffle::clone(res));
138 
139  // return replacement
140  return std::unique_ptr<AstNode>(res);
141  }));
142  }
143  }
144 
145  // -- build relation --
146 
147  auto* rel = new AstRelation();
148  rel->setQualifiedName(relName);
149  // add attributes
150  std::map<const AstArgument*, TypeSet> argTypes =
151  TypeAnalysis::analyseTypes(env, *aggClause, program);
152  for (const auto& cur : head->getArguments()) {
153  rel->addAttribute(std::make_unique<AstAttribute>(toString(*cur),
154  (isOfKind(argTypes[cur], TypeAttribute::Signed)) ? "number" : "symbol"));
155  }
156 
157  program.addClause(std::unique_ptr<AstClause>(aggClause));
158  program.addRelation(std::unique_ptr<AstRelation>(rel));
159 
160  // -- update aggregate --
161 
162  // count the usage of variables in the clause
163  // outside of aggregates. Note that the visitor
164  // is exhaustive hence double counting occurs
165  // which needs to be deducted for variables inside
166  // the aggregators and variables in the expression
167  // of aggregate need to be added. Counter is zero
168  // if the variable is local to the aggregate
169  std::map<std::string, int> varCtr;
170  visitDepthFirst(clause, [&](const AstArgument& arg) {
171  if (const auto* a = dynamic_cast<const AstAggregator*>(&arg)) {
172  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]--; });
173  if (a->getTargetExpression() != nullptr) {
174  visitDepthFirst(*a->getTargetExpression(),
175  [&](const AstVariable& var) { varCtr[var.getName()]++; });
176  }
177  } else {
178  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]++; });
179  }
180  });
181  std::vector<std::unique_ptr<AstArgument>> args;
182  for (auto arg : head->getArguments()) {
183  if (auto* var = dynamic_cast<AstVariable*>(arg)) {
184  // replace local variable by underscore if local
185  if (varCtr[var->getName()] == 0) {
186  args.emplace_back(new AstUnnamedVariable());
187  continue;
188  }
189  }
190  args.emplace_back(arg->clone());
191  }
192  auto aggAtom =
193  std::make_unique<AstAtom>(head->getQualifiedName(), std::move(args), head->getSrcLoc());
194 
195  std::vector<std::unique_ptr<AstLiteral>> newBody;
196  newBody.push_back(std::move(aggAtom));
197  const_cast<AstAggregator&>(agg).setBody(std::move(newBody));
198  });
199  });
200  return changed;
201 }

203 bool MaterializeAggregationQueriesTransformer::needsMaterializedRelation(
204  AstAggregator& agg, AstProgram& program) {
205  // everything with more than 1 body literal => materialize
206  int countAtoms = 0;
207  const AstAtom* atom = nullptr;
208  for (const auto& literal : agg.getBodyLiterals()) {
209  const AstAtom* currentAtom = dynamic_cast<const AstAtom*>(literal);
210  if (currentAtom != nullptr) {
211  ++countAtoms;
212  atom = currentAtom;
213  }
214  }
215 
216  if (countAtoms > 1) {
217  return true;
218  }
219 
220  auto newAtom = std::make_unique<AstAtom>();
221  // Create a dummy true atom so that an aggregate without any atom
222  // can be evaluated without materialising
223  if (atom == nullptr) {
224  std::string relName = "+Tautology";
225  newAtom->setQualifiedName(relName);
226  auto clause = std::make_unique<AstClause>();
227  clause->setHead(souffle::clone(newAtom));
228  program.addClause(std::move(clause));
229  if (getRelation(program, relName) == nullptr) {
230  auto tautologyRel = std::make_unique<AstRelation>();
231  tautologyRel->setQualifiedName(relName);
232  program.addRelation(std::move(tautologyRel));
233  }
234  VecOwn<AstLiteral> newBody;
235  for (const auto& lit : agg.getBodyLiterals()) {
236  newBody.push_back(souffle::clone(lit));
237  }
238  newBody.push_back(souffle::clone(newAtom));
239  agg.setBody(std::move(newBody));
240  atom = newAtom.get();
241  }
242 
243  // If the same variable occurs several times => materialize
244  bool duplicates = false;
245  std::set<std::string> vars;
246  visitDepthFirst(*atom,
247  [&](const AstVariable& var) { duplicates = duplicates || !vars.insert(var.getName()).second; });
248 
249  // If there are duplicates a materialization is required
250  // for all others the materialization can be skipped
251  return duplicates;
252 }
