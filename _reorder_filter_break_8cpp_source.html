32  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
33  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
34  // find filter-break nesting
35  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
36  if (const RamBreak* br = dynamic_cast<RamBreak*>(&filter->getOperation())) {

43  }
