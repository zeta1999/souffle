33  bool changed = false;
34 
35  // parallelize the most outer loop only
36  // most outer loops can be scan/choice/indexScan/indexChoice
37  visitDepthFirst(program, [&](const RamQuery& query) {
38  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> parallelRewriter =
39  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
40  if (const RamScan* scan = dynamic_cast<RamScan*>(node.get())) {
41  if (scan->getTupleId() == 0 && scan->getRelation().getArity() > 0) {
42  if (nullptr == dynamic_cast<RamProject*>(&scan->getOperation())) {
43  changed = true;
44  return std::make_unique<RamParallelScan>(
45  std::make_unique<RamRelationReference>(&scan->getRelation()),
46  scan->getTupleId(), souffle::clone(&scan->getOperation()),

54  std::make_unique<RamRelationReference>(&choice->getRelation()),
55  choice->getTupleId(), souffle::clone(&choice->getCondition()),

57  }
