36 void SCCGraphAnalysis::run(const AstTranslationUnit& translationUnit) {
37  precedenceGraph = translationUnit.getAnalysis<PrecedenceGraphAnalysis>();
38  ioType = translationUnit.getAnalysis<IOType>();
39  sccToRelation.clear();
40  relationToScc.clear();
41  predecessors.clear();
42  successors.clear();
43 
44  /* Compute SCC */
45  std::vector<AstRelation*> relations = translationUnit.getProgram()->getRelations();
46  size_t counter = 0;
47  size_t numSCCs = 0;
48  std::stack<const AstRelation*> S;
49  std::stack<const AstRelation*> P;
50  std::map<const AstRelation*, size_t> preOrder; // Pre-order number of a node (for Gabow's Algo)
51  for (const AstRelation* relation : relations) {
52  relationToScc[relation] = preOrder[relation] = (size_t)-1;
53  }
54  for (const AstRelation* relation : relations) {
55  if (preOrder[relation] == (size_t)-1) {
56  scR(relation, preOrder, counter, S, P, numSCCs);
57  }
58  }
59 
60  /* Build SCC graph */
61  successors.resize(numSCCs);
62  predecessors.resize(numSCCs);
63  for (const AstRelation* u : relations) {
64  for (const AstRelation* v : precedenceGraph->graph().predecessors(u)) {
65  auto scc_u = relationToScc[u];
66  auto scc_v = relationToScc[v];
67  assert(scc_u < numSCCs && "Wrong range");
68  assert(scc_v < numSCCs && "Wrong range");
69  if (scc_u != scc_v) {
70  predecessors[scc_u].insert(scc_v);
71  successors[scc_v].insert(scc_u);
72  }
73  }
74  }
75 
76  /* Store the relations for each SCC */
77  sccToRelation.resize(numSCCs);
78  for (const AstRelation* relation : relations) {
79  sccToRelation[relationToScc[relation]].insert(relation);
80  }
81 }

83 /* Compute strongly connected components using Gabow's algorithm (cf. Algorithms in
84  * Java by Robert Sedgewick / Part 5 / Graph * algorithms). The algorithm has linear
85  * runtime. */
86 void SCCGraphAnalysis::scR(const AstRelation* w, std::map<const AstRelation*, size_t>& preOrder,
87  size_t& counter, std::stack<const AstRelation*>& S, std::stack<const AstRelation*>& P,
88  size_t& numSCCs) {
89  preOrder[w] = counter++;
90  S.push(w);
91  P.push(w);
92  for (const AstRelation* t : precedenceGraph->graph().predecessors(w)) {
93  if (preOrder[t] == (size_t)-1) {
94  scR(t, preOrder, counter, S, P, numSCCs);
95  } else if (relationToScc[t] == (size_t)-1) {
96  while (preOrder[P.top()] > preOrder[t]) {
97  P.pop();
98  }
99  }
100  }
101  if (P.top() == w) {
102  P.pop();
103  } else {
104  return;
105  }
106 
107  const AstRelation* v;
108  do {
109  v = S.top();
110  S.pop();
111  relationToScc[v] = numSCCs;
112  } while (v != w);
113  numSCCs++;
114 }

116 void SCCGraphAnalysis::print(std::ostream& os) const {
117  const std::string& name = Global::config().get("name");
118  std::stringstream ss;
119  /* Print SCC graph */
120  ss << "digraph {" << std::endl;
121  /* Print nodes of SCC graph */

123  ss << "\t" << name << "_" << scc << "[label = \"";
124  ss << join(getInternalRelations(scc), ",\\n",
125  [](std::ostream& out, const AstRelation* rel) { out << rel->getQualifiedName(); });
126  ss << "\" ];" << std::endl;
127  }
128  for (size_t scc = 0; scc < getNumberOfSCCs(); scc++) {
129  for (auto succ : getSuccessorSCCs(scc)) {
130  ss << "\t" << name << "_" << scc << " -> " << name << "_" << succ << ";" << std::endl;
131  }
132  }
133  ss << "}";
134  printHTMLGraph(os, ss.str(), getName());
135 }
