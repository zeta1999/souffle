54  RamProgram(std::vector<std::unique_ptr<RamRelation>> rels, std::unique_ptr<RamStatement> main,
55  std::map<std::string, std::unique_ptr<RamStatement>> subs)
56  : relations(std::move(rels)), main(std::move(main)), subroutines(std::move(subs)) {
57  assert(this->main != nullptr && "Main program is a null-pointer");
58  for (const auto& rel : relations) {

102  RamProgram* clone() const override {
103  auto* res = new RamProgram();
104  res->main = souffle::clone(main);
105  for (auto& rel : relations) {
106  res->relations.push_back(souffle::clone(rel));
107  }
108  for (auto& sub : subroutines) {
109  res->subroutines[sub.first] = souffle::clone(sub.second);
110  }
111  std::map<const RamRelation*, const RamRelation*> refMap;
112  res->apply(makeLambdaRamMapper([&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
113  // rewire relation references to newly cloned relations
114  if (const RamRelationReference* relRef = dynamic_cast<RamRelationReference*>(node.get())) {
115  const RamRelation* rel = refMap[relRef->get()];
116  assert(rel != nullptr && "dangling RAM relation reference");
117  return std::make_unique<RamRelationReference>(rel);
118  } else {
119  return node;
120  }
121  }));
122  return res;
123  }
