32 bool ReorderConditionsTransformer::reorderConditions(RamProgram& program) {
33  bool changed = false;
34  visitDepthFirst(program, [&](const RamQuery& query) {
35  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
36  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
37  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
38  const RamCondition* condition = &filter->getCondition();
39  std::vector<std::unique_ptr<RamCondition>> sortedConds;
40  std::vector<std::unique_ptr<RamCondition>> condList = toConjunctionList(condition);
41  for (auto& cond : condList) {
42  sortedConds.emplace_back(cond->clone());
43  }
44  std::sort(sortedConds.begin(), sortedConds.end(),
45  [&](std::unique_ptr<RamCondition>& a, std::unique_ptr<RamCondition>& b) {
46  return rca->getComplexity(a.get()) < rca->getComplexity(b.get());
47  });
48 
49  if (!std::equal(sortedConds.begin(), sortedConds.end(), condList.begin(),
50  [](std::unique_ptr<RamCondition>& a, std::unique_ptr<RamCondition>& b) {
51  return *a == *b;
52  })) {
53  changed = true;
54  node = std::make_unique<RamFilter>(
55  std::unique_ptr<RamCondition>(toCondition(sortedConds)),
56  souffle::clone(&filter->getOperation()));
57  }
58  }
59  node->apply(makeLambdaRamMapper(filterRewriter));
60  return node;
61  };
62  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
63  });
64  return changed;
65 }
