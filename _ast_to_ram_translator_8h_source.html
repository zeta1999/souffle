98  /**
99  * Concrete attribute
100  */
101  struct Location {
102  int identifier{};
103  int element{};
104  std::unique_ptr<RamRelationReference> relation{nullptr};
105 
106  Location() = default;
107 
108  Location(int ident, int elem, std::unique_ptr<RamRelationReference> rel = nullptr)
109  : identifier(ident), element(elem), relation(std::move(rel)) {}
110 
111  Location(const Location& l) : identifier(l.identifier), element(l.element) {
112  if (l.relation != nullptr) {
113  relation = souffle::clone(l.relation);
114  }
115  }
116 
117  Location& operator=(Location other) {
118  identifier = other.identifier;
119  element = other.element;
120  relation = std::move(other.relation);
121  return *this;
122  }
123 
124  bool operator==(const Location& loc) const {
125  return identifier == loc.identifier && element == loc.element;
126  }
127 
128  bool operator!=(const Location& loc) const {
129  return !(*this == loc);
130  }
131 
132  bool operator<(const Location& loc) const {
133  return identifier < loc.identifier || (identifier == loc.identifier && element < loc.element);
134  }
135 
136  void print(std::ostream& out) const {
137  out << "(" << identifier << "," << element << ")";
138  }
139 
140  friend std::ostream& operator<<(std::ostream& out, const Location& loc) {
141  loc.print(out);
142  return out;
143  }
144  };

151  class ValueIndex {
152  /**
153  * The type mapping variables (referenced by their names) to the
154  * locations where they are used.
155  */
156  using variable_reference_map = std::map<std::string, std::set<Location>>;
157 
158  /**
159  * The type mapping record init expressions to their definition points,
160  * hence the point where they get grounded/bound.
161  */
162  using record_definition_map = std::map<const AstRecordInit*, Location>;
163 
164  /**
165  * A map from generative `AstArgument`s to storage locations. Note,
166  * since in this case AstArgument are indexed by their values (not their
167  * address) no standard map can be utilized.
168  * (By-value indexing induces an ad-hoc form of CSE.)
169  */
170  using generator_location_map = std::vector<std::pair<const AstArgument*, Location>>;
171 
172  /** The index of variable accesses */
173  variable_reference_map var_references;
174 
175  /** The index of record definition points */
176  record_definition_map record_definitions;
177 
178  /** The level of a nested ram operation that is handling a generator operation */
179  generator_location_map arg_generator_locations;
180 
181  public:
182  // -- variables --
183 
184  void addVarReference(const AstVariable& var, const Location& l) {
185  std::set<Location>& locs = var_references[var.getName()];
186  locs.insert(l);
187  }
188 
189  void addVarReference(const AstVariable& var, int ident, int pos,
190  std::unique_ptr<RamRelationReference> rel = nullptr) {
191  addVarReference(var, Location({ident, pos, std::move(rel)}));
192  }
193 
194  bool isDefined(const AstVariable& var) const {
195  return var_references.find(var.getName()) != var_references.end();
196  }
197 
198  const Location& getDefinitionPoint(const AstVariable& var) const {
199  auto pos = var_references.find(var.getName());
200  assert(pos != var_references.end() && "Undefined variable referenced!");
201  return *pos->second.begin();
202  }
203 
204  const variable_reference_map& getVariableReferences() const {
205  return var_references;
206  }
207 
208  // -- records --
209 
210  // - definition -
211 
212  void setRecordDefinition(const AstRecordInit& init, const Location& l) {
213  record_definitions[&init] = l;
214  }
215 
216  void setRecordDefinition(const AstRecordInit& init, int ident, int pos,
217  std::unique_ptr<RamRelationReference> rel = nullptr) {
218  setRecordDefinition(init, Location({ident, pos, std::move(rel)}));
219  }
220 
221  const Location& getDefinitionPoint(const AstRecordInit& init) const {
222  auto pos = record_definitions.find(&init);
223  if (pos != record_definitions.end()) {
224  return pos->second;
225  }
226 
227  fatal("requested location for undefined record!");
228  }
229 
230  // -- generators (aggregates & some functors) --
231 
232  void setGeneratorLoc(const AstArgument& agg, const Location& loc) {
233  arg_generator_locations.push_back(std::make_pair(&agg, loc));
234  }
235 
236  const Location& getGeneratorLoc(const AstArgument& arg) const {
237  // search list
238  for (const auto& cur : arg_generator_locations) {
239  if (*cur.first == arg) {
240  return cur.second;
241  }
242  }
243 
244  fatal("arg `%s` has no generator location", arg);
245  }
246 
247  // -- others --
248 
249  bool isGenerator(const int level) const {
250  // check for aggregator definitions
251  return any_of(arg_generator_locations,
252  [&level](const auto& location) { return location.second.identifier == level; });
253  }
254 
255  bool isSomethingDefinedOn(int level) const {
256  // check for variable definitions
257  for (const auto& cur : var_references) {
258  if (cur.second.begin()->identifier == level) {
259  return true;
260  }
261  }
262  // check for record definitions
263  for (const auto& cur : record_definitions) {
264  if (cur.second.identifier == level) {
265  return true;
266  }
267  }
268  // nothing defined on this level
269  return false;
270  }

399  /**
400  * Get ram representation of constant.
401  */
402  RamDomain getConstantRamRepresentation(const AstConstant& constant) {
403  if (auto strConstant = dynamic_cast<const AstStringConstant*>(&constant)) {
404  return getSymbolTable().lookup(strConstant->getConstant());
405  } else if (dynamic_cast<const AstNilConstant*>(&constant) != nullptr) {
406  return 0;
407  } else if (auto* numConstant = dynamic_cast<const AstNumericConstant*>(&constant)) {
408  assert(numConstant->getType().has_value());
409  switch (*numConstant->getType()) {
410  case AstNumericConstant::Type::Int:
411  return RamSignedFromString(numConstant->getConstant(), nullptr, 0);
412  case AstNumericConstant::Type::Uint:
413  return RamUnsignedFromString(numConstant->getConstant(), nullptr, 0);
414  case AstNumericConstant::Type::Float: return RamFloatFromString(numConstant->getConstant());
415  }
416  }
417 
418  fatal("unaccounted-for constant");
419  }
