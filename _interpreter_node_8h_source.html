9 /************************************************************************
10  *
11  * @file InterpreterNode.h
12  *
13  * Declares the Interpreter Node class. The interpreter node class
14  * is a compact executable representation of RAM nodes for interpretation.
15  * There are two main reasons for the class:
16  * - node types are exposed in form of enums so that fast switch-statements
17  * can be employed for interpretation (visitor patterns with their
18  * double-dispatch are too slow).
19  * - nodes are decorated with data so that frequent on-the-fly data-structure
20  * lookups are avoided.
21  * Every interpreter node is associated with a unique RAM node.
22  ***********************************************************************/

89 /**
90  * @class InterpreterNode
91  * @brief This is a shadow node for a RamNode that is enriched for
92  * with local information so that the interpreter is executing
93  * quickly.
94  */
95 
96 class InterpreterNode {
97 public:
98  using RelationHandle = std::unique_ptr<InterpreterRelation>;
99 
100  InterpreterNode(enum InterpreterNodeType ty, const RamNode* sdw, RelationHandle* relHandle = nullptr)
101  : type(ty), shadow(sdw), relHandle(relHandle) {}
102  virtual ~InterpreterNode() = default;
103 
104  /** @brief get node type */
105  inline enum InterpreterNodeType getType() const {
106  return type;
107  }
108 
109  /** @brief get shadow node, i.e., RAM node */
110  inline const RamNode* getShadow() const {
111  return shadow;
112  }
113 
114  /** @brief get relation from handle */
115  InterpreterRelation* getRelation() const {
116  assert(relHandle && "No relation cached\n");
117  return (*relHandle).get();
118  }
119 
120 protected:
121  enum InterpreterNodeType type;
122  const RamNode* shadow;
123  RelationHandle* const relHandle;
124 };

196 /**
197  * @class InterpreterSuperInstruction
198  * @brief This class encodes information for a super-instruction, which is
199  * used to eliminate Number and TupleElement in index/project/existence operation.
200  */
201 class InterpreterSuperInstruction {
202 public:
203  InterpreterSuperInstruction(size_t i) {
204  first.resize(i);
205  second.resize(i);
206  }
207 
208  InterpreterSuperInstruction(const InterpreterSuperInstruction&) = delete;
209  InterpreterSuperInstruction& operator=(const InterpreterSuperInstruction&) = delete;
210  InterpreterSuperInstruction(InterpreterSuperInstruction&&) = default;
211 
212  /** @brief constant value in the lower bound/pattern */
213  std::vector<RamDomain> first;
214  /** @brief constant value in the upper bound */
215  std::vector<RamDomain> second;
216  /** @brief Encoded tupleElement expressions in the lower bound/pattern */
217  std::vector<std::array<size_t, 3>> tupleFirst;
218  /** @brief Encoded tupleElement expressions in the upper bound */
219  std::vector<std::array<size_t, 3>> tupleSecond;
220  /** @brief Generic expressions in the lower bound/pattern */
221  std::vector<std::pair<size_t, std::unique_ptr<InterpreterNode>>> exprFirst;
222  /** @brief Generic expressions in the upper bound */
223  std::vector<std::pair<size_t, std::unique_ptr<InterpreterNode>>> exprSecond;
224 };
