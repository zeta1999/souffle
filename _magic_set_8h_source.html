122  friend std::ostream& operator<<(std::ostream& out, const AdornedClause& arg) {
123  size_t currpos = 0;
124  bool firstAdded = true;
125  out << arg.clause->getHead()->getQualifiedName() << "{" << arg.headAdornment << "} :- ";
126 
127  std::vector<AstLiteral*> bodyLiterals = arg.clause->getBodyLiterals();
128  for (AstLiteral* literal : bodyLiterals) {
129  if (auto* atom = dynamic_cast<AstAtom*>(literal)) {
130  if (!firstAdded) {
131  out << ", ";
132  }
133  firstAdded = false;
134  out << atom->getQualifiedName() << "{_}";
135  } else if (auto* neg = dynamic_cast<AstNegation*>(literal)) {
136  if (!firstAdded) {
137  out << ", ";
138  }
139  firstAdded = false;
140  if (currpos < arg.bodyAdornment.size()) {
141  out << neg->getAtom()->getQualifiedName() << "{" << arg.bodyAdornment[currpos++] << "}";
142  } else {
143  out << neg->getAtom()->getQualifiedName() << "{__}";
144  ++currpos;
145  }
146  }
147  }
148  out << ". [order: " << arg.ordering << "]";
149 
150  return out;
151  }

169  void addBinding(const std::string& newVariableName, const AstArgument* arg) {
170  originalArguments[newVariableName] = souffle::clone(arg);
171 
172  // find the variable dependencies
173  std::set<std::string> dependencies;
174  visitDepthFirst(*arg, [&](const AstVariable& var) { dependencies.insert(var.getName()); });
175  varDependencies[newVariableName] = dependencies;
176  }
