125  std::set<const AstRelation*> bodyRelations;

127  visitDepthFirst(

137 size_t getClauseNum(const AstProgram* program, const AstClause* clause) {
138  // TODO (azreika): This number might change between the provenance transformer and the AST->RAM
139  // translation. Might need a better way to assign IDs to clauses... (see PR #1288).
140  const AstRelation* rel = getRelation(*program, clause->getHead()->getQualifiedName());
141  assert(rel != nullptr && "clause relation does not exist");
142 
143  size_t clauseNum = 1;
144  for (const auto* cur : getClauses(*program, *rel)) {
145  bool isFact = cur->getBodyLiterals().empty();
146  if (cur == clause) {
147  return isFact ? 0 : clauseNum;
148  }
149 
150  if (!isFact) {
151  clauseNum++;
152  }
153  }
154 
155  fatal("clause does not exist");

201 bool isFact(const AstClause& clause) {
202  // there must be a head
203  if (clause.getHead() == nullptr) {
204  return false;
205  }
206  // there must not be any body clauses
207  if (!clause.getBodyLiterals().empty()) {
208  return false;
209  }
210 
211  // and there are no aggregates
212  bool hasAggregatesOrMultiResultFunctor = false;
213  visitDepthFirst(*clause.getHead(), [&](const AstArgument& arg) {
214  if (dynamic_cast<const AstAggregator*>(&arg)) {
215  hasAggregatesOrMultiResultFunctor = true;
216  }
217 
218  auto func = dynamic_cast<const AstIntrinsicFunctor*>(&arg);
219  auto info = func ? func->getFunctionInfo() : nullptr;
220  if (info && info->multipleResults) {
221  hasAggregatesOrMultiResultFunctor = true;
222  }
223  });
224  return !hasAggregatesOrMultiResultFunctor;
225 }

241 AstClause* reorderAtoms(const AstClause* clause, const std::vector<unsigned int>& newOrder) {
242  // Find all atom positions
243  std::vector<unsigned int> atomPositions;
244  std::vector<AstLiteral*> bodyLiterals = clause->getBodyLiterals();
245  for (unsigned int i = 0; i < bodyLiterals.size(); i++) {
246  if (dynamic_cast<AstAtom*>(bodyLiterals[i]) != nullptr) {
247  atomPositions.push_back(i);
248  }
249  }
250 
251  // Validate given order
252  assert(newOrder.size() == atomPositions.size());
253  std::vector<unsigned int> nopOrder;
254  for (unsigned int i = 0; i < atomPositions.size(); i++) {
255  nopOrder.push_back(i);
256  }
257  assert(std::is_permutation(nopOrder.begin(), nopOrder.end(), newOrder.begin()));
258 
259  // Create a new clause with the given atom order, leaving the rest unchanged
260  AstClause* newClause = cloneHead(clause);
261  unsigned int currentAtom = 0;
262  for (unsigned int currentLiteral = 0; currentLiteral < bodyLiterals.size(); currentLiteral++) {
263  AstLiteral* literalToAdd = bodyLiterals[currentLiteral];
264  if (dynamic_cast<AstAtom*>(literalToAdd) != nullptr) {
265  // Atoms should be reordered
266  literalToAdd = bodyLiterals[atomPositions[newOrder[currentAtom++]]];
267  }
268  newClause->addToBody(souffle::clone(literalToAdd));
269  }
270 
271  return newClause;
272 }

284 IntrinsicFunctors validOverloads(const TypeAnalysis& typing, const AstIntrinsicFunctor& func) {
285  auto typeAttrs = [&](const AstArgument* arg) -> std::set<TypeAttribute> {
286  auto&& types = typing.getTypes(arg);
287  if (types.isAll())
288  return {TypeAttribute::Signed, TypeAttribute::Unsigned, TypeAttribute::Float,
289  TypeAttribute::Symbol, TypeAttribute::Record};
290 
291  std::set<TypeAttribute> tyAttrs;
292  for (auto&& ty : types)
293  tyAttrs.insert(getTypeAttribute(ty));
294  return tyAttrs;
295  };
296  auto retTys = typeAttrs(&func);
297  auto argTys = map(func.getArguments(), typeAttrs);
298 
299  auto candidates = filterNot(functorBuiltIn(func.getFunction()), [&](const IntrinsicFunctor& x) -> bool {
300  if (!x.variadic && argTys.size() != x.params.size()) return true; // arity mismatch?
301 
302  for (size_t i = 0; i < argTys.size(); ++i)
303  if (!contains(argTys[i], x.params[x.variadic ? 0 : i])) return true;
304 
305  return !contains(retTys, x.result);
306  });
307 
308  std::sort(
309  candidates.begin(), candidates.end(), [&](const IntrinsicFunctor& a, const IntrinsicFunctor& b) {
310  if (a.result != b.result) return a.result < b.result;
311  if (a.variadic != b.variadic) return a.variadic < b.variadic;
312  return std::lexicographical_compare(
313  a.params.begin(), a.params.end(), b.params.begin(), b.params.end());
314  });
315  return candidates;
316 }
