32 /** @brief Bidirectional mappping between records and record references */
33 class RecordMap {
34  /** arity of record */
35  const size_t arity;

37  /** hash function for unordered record map */
38  struct RecordHash {
39  std::size_t operator()(std::vector<RamDomain> record) const {
40  std::size_t seed = 0;
41  std::hash<RamDomain> domainHash;
42  for (RamDomain value : record) {
43  seed ^= domainHash(value) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
44  }
45  return seed;
46  }
47  };

49  /** map from records to references */
50  // TODO (b-scholz): replace vector<RamDomain> with something more memory-frugal
51  std::unordered_map<std::vector<RamDomain>, RamDomain, RecordHash> recordToIndex;
52 
53  /** array of records; index represents record reference */
54  // TODO (b-scholz): replace vector<RamDomain> with something more memory-frugal
55  std::vector<std::vector<RamDomain>> indexToRecord;

60  /** @brief converts record to a record reference */
61  // TODO (b-scholz): replace vector<RamDomain> with something more memory-frugal
62  RamDomain pack(const std::vector<RamDomain>& vector) {
63  RamDomain index;
64 #pragma omp critical(record_pack)
65  {
66  auto pos = recordToIndex.find(vector);
67  if (pos != recordToIndex.end()) {
68  index = pos->second;
69  } else {
70 #pragma omp critical(record_unpack)
71  {
72  indexToRecord.push_back(vector);
73  index = indexToRecord.size() - 1;
74  recordToIndex[vector] = index;
75 
76  // assert that new index is smaller than the range
77  assert(index != std::numeric_limits<RamDomain>::max());
78  }
79  }
80  }
81  return index;
82  }
83 
84  /** @brief convert record pointer to a record reference */
85  RamDomain pack(const RamDomain* tuple) {
86  // TODO (b-scholz): data is unnecessarily copied
87  // for a successful lookup. To avoid this, we should
88  // compute a hash of the pointer-array and traverse through
89  // the bucket list of the unordered map finding the record.
90  // Note that in case of non-existence, the record still needs to be
91  // copied for the newly created entry but this will be the less
92  // frequent case.
93  std::vector<RamDomain> tmp(arity);
94  for (size_t i = 0; i < arity; i++) {
95  tmp[i] = tuple[i];
96  }
97  return pack(tmp);
98  }
99 
100  /** @brief convert record reference to a record pointer */
101  const RamDomain* unpack(RamDomain index) const {
102  const RamDomain* res;
103 #pragma omp critical(record_unpack)
104  res = indexToRecord[index].data();
105  return res;
106  }

109 class RecordTable {
110 public:
111  RecordTable() = default;
112  virtual ~RecordTable() = default;
113 
114  /** @brief convert record to record reference */
115  RamDomain pack(RamDomain* tuple, size_t arity) {
116  return lookupArity(arity).pack(tuple);
117  }
118  /** @brief convert record reference to a record */
119  const RamDomain* unpack(RamDomain ref, size_t arity) const {
120  std::unordered_map<size_t, RecordMap>::const_iterator iter;
121 #pragma omp critical(RecordTableGetForArity)
122  {
123  // Find a previously emplaced map
124  iter = maps.find(arity);
125  }
126  assert(iter != maps.end() && "Attempting to unpack record for non-existing arity");
127  return (iter->second).unpack(ref);
128  }
129 
130 private:
131  /** @brief lookup RecordMap for a given arity; if it does not exist, create new RecordMap */
132  RecordMap& lookupArity(size_t arity) {
133  std::unordered_map<size_t, RecordMap>::iterator mapsIterator;
134 #pragma omp critical(RecordTableGetForArity)
135  {
136  // This will create a new map if it doesn't exist yet.
137  mapsIterator = maps.emplace(arity, arity).first;
138  }
139  return mapsIterator->second;
140  }
141 
142  /** Arity/RecordMap association */
143  std::unordered_map<size_t, RecordMap> maps;
144 };
