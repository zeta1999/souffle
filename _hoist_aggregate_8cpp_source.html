30 bool HoistAggregateTransformer::hoistAggregate(RamProgram& program) {
31  bool changed = false;
32 
33  // There are two cases: aggregates that have no data-dependencies on
34  // other RAM operations, and aggregates that have data-dependencies.
35  // A rewriter has two tasks: (1) identify a single aggregate that
36  // can be hoisted and (2) insert it at the outermost level.
37  // We assume all RamOperations are renumbered for this transformation.
38 
39  // Hoist a single aggregate to an outer scope that is data-independent.
40  visitDepthFirst(program, [&](const RamQuery& query) {
41  std::unique_ptr<RamNestedOperation> newAgg;
42  bool priorTupleOp = false;
43  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRewriter =
44  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
45  if (nullptr != dynamic_cast<RamAggregate*>(node.get())) {
46  auto* tupleOp = dynamic_cast<RamTupleOperation*>(node.get());
47  assert(tupleOp != nullptr && "aggregate conversion to tuple operation failed");
48  if (rla->getLevel(tupleOp) == -1 && !priorTupleOp) {
49  changed = true;
50  newAgg = souffle::clone(tupleOp);
51  assert(newAgg != nullptr && "failed to make a clone");
52  return souffle::clone(&tupleOp->getOperation());
53  }
54  } else if (nullptr != dynamic_cast<RamTupleOperation*>(node.get())) {
55  // tuple operation that is a non-aggregate
56  priorTupleOp = true;
57  }
58  node->apply(makeLambdaRamMapper(aggRewriter));
59  return node;
60  };
61  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRewriter));
62  if (newAgg != nullptr) {
63  newAgg->rewrite(&newAgg->getOperation(), souffle::clone(&query.getOperation()));
64  const_cast<RamQuery*>(&query)->rewrite(&query.getOperation(), std::move(newAgg));
65  }
66  });
67 
68  // hoist a single aggregate to an outer scope that is data-dependent on a prior operation.
69  visitDepthFirst(program, [&](const RamQuery& query) {
70  int newLevel = -1;
71  std::unique_ptr<RamNestedOperation> newAgg;
72  int priorOpLevel = -1;
73 
74  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRewriter =
75  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
76  if (nullptr != dynamic_cast<RamAbstractAggregate*>(node.get())) {
77  auto* tupleOp = dynamic_cast<RamTupleOperation*>(node.get());
78  assert(tupleOp != nullptr && "aggregate conversion to nested operation failed");
79  int dataDepLevel = rla->getLevel(tupleOp);
80  if (dataDepLevel != -1 && dataDepLevel < tupleOp->getTupleId() - 1) {
81  // If all tuple ops between the data-dependence level and agg
82  // are aggregates, then we do not hoist, i.e., we would
83  // continuously swap their positions.
84  if (dataDepLevel != priorOpLevel) {
85  changed = true;
86  newLevel = dataDepLevel;
87  newAgg = souffle::clone(tupleOp);
88  assert(newAgg != nullptr && "failed to make a clone");
89  return souffle::clone(&tupleOp->getOperation());
90  }
91  }
92  } else if (const RamTupleOperation* tupleOp = dynamic_cast<RamTupleOperation*>(node.get())) {
93  priorOpLevel = tupleOp->getTupleId();
94  }
95  node->apply(makeLambdaRamMapper(aggRewriter));
96  if (auto* search = dynamic_cast<RamTupleOperation*>(node.get())) {
97  if (newAgg != nullptr && search->getTupleId() == newLevel) {
98  newAgg->rewrite(&newAgg->getOperation(), souffle::clone(&search->getOperation()));
99  search->rewrite(&search->getOperation(), std::move(newAgg));
100  }
101  }
102  return node;
103  };
104  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRewriter));
105  });
106  return changed;
