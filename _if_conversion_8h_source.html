28 /**
29  * @class IfConversionTransformer
30  * @brief Convert IndexScan operations to Filter/Existence Checks
31 
32  * If there exists IndexScan operations in the RAM, and their tuples
33  * are not further used in subsequent operations, the IndexScan operations
34  * will be rewritten to Filter/Existence Checks.
35  *
36  * For example,
37  *
38  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
39  * QUERY
40  * ...
41  * FOR t1 IN X ON INDEX t1.x = 10 AND t1.y = 20
42  * ... // no occurrence of t1
43  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
44  *
45  * will be rewritten to
46  *
47  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
48  * QUERY
49  * ...
50  * IF (10,20) NOT IN A
51  * ...
52  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
53  *
54  */

56 public:
57  std::string getName() const override {
