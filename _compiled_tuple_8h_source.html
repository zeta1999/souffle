27 /**
28  * The type of object stored within relations representing the actual
29  * tuple value. Each tuple consists of a constant number of components.
30  *
31  * @tparam Domain the domain of the component values
32  * @tparam arity the number of components within an instance
33  */
34 template <typename Domain, std::size_t _arity>
35 struct Tuple {
36  // some features for template meta programming
37  using value_type = Domain;
38  static constexpr size_t arity = _arity;
39 
40  // the stored data
41  Domain data[arity];
42 
43  // constructors, destructors and assignment are default
44 
45  // provide access to components
46  const Domain& operator[](std::size_t index) const {
47  return data[index];
48  }
49 
50  // provide access to components
51  Domain& operator[](std::size_t index) {
52  return data[index];
53  }
54 
55  // a comparison operation
56  bool operator==(const Tuple& other) const {
57  for (std::size_t i = 0; i < arity; i++) {
58  if (data[i] != other.data[i]) return false;
59  }
60  return true;
61  }
62 
63  // inequality comparison
64  bool operator!=(const Tuple& other) const {
65  return !(*this == other);
66  }
67 
68  // required to put tuples into e.g. a std::set container
69  bool operator<(const Tuple& other) const {
70  for (std::size_t i = 0; i < arity; ++i) {
71  if (data[i] < other.data[i]) return true;
72  if (data[i] > other.data[i]) return false;
73  }
74  return false;
75  }
76 
77  // required to put tuples into e.g. a btree container
78  bool operator>(const Tuple& other) const {
79  for (std::size_t i = 0; i < arity; ++i) {
80  if (data[i] > other.data[i]) return true;
81  if (data[i] < other.data[i]) return false;
82  }
83  return false;
84  }
85 
86  // allow tuples to be printed
87  friend std::ostream& operator<<(std::ostream& out, const Tuple& tuple) {
88  if (arity == 0) return out << "[]";
89  out << "[";
90  for (std::size_t i = 0; i < (std::size_t)(arity - 1); ++i) {
91  out << tuple.data[i];
92  out << ",";
93  }
94  return out << tuple.data[arity - 1] << "]";
