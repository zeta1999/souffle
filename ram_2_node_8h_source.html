32  * @class RamNodeMapper
33  * @brief An abstract class for manipulating RAM Nodes by substitution
34  */
35 class RamNodeMapper {
36 public:
37  virtual ~RamNodeMapper() = default;
38 
39  /**
40  * @brief Abstract replacement method for a node.
41  *
42  * If the given nodes is to be replaced, the handed in node
43  * will be destroyed by the mapper and the returned node
44  * will become owned by the caller.
45  */
46  virtual std::unique_ptr<RamNode> operator()(std::unique_ptr<RamNode> node) const = 0;
47 
48  /**
49  * @brief Wrapper for any subclass of the RAM node hierarchy performing type casts.
50  */
51  template <typename T>
52  std::unique_ptr<T> operator()(std::unique_ptr<T> node) const {
53  std::unique_ptr<RamNode> resPtr =
54  (*this)(std::unique_ptr<RamNode>(static_cast<RamNode*>(node.release())));
55  assert(nullptr != dynamic_cast<T*>(resPtr.get()) && "Invalid target node!");
56  return std::unique_ptr<T>(dynamic_cast<T*>(resPtr.release()));
57  }
58 };
