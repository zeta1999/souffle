41 bool NormaliseConstraintsTransformer::transform(AstTranslationUnit& translationUnit) {
42  bool changed = false;
43 
44  // set a prefix for variables bound by magic-set for identification later
45  // prepended by + to avoid conflict with user-defined variables
46  static constexpr const char* boundPrefix = "+abdul";
47 
48  AstProgram& program = *translationUnit.getProgram();
49 
50  /* Create a node mapper that recursively replaces all constants and underscores
51  * with named variables.
52  *
53  * The mapper keeps track of constraints that should be added to the original
54  * clause it is being applied on in a given constraint set.
55  */
56  struct constraintNormaliser : public AstNodeMapper {
57  std::set<AstBinaryConstraint*>& constraints;
58  mutable int changeCount;
59 
60  constraintNormaliser(std::set<AstBinaryConstraint*>& constraints, int changeCount)
61  : constraints(constraints), changeCount(changeCount) {}
62 
63  bool hasChanged() const {
64  return changeCount > 0;
65  }
66 
67  int getChangeCount() const {
68  return changeCount;
69  }
70 
71  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
72  if (auto* stringConstant = dynamic_cast<AstStringConstant*>(node.get())) {
73  // string constant found
74  changeCount++;
75 
76  // create new variable name (with appropriate suffix)
77  std::string constantValue = stringConstant->getConstant();
78  std::stringstream newVariableName;
79  newVariableName << boundPrefix << changeCount << "_" << constantValue << "_s";
80 
81  // create new constraint (+abdulX = constant)
82  auto newVariable = std::make_unique<AstVariable>(newVariableName.str());
83  constraints.insert(new AstBinaryConstraint(
84  BinaryConstraintOp::EQ, souffle::clone(newVariable), souffle::clone(stringConstant)));
85 
86  // update constant to be the variable created
87  return newVariable;
88  } else if (auto* numberConstant = dynamic_cast<AstNumericConstant*>(node.get())) {
89  // number constant found
90  changeCount++;
91 
92  // create new variable name (with appropriate suffix)
93  std::stringstream newVariableName;
94  newVariableName << boundPrefix << changeCount << "_" << numberConstant->getConstant() << "_n";
95 
96  assert(numberConstant->getType() && "numeric constant hasn't been poly-constrained");
97  auto opEq = *numberConstant->getType() == AstNumericConstant::Type::Float
98  ? BinaryConstraintOp::FEQ
99  : BinaryConstraintOp::EQ;
100 
101  // create new constraint (+abdulX = constant)
102  auto newVariable = std::make_unique<AstVariable>(newVariableName.str());
103  constraints.insert(new AstBinaryConstraint(
104  opEq, souffle::clone(newVariable), souffle::clone(numberConstant)));
105 
106  // update constant to be the variable created
107  return newVariable;
108  } else if (dynamic_cast<AstUnnamedVariable*>(node.get()) != nullptr) {
109  // underscore found
110  changeCount++;
111 
112  // create new variable name
113  std::stringstream newVariableName;
114  newVariableName << "+underscore" << changeCount;
115 
116  return std::make_unique<AstVariable>(newVariableName.str());
117  }
118 
119  node->apply(*this);
120  return node;
121  }
122  };
123 
124  int changeCount = 0; // number of constants and underscores seen so far
125 
126  // apply the change to all clauses in the program
127  for (AstRelation* rel : program.getRelations()) {
128  for (AstClause* clause : getClauses(program, *rel)) {
129  if (isFact(*clause)) {
130  continue; // don't normalise facts
131  }
132 
133  std::set<AstBinaryConstraint*> constraints;
134  constraintNormaliser update(constraints, changeCount);
135  clause->apply(update);
136 
137  changeCount = update.getChangeCount();
138  changed = changed || update.hasChanged();
139 
140  for (AstBinaryConstraint* constraint : constraints) {
141  clause->addToBody(std::unique_ptr<AstBinaryConstraint>(constraint));
142  }
143  }
144  }
145 
146  return changed;
147 }
