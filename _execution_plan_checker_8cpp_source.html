36 bool AstExecutionPlanChecker::transform(AstTranslationUnit& translationUnit) {
37  auto* relationSchedule = translationUnit.getAnalysis<RelationScheduleAnalysis>();
38  auto* recursiveClauses = translationUnit.getAnalysis<RecursiveClausesAnalysis>();
39  auto&& report = translationUnit.getErrorReport();
40 
41  for (const RelationScheduleAnalysisStep& step : relationSchedule->schedule()) {
42  const std::set<const AstRelation*>& scc = step.computed();
43  for (const AstRelation* rel : scc) {
44  for (const AstClause* clause : getClauses(*translationUnit.getProgram(), *rel)) {
45  if (!recursiveClauses->recursive(clause)) {
46  continue;
47  }
48  if (clause->getExecutionPlan() == nullptr) {
49  continue;
50  }
51  int version = 0;
52  for (const auto* atom : getBodyLiterals<AstAtom>(*clause)) {
53  if (scc.count(getAtomRelation(atom, translationUnit.getProgram())) != 0u) {
54  version++;
55  }
56  }
57  int maxVersion = -1;
58  for (auto const& cur : clause->getExecutionPlan()->getOrders()) {
59  maxVersion = std::max(cur.first, maxVersion);
60  }
61 
62  if (version <= maxVersion) {
63  for (const auto& cur : clause->getExecutionPlan()->getOrders()) {
64  if (cur.first >= version) {
65  report.addDiagnostic(Diagnostic(Diagnostic::Type::ERROR,
66  DiagnosticMessage(
67  "execution plan for version " + std::to_string(cur.first),
68  cur.second->getSrcLoc()),
69  {DiagnosticMessage("only versions 0.." + std::to_string(version - 1) +
70  " permitted")}));
71  }
72  }
73  }
74  }
75  }
76  }
77  return false;
78 }
