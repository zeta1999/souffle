99  */
100  const RamDomain* getBase() const {
101  return base;
102  }
103

105  * Provides access to tuple components.
106  */
107  const RamDomain& operator[](std::size_t i) const {
108  return base[i];
109  }
110 
111  bool operator==(const TupleRef& other) const {
112  if (arity != other.arity) {
113  return false;
114  }
115  for (size_t i = 0; i < arity; ++i) {
116  if (base[i] != other[i]) {
117  return false;
118  }
119  }

121  }

143  static Order create(size_t arity);

146

148  * Determines whether this order is a valid order.

150  bool valid() const;
151 
152  template <std::size_t Arity>
153  Tuple<RamDomain, Arity> encode(const Tuple<RamDomain, Arity>& entry) const {
154  Tuple<RamDomain, Arity> res{};
155  for (std::size_t i = 0; i < Arity; ++i) {
156  res[i] = entry[order[i]];

162  Tuple<RamDomain, Arity> decode(const Tuple<RamDomain, Arity>& entry) const {
163  Tuple<RamDomain, Arity> res{};
164  for (std::size_t i = 0; i < Arity; ++i) {
165  res[order[i]] = entry[i];
166  }
167  return res;
168  }
169 
170  const AttributeOrder& getOrder() const {
171  return this->order;
172  }

175  bool operator!=(const Order& other) const;
176  bool operator<(const Order& other) const;

181 /**
182  * An abstract perspective on a data range. A stream is a pair of iterators,
183  * referencing the begin and end of a traversible sequence. A stream can only
184  * be traversed once.
185  */
186 class Stream {
187 public:
188  // the size of the internally maintained buffer, corresponding
189  // to the maximum chunk size retrieved from the source
190  constexpr static int BUFFER_SIZE = 128;
191 
192  // the 'interface' for data sources
193  class Source {
194  public:
195  virtual ~Source() = default;
196 
197  /**
198  * Requests the source to retrieve the next set of elements,
199  * to be stored in an array addressed by the first parameter.
200  * The second parameter states an upper limit for the number
201  * of elements to be retrieved.
202  *
203  * @return the number of elements retrieved, 0 if end has reached.
204  */
205  virtual int load(TupleRef* trg, int max) = 0;
206 
207  /**
208  * Fill the target array in the first parameter with references to the current set of elements.
209  * This is useful for cloning references to this Source. The second parameter limits the range
210  * copied.
211  *
212  * @return the number of elements retrieved, 0 if end has reached.
213  */
214  virtual int reload(TupleRef* trg, int max) = 0;
215 
216  /**
217  * Clone a source with the exact same state
218  */
219  virtual std::unique_ptr<Source> clone() = 0;
220  };
221 
222 private:
223  // the source to read data from
224  std::unique_ptr<Source> source = nullptr;
225 
226  // an internal buffer for decoded elements
227  std::array<TupleRef, BUFFER_SIZE> buffer{};
228 
229  // the current position in the buffer
230  int cur = 0;
231 
232  // the end of valid elements in the buffer
233  int limit = 0;
234 
235 public:
236  Stream(std::unique_ptr<Source>&& src) : source(std::move(src)) {
237  loadNext();
238  }
239 
240  Stream() = default;
241 
242  Stream(Stream& other) = delete;
243 
244  Stream& operator=(Stream& other) = delete;
245 
246  Stream(Stream&& other)
247  : source(std::move(other.source)), buffer(other.buffer), cur(other.cur), limit(other.limit) {}
248 
249  Stream& operator=(Stream&& other) {
250  source = std::move(other.source);
251  // only copy important data
252  std::memcpy(
253  &buffer[other.cur], &other.buffer[other.cur], sizeof(TupleRef) * (other.limit - other.cur));
254  cur = other.cur;
255  limit = other.limit;
256  return *this;
257  }
258 
259  template <typename S>
260  Stream(std::unique_ptr<S>&& src) : Stream(std::unique_ptr<Source>(std::move(src))) {}
261 
262  std::unique_ptr<Stream> clone() const {
263  if (source == nullptr) {
264  return std::make_unique<Stream>();
265  }
266  auto newStream = std::make_unique<Stream>(source->clone());
267  newStream->source->reload(&newStream->buffer[0], limit);
268  newStream->cur = cur;
269  newStream->limit = limit;
270  return newStream;
271  }

273  /**
274  * The iterator exposed by this stream to iterate through
275  * its elements using a range-based for.
276  */
277  class Iterator : public std::iterator<std::forward_iterator_tag, TupleRef> {
278  Stream* stream = nullptr;
279 
280  public:
281  Iterator() = default;
282 
283  Iterator(Stream& stream) : stream(&stream) {
284  if (stream.cur >= stream.limit) {
285  this->stream = nullptr;
286  }
287  }
288 
289  Iterator& operator++() {
290  ++stream->cur;
291  if (stream->cur < stream->limit) {
292  return *this;
293  }
294  stream->loadNext();
295  if (stream->cur >= stream->limit) {
296  stream = nullptr;
297  }
298  return *this;
299  }
300 
301  const TupleRef& operator*() const {
302  return stream->buffer[stream->cur];
303  }
304 
305  bool operator!=(const Iterator& other) const {
306  return stream != other.stream;
307  }
308 
309  bool operator==(const Iterator& other) const {
310  return stream == other.stream;
311  }
312  };
313 
314  // support for ranged based for loops
315  Iterator begin() {
316  return Iterator(*this);
317  }
318  Iterator end() const {
319  return Iterator();
320  }
321 
322 private:
323  /**
324  * Retrieves the next chunk of elements from the source.
325  */
326  void loadNext() {
327  limit = source->load(&buffer[0], BUFFER_SIZE);
328  cur = 0;
329  }
330 };
