37 /**
38  * Intermediate representation of a datalog clause.
39  *
40  * A clause can either be:
41  * - a fact - a clause with no body (e.g., X(a,b))
42  * - a rule - a clause with a head and a body (e.g., Y(a,b) -: X(a,b))
43  *
44  * TODO (azreika): make clause abstract and split into two subclasses: Rule and Fact
45  */
46 class AstClause : public AstNode {
47 public:
48  AstClause(Own<AstAtom> head = {}, VecOwn<AstLiteral> bodyLiterals = {}, Own<AstExecutionPlan> plan = {},
49  SrcLocation loc = {})
50  : AstNode(std::move(loc)), head(std::move(head)), bodyLiterals(std::move(bodyLiterals)),
51  plan(std::move(plan)) {}
52 
53  /** Add a Literal to the body of the clause */
54  void addToBody(Own<AstLiteral> literal) {
55  bodyLiterals.push_back(std::move(literal));
56  }
57 
58  /** Set the head of clause to @p h */
59  void setHead(Own<AstAtom> h) {
60  head = std::move(h);
61  }
62 
63  /** Set the bodyLiterals of clause to @p body */
64  void setBodyLiterals(VecOwn<AstLiteral> body) {
65  bodyLiterals = std::move(body);
66  }
67 
68  /** Return the atom that represents the head of the clause */
69  AstAtom* getHead() const {
70  return head.get();
71  }
72 
73  /** Obtains a copy of the internally maintained body literals */
74  std::vector<AstLiteral*> getBodyLiterals() const {
75  return toPtrVector(bodyLiterals);
76  }
77 
78  /** Obtains the execution plan associated to this clause or null if there is none */
79  const AstExecutionPlan* getExecutionPlan() const {
80  return plan.get();
81  }
82 
83  /** Updates the execution plan associated to this clause */
84  void setExecutionPlan(Own<AstExecutionPlan> plan) {
85  this->plan = std::move(plan);
86  }
87 
88  /** Resets the execution plan */
89  void clearExecutionPlan() {
90  plan = nullptr;
91  }
92 
93  AstClause* clone() const override {
94  return new AstClause(
95  souffle::clone(head), souffle::clone(bodyLiterals), souffle::clone(plan), getSrcLoc());
96  }
97 
98  void apply(const AstNodeMapper& map) override {
99  head = map(std::move(head));
100  for (auto& lit : bodyLiterals) {
101  lit = map(std::move(lit));
102  }
103  }
104 
105  std::vector<const AstNode*> getChildNodes() const override {
106  std::vector<const AstNode*> res = {head.get()};
107  for (auto& cur : bodyLiterals) {
108  res.push_back(cur.get());
109  }
110  return res;
111  }
112 
113 protected:
114  void print(std::ostream& os) const override {
115  if (head != nullptr) {
116  os << *head;
117  }
118  if (!bodyLiterals.empty()) {
119  os << " :- \n " << join(bodyLiterals, ",\n ");
120  }
121  os << ".";
122  if (plan != nullptr) {
123  os << *plan;
124  }
125  }
126 
127  bool equal(const AstNode& node) const override {
128  const auto& other = static_cast<const AstClause&>(node);
129  return equal_ptr(head, other.head) && equal_targets(bodyLiterals, other.bodyLiterals) &&
130  equal_ptr(plan, other.plan);
131  }
132 
133  /** head of the clause */
134  Own<AstAtom> head;
135 
136  /** body literals of clause */
137  VecOwn<AstLiteral> bodyLiterals;
138 
139  /** user defined execution plan (if not defined, plan is null) */
140  Own<AstExecutionPlan> plan;
141 };
