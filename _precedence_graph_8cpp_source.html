32 void PrecedenceGraphAnalysis::run(const AstTranslationUnit& translationUnit) {
33  /* Get relations */
34  const AstProgram& program = *translationUnit.getProgram();
35  std::vector<AstRelation*> relations = program.getRelations();
36 
37  for (AstRelation* r : relations) {
38  backingGraph.insert(r);
39  for (const auto& c : getClauses(program, *r)) {
40  const std::set<const AstRelation*>& dependencies =
41  getBodyRelations(c, translationUnit.getProgram());
42  for (auto source : dependencies) {
43  backingGraph.insert(source, r);
44  }
45  }
46  }
47 }

49 void PrecedenceGraphAnalysis::print(std::ostream& os) const {
50  /* Print dependency graph */
51  std::stringstream ss;
52  ss << "digraph {\n";
53  /* Print node of dependence graph */
54  for (const AstRelation* rel : backingGraph.vertices()) {
55  if (rel != nullptr) {
56  ss << "\t\"" << rel->getQualifiedName() << "\" [label = \"" << rel->getQualifiedName()
57  << "\"];\n";
58  }
59  }
60  for (const AstRelation* rel : backingGraph.vertices()) {
61  if (rel != nullptr) {
62  for (const AstRelation* adjRel : backingGraph.successors(rel)) {
63  if (adjRel != nullptr) {
64  ss << "\t\"" << rel->getQualifiedName() << "\" -> \"" << adjRel->getQualifiedName()
65  << "\";\n";
66  }
67  }
68  }
69  }
70  ss << "}\n";
71  printHTMLGraph(os, ss.str(), getName());
