90 /**
91  * Simple implementation of a which tool
92  */
93 inline std::string which(const std::string& name) {
94  // Check if name has path components in it and if so return it immediately
95  if (name.find('/') != std::string::npos) {
96  return name;
97  }
98  // Get PATH from environment, if it exists.
99  const char* syspath = ::getenv("PATH");
100  if (syspath == nullptr) {
101  return "";
102  }
103  char buf[PATH_MAX];
104  std::stringstream sstr;
105  sstr << syspath;
106  std::string sub;
107 
108  // Check for existence of a binary called 'name' in PATH
109  while (std::getline(sstr, sub, ':')) {
110  std::string path = sub + "/" + name;
111  if ((::realpath(path.c_str(), buf) != nullptr) && isExecutable(path) && !existDir(path)) {
112  return buf;
113  }
114  }
115  return "";
116 }

118 /**
119  * C++-style dirname
120  */
121 inline std::string dirName(const std::string& name) {
122  if (name.empty()) {
123  return ".";
124  }
125  size_t lastNotSlash = name.find_last_not_of('/');
126  // All '/'
127  if (lastNotSlash == std::string::npos) {
128  return "/";
129  }
130  size_t leadingSlash = name.find_last_of('/', lastNotSlash);
131  // No '/'
132  if (leadingSlash == std::string::npos) {
133  return ".";
134  }
135  // dirname is '/'
136  if (leadingSlash == 0) {
137  return "/";
138  }
139  return name.substr(0, leadingSlash);
140 }

151 /**
152  * Join two paths together; note that this does not resolve overlaps or relative paths.
153  */
154 inline std::string pathJoin(const std::string& first, const std::string& second) {
155  unsigned firstPos = static_cast<unsigned>(first.size()) - 1;
156  while (first.at(firstPos) == '/') {
157  firstPos--;
158  }
159  unsigned secondPos = 0;
160  while (second.at(secondPos) == '/') {
161  secondPos++;
162  }
163  return first.substr(0, firstPos + 1) + '/' + second.substr(secondPos);
164 }

166 /*
167  * Find out if an executable given by @p tool exists in the path given @p path
168  * relative to the directory given by @ base. A path here refers a
169  * colon-separated list of directories.
170  */
171 inline std::string findTool(const std::string& tool, const std::string& base, const std::string& path) {
172  std::string dir = dirName(base);
173  std::stringstream sstr(path);
174  std::string sub;
175 
176  while (std::getline(sstr, sub, ':')) {
177  std::string subpath = dir + "/" + sub + '/' + tool;
178  if (isExecutable(subpath)) {
179  return absPath(subpath);
180  }
181  }
182  return "";
183 }

185 /*
186  * Get the basename of a fully qualified filename
187  */
188 inline std::string baseName(const std::string& filename) {
189  if (filename.empty()) {
190  return ".";
191  }
192 
193  size_t lastNotSlash = filename.find_last_not_of('/');
194  if (lastNotSlash == std::string::npos) {
195  return "/";
196  }
197 
198  size_t lastSlashBeforeBasename = filename.find_last_of('/', lastNotSlash - 1);
199  if (lastSlashBeforeBasename == std::string::npos) {
200  lastSlashBeforeBasename = static_cast<size_t>(-1);
201  }
202  return filename.substr(lastSlashBeforeBasename + 1, lastNotSlash - lastSlashBeforeBasename);
203 }

205 /**
206  * File name, with extension removed.
207  */
208 inline std::string simpleName(const std::string& path) {
209  std::string name = baseName(path);
210  const size_t lastDot = name.find_last_of('.');
211  // file has no extension
212  if (lastDot == std::string::npos) {
213  return name;
214  }
215  const size_t lastSlash = name.find_last_of('/');
216  // last slash occurs after last dot, so no extension
217  if (lastSlash != std::string::npos && lastSlash > lastDot) {
218  return name;
219  }
220  // last dot after last slash, or no slash
221  return name.substr(0, lastDot);
222 }

237  return std::string();

263 inline std::stringstream execStdOut(char const* cmd) {
264  FILE* in = popen(cmd, "r");
265  std::stringstream data;
266  while (in != nullptr) {
267  char c = fgetc(in);
268  if (feof(in) != 0) {
269  break;
270  }
271  data << c;
272  }
273  pclose(in);
274  return data;
