49  Analysis* getAnalysis() const {
50  static const bool debug = Global::config().has("debug-report");
51  std::string name = Analysis::name;
52  auto it = analyses.find(name);
53  if (it == analyses.end()) {
54  // analysis does not exist yet, create instance and run it.
55  analyses[name] = std::make_unique<Analysis>();
56  analyses[name]->run(*this);
57  if (debug) {
58  std::stringstream ss;
59  analyses[name]->print(ss);
60  if (nullptr == dynamic_cast<PrecedenceGraphAnalysis*>(analyses[name].get()) &&
61  nullptr == dynamic_cast<SCCGraphAnalysis*>(analyses[name].get())) {
62  debugReport.addSection(name, "Ast Analysis [" + name + "]", ss.str());
63  } else {
64  debugReport.addSection(
65  DebugReportSection(name, "Ast Analysis [" + name + "]", {}, ss.str()));
66  }
67  }
68  }
