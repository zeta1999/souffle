38  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {

40  // sum k : { .. } where k is a constant
41  if (auto* agg = dynamic_cast<AstAggregator*>(node.get())) {
42  if (agg->getOperator() == AggregateOp::SUM) {
43  if (const auto* constant =
44  dynamic_cast<const AstNumericConstant*>(agg->getTargetExpression())) {
