259 /**
260  * A utility function visiting all nodes within the RAM fragment rooted by the given node
261  * recursively in a depth-first pre-order fashion applying the given visitor to each
262  * encountered node.
263  *
264  * @param root the root of the RAM fragment to be visited
265  * @param visitor the visitor to be applied on each node
266  * @param args a list of extra parameters to be forwarded to the visitor
267  */
268 template <typename R, typename... Ps, typename... Args>
269 void visitDepthFirstPreOrder(const RamNode& root, RamVisitor<R, Ps...>& visitor, Args&... args) {
270  visitor(root, args...);
271  for (const RamNode* cur : root.getChildNodes()) {
272  if (cur != nullptr) {
273  visitDepthFirstPreOrder(*cur, visitor, args...);
274  }
275  }
276 }

294 /**
295  * A specialized visitor wrapping a lambda function -- an auxiliary type required
296  * for visitor convenience functions.
297  */
298 template <typename R, typename N>
299 struct LambdaRamVisitor : public RamVisitor<void> {
300  std::function<R(const N&)> lambda;
301  LambdaRamVisitor(std::function<R(const N&)> lambda) : lambda(std::move(lambda)) {}
302  void visit(const RamNode& node) override {
303  if (const auto* n = dynamic_cast<const N*>(&node)) {
304  lambda(*n);
305  }
306  }
307 };
308 
309 /**
310  * A factory function for creating LambdaRamVisitor instances.

312 template <typename R, typename N>
313 LambdaRamVisitor<R, N> makeLambdaRamVisitor(const std::function<R(const N&)>& fun) {
314  return LambdaRamVisitor<R, N>(fun);
