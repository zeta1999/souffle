32 std::unique_ptr<RamOperation> ChoiceConversionTransformer::rewriteScan(const RamScan* scan) {
33  bool transformTuple = false;
34 
35  // Check that RamFilter follows the Scan in the loop nest
36  if (const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation())) {
37  // Check that the Filter uses the identifier in the Scan
38  if (rla->getLevel(&filter->getCondition()) == scan->getTupleId()) {
39  transformTuple = true;
40 
41  // Check that the filter is not referred to after
42  const auto* nextNode = dynamic_cast<const RamNode*>(&filter->getOperation());
43 
44  visitDepthFirst(*nextNode, [&](const RamTupleElement& element) {
45  if (element.getTupleId() == scan->getTupleId()) {
46  transformTuple = false;
47  }
48  });
49  }
50  }
51 
52  // Convert the Scan/If pair into a Choice
53  if (transformTuple) {
54  std::vector<std::unique_ptr<RamExpression>> newValues;
55  const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation());
56  const int identifier = scan->getTupleId();
57 
58  return std::make_unique<RamChoice>(std::make_unique<RamRelationReference>(&scan->getRelation()),
59  identifier, souffle::clone(&filter->getCondition()), souffle::clone(&scan->getOperation()),

61  }

114 bool ChoiceConversionTransformer::convertScans(RamProgram& program) {
115  bool changed = false;
116  visitDepthFirst(program, [&](const RamQuery& query) {
117  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> scanRewriter =
118  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
119  if (const RamScan* scan = dynamic_cast<RamScan*>(node.get())) {
120  if (std::unique_ptr<RamOperation> op = rewriteScan(scan)) {
121  changed = true;
122  node = std::move(op);
123  }
124  } else if (const RamIndexScan* indexScan = dynamic_cast<RamIndexScan*>(node.get())) {
125  if (std::unique_ptr<RamOperation> op = rewriteIndexScan(indexScan)) {
126  changed = true;
127  node = std::move(op);
128  }
129  }
130  node->apply(makeLambdaRamMapper(scanRewriter));
131 
132  return node;
133  };
134  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(scanRewriter));
135  });
136 
137  return changed;
