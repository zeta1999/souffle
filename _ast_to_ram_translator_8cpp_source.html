290  const AstLiteral* lit, const ValueIndex& index) {
291  class ConstraintTranslator : public AstVisitor<std::unique_ptr<RamCondition>> {
292  AstToRamTranslator& translator;
293  const ValueIndex& index;
294 
295  public:
296  ConstraintTranslator(AstToRamTranslator& translator, const ValueIndex& index)
297  : translator(translator), index(index) {}
298 
299  /** for atoms */
300  std::unique_ptr<RamCondition> visitAtom(const AstAtom&) override {
301  return nullptr; // covered already within the scan/lookup generation step
302  }
303 
304  /** for binary relations */
305  std::unique_ptr<RamCondition> visitBinaryConstraint(const AstBinaryConstraint& binRel) override {
306  auto valLHS = translator.translateValue(binRel.getLHS(), index);
307  auto valRHS = translator.translateValue(binRel.getRHS(), index);
308  return std::make_unique<RamConstraint>(
309  binRel.getOperator(), std::move(valLHS), std::move(valRHS));
310  }

312  /** for provenance negation */
313  std::unique_ptr<RamCondition> visitProvenanceNegation(const AstProvenanceNegation& neg) override {
314  const auto* atom = neg.getAtom();
315  size_t auxiliaryArity = translator.getEvaluationArity(atom);
316  assert(auxiliaryArity <= atom->getArity() && "auxiliary arity out of bounds");
317  size_t arity = atom->getArity() - auxiliaryArity;
318  std::vector<std::unique_ptr<RamExpression>> values;
319 
320  auto args = atom->getArguments();
321  for (size_t i = 0; i < arity; i++) {
322  values.push_back(translator.translateValue(args[i], index));
323  }
324  // we don't care about the provenance columns when doing the existence check
325  if (Global::config().has("provenance")) {
326  // undefined value for rule number
327  values.push_back(std::make_unique<RamUndefValue>());
328  // add the height annotation for provenanceNotExists
329  for (size_t height = 1; height < auxiliaryArity; height++) {
330  values.push_back(translator.translateValue(args[arity + height], index));
331  }
332  }
333  return std::make_unique<RamNegation>(std::make_unique<RamProvenanceExistenceCheck>(
334  translator.translateRelation(atom), std::move(values)));
335  }
336 
337  /** for negations */
338  std::unique_ptr<RamCondition> visitNegation(const AstNegation& neg) override {
339  const auto* atom = neg.getAtom();
340  size_t auxiliaryArity = translator.getEvaluationArity(atom);
341  assert(auxiliaryArity <= atom->getArity() && "auxiliary arity out of bounds");
342  size_t arity = atom->getArity() - auxiliaryArity;
343 
344  if (arity == 0) {
345  // for a nullary, negation is a simple emptiness check
346  return std::make_unique<RamEmptinessCheck>(translator.translateRelation(atom));
347  }
348 
349  // else, we construct the atom and create a negation
350  std::vector<std::unique_ptr<RamExpression>> values;
351  auto args = atom->getArguments();
352  for (size_t i = 0; i < arity; i++) {
353  values.push_back(translator.translateValue(args[i], index));
354  }
355  for (size_t i = 0; i < auxiliaryArity; i++) {
356  values.push_back(std::make_unique<RamUndefValue>());
357  }
358  return std::make_unique<RamNegation>(std::make_unique<RamExistenceCheck>(
359  translator.translateRelation(atom), std::move(values)));
360  }
361  };
362  return ConstraintTranslator(*this, index)(*lit);
363 }

365 std::unique_ptr<AstClause> AstToRamTranslator::ClauseTranslator::getReorderedClause(
366  const AstClause& clause, const int version) const {
367  const auto plan = clause.getExecutionPlan();
368 
369  // check whether there is an imposed order constraint
370  if (plan == nullptr) {
371  return nullptr;
372  }
373  auto orders = plan->getOrders();
374  if (orders.find(version) == orders.end()) {
375  return nullptr;
376  }
377 
378  // get the imposed order
379  const auto& order = orders[version];
380 
381  // create a copy and fix order
382  std::unique_ptr<AstClause> reorderedClause(clause.clone());
383 
384  // Change order to start at zero
385  std::vector<unsigned int> newOrder(order->getOrder().size());
386  std::transform(order->getOrder().begin(), order->getOrder().end(), newOrder.begin(),
387  [](unsigned int i) -> unsigned int { return i - 1; });
388 
389  // re-order atoms
390  reorderedClause.reset(reorderAtoms(reorderedClause.get(), newOrder));
391 
392  // clear other order and fix plan
393  reorderedClause->clearExecutionPlan();
394 
395  return reorderedClause;
396 }

412 void AstToRamTranslator::ClauseTranslator::indexValues(const AstNode* curNode,
413  std::map<const AstNode*, std::unique_ptr<arg_list>>& nodeArgs,
414  std::map<const arg_list*, int>& arg_level, RamRelationReference* relation) {
415  arg_list* cur = getArgList(curNode, nodeArgs);
416  for (size_t pos = 0; pos < cur->size(); ++pos) {
417  // get argument
418  auto& arg = (*cur)[pos];
419 
420  // check for variable references
421  if (auto var = dynamic_cast<const AstVariable*>(arg)) {
422  if (pos < relation->get()->getArity()) {
423  valueIndex.addVarReference(*var, arg_level[cur], pos, souffle::clone(relation));
424  } else {
425  valueIndex.addVarReference(*var, arg_level[cur], pos);
426  }
427  }
428 
429  // check for nested records
430  if (auto rec = dynamic_cast<const AstRecordInit*>(arg)) {
431  // introduce new nesting level for unpack
432  op_nesting.push_back(rec);
433  arg_level[getArgList(rec, nodeArgs)] = level++;
434 
435  // register location of record
436  valueIndex.setRecordDefinition(*rec, arg_level[cur], pos);
437 
438  // resolve nested components
439  indexValues(rec, nodeArgs, arg_level, relation);
440  }
441  }

445 void AstToRamTranslator::ClauseTranslator::createValueIndex(const AstClause& clause) {
446  for (const auto* atom : getBodyLiterals<AstAtom>(clause)) {
447  // std::map<const arg_list*, int> arg_level;
448  std::map<const AstNode*, std::unique_ptr<arg_list>> nodeArgs;
449 
450  std::map<const arg_list*, int> arg_level;
451  nodeArgs[atom] = std::make_unique<arg_list>(atom->getArguments());
452  // the atom is obtained at the current level
453  // increment nesting level for the atom
454  arg_level[nodeArgs[atom].get()] = level++;
455  op_nesting.push_back(atom);
456 
457  indexValues(atom, nodeArgs, arg_level, translator.translateRelation(atom).get());
458  }
459 
460  // add aggregation functions
461  visitDepthFirstPostOrder(clause, [&](const AstArgument& arg) {
462  // returns the write-location for this generator (or none if an equiv arg was already seen)
463  auto addGenerator = [&]() -> std::optional<int> {
464  // The by-value compare means that we're effectively doing CSE for any
465  // generator args during code-gen. This is a weird place to do this.
466  if (any_of(generators, [&](auto* x) { return *x == arg; })) return {};
467  generators.push_back(&arg);
468 
469  int aggLoc = level++;
470  valueIndex.setGeneratorLoc(arg, Location({aggLoc, 0}));
471  return aggLoc;
472  };
473 
474  if (auto agg = dynamic_cast<const AstAggregator*>(&arg)) {
475  if (auto aggLoc = addGenerator()) {
476  // bind aggregator variables to locations
477  const AstAtom* atom = nullptr;
478  for (auto lit : agg->getBodyLiterals()) {
479  if (atom == nullptr) {
480  atom = dynamic_cast<const AstAtom*>(lit);
481  } else {
482  break;
483  }
484  }
485  if (atom != nullptr) {
486  size_t pos = 0;
487  for (auto* arg : atom->getArguments()) {
488  if (const auto* var = dynamic_cast<const AstVariable*>(arg)) {
489  valueIndex.addVarReference(
490  *var, *aggLoc, (int)pos, translator.translateRelation(atom));
491  }
492  ++pos;
493  }
494  }
495  }
496  }
497 
498  auto func = dynamic_cast<const AstIntrinsicFunctor*>(&arg);
499  if (func && func->getFunctionInfo()->multipleResults) {
500  addGenerator();
501  }
502  });
503 }

574 std::unique_ptr<RamOperation> AstToRamTranslator::ClauseTranslator::filterByConstraints(size_t const level,
575  const std::vector<AstArgument*>& args, std::unique_ptr<RamOperation> op, bool constrainByFunctors) {
576  size_t pos = 0;
577 
578  auto mkFilter = [&](bool isFloatArg, std::unique_ptr<RamExpression> rhs) {
579  return std::make_unique<RamFilter>(
580  std::make_unique<RamConstraint>(isFloatArg ? BinaryConstraintOp::FEQ : BinaryConstraintOp::EQ,
581  std::make_unique<RamTupleElement>(level, pos), std::move(rhs)),
582  std::move(op));
583  };
584 
585  for (auto* a : args) {
586  if (auto* c = dynamic_cast<const AstConstant*>(a)) {
587  auto* const c_num = dynamic_cast<const AstNumericConstant*>(c);
588  assert((!c_num || c_num->getType()) && "numeric constant wasn't bound to a type");
589  op = mkFilter(c_num && *c_num->getType() == AstNumericConstant::Type::Float,
590  translator.translateConstant(*c));
591  } else if (auto* func = dynamic_cast<const AstFunctor*>(a)) {
592  if (constrainByFunctors) {
593  op = mkFilter(func->getReturnType() == TypeAttribute::Float,
594  translator.translateValue(func, valueIndex));
595  }
596  }
597 
598  ++pos;
599  }
600 
601  return op;

604 /** generate RAM code for a clause */
605 std::unique_ptr<RamStatement> AstToRamTranslator::ClauseTranslator::translateClause(
606  const AstClause& clause, const AstClause& originalClause, const int version) {
607  if (auto reorderedClause = getReorderedClause(clause, version)) {
608  // translate reordered clause
609  return translateClause(*reorderedClause, originalClause, version);
610  }
611 
612  // get extract some details
613  const AstAtom* head = clause.getHead();
614 
615  // handle facts
616  if (isFact(clause)) {
617  // translate arguments
618  std::vector<std::unique_ptr<RamExpression>> values;
619  for (auto& arg : head->getArguments()) {
620  values.push_back(translator.translateValue(arg, ValueIndex()));
621  }
622 
623  // create a fact statement
624  return std::make_unique<RamQuery>(
625  std::make_unique<RamProject>(translator.translateRelation(head), std::move(values)));
626  }
627 
628  // the rest should be rules
629  assert(isRule(clause));
630 
631  createValueIndex(clause);
632 
633  // -- create RAM statement --
634 
635  std::unique_ptr<RamOperation> op = createOperation(clause);
636 
637  /* add equivalence constraints imposed by variable binding */
638  for (const auto& cur : valueIndex.getVariableReferences()) {
639  // the first appearance
640  const Location& first = *cur.second.begin();
641  // all other appearances
642  for (const Location& loc : cur.second) {
643  if (first != loc && !valueIndex.isGenerator(loc.identifier)) {
644  // FIXME: equiv' for float types (`FEQ`)
645  op = std::make_unique<RamFilter>(
646  std::make_unique<RamConstraint>(
647  BinaryConstraintOp::EQ, makeRamTupleElement(first), makeRamTupleElement(loc)),
648  std::move(op));
649  }
650  }
651  }
652 
653  /* add conditions caused by atoms, negations, and binary relations */
654  for (const auto& lit : clause.getBodyLiterals()) {
655  if (auto condition = translator.translateConstraint(lit, valueIndex)) {
656  op = std::make_unique<RamFilter>(std::move(condition), std::move(op));
657  }
658  }
659 
660  // add aggregator conditions
661  size_t curLevel = op_nesting.size() - 1;
662  for (auto it = op_nesting.rbegin(); it != op_nesting.rend(); ++it, --curLevel) {
663  const AstNode* cur = *it;
664 
665  if (const auto* atom = dynamic_cast<const AstAtom*>(cur)) {
666  // add constraints
667  size_t pos = 0;
668  for (auto arg : atom->getArguments()) {
669  if (auto* agg = dynamic_cast<AstAggregator*>(arg)) {
670  auto loc = valueIndex.getGeneratorLoc(*agg);
671  // FIXME: equiv' for float types (`FEQ`)
672  op = std::make_unique<RamFilter>(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ,
673  std::make_unique<RamTupleElement>(curLevel, pos),
674  makeRamTupleElement(loc)),
675  std::move(op));
676  }
677  ++pos;
678  }
679  }
680  }
681 
682  // add generator levels
683  --level;
684  for (auto* cur : reverse(generators)) {
685  if (auto agg = dynamic_cast<const AstAggregator*>(cur)) {
686  // condition for aggregate and helper function to add terms
687  std::unique_ptr<RamCondition> aggCond;
688  auto addAggCondition = [&](std::unique_ptr<RamCondition> arg) {
689  aggCond = aggCond ? std::make_unique<RamConjunction>(std::move(aggCond), std::move(arg))
690  : std::move(arg);
691  };
692 
693  // translate constraints of sub-clause
694  for (auto&& lit : agg->getBodyLiterals()) {
695  if (auto newCondition = translator.translateConstraint(lit, valueIndex)) {
696  addAggCondition(std::move(newCondition));
697  }
698  }
699 
700  // get the first predicate of the sub-clause
701  // NB: at most one atom is permitted in a sub-clause
702  const AstAtom* atom = nullptr;
703  for (auto&& lit : agg->getBodyLiterals()) {
704  if (atom == nullptr) {
705  atom = dynamic_cast<const AstAtom*>(lit);
706  } else {
707  assert(dynamic_cast<const AstAtom*>(lit) == nullptr &&
708  "Unsupported complex aggregation body encountered!");
709  }
710  }
711 
712  // translate arguments's of atom (if exists) to conditions
713  if (atom != nullptr) {
714  size_t pos = 0;
715  auto addAggEqCondition = [&](std::unique_ptr<RamExpression> value) {
716  if (isRamUndefValue(value.get())) return;
717 
718  // FIXME: equiv' for float types (`FEQ`)
719  addAggCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ,
720  std::make_unique<RamTupleElement>(level, pos), std::move(value)));
721  };
722  for (auto* arg : atom->getArguments()) {
723  // variable bindings are issued differently since we don't want self
724  // referential variable bindings
725  if (auto* var = dynamic_cast<const AstVariable*>(arg)) {
726  for (auto&& loc : valueIndex.getVariableReferences().find(var->getName())->second) {
727  if (level != loc.identifier || (int)pos != loc.element) {
728  addAggEqCondition(makeRamTupleElement(loc));
729  break;
730  }
731  }
732  } else if (auto value = translator.translateValue(arg, valueIndex)) {
733  addAggEqCondition(std::move(value));
734  }
735  ++pos;
736  }
737  }
738 
739  // translate aggregate expression
740  auto expr = translator.translateValue(agg->getTargetExpression(), valueIndex);
741 
742  // add Ram-Aggregation layer
743  op = std::make_unique<RamAggregate>(std::move(op), agg->getOperator(),
744  translator.translateRelation(atom),
745  expr ? std::move(expr) : std::make_unique<RamUndefValue>(),
746  aggCond ? std::move(aggCond) : std::make_unique<RamTrue>(), level);
747  } else if (const auto* func = dynamic_cast<const AstIntrinsicFunctor*>(cur)) {
748  std::vector<std::unique_ptr<RamExpression>> args;
749  for (auto&& x : func->getArguments()) {
750  args.push_back(translator.translateValue(x, valueIndex));
751  }
752 
753  auto func_op = [&]() -> RamNestedIntrinsicOp {
754  switch (func->getFunctionInfo()->op) {
755  case FunctorOp::RANGE: return RamNestedIntrinsicOp::RANGE;
756  case FunctorOp::URANGE: return RamNestedIntrinsicOp::URANGE;
757  case FunctorOp::FRANGE: return RamNestedIntrinsicOp::FRANGE;
758 
759  default:
760  assert(func->getFunctionInfo()->multipleResults);
761  fatal("missing case handler or bad code-gen");
762  }
763  };
764 
765  op = std::make_unique<RamNestedIntrinsicOperator>(
766  func_op(), std::move(args), std::move(op), level);
767  }
768 
769  --level;
770  }
771 
772  // build operation bottom-up
773  while (!op_nesting.empty()) {
774  // get next operator
775  const AstNode* cur = op_nesting.back();
776  op_nesting.pop_back();
777 
778  // get current nesting level
779  auto level = op_nesting.size();
780 
781  if (const auto* atom = dynamic_cast<const AstAtom*>(cur)) {
782  // add constraints
783  // TODO: do we wish to enable constraints by header functor? record inits do so...
784  op = filterByConstraints(level, atom->getArguments(), std::move(op), false);
785 
786  // check whether all arguments are unnamed variables
787  bool isAllArgsUnnamed = true;
788  for (auto* argument : atom->getArguments()) {
789  if (dynamic_cast<AstUnnamedVariable*>(argument) == nullptr) {
790  isAllArgsUnnamed = false;
791  }
792  }
793 
794  // add check for emptiness for an atom
795  op = std::make_unique<RamFilter>(
796  std::make_unique<RamNegation>(
797  std::make_unique<RamEmptinessCheck>(translator.translateRelation(atom))),
798  std::move(op));
799 
800  // add a scan level
801  if (atom->getArity() != 0 && !isAllArgsUnnamed) {
802  if (head->getArity() == 0) {
803  op = std::make_unique<RamBreak>(
804  std::make_unique<RamNegation>(
805  std::make_unique<RamEmptinessCheck>(translator.translateRelation(head))),
806  std::move(op));
807  }
808  if (Global::config().has("profile")) {
809  std::stringstream ss;
810  ss << head->getQualifiedName();
811  ss.str("");
812  ss << "@frequency-atom" << ';';
813  ss << originalClause.getHead()->getQualifiedName() << ';';
814  ss << version << ';';
815  ss << stringify(toString(clause)) << ';';
816  ss << stringify(toString(*atom)) << ';';
817  ss << stringify(toString(originalClause)) << ';';
818  ss << level << ';';
819  op = std::make_unique<RamScan>(
820  translator.translateRelation(atom), level, std::move(op), ss.str());
821  } else {
822  op = std::make_unique<RamScan>(translator.translateRelation(atom), level, std::move(op));
823  }
824  }
825 
826  // TODO: support constants in nested records!
827  } else if (const auto* rec = dynamic_cast<const AstRecordInit*>(cur)) {
828  // add constant constraints
829  op = filterByConstraints(level, rec->getArguments(), std::move(op));
830 
831  // add an unpack level
832  const Location& loc = valueIndex.getDefinitionPoint(*rec);
833  op = std::make_unique<RamUnpackRecord>(
834  std::move(op), level, makeRamTupleElement(loc), rec->getArguments().size());
835  } else {
836  fatal("Unsupported AST node for creation of scan-level!");
837  }
838  }
839 
840  /* generate the final RAM Insert statement */
841  std::unique_ptr<RamCondition> cond = createCondition(originalClause);
842  if (cond != nullptr) {
843  return std::make_unique<RamQuery>(std::make_unique<RamFilter>(std::move(cond), std::move(op)));
844  } else {
845  return std::make_unique<RamQuery>(std::move(op));
846  }
847 }

962 /** generate RAM code for recursive relations in a strongly-connected component */
963 std::unique_ptr<RamStatement> AstToRamTranslator::translateRecursiveRelation(
964  const std::set<const AstRelation*>& scc, const RecursiveClausesAnalysis* recursiveClauses) {
965  // initialize sections
966  std::vector<std::unique_ptr<RamStatement>> preamble;
967  std::vector<std::unique_ptr<RamStatement>> updateTable;
968  std::vector<std::unique_ptr<RamStatement>> postamble;
969 
970  auto genMerge = [](const RamRelationReference* dest,
971  const RamRelationReference* src) -> std::unique_ptr<RamStatement> {
972  std::vector<std::unique_ptr<RamExpression>> values;
973  if (src->get()->getArity() == 0) {
974  return std::make_unique<RamQuery>(std::make_unique<RamFilter>(
975  std::make_unique<RamNegation>(std::make_unique<RamEmptinessCheck>(souffle::clone(src))),
976  std::make_unique<RamProject>(souffle::clone(dest), std::move(values))));
977  }
978  for (std::size_t i = 0; i < dest->get()->getArity(); i++) {
979  values.push_back(std::make_unique<RamTupleElement>(0, i));
980  }
981  auto stmt = std::make_unique<RamQuery>(std::make_unique<RamScan>(souffle::clone(src), 0,
982  std::make_unique<RamProject>(souffle::clone(dest), std::move(values))));
983  if (dest->get()->getRepresentation() == RelationRepresentation::EQREL) {
984  return std::make_unique<RamSequence>(
985  std::make_unique<RamExtend>(souffle::clone(dest), souffle::clone(src)), std::move(stmt));
986  }
987  return stmt;
988  };
989 
990  // --- create preamble ---
991 
992  /* Compute non-recursive clauses for relations in scc and push
993  the results in their delta tables. */
994  for (const AstRelation* rel : scc) {
995  /* create update statements for fixpoint (even iteration) */
996  std::unique_ptr<RamStatement> updateRelTable = std::make_unique<RamSequence>(
997  genMerge(translateRelation(rel).get(), translateNewRelation(rel).get()),
998  std::make_unique<RamSwap>(translateDeltaRelation(rel), translateNewRelation(rel)),
999  std::make_unique<RamClear>(translateNewRelation(rel)));
1000 
1001  /* measure update time for each relation */
1002  if (Global::config().has("profile")) {
1003  updateRelTable = std::make_unique<RamLogRelationTimer>(std::move(updateRelTable),
1004  LogStatement::cRecursiveRelation(toString(rel->getQualifiedName()), rel->getSrcLoc()),
1005  translateNewRelation(rel));
1006  }
1007 
1008  /* drop temporary tables after recursion */
1009  appendStmt(postamble, std::make_unique<RamClear>(translateDeltaRelation(rel)));
1010  appendStmt(postamble, std::make_unique<RamClear>(translateNewRelation(rel)));
1011 
1012  /* Generate code for non-recursive part of relation */
1013  /* Generate merge operation for temp tables */
1014  appendStmt(preamble, translateNonRecursiveRelation(*rel, recursiveClauses));
1015  appendStmt(preamble, genMerge(translateDeltaRelation(rel).get(), translateRelation(rel).get()));
1016 
1017  /* Add update operations of relations to parallel statements */
1018  appendStmt(updateTable, std::move(updateRelTable));
1019  }
1020 
1021  // --- build main loop ---
1022 
1023  std::vector<std::unique_ptr<RamStatement>> loopSeq;
1024 
1025  // create a utility to check SCC membership
1026  auto isInSameSCC = [&](const AstRelation* rel) {
1027  return std::find(scc.begin(), scc.end(), rel) != scc.end();
1028  };

1031  for (const AstRelation* rel : scc) {

1033

1035  for (const auto& cl : getClauses(*program, *rel)) {

1038  continue;

1047

1063  }

1067  // atoms)
1068  nameUnnamedVariables(r1.get());

1071  for (size_t k = j + 1; k < atoms.size(); k++) {
1072  if (isInSameSCC(getAtomRelation(atoms[k], program))) {
1073  auto cur = souffle::clone(getBodyLiterals<AstAtom>(*r1)[k]);
1074  cur->setQualifiedName(
1075  translateDeltaRelation(getAtomRelation(atoms[k], program))->get()->getName());
1076  r1->addToBody(std::make_unique<AstNegation>(std::move(cur)));
1077  }

1089  LogStatement::tRecursiveRule(relationName, version, srcLocation, clauseText);

1121  continue;

1132  translateNewRelation(rel));

1134  appendStmt(loopRelSeq, std::move(newStmt));

1138  appendStmt(loopSeq, std::make_unique<RamSequence>(std::move(loopRelSeq)));

1145  : std::move(clause));

1160  std::make_unique<RamExit>(std::move(exitCond)),
1161  std::make_unique<RamSequence>(std::move(updateTable)))));
1162  }
1163  if (postamble.size() > 0) {
1164  appendStmt(res, std::make_unique<RamSequence>(std::move(postamble)));
1165  }

1167  return std::make_unique<RamSequence>(std::move(res));
