33 /**
34  * @class MakeIndexTransformer
35  * @brief Make indexable operations to indexed operations.
36  *
37  * The transformer assumes that the RAM has been levelled before.
38  * The conditions that could be used for an index must be located
39  * immediately after the scan or aggregate operation.
40  *
41  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
42  * QUERY
43  * ...
44  * FOR t1 IN A
45  * IF t1.x = 10 /\ t1.y = 20 /\ C
46  * ...
47  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
48  *
49  * will be rewritten to
50  *
51  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
52  * QUERY
53  * ...
54  * SEARCH t1 IN A INDEX t1.x=10 AND t1.y = 20
55  * IF C
56  * ...
57  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
58  */
