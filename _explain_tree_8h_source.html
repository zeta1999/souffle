32 class ScreenBuffer {
33 public:
34  // constructor
35  ScreenBuffer(uint32_t w, uint32_t h) : width(w), height(h), buffer(nullptr) {
36  assert(width > 0 && height > 0 && "wrong dimensions");
37  buffer = new char[width * height];
38  memset(buffer, ' ', width * height);
39  }
40 
41  ~ScreenBuffer() {
42  delete[] buffer;
43  }
44 
45  // write into screen buffer at a specific location
46  void write(uint32_t x, uint32_t y, const std::string& s) {
47  assert(x < width && "wrong x dimension");
48  assert(y < height && "wrong y dimension");
49  assert(x + s.length() <= width && "string too long");
50  for (size_t i = 0; i < s.length(); i++) {
51  buffer[y * width + x + i] = s[i];
52  }
53  }
54 
55  std::string getString() {
56  std::stringstream ss;
57  print(ss);
58  return ss.str();
59  }
60 
61  // print screen buffer
62  void print(std::ostream& os) {
63  if (height > 0 && width > 0) {
64  for (int i = height - 1; i >= 0; i--) {
65  for (size_t j = 0; j < width; j++) {

69  }
70  }
71  }

75  uint32_t height; // height of the screen buffer

136  // place node and its sub-trees
137  void place(uint32_t x, uint32_t y) override {
138  // there must exist at least one kid
139  assert(!children.empty() && "no children");
140 
141  // set x/y pos
142  xpos = x;
143  ypos = y;
144 
145  height = 0;
146 
147  // compute size of bounding box
148  for (const std::unique_ptr<TreeNode>& k : children) {
149  k->place(x, y + 2);
150  x += k->getWidth() + 1;
151  width += k->getWidth() + 1;
152  height = std::max(height, k->getHeight());
153  }
154  height += 2;
155 
156  // text of inner node is longer than all its sub-trees
157  if (width < txt.length()) {
158  width = txt.length();
159  }
160  };
161 
162  // render node text and separator line
163  void render(ScreenBuffer& s) override {
164  s.write(xpos + (width - txt.length()) / 2, ypos, txt);
165  for (const std::unique_ptr<TreeNode>& k : children) {
166  k->render(s);
167  }
168  std::string separator(width - label.length(), '-');
169  separator += label;
170  s.write(xpos, ypos + 1, separator);
171  }
