73 std::unique_ptr<AstRelation> makeInfoRelation(
74  AstClause& originalClause, size_t originalClauseNum, AstTranslationUnit& translationUnit) {
75  AstQualifiedName name =
76  makeRelationName(originalClause.getHead()->getQualifiedName(), "@info", originalClauseNum);
77 
78  // initialise info relation
79  auto infoRelation = new AstRelation();
80  infoRelation->setQualifiedName(name);
81  // set qualifier to INFO_RELATION
82  infoRelation->setRepresentation(RelationRepresentation::INFO);
83 
84  // create new clause containing a single fact
85  auto infoClause = new AstClause();
86  auto infoClauseHead = new AstAtom();
87  infoClauseHead->setQualifiedName(name);
88 
89  // (darth_tytus): Can this be unsigned?
90  infoRelation->addAttribute(std::make_unique<AstAttribute>("clause_num", AstQualifiedName("number")));
91  infoClauseHead->addArgument(std::make_unique<AstNumericConstant>(originalClauseNum));
92 
93  // add head relation as meta info
94  std::vector<std::string> headVariables;
95 
96  // a method to stringify an AstArgument, translating functors and aggregates
97  // keep a global counter of functor and aggregate numbers, which increment for each unique
98  // functor/aggregate
99  int functorNumber = 0;
100  int aggregateNumber = 0;
101  auto getArgInfo = [&](AstArgument* arg) -> std::string {
102  if (auto* var = dynamic_cast<AstVariable*>(arg)) {
103  return toString(*var);
104  } else if (auto* constant = dynamic_cast<AstConstant*>(arg)) {
105  return toString(*constant);
106  }
107  if (nullptr != dynamic_cast<AstUnnamedVariable*>(arg)) {
108  return "_";
109  }
110  if (nullptr != dynamic_cast<AstFunctor*>(arg)) {
111  return tfm::format("functor_%d", functorNumber++);
112  }
113  if (nullptr != dynamic_cast<AstAggregator*>(arg)) {
114  return tfm::format("agg_%d", aggregateNumber++);
115  }
116 
117  fatal("Unhandled argument type");
118  };
119 
120  // add head arguments
121  for (auto& arg : originalClause.getHead()->getArguments()) {
122  headVariables.push_back(getArgInfo(arg));
123  }
124 
125  // join variables in the head with commas
126  std::stringstream headVariableString;
127  headVariableString << join(headVariables, ",");
128 
129  // add an attribute to infoRelation for the head of clause
130  infoRelation->addAttribute(
131  std::make_unique<AstAttribute>(std::string("head_vars"), AstQualifiedName("symbol")));
132  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(toString(join(headVariables, ","))));
133 
134  // visit all body literals and add to info clause head
135  for (size_t i = 0; i < originalClause.getBodyLiterals().size(); i++) {
136  auto lit = originalClause.getBodyLiterals()[i];
137 
138  const AstAtom* atom = nullptr;
139  if (dynamic_cast<AstAtom*>(lit) != nullptr) {
140  atom = static_cast<AstAtom*>(lit);
141  } else if (dynamic_cast<AstNegation*>(lit) != nullptr) {
142  atom = static_cast<AstNegation*>(lit)->getAtom();
143  } else if (dynamic_cast<AstProvenanceNegation*>(lit) != nullptr) {
144  atom = static_cast<AstProvenanceNegation*>(lit)->getAtom();
145  }
146 
147  // add an attribute for atoms and binary constraints
148  if (atom != nullptr || dynamic_cast<AstBinaryConstraint*>(lit) != nullptr) {
149  infoRelation->addAttribute(std::make_unique<AstAttribute>(
150  std::string("rel_") + std::to_string(i), AstQualifiedName("symbol")));
151  }
152 
153  if (atom != nullptr) {
154  std::string relName = toString(atom->getQualifiedName());
155 
156  // for an atom, add its name and variables (converting aggregates to variables)
157  if (dynamic_cast<AstAtom*>(lit) != nullptr) {
158  std::string atomDescription = relName;
159 
160  for (auto& arg : atom->getArguments()) {
161  atomDescription.append("," + getArgInfo(arg));
162  }
163 
164  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(atomDescription));
165  // for a negation, add a marker with the relation name
166  } else if (dynamic_cast<AstNegation*>(lit) != nullptr) {
167  infoClauseHead->addArgument(std::make_unique<AstStringConstant>("!" + relName));
168  }
169  }
170  }
171 
172  // visit all body constraints and add to info clause head
173  for (size_t i = 0; i < originalClause.getBodyLiterals().size(); i++) {
174  auto lit = originalClause.getBodyLiterals()[i];
175 
176  if (auto con = dynamic_cast<AstBinaryConstraint*>(lit)) {
177  // for a constraint, add the constraint symbol and LHS and RHS
178  std::string constraintDescription = toBinaryConstraintSymbol(con->getOperator());
179 
180  constraintDescription.append("," + getArgInfo(con->getLHS()));
181  constraintDescription.append("," + getArgInfo(con->getRHS()));
182 
183  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(constraintDescription));
184  }
185  }
186 
187  infoRelation->addAttribute(std::make_unique<AstAttribute>("clause_repr", AstQualifiedName("symbol")));
188  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(toString(originalClause)));
189 
190  // set clause head and add clause to info relation
191  infoClause->setHead(std::unique_ptr<AstAtom>(infoClauseHead));
192  translationUnit.getProgram()->addClause(std::unique_ptr<AstClause>(infoClause));
193 
194  return std::unique_ptr<AstRelation>(infoRelation);
195 }

197 /** Transform eqrel relations to explicitly define equivalence relations */
198 void transformEqrelRelation(AstProgram& program, AstRelation& rel) {
199  assert(rel.getRepresentation() == RelationRepresentation::EQREL &&
200  "attempting to transform non-eqrel relation");
201  assert(rel.getArity() == 2 && "eqrel relation not binary");
202 
203  rel.setRepresentation(RelationRepresentation::BTREE);
204 
205  // transitivity
206  // transitive clause: A(x, z) :- A(x, y), A(y, z).
207  auto transitiveClause = new AstClause();
208  auto transitiveClauseHead = new AstAtom(rel.getQualifiedName());
209  transitiveClauseHead->addArgument(std::make_unique<AstVariable>("x"));
210  transitiveClauseHead->addArgument(std::make_unique<AstVariable>("z"));
211 
212  auto transitiveClauseBody = new AstAtom(rel.getQualifiedName());
213  transitiveClauseBody->addArgument(std::make_unique<AstVariable>("x"));
214  transitiveClauseBody->addArgument(std::make_unique<AstVariable>("y"));
215 
216  auto transitiveClauseBody2 = new AstAtom(rel.getQualifiedName());
217  transitiveClauseBody2->addArgument(std::make_unique<AstVariable>("y"));
218  transitiveClauseBody2->addArgument(std::make_unique<AstVariable>("z"));
219 
220  transitiveClause->setHead(std::unique_ptr<AstAtom>(transitiveClauseHead));
221  transitiveClause->addToBody(std::unique_ptr<AstLiteral>(transitiveClauseBody));
222  transitiveClause->addToBody(std::unique_ptr<AstLiteral>(transitiveClauseBody2));
223  program.addClause(std::unique_ptr<AstClause>(transitiveClause));
224 
225  // symmetric
226  // symmetric clause: A(x, y) :- A(y, x).
227  auto symClause = new AstClause();
228  auto symClauseHead = new AstAtom(rel.getQualifiedName());
229  symClauseHead->addArgument(std::make_unique<AstVariable>("x"));
230  symClauseHead->addArgument(std::make_unique<AstVariable>("y"));
231 
232  auto symClauseBody = new AstAtom(rel.getQualifiedName());
233  symClauseBody->addArgument(std::make_unique<AstVariable>("y"));
234  symClauseBody->addArgument(std::make_unique<AstVariable>("x"));
235 
236  symClause->setHead(std::unique_ptr<AstAtom>(symClauseHead));
237  symClause->addToBody(std::unique_ptr<AstLiteral>(symClauseBody));
238  program.addClause(std::unique_ptr<AstClause>(symClause));
239 
240  // reflexivity
241  // reflexive clause: A(x, x) :- A(x, _).
242  auto reflexiveClause = new AstClause();
243  auto reflexiveClauseHead = new AstAtom(rel.getQualifiedName());
244  reflexiveClauseHead->addArgument(std::make_unique<AstVariable>("x"));
245  reflexiveClauseHead->addArgument(std::make_unique<AstVariable>("x"));
246 
247  auto reflexiveClauseBody = new AstAtom(rel.getQualifiedName());
248  reflexiveClauseBody->addArgument(std::make_unique<AstVariable>("x"));
249  reflexiveClauseBody->addArgument(std::make_unique<AstUnnamedVariable>());
250 
251  reflexiveClause->setHead(std::unique_ptr<AstAtom>(reflexiveClauseHead));
252  reflexiveClause->addToBody(std::unique_ptr<AstLiteral>(reflexiveClauseBody));
253  program.addClause(std::unique_ptr<AstClause>(reflexiveClause));
254 }

257 std::unique_ptr<AstArgument> getNextLevelNumber(const std::vector<AstArgument*>& levels) {
258  if (levels.empty()) return mk<AstNumericConstant>(0);
259 
260  auto max = levels.size() == 1
261  ? std::unique_ptr<AstArgument>(levels[0])
262  : mk<AstIntrinsicFunctor>("max",
263  map(levels, [](auto&& x) { return std::unique_ptr<AstArgument>(x); }));
264 
265  return mk<AstIntrinsicFunctor>("+", std::move(max), mk<AstNumericConstant>(1));
266 }

269 bool ProvenanceTransformer::transformMaxHeight(AstTranslationUnit& translationUnit) {
270  auto program = translationUnit.getProgram();
271 
272  for (auto relation : program->getRelations()) {
273  if (relation->getRepresentation() == RelationRepresentation::EQREL) {
274  // Explicitly expand eqrel relation
275  transformEqrelRelation(*program, *relation);
276  }
277  }
278 
279  for (auto relation : program->getRelations()) {
280  // generate info relations for each clause
281  // do this before all other transformations so that we record
282  // the original rule without any instrumentation
283  for (auto clause : getClauses(*program, *relation)) {
284  if (!isFact(*clause)) {
285  // add info relation
286  program->addRelation(
287  makeInfoRelation(*clause, getClauseNum(program, clause), translationUnit));
288  }
289  }
290 
291  relation->addAttribute(
292  std::make_unique<AstAttribute>(std::string("@rule_number"), AstQualifiedName("number")));
293  relation->addAttribute(
294  std::make_unique<AstAttribute>(std::string("@level_number"), AstQualifiedName("number")));
295 
296  for (auto clause : getClauses(*program, *relation)) {
297  size_t clauseNum = getClauseNum(program, clause);
298 
299  // mapper to add two provenance columns to atoms
300  struct M : public AstNodeMapper {
301  using AstNodeMapper::operator();
302 
303  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
304  // add provenance columns
305  if (auto atom = dynamic_cast<AstAtom*>(node.get())) {
306  atom->addArgument(std::make_unique<AstUnnamedVariable>());
307  atom->addArgument(std::make_unique<AstUnnamedVariable>());
308  } else if (auto neg = dynamic_cast<AstNegation*>(node.get())) {
309  auto atom = neg->getAtom();
310  atom->addArgument(std::make_unique<AstUnnamedVariable>());
311  atom->addArgument(std::make_unique<AstUnnamedVariable>());
312  }
313 
314  // otherwise - apply mapper recursively
315  node->apply(*this);
316  return node;
317  }
318  };
319 
320  // add unnamed vars to each atom nested in arguments of head
321  clause->getHead()->apply(M());
322 
323  // if fact, level number is 0
324  if (isFact(*clause)) {
325  clause->getHead()->addArgument(std::make_unique<AstNumericConstant>(0));
326  clause->getHead()->addArgument(std::make_unique<AstNumericConstant>(0));
327  } else {
328  std::vector<AstArgument*> bodyLevels;
329 
330  for (size_t i = 0; i < clause->getBodyLiterals().size(); i++) {
331  auto lit = clause->getBodyLiterals()[i];
332 
333  // add unnamed vars to each atom nested in arguments of lit
334  lit->apply(M());
335 
336  // add two provenance columns to lit; first is rule num, second is level num
337  if (auto atom = dynamic_cast<AstAtom*>(lit)) {
338  atom->addArgument(std::make_unique<AstUnnamedVariable>());
339  atom->addArgument(std::make_unique<AstVariable>("@level_num_" + std::to_string(i)));
340  bodyLevels.push_back(new AstVariable("@level_num_" + std::to_string(i)));
341  }
342  }
343 
344  // add two provenance columns to head lit
345  clause->getHead()->addArgument(std::make_unique<AstNumericConstant>(clauseNum));
346  clause->getHead()->addArgument(getNextLevelNumber(bodyLevels));
347  }
348  }
349  }
350  return true;
351 }
