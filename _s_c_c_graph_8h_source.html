69  /** Get all SCCs containing a successor of a given relation. */
70  std::set<size_t> getSuccessorSCCs(const AstRelation* relation) const {
71  std::set<size_t> successorSccs;
72  const auto scc = relationToScc.at(relation);
73  for (const auto& successor : precedenceGraph->graph().successors(relation)) {
74  const auto successorScc = relationToScc.at(successor);

191  /** Return if the given SCC is recursive. */
192  bool isRecursive(const size_t scc) const {
193  const std::set<const AstRelation*>& sccRelations = sccToRelation.at(scc);
194  if (sccRelations.size() == 1) {
195  const AstRelation* singleRelation = *sccRelations.begin();
196  if (precedenceGraph->graph().predecessors(singleRelation).count(singleRelation) == 0u) {
197  return false;
198  }
