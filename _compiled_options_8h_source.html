112  /**
113  * Parses the given command line parameters, handles -h help requests or errors
114  * and returns whether the parsing was successful or not.
115  */
116  bool parse(int argc, char** argv) {
117  // get executable name
118  std::string exec_name = "analysis";
119  if (argc > 0) {
120  exec_name = argv[0];
121  }
122 
123  // local options
124  std::string fact_dir = input_dir;
125  std::string out_dir = output_dir;
126 
127 // avoid warning due to Solaris getopt.h
128 #pragma GCC diagnostic push
129 #pragma GCC diagnostic ignored "-Wwrite-strings"
130  // long options
131  option longOptions[] = {{"facts", true, nullptr, 'F'}, {"output", true, nullptr, 'D'},
132  {"profile", true, nullptr, 'p'}, {"jobs", true, nullptr, 'j'}, {"index", true, nullptr, 'i'},
133  // the terminal option -- needs to be null
134  {nullptr, false, nullptr, 0}};
135 #pragma GCC diagnostic pop
136 
137  // check whether all options are fine
138  bool ok = true;
139 
140  int c; /* command-line arguments processing */
141  while ((c = getopt_long(argc, argv, "D:F:hp:j:i:", longOptions, nullptr)) != EOF) {
142  switch (c) {
143  /* Fact directories */
144  case 'F':
145  if (!existDir(optarg)) {
146  printf("Fact directory %s does not exists!\n", optarg);
147  ok = false;
148  }
149  fact_dir = optarg;
150  break;
151  /* Output directory for resulting .csv files */
152  case 'D':
153  if (*optarg && !existDir(optarg)) {
154  printf("Output directory %s does not exists!\n", optarg);
155  ok = false;
156  }
157  out_dir = optarg;
158  break;
159  case 'p':
160  if (!profiling) {
161  std::cerr << "\nError: profiling was not enabled in compilation\n\n";
162  printHelpPage(exec_name);
163  exit(EXIT_FAILURE);
164  }
165  profile_name = optarg;
166  break;
167  case 'j':
168 #ifdef _OPENMP
169  if (std::string(optarg) == "auto") {
170  num_jobs = 0;
171  } else {
172  int num = atoi(optarg);
173  if (num > 0) {
174  num_jobs = num;
175  } else {
176  std::cerr << "Invalid number of jobs [-j]: " << optarg << "\n";
177  ok = false;
178  }
179  }
180 #else
181  std::cerr << "\nWarning: OpenMP was not enabled in compilation\n\n";
182 #endif
183  break;
184  default: printHelpPage(exec_name); return false;
185  }
186  }
187 
188  // update member fields
189  input_dir = fact_dir;
190  output_dir = out_dir;
191 
192  // return success state
193  return ok;
194  }

201  std::cerr << "====================================================================\n";

206  std::cerr << " (default: " << output_dir << ")\n";
207  std::cerr << " (suppress output with \"\")\n";
208  std::cerr << " -F <DIR>, --facts=<DIR> -- Specify directory for fact files\n";
209  std::cerr << " (default: " << input_dir << ")\n";
210  if (profiling) {
211  std::cerr << " -p <file>, --profile=<file> -- Specify filename for profiling\n";
212  std::cerr << " (default: " << profile_name << ")\n";
213  }
214 #ifdef _OPENMP
215  std::cerr << " -j <NUM>, --jobs=<NUM> -- Specify number of threads\n";
216  if (num_jobs > 0) {

218  } else {

220  }

225  std::cout << " Copyright (c) 2013-16 Oracle and/or its affiliates." << std::endl;
226  std::cerr << " All rights reserved.\n";
227  std::cerr << "====================================================================\n";
228  }
