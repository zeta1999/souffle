51  /** @brief templated method to compute/retrieve an analysis for a translation unit */
52  template <class Analysis>
53  Analysis* getAnalysis() const {
54  static const bool debug = Global::config().has("debug-report");
55  std::string name = Analysis::name;
56  auto it = analyses.find(name);
57  if (it == analyses.end()) {
58  // analysis does not exist yet, create instance and run it.
59  auto analysis = std::make_unique<Analysis>(Analysis::name);
60  analysis->run(*this);
61  // output analysis in debug report
62  if (debug) {
63  std::stringstream ramAnalysisStr;
64  ramAnalysisStr << *analysis;
65  if (!ramAnalysisStr.str().empty()) {
66  debugReport.addSection(
67  analysis->getName(), "RAM Analysis " + analysis->getName(), ramAnalysisStr.str());
68  }
69  }
70  // check it hasn't been created by someone else, and insert if not
71  it = analyses.find(name);
72  if (it == analyses.end()) {
73  analyses[name] = std::move(analysis);
74  }
75  }
76  return dynamic_cast<Analysis*>(analyses[name].get());
77  }
