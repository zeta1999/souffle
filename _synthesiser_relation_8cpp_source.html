25 std::unique_ptr<SynthesiserRelation> SynthesiserRelation::getSynthesiserRelation(
26  const RamRelation& ramRel, const MinIndexSelection& indexSet, bool isProvenance) {
27  SynthesiserRelation* rel;
28 
29  // Handle the qualifier in souffle code
30  if (isProvenance) {
31  rel = new SynthesiserDirectRelation(ramRel, indexSet, isProvenance);
32  } else if (ramRel.isNullary()) {
33  rel = new SynthesiserNullaryRelation(ramRel, indexSet, isProvenance);
34  } else if (ramRel.getRepresentation() == RelationRepresentation::BTREE) {
35  rel = new SynthesiserDirectRelation(ramRel, indexSet, isProvenance);
36  } else if (ramRel.getRepresentation() == RelationRepresentation::BRIE) {
37  rel = new SynthesiserBrieRelation(ramRel, indexSet, isProvenance);
38  } else if (ramRel.getRepresentation() == RelationRepresentation::EQREL) {
39  rel = new SynthesiserEqrelRelation(ramRel, indexSet, isProvenance);
40  } else if (ramRel.getRepresentation() == RelationRepresentation::INFO) {
41  rel = new SynthesiserInfoRelation(ramRel, indexSet, isProvenance);
42  } else {
43  // Handle the data structure command line flag
44  if (ramRel.getArity() > 6) {
45  rel = new SynthesiserIndirectRelation(ramRel, indexSet, isProvenance);
46  } else {
47  rel = new SynthesiserDirectRelation(ramRel, indexSet, isProvenance);
48  }
49  }
50 
51  assert(rel != nullptr && "relation type not specified");
52 
53  // generate index set
54  rel->computeIndices();
55 
56  return std::unique_ptr<SynthesiserRelation>(rel);
57 }

97 /** Generate index set for a direct indexed relation */
98 void SynthesiserDirectRelation::computeIndices() {
99  // Generate and set indices
100  MinIndexSelection::OrderCollection inds = indices.getAllOrders();
101 
102  // generate a full index if no indices exist
103  assert(!inds.empty() && "no full index in relation");
104 
105  size_t index_nr = 0;
106  // expand all search orders to be full
107  for (auto& ind : inds) {
108  // use a set as a cache for fast lookup
109  std::set<int> curIndexElems(ind.begin(), ind.end());
110 
111  // If this relation is used with provenance,
112  // we must expand all search orders to be full indices,
113  // since weak/strong comparators and updaters need this,
114  // and also add provenance annotations to the indices
115  if (isProvenance) {
116  // expand index to be full
117  for (size_t i = 0; i < getArity() - relation.getAuxiliaryArity(); i++) {
118  if (curIndexElems.find(i) == curIndexElems.end()) {
119  ind.push_back(i);
120  }
121  }
122  // remove any provenance annotations already in the index order
123  if (curIndexElems.find(getArity() - relation.getAuxiliaryArity() + 1) != curIndexElems.end()) {
124  ind.erase(std::find(ind.begin(), ind.end(), getArity() - relation.getAuxiliaryArity() + 1));
125  }
126 
127  if (curIndexElems.find(getArity() - relation.getAuxiliaryArity()) != curIndexElems.end()) {
128  ind.erase(std::find(ind.begin(), ind.end(), getArity() - relation.getAuxiliaryArity()));
129  }
130 
131  // add provenance annotations to the index, but in reverse order
132  ind.push_back(getArity() - relation.getAuxiliaryArity() + 1);
133  ind.push_back(getArity() - relation.getAuxiliaryArity());
134  masterIndex = 0;
135  } else if (ind.size() == getArity()) {
136  masterIndex = index_nr;
137  }
138  index_nr++;
139  }
140  assert(masterIndex < inds.size() && "no full index in relation");
141  computedIndices = inds;
142 }

160 /** Generate type struct of a direct indexed relation */
161 void SynthesiserDirectRelation::generateTypeStruct(std::ostream& out) {
162  size_t arity = getArity();
163  size_t auxiliaryArity = relation.getAuxiliaryArity();
164  const auto& inds = getIndices();
165  size_t numIndexes = inds.size();
166  std::map<MinIndexSelection::LexOrder, int> indexToNumMap;
167 
168  // struct definition
169  out << "struct " << getTypeName() << " {\n";
170 
171  // stored tuple type
172  out << "using t_tuple = Tuple<RamDomain, " << arity << ">;\n";
173 
174  // generate an updater class for provenance
175  if (isProvenance) {
176  out << "struct updater_" << getTypeName() << " {\n";
177  out << "void update(t_tuple& old_t, const t_tuple& new_t) {\n";
178 
179  for (size_t i = arity - auxiliaryArity; i < arity; i++) {
180  out << "old_t[" << i << "] = new_t[" << i << "];\n";
181  }
182 
183  out << "}\n";
184  out << "};\n";
185  }
186 
187  // generate the btree type for each relation
188  for (size_t i = 0; i < inds.size(); i++) {
189  auto& ind = inds[i];
190 
191  if (i < getMinIndexSelection().getAllOrders().size()) {
192  indexToNumMap[getMinIndexSelection().getAllOrders()[i]] = i;
193  }
194 
195  auto genstruct = [&](std::string name, size_t bound) {
196  out << "struct " << name << "{\n";
197  out << " int operator()(const t_tuple& a, const t_tuple& b) const {\n";
198  out << " return ";
199  std::function<void(size_t)> gencmp = [&](size_t i) {
200  size_t attrib = ind[i];
201  out << "(a[" << attrib << "] < b[" << attrib << "]) ? -1 : ((a[" << attrib << "] > b["
202  << attrib << "]) ? 1 :(";
203  if (i + 1 < bound) {
204  gencmp(i + 1);
205  } else {
206  out << "0";
207  }
208  out << "))";
209  };
210  gencmp(0);
211  out << ";\n }\n";
212  out << "bool less(const t_tuple& a, const t_tuple& b) const {\n";
213  out << " return ";
214  std::function<void(size_t)> genless = [&](size_t i) {
215  size_t attrib = ind[i];
216  out << " a[" << attrib << "] < b[" << attrib << "]";
217  if (i + 1 < bound) {
218  out << "|| (a[" << attrib << "] == b[" << attrib << "] && (";
219  genless(i + 1);
220  out << "))";
221  }
222  };
223  genless(0);
224  out << ";\n }\n";
225  out << "bool equal(const t_tuple& a, const t_tuple& b) const {\n";
226  out << "return ";
227  std::function<void(size_t)> geneq = [&](size_t i) {
228  size_t attrib = ind[i];
229  out << "a[" << attrib << "] == b[" << attrib << "]";
230  if (i + 1 < bound) {
231  out << "&&";
232  geneq(i + 1);
233  }
234  };
235  geneq(0);
236  out << ";\n }\n";
237  out << "};\n";
238  };
239 
240  std::string comparator = "t_comparator_" + std::to_string(i);
241  genstruct(comparator, ind.size());
242 
243  // for provenance, all indices must be full so we use btree_set
244  // also strong/weak comparators and updater methods
245  if (isProvenance) {
246  std::string comparator_aux;
247  if (provenanceIndexNumbers.find(i) == provenanceIndexNumbers.end()) {
248  // index for bottom up phase
249  comparator_aux = "t_comparator_" + std::to_string(i) + "_aux";
250  genstruct(comparator_aux, ind.size() - 2);
251  } else {
252  // index for top down phase
253  comparator_aux = comparator;
254  }
255  out << "using t_ind_" << i << " = btree_set<t_tuple," << comparator
256  << ",std::allocator<t_tuple>,256,typename "
257  "souffle::detail::default_strategy<t_tuple>::type,"
258  << comparator_aux << ",updater_" << getTypeName() << ">;\n";
259  } else {
260  if (ind.size() == arity) {
261  out << "using t_ind_" << i << " = btree_set<t_tuple," << comparator << ">;\n";
262  } else {
263  // without provenance, some indices may be not full, so we use btree_multiset for those
264  out << "using t_ind_" << i << " = btree_multiset<t_tuple," << comparator << ">;\n";
265  }
266  }
267  out << "t_ind_" << i << " ind_" << i << ";\n";
268  }
269 
270  // typedef master index iterator to be struct iterator
271  out << "using iterator = t_ind_" << masterIndex << "::iterator;\n";
272 
273  // create a struct storing hints for each btree
274  out << "struct context {\n";
275  for (size_t i = 0; i < numIndexes; i++) {
276  out << "t_ind_" << i << "::operation_hints hints_" << i << ";\n";
277  }
278  out << "};\n";
279  out << "context createContext() { return context(); }\n";
280 
281  // insert methods
282  out << "bool insert(const t_tuple& t) {\n";
283  out << "context h;\n";
284  out << "return insert(t, h);\n";
285  out << "}\n"; // end of insert(t_tuple&)
286 
287  out << "bool insert(const t_tuple& t, context& h) {\n";
288  out << "if (ind_" << masterIndex << ".insert(t, h.hints_" << masterIndex << ")) {\n";
289  for (size_t i = 0; i < numIndexes; i++) {
290  if (i != masterIndex && provenanceIndexNumbers.find(i) == provenanceIndexNumbers.end()) {
291  out << "ind_" << i << ".insert(t, h.hints_" << i << ");\n";
292  }
293  }
294  out << "return true;\n";
295  out << "} else return false;\n";
296  out << "}\n"; // end of insert(t_tuple&, context&)
297 
298  out << "bool insert(const RamDomain* ramDomain) {\n";
299  out << "RamDomain data[" << arity << "];\n";
300  out << "std::copy(ramDomain, ramDomain + " << arity << ", data);\n";
301  out << "const t_tuple& tuple = reinterpret_cast<const t_tuple&>(data);\n";
302  out << "context h;\n";
303  out << "return insert(tuple, h);\n";
304  out << "}\n"; // end of insert(RamDomain*)
305 
306  std::vector<std::string> decls;
307  std::vector<std::string> params;
308  for (size_t i = 0; i < arity; i++) {
309  decls.push_back("RamDomain a" + std::to_string(i));
310  params.push_back("a" + std::to_string(i));
311  }
312  out << "bool insert(" << join(decls, ",") << ") {\n";
313  out << "RamDomain data[" << arity << "] = {" << join(params, ",") << "};\n";
314  out << "return insert(data);\n";
315  out << "}\n"; // end of insert(RamDomain x1, RamDomain x2, ...)
316 
317  // contains methods
318  out << "bool contains(const t_tuple& t, context& h) const {\n";
319  out << "return ind_" << masterIndex << ".contains(t, h.hints_" << masterIndex << ");\n";
320  out << "}\n";
321 
322  out << "bool contains(const t_tuple& t) const {\n";
323  out << "context h;\n";
324  out << "return contains(t, h);\n";
325  out << "}\n";
326 
327  // size method
328  out << "std::size_t size() const {\n";
329  out << "return ind_" << masterIndex << ".size();\n";
330  out << "}\n";
331 
332  // find methods
333  out << "iterator find(const t_tuple& t, context& h) const {\n";
334  out << "return ind_" << masterIndex << ".find(t, h.hints_" << masterIndex << ");\n";
335  out << "}\n";
336 
337  out << "iterator find(const t_tuple& t) const {\n";
338  out << "context h;\n";
339  out << "return find(t, h);\n";
340  out << "}\n";
341 
342  // empty lowerUpperRange method
343  out << "range<iterator> lowerUpperRange_" << SearchSignature(arity)
344  << "(const t_tuple& lower, const t_tuple& upper, context& h) const "
345  "{\n";
346 
347  out << "return range<iterator>(ind_" << masterIndex << ".begin(),ind_" << masterIndex << ".end());\n";
348  out << "}\n";
349 
350  out << "range<iterator> lowerUpperRange_" << SearchSignature(arity)
351  << "(const t_tuple& lower, const t_tuple& upper) const {\n";
352 
353  out << "return range<iterator>(ind_" << masterIndex << ".begin(),ind_" << masterIndex << ".end());\n";
354  out << "}\n";
355 
356  // lowerUpperRange methods for each pattern which is used to search this relation
357  for (auto search : getMinIndexSelection().getSearches()) {
358  auto& lexOrder = getMinIndexSelection().getLexOrder(search);
359  size_t indNum = indexToNumMap[lexOrder];
360 
361  out << "range<t_ind_" << indNum << "::iterator> lowerUpperRange_" << search;
362  out << "(const t_tuple& lower, const t_tuple& upper, context& h) const {\n";
363 
364  // count size of search pattern
365  size_t indSize = 0;
366  for (size_t column = 0; column < arity; column++) {
367  if (search[column] != AttributeConstraint::None) {
368  indSize++;
369  }
370  }
371 
372  // use the more efficient find() method if the search pattern is full
373  if (indSize == arity) {
374  out << "auto pos = ind_" << indNum << ".find(lower, h.hints_" << indNum << ");\n";
375  out << "auto fin = ind_" << indNum << ".end();\n";
376  out << "if (pos != fin) {fin = pos; ++fin;}\n";
377  out << "return make_range(pos, fin);\n";
378  } else {
379  // generate lower and upper bounds for range search
380  out << "t_tuple low(lower); t_tuple high(lower);\n";
381  // check which indices to pad out
382  for (size_t column = 0; column < arity; column++) {
383  // if bit number column is not set
384  if (search[column] == AttributeConstraint::None) {
385  out << "low[" << column << "] = MIN_RAM_SIGNED;\n";
386  out << "high[" << column << "] = MAX_RAM_SIGNED;\n";
387  }
388  }
389  out << "return make_range(ind_" << indNum << ".lower_bound(low, h.hints_" << indNum << "), ind_"
390  << indNum << ".upper_bound(high, h.hints_" << indNum << "));\n";
391  }
392  out << "}\n";
393 
394  out << "range<t_ind_" << indNum << "::iterator> lowerUpperRange_" << search;
395  out << "(const t_tuple& lower, const t_tuple& upper) const {\n";
396 
397  out << "context h;\n";
398  out << "return lowerUpperRange_" << search << "(lower,upper,h);\n";
399  out << "}\n";
400  }
401 
402  // empty method
403  out << "bool empty() const {\n";
404  out << "return ind_" << masterIndex << ".empty();\n";
405  out << "}\n";
406 
407  // partition method for parallelism
408  out << "std::vector<range<iterator>> partition() const {\n";
409  out << "return ind_" << masterIndex << ".getChunks(400);\n";
410  out << "}\n";
411

414  for (size_t i = 0; i < numIndexes; i++) {
415  out << "ind_" << i << ".clear();\n";
416  }
417  out << "}\n";
418

421  out << "return ind_" << masterIndex << ".begin();\n";

425  out << "return ind_" << masterIndex << ".end();\n";

755 /** Generate index set for a brie relation */
756 void SynthesiserBrieRelation::computeIndices() {
757  assert(!isProvenance && "bries cannot be used with provenance");
758 
759  // Generate and set indices
760  MinIndexSelection::OrderCollection inds = indices.getAllOrders();
761 
762  // generate a full index if no indices exist
763  assert(!inds.empty() && "No full index in relation");
764 
765  // expand all indexes to be full
766  for (auto& ind : inds) {
767  if (ind.size() != getArity()) {
768  // use a set as a cache for fast lookup
769  std::set<int> curIndexElems(ind.begin(), ind.end());
770 
771  // expand index to be full
772  for (size_t i = 0; i < getArity(); i++) {
773  if (curIndexElems.find(i) == curIndexElems.end()) {
774  ind.push_back(i);
775  }
776  }
777  }
778 
779  assert(ind.size() == getArity() && "index is not a full");
780  }
781  masterIndex = 0;
782 
783  computedIndices = inds;
784 }
