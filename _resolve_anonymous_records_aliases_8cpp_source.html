40 std::map<std::string, const AstRecordInit*> ResolveAnonymousRecordsAliases::findVariablesRecordMapping(
41  AstTranslationUnit& tu, const AstClause& clause) {
42  std::map<std::string, const AstRecordInit*> variableRecordMap;
43 
44  auto isVariable = [](AstNode* node) -> bool { return dynamic_cast<AstVariable*>(node) != nullptr; };
45  auto isRecord = [](AstNode* node) -> bool { return dynamic_cast<AstRecordInit*>(node) != nullptr; };
46 
47  auto& typeAnalysis = *tu.getAnalysis<TypeAnalysis>();
48  auto groundedTerms = getGroundedTerms(tu, clause);
49 
50  for (auto* literal : clause.getBodyLiterals()) {
51  if (auto constraint = dynamic_cast<AstBinaryConstraint*>(literal)) {
52  if (!isEqConstraint(constraint->getOperator())) {
53  continue;
54  }
55 
56  auto left = constraint->getLHS();
57  auto right = constraint->getRHS();
58 
59  if (!isVariable(left) && !isVariable(right)) {
60  continue;
61  }
62 
63  if (!isRecord(left) && !isRecord(right)) {
64  continue;
65  }
66 
67  // TODO (darth_tytus): This should change in the future.
68  // Currently type system assigns to anonymous records {- All types - }
69  // which is inelegant.
70  if (!typeAnalysis.getTypes(left).isAll()) {
71  continue;
72  }
73 
74  auto* variable = static_cast<AstVariable*>(isVariable(left) ? left : right);
75  const auto& variableName = variable->getName();
76 
77  if (!groundedTerms.find(variable)->second) {
78  continue;
79  }
80 
81  // We are interested only in the first mapping.
82  if (variableRecordMap.find(variableName) != variableRecordMap.end()) {
83  continue;
84  }
85 
86  auto* record = static_cast<AstRecordInit*>(isRecord(left) ? left : right);
87 
88  variableRecordMap.insert({variableName, record});
89  }
90  }
91 
92  return variableRecordMap;
93 }

95 bool ResolveAnonymousRecordsAliases::replaceNamedVariables(AstTranslationUnit& tu, AstClause& clause) {
96  struct ReplaceVariables : public AstNodeMapper {
97  std::map<std::string, const AstRecordInit*> varToRecordMap;
98 
99  ReplaceVariables(std::map<std::string, const AstRecordInit*> varToRecordMap)
100  : varToRecordMap(std::move(varToRecordMap)){};
101 
102  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
103  if (auto variable = dynamic_cast<AstVariable*>(node.get())) {
104  auto iteratorToRecord = varToRecordMap.find(variable->getName());
105  if (iteratorToRecord != varToRecordMap.end()) {
106  return souffle::clone(iteratorToRecord->second);
107  }
108  }
109 
110  node->apply(*this);
111 
112  return node;
113  }
114  };
115 
116  auto variableToRecordMap = findVariablesRecordMapping(tu, clause);
117  bool changed = variableToRecordMap.size() > 0;
118  if (changed) {
119  ReplaceVariables update(std::move(variableToRecordMap));
120  clause.apply(update);
121  }
122  return changed;
123 }

125 bool ResolveAnonymousRecordsAliases::replaceUnnamedVariable(AstClause& clause) {
126  struct ReplaceUnnamed : public AstNodeMapper {
127  mutable bool changed{false};
128  ReplaceUnnamed() = default;
129 
130  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
131  auto isUnnamed = [](AstNode* node) -> bool {
132  return dynamic_cast<AstUnnamedVariable*>(node) != nullptr;
133  };
134  auto isRecord = [](AstNode* node) -> bool {
135  return dynamic_cast<AstRecordInit*>(node) != nullptr;
136  };
137 
138  if (auto constraint = dynamic_cast<AstBinaryConstraint*>(node.get())) {
139  auto left = constraint->getLHS();
140  auto right = constraint->getRHS();
141  bool hasUnnamed = isUnnamed(left) || isUnnamed(right);
142  bool hasRecord = isRecord(left) || isRecord(right);
143  auto op = constraint->getOperator();
144  if (hasUnnamed && hasRecord && isEqConstraint(op)) {
145  return std::make_unique<AstBooleanConstraint>(true);
146  }
147  }
148 
149  node->apply(*this);
150 
151  return node;
152  }
153  };
154 
155  ReplaceUnnamed update;
156  clause.apply(update);
157 
158  return update.changed;
159 }

167
