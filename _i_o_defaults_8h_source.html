45 
46  changed |= setDefaults(translationUnit);

51  /**
52  * Set IO defaults.
53  *
54  * If no IO type is specified, use 'file'
55  * If no name is set, use the relation name.
56  * Add the operation type to the directives list.
57  * If a global fact directory is specified, add to the directives list.
58  * If a global output directory is specified, add to the directives list.
59  * If stdout is requested at the command line ('-D-'), change all output to stdout.
60  * If a printsize operation is requested, set IO type and operation accordingly.
61  *
62  * @param translationUnit
63  * @return true if any changes were made
64  */
65  bool setDefaults(AstTranslationUnit& translationUnit) {
66  bool changed = false;
67  AstProgram* program = translationUnit.getProgram();
68  for (AstIO* io : program->getIOs()) {
69  // Set a default IO of file
70  if (!io->hasDirective("IO")) {
71  io->addDirective("IO", "file");
72  changed = true;
73  }
74 
75  // Set the relation name
76  if (!io->hasDirective("name")) {
77  io->addDirective("name", getRelationName(io));
78  changed = true;
79  }
80 
81  // Set the operation type (input/output/printsize)
82  if (!io->hasDirective("operation")) {
83  if (io->getType() == AstIoType::input) {
84  io->addDirective("operation", "input");
85  changed = true;
86  // Configure input directory
87  if (Global::config().has("fact-dir")) {
88  io->addDirective("fact-dir", Global::config().get("fact-dir"));
89  }
90  } else if (io->getType() == AstIoType::output) {
91  io->addDirective("operation", "output");
92  changed = true;
93  // Configure output directory
94  if (Global::config().has("output-dir")) {
95  if (Global::config().has("output-dir", "-")) {
96  io->addDirective("IO", "stdout");
97  io->addDirective("headers", "true");
98  } else {
99  io->addDirective("output-dir", Global::config().get("output-dir"));
100  }
101  }
102  } else if (io->getType() == AstIoType::printsize) {
103  io->addDirective("operation", "printsize");
104  io->addDirective("IO", "stdoutprintsize");
105  changed = true;
106  }
107  }
108  }
109 
110  return changed;
111  }
112 
113  /**
114  * Get the relation name from the qualified name.
115  *
116  * @return Valid relation name from the concatenated qualified name.
117  */
118  std::string getRelationName(const AstIO* node) {
119  return toString(join(node->getQualifiedName().getQualifiers(), "."));
120  }
121 };
