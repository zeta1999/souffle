59  bool setAttributeParams(AstTranslationUnit& translationUnit) {
60  bool changed = false;
61  AstProgram* program = translationUnit.getProgram();
62  auto auxArityAnalysis = translationUnit.getAnalysis<AuxiliaryArity>();
63 
64  for (AstIO* io : program->getIOs()) {
65  AstRelation* rel = getRelation(*translationUnit.getProgram(), io->getQualifiedName());
66  // Prepare type system information.
67  std::vector<std::string> attributesParams;
68 
69  for (const auto* attribute : rel->getAttributes()) {
70  attributesParams.push_back(attribute->getName());
71  }
72 
73  // Casting due to json11.h type requirements.
74  long long arity{static_cast<long long>(rel->getArity() - auxArityAnalysis->getArity(rel))};
75  long long auxArity{static_cast<long long>(auxArityAnalysis->getArity(rel))};
76 
77  json11::Json relJson = json11::Json::object{{"arity", arity}, {"auxArity", auxArity},
78  {"params", json11::Json::array(attributesParams.begin(), attributesParams.end())}};
79 
80  json11::Json params = json11::Json::object{
81  {"relation", relJson}, {"records", getRecordsParams(translationUnit)}};
82 
83  io->addDirective("params", params.dump());
84  changed = true;
85  }
86  return changed;
87  }

89  bool setAttributeNames(AstTranslationUnit& translationUnit) {
90  bool changed = false;
91  AstProgram* program = translationUnit.getProgram();
92  for (AstIO* io : program->getIOs()) {
93  if (io->hasDirective("attributeNames")) {
94  continue;
95  }
96  AstRelation* rel = getRelation(*translationUnit.getProgram(), io->getQualifiedName());
97  std::string delimiter("\t");
98  if (io->hasDirective("delimiter")) {
99  delimiter = io->getDirective("delimiter");
100  }
101 
102  std::vector<std::string> attributeNames;
103  for (const auto* attribute : rel->getAttributes()) {
104  attributeNames.push_back(attribute->getName());
105  }
106 
107  if (Global::config().has("provenance")) {
108  auto auxArityAnalysis = translationUnit.getAnalysis<AuxiliaryArity>();
109  std::vector<std::string> originalAttributeNames(
110  attributeNames.begin(), attributeNames.end() - auxArityAnalysis->getArity(rel));
111  io->addDirective("attributeNames", toString(join(originalAttributeNames, delimiter)));
112  } else {
113  io->addDirective("attributeNames", toString(join(attributeNames, delimiter)));
114  }
115  changed = true;
116  }
117  return changed;
118  }

120  bool setAttributeTypes(AstTranslationUnit& translationUnit) {
121  bool changed = false;
122  AstProgram* program = translationUnit.getProgram();
123  auto auxArityAnalysis = translationUnit.getAnalysis<AuxiliaryArity>();
124  auto typeEnv = &translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
125 
126  for (AstIO* io : program->getIOs()) {
127  AstRelation* rel = getRelation(*translationUnit.getProgram(), io->getQualifiedName());
128  // Prepare type system information.
129  std::vector<std::string> attributesTypes;
130 
131  for (const auto* attribute : rel->getAttributes()) {
132  auto typeName = attribute->getTypeName();
133  auto type = getTypeQualifier(typeEnv->getType(typeName));
134  attributesTypes.push_back(type);
135  }
136 
137  // Casting due to json11.h type requirements.
138  long long arity{static_cast<long long>(rel->getArity() - auxArityAnalysis->getArity(rel))};
139  long long auxArity{static_cast<long long>(auxArityAnalysis->getArity(rel))};
140 
141  json11::Json relJson = json11::Json::object{{"arity", arity}, {"auxArity", auxArity},
142  {"types", json11::Json::array(attributesTypes.begin(), attributesTypes.end())}};
143 
144  json11::Json types = json11::Json::object{
145  {"relation", relJson}, {"records", getRecordsTypes(translationUnit)}};
146 
147  io->addDirective("types", types.dump());
148  changed = true;
149  }
150  return changed;
151  }

157  json11::Json getRecordsTypes(AstTranslationUnit& translationUnit) const {
158  static json11::Json ramRecordTypes;
159  // Check if the types where already constructed
160  if (!ramRecordTypes.is_null()) {
161  return ramRecordTypes;
162  }
163 
164  AstProgram* program = translationUnit.getProgram();
165  auto typeEnv = &translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
166  std::vector<std::string> elementTypes;
167  std::map<std::string, json11::Json> records;
168 
169  // Iterate over all record types in the program populating the records map.
170  for (auto* astType : program->getTypes()) {
171  const auto& type = typeEnv->getType(astType->getQualifiedName());
172  if (isA<RecordType>(type)) {
173  elementTypes.clear();
174 
175  for (const Type* field : as<RecordType>(type)->getFields()) {
176  elementTypes.push_back(getTypeQualifier(*field));
177  }
178  const size_t recordArity = elementTypes.size();
179  json11::Json recordInfo = json11::Json::object{
180  {"types", std::move(elementTypes)}, {"arity", static_cast<long long>(recordArity)}};
181  records.emplace(getTypeQualifier(type), std::move(recordInfo));
182  }
183  }
184 
185  ramRecordTypes = json11::Json(records);
186  return ramRecordTypes;
187  }

189  json11::Json getRecordsParams(AstTranslationUnit& translationUnit) const {
190  static json11::Json ramRecordParams;
191  // Check if the types where already constructed
192  if (!ramRecordParams.is_null()) {
193  return ramRecordParams;
194  }
195 
196  AstProgram* program = translationUnit.getProgram();
197  std::vector<std::string> elementParams;
198  std::map<std::string, json11::Json> records;
199 
200  // Iterate over all record types in the program populating the records map.
201  for (auto* astType : program->getTypes()) {
202  if (isA<AstRecordType>(astType)) {
203  elementParams.clear();
204 
205  for (const auto field : as<AstRecordType>(astType)->getFields()) {
206  elementParams.push_back(field->getName());
207  }
208  const size_t recordArity = elementParams.size();
209  json11::Json recordInfo = json11::Json::object{
210  {"params", std::move(elementParams)}, {"arity", static_cast<long long>(recordArity)}};
211  records.emplace(astType->getQualifiedName().toString(), std::move(recordInfo));
212  }
213  }
214 
215  ramRecordParams = json11::Json(records);
216  return ramRecordParams;
217  }
