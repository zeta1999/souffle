36 bool PartitionBodyLiteralsTransformer::transform(AstTranslationUnit& translationUnit) {
37  bool changed = false;
38  AstProgram& program = *translationUnit.getProgram();
39 
40  /* Process:
41  * Go through each clause and construct a variable dependency graph G.
42  * The nodes of G are the variables. A path between a and b exists iff
43  * a and b appear in a common body literal.
44  *
45  * Using the graph, we can extract the body literals that are not associated
46  * with the arguments in the head atom into new relations. Depending on
47  * variable dependencies among these body literals, the literals can
48  * be partitioned into multiple separate new propositional clauses.
49  *
50  * E.g. a(x) <- b(x), c(y), d(y), e(z), f(z). can be transformed into:
51  * - a(x) <- b(x), newrel0(), newrel1().
52  * - newrel0() <- c(y), d(y).
53  * - newrel1() <- e(z), f(z).
54  *
55  * Note that only one pass through the clauses is needed:
56  * - All arguments in the body literals of the transformed clause cannot be
57  * independent of the head arguments (by construction)
58  * - The new relations holding the disconnected body literals are propositional,
59  * hence having no head arguments, and so the transformation does not apply.
60  */
61 
62  // Store clauses to add and remove after analysing the program
63  std::vector<AstClause*> clausesToAdd;
64  std::vector<const AstClause*> clausesToRemove;
65 
66  // The transformation is local to each rule, so can visit each independently
67  visitDepthFirst(program, [&](const AstClause& clause) {
68  // Create the variable dependency graph G
69  Graph<std::string> variableGraph = Graph<std::string>();
70  std::set<std::string> ruleVariables;
71 
72  // Add in the nodes
73  // The nodes of G are the variables in the rule
74  visitDepthFirst(clause, [&](const AstVariable& var) {
75  variableGraph.insert(var.getName());
76  ruleVariables.insert(var.getName());
77  });
78 
79  // Add in the edges
80  // Since we are only looking at reachability in the final graph, it is
81  // enough to just add in an (undirected) edge from the first variable
82  // in the literal to each of the other variables.
83  std::vector<AstLiteral*> literalsToConsider = clause.getBodyLiterals();
84  literalsToConsider.push_back(clause.getHead());
85 
86  for (AstLiteral* clauseLiteral : literalsToConsider) {
87  std::set<std::string> literalVariables;
88 
89  // Store all variables in the literal
90  visitDepthFirst(
91  *clauseLiteral, [&](const AstVariable& var) { literalVariables.insert(var.getName()); });
92 
93  // No new edges if only one variable is present
94  if (literalVariables.size() > 1) {
95  std::string firstVariable = *literalVariables.begin();
96  literalVariables.erase(literalVariables.begin());
97 
98  // Create the undirected edge
99  for (const std::string& var : literalVariables) {
100  variableGraph.insert(firstVariable, var);
101  variableGraph.insert(var, firstVariable);
102  }
103  }
104  }
105 
106  // Find the connected components of G
107  std::set<std::string> seenNodes;
108 
109  // Find the connected component associated with the head
110  std::set<std::string> headComponent;
111  visitDepthFirst(
112  *clause.getHead(), [&](const AstVariable& var) { headComponent.insert(var.getName()); });
113 
114  if (!headComponent.empty()) {
115  variableGraph.visitDepthFirst(*headComponent.begin(), [&](const std::string& var) {
116  headComponent.insert(var);
117  seenNodes.insert(var);
118  });
119  }
120 
121  // Compute all other connected components in the graph G
122  std::set<std::set<std::string>> connectedComponents;
123 
124  for (std::string var : ruleVariables) {
125  if (seenNodes.find(var) != seenNodes.end()) {
126  // Node has already been added to a connected component
127  continue;
128  }
129 
130  // Construct the connected component
131  std::set<std::string> component;
132  variableGraph.visitDepthFirst(var, [&](const std::string& child) {
133  component.insert(child);
134  seenNodes.insert(child);
135  });
136  connectedComponents.insert(component);
137  }
138 
139  if (connectedComponents.empty()) {
140  // No separate connected components, so no point partitioning
141  return;
142  }
143 
144  // Need to extract some disconnected lits!
145  changed = true;
146  std::vector<AstAtom*> replacementAtoms;
147 
148  // Construct the new rules
149  for (const std::set<std::string>& component : connectedComponents) {
150  // Come up with a unique new name for the relation
151  static int disconnectedCount = 0;
152  std::stringstream nextName;
153  nextName << "+disconnected" << disconnectedCount;
154  AstQualifiedName newRelationName = nextName.str();
155  disconnectedCount++;
156 
157  // Create the extracted relation and clause for the component
158  // newrelX() <- disconnectedLiterals(x).
159  auto newRelation = std::make_unique<AstRelation>();
160  newRelation->setQualifiedName(newRelationName);
161  program.addRelation(std::move(newRelation));
162 
163  auto* disconnectedClause = new AstClause();
164  disconnectedClause->setSrcLoc(clause.getSrcLoc());
165  disconnectedClause->setHead(std::make_unique<AstAtom>(newRelationName));
166 
167  // Find the body literals for this connected component
168  std::vector<AstLiteral*> associatedLiterals;
169  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
170  bool associated = false;
171  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) {
172  if (component.find(var.getName()) != component.end()) {
173  associated = true;
174  }
175  });
176  if (associated) {
177  disconnectedClause->addToBody(souffle::clone(bodyLiteral));
178  }
179  }
180 
181  // Create the atom to replace all these literals
182  replacementAtoms.push_back(new AstAtom(newRelationName));
183 
184  // Add the clause to the program
185  clausesToAdd.push_back(disconnectedClause);
186  }
187 
188  // Create the replacement clause
189  // a(x) <- b(x), c(y), d(z). --> a(x) <- newrel0(), newrel1(), b(x).
190  auto* replacementClause = new AstClause();
191  replacementClause->setSrcLoc(clause.getSrcLoc());
192  replacementClause->setHead(souffle::clone(clause.getHead()));
193 
194  // Add the new propositions to the clause first
195  for (AstAtom* newAtom : replacementAtoms) {
196  replacementClause->addToBody(std::unique_ptr<AstLiteral>(newAtom));
197  }
198 
199  // Add the remaining body literals to the clause
200  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
201  bool associated = false;
202  bool hasVariables = false;
203  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) {
204  hasVariables = true;
205  if (headComponent.find(var.getName()) != headComponent.end()) {
206  associated = true;
207  }
208  });
209  if (associated || !hasVariables) {
210  replacementClause->addToBody(souffle::clone(bodyLiteral));
211  }
212  }
213 
214  // Replace the old clause with the new one
215  clausesToRemove.push_back(&clause);
216  clausesToAdd.push_back(replacementClause);
217  });
218 
219  // Adjust the program
220  for (AstClause* newClause : clausesToAdd) {
221  program.addClause(std::unique_ptr<AstClause>(newClause));
222  }
223 
224  for (const AstClause* oldClause : clausesToRemove) {
225  program.removeClause(oldClause);
226  }
227 
228  return changed;
229 }
