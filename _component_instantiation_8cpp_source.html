54  * A container type for the (instantiated) content of a component.
55  */
56 struct ComponentContent {
57  std::vector<std::unique_ptr<AstType>> types;
58  std::vector<std::unique_ptr<AstRelation>> relations;
59  std::vector<std::unique_ptr<AstIO>> ios;
60  std::vector<std::unique_ptr<AstClause>> clauses;
61 
62  void add(std::unique_ptr<AstType>& type, ErrorReport& report) {
63  // add to result content (check existence first)
64  auto foundItem =
65  std::find_if(types.begin(), types.end(), [&](const std::unique_ptr<AstType>& element) {
66  return (element->getQualifiedName() == type->getQualifiedName());
67  });
68  if (foundItem != types.end()) {
69  Diagnostic err(Diagnostic::Type::ERROR,
70  DiagnosticMessage(
71  "Redefinition of type " + toString(type->getQualifiedName()), type->getSrcLoc()),
72  {DiagnosticMessage("Previous definition", (*foundItem)->getSrcLoc())});
73  report.addDiagnostic(err);
74  }
75  types.push_back(std::move(type));
76  }
77 
78  void add(std::unique_ptr<AstRelation>& rel, ErrorReport& report) {
79  // add to result content (check existence first)
80  auto foundItem = std::find_if(
81  relations.begin(), relations.end(), [&](const std::unique_ptr<AstRelation>& element) {
82  return (element->getQualifiedName() == rel->getQualifiedName());
83  });
84  if (foundItem != relations.end()) {
85  Diagnostic err(Diagnostic::Type::ERROR,
86  DiagnosticMessage("Redefinition of relation " + toString(rel->getQualifiedName()),
87  rel->getSrcLoc()),
88  {DiagnosticMessage("Previous definition", (*foundItem)->getSrcLoc())});
89  report.addDiagnostic(err);
90  }
91  relations.push_back(std::move(rel));
92  }
93 
94  void add(std::unique_ptr<AstClause>& clause, ErrorReport& /* report */) {
95  clauses.push_back(std::move(clause));
96  }
97 
98  void add(std::unique_ptr<AstIO>& newIO, ErrorReport& report) {
99  // Check if i/o directive already exists
100  auto foundItem = std::find_if(ios.begin(), ios.end(), [&](const std::unique_ptr<AstIO>& io) {
101  return io->getQualifiedName() == newIO->getQualifiedName();
102  });
103  // if yes, add error
104  if (foundItem != ios.end()) {
105  auto type = (*foundItem)->getType();
106  if (type == newIO->getType() && newIO->getType() != AstIoType::output) {
107  Diagnostic err(Diagnostic::Type::ERROR,
108  DiagnosticMessage("Redefinition I/O operation " + toString(newIO->getQualifiedName()),
109  newIO->getSrcLoc()),
110  {DiagnosticMessage("Previous definition", (*foundItem)->getSrcLoc())});
111  report.addDiagnostic(err);
112  }
113  }
114  // if not, add it
115  ios.push_back(std::move(newIO));
116  }
117 };

128 /**
129  * Collects clones of all the content in the given component and its base components.
130  */
131 void collectContent(AstProgram& program, const AstComponent& component, const TypeBinding& binding,
132  const AstComponent* enclosingComponent, const ComponentLookup& componentLookup, ComponentContent& res,
133  std::vector<std::unique_ptr<AstClause>>& orphans, const std::set<std::string>& overridden,
134  ErrorReport& report, unsigned int maxInstantiationDepth) {
135  // start with relations and clauses of the base components
136  for (const auto& base : component.getBaseComponents()) {
137  const AstComponent* comp = componentLookup.getComponent(enclosingComponent, base->getName(), binding);
138  if (comp != nullptr) {
139  // link formal with actual type parameters
140  const auto& formalParams = comp->getComponentType()->getTypeParameters();
141  const auto& actualParams = base->getTypeParameters();
142 
143  // update type binding
144  TypeBinding activeBinding = binding.extend(formalParams, actualParams);
145 
146  for (const auto& cur : comp->getInstantiations()) {
147  // instantiate sub-component
148  ComponentContent content = getInstantiatedContent(program, *cur, enclosingComponent,
149  componentLookup, orphans, report, activeBinding, maxInstantiationDepth - 1);
150 
151  // process types
152  for (auto& type : content.types) {
153  res.add(type, report);
154  }
155 
156  // process relations
157  for (auto& rel : content.relations) {
158  res.add(rel, report);
159  }
160 
161  // process clauses
162  for (auto& clause : content.clauses) {
163  res.add(clause, report);
164  }
165 
166  // process io directives
167  for (auto& io : content.ios) {
168  res.add(io, report);
169  }
170  }
171 
172  // collect definitions from base type
173  std::set<std::string> superOverridden;
174  superOverridden.insert(overridden.begin(), overridden.end());
175  superOverridden.insert(component.getOverridden().begin(), component.getOverridden().end());
176  collectContent(program, *comp, activeBinding, comp, componentLookup, res, orphans,
177  superOverridden, report, maxInstantiationDepth);
178  }
179  }
180 
181  // and continue with the local types
182  for (const auto& cur : component.getTypes()) {
183  // create a clone
184  std::unique_ptr<AstType> type(cur->clone());
185 
186  // instantiate elements of union types
187  visitDepthFirst(*type, [&](const AstUnionType& type) {
188  for (const auto& name : type.getTypes()) {
189  AstQualifiedName newName = binding.find(name);
190  if (!newName.empty()) {
191  const_cast<AstQualifiedName&>(name) = newName;
192  }
193  }
194  });
195 
196  // instantiate elements of record types
197  visitDepthFirst(*type, [&](const AstRecordType& type) {
198  for (const auto& field : type.getFields()) {
199  auto&& newName = binding.find(field->getTypeName());
200  if (!newName.empty()) {
201  const_cast<AstAttribute&>(*field).setTypeName(newName);
202  }
203  }
204  });
205 
206  // add to result list (check existence first)
207  res.add(type, report);
208  }
209 
210  // and the local relations
211  for (const auto& cur : component.getRelations()) {
212  // create a clone
213  std::unique_ptr<AstRelation> rel(cur->clone());
214 
215  // update attribute types
216  for (AstAttribute* attr : rel->getAttributes()) {
217  AstQualifiedName forward = binding.find(attr->getTypeName());
218  if (!forward.empty()) {
219  attr->setTypeName(forward);
220  }
221  }
222 
223  // add to result list (check existence first)
224  res.add(rel, report);
225  }
226 
227  // and the local io directives
228  for (const auto& io : component.getIOs()) {
229  // create a clone
230  std::unique_ptr<AstIO> instantiatedIO(io->clone());
231 
232  res.add(instantiatedIO, report);
233  }
234 
235  // index the available relations
236  std::map<AstQualifiedName, AstRelation*> index;
237  for (const auto& cur : res.relations) {
238  index[cur->getQualifiedName()] = cur.get();
239  }
240 
241  // add the local clauses
242  // TODO: check orphans
243  for (const auto& cur : component.getClauses()) {
244  if (overridden.count(cur->getHead()->getQualifiedName().getQualifiers()[0]) == 0) {
245  AstRelation* rel = index[cur->getHead()->getQualifiedName()];
246  if (rel != nullptr) {
247  std::unique_ptr<AstClause> instantiatedClause(cur->clone());
248  res.add(instantiatedClause, report);
249  } else {
250  orphans.emplace_back(cur->clone());
251  }
252  }
253  }
254 
255  // add orphan clauses at the current level if they can be resolved
256  for (auto iter = orphans.begin(); iter != orphans.end();) {
257  auto& cur = *iter;
258  AstRelation* rel = index[cur->getHead()->getQualifiedName()];
259  if (rel != nullptr) {
260  // add orphan to current instance and delete from orphan list
261  std::unique_ptr<AstClause> instantiatedClause(cur->clone());
262  res.add(instantiatedClause, report);
263  iter = orphans.erase(iter);
264  } else {
265  ++iter;
266  }
267  }
268 }

270 ComponentContent getInstantiatedContent(AstProgram& program, const AstComponentInit& componentInit,
271  const AstComponent* enclosingComponent, const ComponentLookup& componentLookup,
272  std::vector<std::unique_ptr<AstClause>>& orphans, ErrorReport& report, const TypeBinding& binding,
273  unsigned int maxDepth) {
274  // start with an empty list
275  ComponentContent res;
276 
277  if (maxDepth == 0) {
278  report.addError("Component instantiation limit reached", componentInit.getSrcLoc());
279  return res;
280  }
281 
282  // get referenced component
283  const AstComponent* component = componentLookup.getComponent(
284  enclosingComponent, componentInit.getComponentType()->getName(), binding);
285  if (component == nullptr) {
286  // this component is not defined => will trigger a semantic error
287  return res;
288  }
289 
290  // update type biding
291  const auto& formalParams = component->getComponentType()->getTypeParameters();
292  const auto& actualParams = componentInit.getComponentType()->getTypeParameters();
293  TypeBinding activeBinding = binding.extend(formalParams, actualParams);
294 
295  // instantiated nested components
296  for (const auto& cur : component->getInstantiations()) {
297  // get nested content
298  ComponentContent nestedContent = getInstantiatedContent(
299  program, *cur, component, componentLookup, orphans, report, activeBinding, maxDepth - 1);
300 
301  // add types
302  for (auto& type : nestedContent.types) {
303  res.add(type, report);
304  }
305 
306  // add relations
307  for (auto& rel : nestedContent.relations) {
308  res.add(rel, report);
309  }
310 
311  // add clauses
312  for (auto& clause : nestedContent.clauses) {
313  res.add(clause, report);
314  }
315 
316  // add IO directives
317  for (auto& io : nestedContent.ios) {
318  res.add(io, report);
319  }
320  }
321 
322  // collect all content in this component
323  std::set<std::string> overridden;
324  collectContent(program, *component, activeBinding, enclosingComponent, componentLookup, res, orphans,
325  overridden, report, maxDepth);
326 
327  // update type names
328  std::map<AstQualifiedName, AstQualifiedName> typeNameMapping;
329  for (const auto& cur : res.types) {
330  auto newName = componentInit.getInstanceName() + cur->getQualifiedName();
331  typeNameMapping[cur->getQualifiedName()] = newName;
332  cur->setQualifiedName(newName);
333  }
334 
335  // update relation names
336  std::map<AstQualifiedName, AstQualifiedName> relationNameMapping;
337  for (const auto& cur : res.relations) {
338  auto newName = componentInit.getInstanceName() + cur->getQualifiedName();
339  relationNameMapping[cur->getQualifiedName()] = newName;
340  cur->setQualifiedName(newName);
341  }
342 
343  // create a helper function fixing type and relation references
344  auto fixNames = [&](const AstNode& node) {
345  // rename attribute types in headers
346  visitDepthFirst(node, [&](const AstAttribute& attr) {
347  auto pos = typeNameMapping.find(attr.getTypeName());
348  if (pos != typeNameMapping.end()) {
349  const_cast<AstAttribute&>(attr).setTypeName(pos->second);
350  }
351  });
352 
353  // rename atoms in clauses
354  visitDepthFirst(node, [&](const AstAtom& atom) {
355  auto pos = relationNameMapping.find(atom.getQualifiedName());
356  if (pos != relationNameMapping.end()) {
357  const_cast<AstAtom&>(atom).setQualifiedName(pos->second);
358  }
359  });
360 
361  // rename IO directives
362  visitDepthFirst(node, [&](const AstIO& io) {
363  auto pos = relationNameMapping.find(io.getQualifiedName());
364  if (pos != relationNameMapping.end()) {
365  const_cast<AstIO&>(io).setQualifiedName(pos->second);
366  }
367  });
368 
369  // rename field types in records
370  visitDepthFirst(node, [&](const AstRecordType& recordType) {
371  auto&& fields = recordType.getFields();
372  for (size_t i = 0; i < fields.size(); i++) {
373  auto& field = fields[i];
374  auto pos = typeNameMapping.find(field->getTypeName());
375  if (pos != typeNameMapping.end()) {
376  const_cast<AstRecordType&>(recordType).setFieldType(i, pos->second);
377  }
378  }
379  });
380 
381  // rename variant types in unions
382  visitDepthFirst(node, [&](const AstUnionType& unionType) {
383  auto& variants = unionType.getTypes();
384  for (size_t i = 0; i < variants.size(); i++) {
385  auto pos = typeNameMapping.find(variants[i]);
386  if (pos != typeNameMapping.end()) {
387  const_cast<AstUnionType&>(unionType).setVariantType(i, pos->second);
388  }
389  }
390  });
391 
392  // rename type information in typecast
393  visitDepthFirst(node, [&](const AstTypeCast& cast) {
394  auto pos = typeNameMapping.find(cast.getType());
395  if (pos != typeNameMapping.end()) {
396  const_cast<AstTypeCast&>(cast).setType(pos->second);
397  }
398  });
399  };
400 
401  // rename attributes in relation decls
402  for (const auto& cur : res.relations) {
403  fixNames(*cur);
404  }
405 
406  // rename added clauses
407  for (const auto& cur : res.clauses) {
408  fixNames(*cur);
409  }
410 
411  // rename orphans
412  for (const auto& cur : orphans) {
413  fixNames(*cur);
414  }
415 
416  // rename io directives
417  for (const auto& cur : res.ios) {
418  fixNames(*cur);
419  }
420 
421  // rename subtypes
422  for (const auto& cur : res.types) {
423  fixNames(*cur);
424  }
425 
426  // done
427  return res;
428 }

431 bool ComponentInstantiationTransformer::transform(AstTranslationUnit& translationUnit) {
432  // TODO: Do this without being a friend class of AstProgram
433 
434  AstProgram& program = *translationUnit.getProgram();
435 
436  auto* componentLookup = translationUnit.getAnalysis<ComponentLookup>();
437 
438  for (const auto& cur : program.instantiations) {
439  std::vector<std::unique_ptr<AstClause>> orphans;
440 
441  ComponentContent content = getInstantiatedContent(
442  program, *cur, nullptr, *componentLookup, orphans, translationUnit.getErrorReport());
443  for (auto& type : content.types) {
444  program.types.push_back(std::move(type));
445  }
446  for (auto& rel : content.relations) {
447  program.relations.push_back(std::move(rel));
448  }
449  for (auto& clause : content.clauses) {
450  program.clauses.push_back(std::move(clause));
451  }
452  for (auto& orphan : orphans) {
453  program.clauses.push_back(std::move(orphan));
454  }
455  for (auto& io : content.ios) {
456  program.ios.push_back(std::move(io));
457  }
458  }
459 
460  // delete components and instantiations
461  program.instantiations.clear();
462  program.components.clear();
463 
464  return true;
465 }
