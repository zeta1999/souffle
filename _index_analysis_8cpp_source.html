64 bool MaxMatching::bfSearch() {
65  Node u;
66  std::queue<Node> bfQueue;
67  // Build layers
68  for (auto& it : graph) {
69  if (getMatch(it.first) == NullVertex) {
70  distance[it.first] = 0;
71  bfQueue.push(it.first);
72  } else {
73  distance[it.first] = InfiniteDistance;
74  }
75  }
76 
77  distance[NullVertex] = InfiniteDistance;
78  while (!bfQueue.empty()) {
79  u = bfQueue.front();
80  bfQueue.pop();
81  assert(u != NullVertex);
82  const Edges& children = graph[u];
83  for (auto it : children) {
84  Node mv = getMatch(it);
85  if (getDistance(mv) == InfiniteDistance) {
86  distance[mv] = getDistance(u) + 1;
87  if (mv != NullVertex) {
88  bfQueue.push(mv);
89  }
90  }
91  }
92  }
93  return (getDistance(NullVertex) != InfiniteDistance);
94 }

101  if (dfSearch(getMatch(v))) {
102  match[u] = v;
103  match[v] = u;

107  }

110  return false;
111  }
112  return true;

126 void MinIndexSelection::solve() {
127  // map the keys in the key set to lexicographical order
128  if (searches.empty()) {
129  return;
130  }
131 
132  // map the signatures of each searches to a unique index for the matching problem
133  AttributeIndex currentIndex = 1;
134  for (SearchSignature s : searches) {
135  // we skip if the search is empty
136  if (s.empty()) {
137  continue;
138  }
139 
140  // map the signature to its unique index in each set
141  signatureToIndexA.insert({s, currentIndex});
142  signatureToIndexB.insert({s, currentIndex + 1});
143  // map each index back to the search signature
144  indexToSignature.insert({currentIndex, s});
145  indexToSignature.insert({currentIndex + 1, s});
146  currentIndex += 2;
147  }
148 
149  // Construct the matching poblem
150  for (auto search : searches) {
151  // For this node check if other nodes are strict subsets
152  for (auto itt : searches) {
153  if (search.isStrictSubset(itt)) {
154  matching.addEdge(signatureToIndexA[search], signatureToIndexB[itt]);
155  }
156  }
157  }
158 
159  // Perform the Hopcroft-Karp on the graph and receive matchings (mapped A->B and B->A)
160  // Assume: alg.calculate is not called on an empty graph
161  assert(!searches.empty());
162  const MaxMatching::Matchings& matchings = matching.solve();
163 
164  // Extract the chains given the nodes and matchings
165  const ChainOrderMap chains = getChainsFromMatching(matchings, searches);
166 
167  // Should never get no chains back as we never call calculate on an empty graph
168  assert(!chains.empty());
169  for (const auto& chain : chains) {
170  std::vector<uint32_t> ids;
171  SearchSignature initDelta = *(chain.begin());
172  insertIndex(ids, initDelta);
173 
174  for (auto iit = chain.begin(); next(iit) != chain.end(); ++iit) {
175  SearchSignature delta = SearchSignature::getDelta(*next(iit), *iit);
176  insertIndex(ids, delta);
177  }
178 
179  assert(!ids.empty());
180 
181  orders.push_back(ids);
182  }
183 
184  // Construct the matching poblem
185  for (auto search : searches) {
186  int idx = map(search);
187  size_t l = card(search);
188  SearchSignature k(search.arity());
189  for (size_t i = 0; i < l; i++) {
190  k.set(orders[idx][i], AttributeConstraint::Equal);
191  }
192  assert(k == search && "incorrect lexicographical order");
193  }
194 }

196 MinIndexSelection::Chain MinIndexSelection::getChain(
197  const SearchSignature umn, const MaxMatching::Matchings& match) {
198  SearchSignature start = umn; // start at an unmatched node
199  Chain chain;
200  // given an unmapped node from set A we follow it from set B until it cannot be matched from B
201  // if not mateched from B then umn is a chain
202  //
203  // Assume : no circular mappings, i.e. a in A -> b in B -> ........ -> a in A is not allowed.
204  // Given this, the loop will terminate
205  while (true) {
206  auto mit = match.find(signatureToIndexB[start]); // we start from B side
207  // on each iteration we swap sides when collecting the chain so we use the corresponding index map
208  chain.insert(start);
209 
210  if (mit == match.end()) {
211  return chain;
212  }
213 
214  SearchSignature a = indexToSignature.at(mit->second);
215  chain.insert(a);
216  start = a;
217  }
218 }

220 const MinIndexSelection::ChainOrderMap MinIndexSelection::getChainsFromMatching(
221  const MaxMatching::Matchings& match, const SearchSet& nodes) {
222  assert(!nodes.empty());
223 
224  // Get all unmatched nodes from A
225  const SearchSet& umKeys = getUnmatchedKeys(match, nodes);
226 
227  // Case: if no unmatched nodes then we have an anti-chain
228  if (umKeys.empty()) {
229  for (auto node : nodes) {
230  std::set<SearchSignature> a;
231  a.insert(node);
232  chainToOrder.push_back(a);
233  return chainToOrder;
234  }
235  }
236 
237  assert(!umKeys.empty());
238 
239  // A worklist of used nodes
240  SearchSet usedKeys;
241 
242  // Case: nodes < umKeys or if nodes == umKeys then anti chain - this is handled by this loop
243  for (auto umKey : umKeys) {
244  Chain c = getChain(umKey, match);
245  assert(!c.empty());
246  chainToOrder.push_back(c);
247  }
248 
249  assert(!chainToOrder.empty());
250 
251  return chainToOrder;
252 }

254 void RamIndexAnalysis::run(const RamTranslationUnit& translationUnit) {
255  // After complete:
256  // 1. All relations should have at least one index (for full-order search).
257  // 2. Two relations involved in a swap operation will have same set of indices.
258  // 3. A 0-arity relation will have only one index where LexOrder is defined as empty. A comparator using
259  // an empty order should regard all elements as equal and therefore only allow one arbitrary tuple to be
260  // inserted.
261  //
262  // TODO:
263  // 0-arity relation in a provenance program still need to be revisited.
264 
265  // visit all nodes to collect searches of each relation
266  visitDepthFirst(translationUnit.getProgram(), [&](const RamNode& node) {
267  if (const auto* indexSearch = dynamic_cast<const RamIndexOperation*>(&node)) {
268  MinIndexSelection& indexes = getIndexes(indexSearch->getRelation());
269  indexes.addSearch(getSearchSignature(indexSearch));
270  } else if (const auto* exists = dynamic_cast<const RamExistenceCheck*>(&node)) {
271  MinIndexSelection& indexes = getIndexes(exists->getRelation());
272  indexes.addSearch(getSearchSignature(exists));
273  } else if (const auto* provExists = dynamic_cast<const RamProvenanceExistenceCheck*>(&node)) {
274  MinIndexSelection& indexes = getIndexes(provExists->getRelation());
275  indexes.addSearch(getSearchSignature(provExists));
276  } else if (const auto* ramRel = dynamic_cast<const RamRelation*>(&node)) {
277  MinIndexSelection& indexes = getIndexes(*ramRel);
278  indexes.addSearch(getSearchSignature(ramRel));
279  }
280  });
281 
282  // A swap happen between rel A and rel B indicates A should include all indices of B, vice versa.
283  visitDepthFirst(translationUnit.getProgram(), [&](const RamSwap& swap) {
284  // Note: this naive approach will not work if there exists chain or cyclic swapping.
285  // e.g. swap(relA, relB) swap(relB, relC) swap(relC, relA)
286  // One need to keep merging the search set until a fixed point where no more index is introduced
287  // in any of the relation in a complete iteration.
288  //
289  // Currently RAM does not have such situation.
290  const RamRelation& relA = swap.getFirstRelation();
291  const RamRelation& relB = swap.getSecondRelation();
292 
293  MinIndexSelection& indexesA = getIndexes(relA);
294  MinIndexSelection& indexesB = getIndexes(relB);
295  // Add all searchSignature of A into B
296  for (const auto& signature : indexesA.getSearches()) {
297  indexesB.addSearch(signature);
298  }
299 
300  // Add all searchSignature of B into A
301  for (const auto& signature : indexesB.getSearches()) {
302  indexesA.addSearch(signature);
303  }
304  });
305 
306  // find optimal indexes for relations
307  for (auto& cur : minIndexCover) {
308  MinIndexSelection& indexes = cur.second;
309  indexes.solve();
310  }
311 
312  // Only case where indexSet is still empty is when relation has arity == 0
313  for (auto& cur : minIndexCover) {
314  MinIndexSelection& indexes = cur.second;
315  if (indexes.getAllOrders().empty()) {
316  indexes.insertDefaultTotalIndex(0);
317  }
318  }
319 }

388 SearchSignature RamIndexAnalysis::getSearchSignature(const RamIndexOperation* search) const {
389  size_t arity = search->getRelation().getArity();
390 
391  auto lower = search->getRangePattern().first;
392  auto upper = search->getRangePattern().second;
393 
394  SearchSignature keys(arity);
395  for (size_t i = 0; i < arity; ++i) {
396  // if both bounds are undefined
397  if (isRamUndefValue(lower[i]) && isRamUndefValue(upper[i])) {
398  keys.set(i, AttributeConstraint::None);
399  // if bounds are equal we have an equality
400  } else if (*lower[i] == *upper[i]) {
401  keys.set(i, AttributeConstraint::Equal);
402  } else {
403  keys.set(i, AttributeConstraint::Inequal);
404  }
405  }
406  return keys;
407 }

409 SearchSignature RamIndexAnalysis::getSearchSignature(
410  const RamProvenanceExistenceCheck* provExistCheck) const {
411  const auto values = provExistCheck->getValues();
412  auto auxiliaryArity = provExistCheck->getRelation().getAuxiliaryArity();
413 
414  SearchSignature keys(values.size());
415 
416  // all payload attributes should be equalities
417  for (size_t i = 0; i < values.size() - auxiliaryArity; i++) {
418  if (!isRamUndefValue(values[i])) {
419  keys.set(i, AttributeConstraint::Equal);
420  }
421  }
422 
423  // all auxiliary attributes should be free
424  for (size_t i = values.size() - auxiliaryArity; i < values.size(); i++) {
425  keys.set(i, AttributeConstraint::None);
426  }
427 
428  return keys;
429 }
