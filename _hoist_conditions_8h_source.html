26 /**
27  * @class HoistConditionsTransformer
28  * @brief Hosts conditions in a loop-nest to the most-outer/semantically-correct loop
29  *
30  * Hoists the conditions to the earliest point in the loop nest where their
31  * evaluation is still semantically correct.
32  *
33  * The transformations assumes that filter operations are stored verbose,
34  * i.e. a conjunction is expressed by two consecutive filter operations.
35  * For example ..
36  *
37  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
38  * QUERY
39  * ...
40  * IF C1 /\ C2 then
41  * ...
42  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
43  *
44  * should be rewritten / or produced by the translator as
45  *
46  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
47  * QUERY
48  * ...
49  * IF C1
50  * IF C2
51  * ...
52  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
53  *
54  * otherwise the levelling becomes imprecise, i.e., for both conditions
55  * the most outer-level is sought rather than considered separately.
56  *
57  * If there are transformers prior to hoistConditions() that introduce
58  * conjunction, another transformer is required that splits the
59  * filter operations. However, at the moment this is not necessary
60  * because the translator delivers already the right RAM format.
61  *
62  * TODO: break-up conditions while transforming so that this requirement
63  * is removed.
64  */
65 class HoistConditionsTransformer : public RamTransformer {
66 public:
67  std::string getName() const override {
68  return "HoistConditionsTransformer";
69  }
70 
71  /**
72  * @brief Hoist filter operations.
73  * @param program that is transformed
74  * @return Flag showing whether the program has been changed by the transformation
75  *
76  * There are two types of conditions in
77  * filter operations. The first type depends on tuples of
78  * RamTupleOperation operations. The second type are independent of
79  * tuple access. Both types of conditions will be hoisted to
80  * the most out-scope such that the program is still valid.
81  */
82  bool hoistConditions(RamProgram& program);
83 
84 protected:
85  RamLevelAnalysis* rla{nullptr};
86 
87  bool transform(RamTranslationUnit& translationUnit) override {
88  rla = translationUnit.getAnalysis<RamLevelAnalysis>();
89  return hoistConditions(translationUnit.getProgram());
90  }
91 };
