30 bool HoistConditionsTransformer::hoistConditions(RamProgram& program) {
31  bool changed = false;
32 
33  // helper for collecting conditions from filter operations
34  auto addCondition = [](std::unique_ptr<RamCondition> condition,
35  std::unique_ptr<RamCondition> c) -> std::unique_ptr<RamCondition> {
36  if (condition == nullptr) {
37  return c;
38  } else {
39  return std::make_unique<RamConjunction>(std::move(condition), std::move(c));
40  }
41  };
42 
43  // hoist conditions to the most outer scope if they
44  // don't depend on RamTupleOperations
45  visitDepthFirst(program, [&](const RamQuery& query) {
46  std::unique_ptr<RamCondition> newCondition;
47  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
48  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
49  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
50  const RamCondition& condition = filter->getCondition();
51  // if filter condition is independent of any RamTupleOperation,
52  // delete the filter operation and collect condition
53  if (rla->getLevel(&condition) == -1) {
54  changed = true;
55  newCondition = addCondition(std::move(newCondition), souffle::clone(&condition));
56  node->apply(makeLambdaRamMapper(filterRewriter));
57  return souffle::clone(&filter->getOperation());
58  }
59  }
60  node->apply(makeLambdaRamMapper(filterRewriter));
61  return node;
62  };
63  auto* mQuery = const_cast<RamQuery*>(&query);
64  mQuery->apply(makeLambdaRamMapper(filterRewriter));
65  if (newCondition != nullptr) {
66  // insert new filter operation at outer-most level of the query
67  changed = true;
68  auto* nestedOp = const_cast<RamOperation*>(&mQuery->getOperation());
69  mQuery->rewrite(
70  nestedOp, std::make_unique<RamFilter>(std::move(newCondition), souffle::clone(nestedOp)));
71  }
72  });
73 
74  // hoist conditions for each RamTupleOperation operation
75  visitDepthFirst(program, [&](const RamTupleOperation& search) {
76  std::unique_ptr<RamCondition> newCondition;
77  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
78  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
79  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
80  const RamCondition& condition = filter->getCondition();
81  // if filter condition matches level of RamTupleOperation,
82  // delete the filter operation and collect condition
83  if (rla->getLevel(&condition) == search.getTupleId()) {
84  changed = true;
85  newCondition = addCondition(std::move(newCondition), souffle::clone(&condition));
86  node->apply(makeLambdaRamMapper(filterRewriter));
87  return souffle::clone(&filter->getOperation());
88  }
89  }
90  node->apply(makeLambdaRamMapper(filterRewriter));
91  return node;
92  };
93  auto* tupleOp = const_cast<RamTupleOperation*>(&search);
94  tupleOp->apply(makeLambdaRamMapper(filterRewriter));
95  if (newCondition != nullptr) {
96  // insert new filter operation after the search operation
97  changed = true;
98  tupleOp->rewrite(&tupleOp->getOperation(), std::make_unique<RamFilter>(std::move(newCondition),
99  souffle::clone(&tupleOp->getOperation())));
100  }
101  });
102  return changed;
103 }
