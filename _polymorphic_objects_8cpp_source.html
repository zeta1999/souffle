41 bool PolymorphicObjectsTransformer::transform(AstTranslationUnit& translationUnit) {
42  struct TypeRewriter : public AstNodeMapper {
43  mutable bool changed{false};
44  const TypeAnalysis& typeAnalysis;
45  ErrorReport& report;
46 
47  TypeRewriter(const TypeAnalysis& typeAnalysis, ErrorReport& report)
48  : typeAnalysis(typeAnalysis), report(report) {}
49 
50  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
51  // Utility lambdas to determine if all args are of the same type.
52  auto isFloat = [&](const AstArgument* argument) {
53  return isOfKind(typeAnalysis.getTypes(argument), TypeAttribute::Float);
54  };
55  auto isUnsigned = [&](const AstArgument* argument) {
56  return isOfKind(typeAnalysis.getTypes(argument), TypeAttribute::Unsigned);
57  };
58  auto isSymbol = [&](const AstArgument* argument) {
59  return isOfKind(typeAnalysis.getTypes(argument), TypeAttribute::Symbol);
60  };
61 
62  // rewrite sub-expressions first
63  node->apply(*this);
64 
65  // It's possible that at this stage we get an undeclared clause.
66  // In this case types can't be assigned to it, and the procedure getTypes can fail
67  try {
68  // Handle numeric constant
69  if (auto* numericConstant = dynamic_cast<AstNumericConstant*>(node.get())) {
70  // Check if there is no value yet.
71  if (!numericConstant->getType().has_value()) {
72  TypeSet types = typeAnalysis.getTypes(numericConstant);
73 
74  auto hasOfKind = [&](TypeAttribute kind) -> bool {
75  return any_of(types, [&](const Type& type) { return isOfKind(type, kind); });
76  };
77  if (hasOfKind(TypeAttribute::Signed)) {
78  numericConstant->setType(AstNumericConstant::Type::Int);
79  changed = true;
80  } else if (hasOfKind(TypeAttribute::Unsigned)) {
81  numericConstant->setType(AstNumericConstant::Type::Uint);
82  changed = true;
83  } else if (hasOfKind(TypeAttribute::Float)) {
84  numericConstant->setType(AstNumericConstant::Type::Float);
85  changed = true;
86  }
87  }
88  }
89 
90  // Handle functor
91  auto* functor = as<AstIntrinsicFunctor>(node);
92  if (functor && !functor->getFunctionInfo()) {
93  // any valid candidate will do. pick the first.
94  auto candidates = validOverloads(typeAnalysis, *functor);
95  if (!candidates.empty()) {
96  functor->setFunctionInfo(candidates.front().get());
97  changed = true;
98  }
99  }
100 
101  // Handle binary constraint
102  if (auto* binaryConstraint = dynamic_cast<AstBinaryConstraint*>(node.get())) {
103  if (isOverloaded(binaryConstraint->getOperator())) {
104  // Get arguments
105  auto* leftArg = binaryConstraint->getLHS();
106  auto* rightArg = binaryConstraint->getRHS();
107  auto oldOp = binaryConstraint->getOperator();
108 
109  // Both args must be of the same type
110  if (isFloat(leftArg) && isFloat(rightArg)) {
111  binaryConstraint->setOperator(convertOverloadedConstraint(
112  binaryConstraint->getOperator(), TypeAttribute::Float));
113  } else if (isUnsigned(leftArg) && isUnsigned(rightArg)) {
114  binaryConstraint->setOperator(convertOverloadedConstraint(
115  binaryConstraint->getOperator(), TypeAttribute::Unsigned));
116  } else if (isSymbol(leftArg) && isSymbol(rightArg)) {
117  binaryConstraint->setOperator(convertOverloadedConstraint(
118  binaryConstraint->getOperator(), TypeAttribute::Symbol));
119  }
120 
121  changed |= binaryConstraint->getOperator() != oldOp;
122  }
123  }
124 
125  if (auto* aggregator = dynamic_cast<AstAggregator*>(node.get())) {
126  if (isOverloadedAggregator(aggregator->getOperator())) {
127  auto* targetExpression = aggregator->getTargetExpression();
128  auto oldOp = aggregator->getOperator();
129 
130  if (isFloat(targetExpression)) {
131  aggregator->setOperator(convertOverloadedAggregator(
132  aggregator->getOperator(), TypeAttribute::Float));
133  } else if (isUnsigned(targetExpression)) {
134  aggregator->setOperator(convertOverloadedAggregator(
135  aggregator->getOperator(), TypeAttribute::Unsigned));
136  }
137 
138  changed |= aggregator->getOperator() != oldOp;
139  }
140  }
141  } catch (std::out_of_range&) {
142  // No types to convert in undeclared clauses
143  }
144 
145  return node;
146  }
147  };
148  const TypeAnalysis& typeAnalysis = *translationUnit.getAnalysis<TypeAnalysis>();
149  TypeRewriter update(typeAnalysis, translationUnit.getErrorReport());
150  translationUnit.getProgram()->apply(update);
151  return update.changed;
152 }
