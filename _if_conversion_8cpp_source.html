33 std::unique_ptr<RamOperation> IfConversionTransformer::rewriteIndexScan(const RamIndexScan* indexScan) {
34  // check whether tuple is used in subsequent operations
35  bool tupleNotUsed = true;
36  visitDepthFirst(*indexScan, [&](const RamTupleElement& element) {
37  if (element.getTupleId() == indexScan->getTupleId()) {
38  tupleNotUsed = false;
39  }
40  });
41 
42  // if not used, transform the IndexScan operation to an existence check
43  if (tupleNotUsed) {
44  // replace IndexScan with an Filter/Existence check
45  std::vector<std::unique_ptr<RamExpression>> newValues;
46 
47  size_t arity = indexScan->getRelation().getArity();
48  for (size_t i = 0; i < arity; ++i) {
49  if (*(indexScan->getRangePattern().first[i]) != *(indexScan->getRangePattern().second[i])) {
50  fatal("Inequal upper and lower bounds not supported while rewriting index scan in "
51  "IfConversionTransformer");
52  }
53  }
54 
55  for (auto& cur : indexScan->getRangePattern().second) {
56  RamExpression* val = nullptr;
57  if (cur != nullptr) {
58  val = cur->clone();
59  }
60  newValues.emplace_back(val);
61  }
62 
63  // check if there is a break statement nested in the Scan - if so, remove it
64  RamOperation* newOp;
65  if (const auto* breakOp = dynamic_cast<const RamBreak*>(&indexScan->getOperation())) {
66  newOp = breakOp->getOperation().clone();
67  } else {
68  newOp = indexScan->getOperation().clone();
69  }
70 
71  return std::make_unique<RamFilter>(
72  std::make_unique<RamExistenceCheck>(
73  std::make_unique<RamRelationReference>(&indexScan->getRelation()),
74  std::move(newValues)),
75  std::unique_ptr<RamOperation>(newOp), indexScan->getProfileText());
76  }
77  return nullptr;
78 }

80 bool IfConversionTransformer::convertIndexScans(RamProgram& program) {
81  bool changed = false;
82  visitDepthFirst(program, [&](const RamQuery& query) {
83  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> scanRewriter =
84  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
85  if (const RamIndexScan* scan = dynamic_cast<RamIndexScan*>(node.get())) {
86  if (std::unique_ptr<RamOperation> op = rewriteIndexScan(scan)) {
87  changed = true;
88  node = std::move(op);
89  }
90  }
91  node->apply(makeLambdaRamMapper(scanRewriter));
92  return node;
93  };
94  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(scanRewriter));
95  });
96  return changed;
97 }
