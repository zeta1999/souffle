45 /**
46  * A auxiliary class to be returned by the join function aggregating the information
47  * required to print a list of elements as well as the implementation of the printing
48  * itself.
49  */
50 template <typename Iter, typename Printer>
51 class joined_sequence {
52  /** The begin of the range to be printed */
53  Iter begin;
54 
55  /** The end of the range to be printed */
56  Iter end;
57 
58  /** The seperator to be utilized between elements */
59  std::string sep;
60 
61  /** A functor printing an element */
62  Printer p;
63 
64 public:
65  /** A constructor setting up all fields of this class */
66  joined_sequence(const Iter& a, const Iter& b, std::string sep, Printer p)
67  : begin(a), end(b), sep(std::move(sep)), p(std::move(p)) {}
68 
69  /** The actual print method */
70  friend std::ostream& operator<<(std::ostream& out, const joined_sequence& s) {
71  auto cur = s.begin;
72  if (cur == s.end) {
73  return out;
74  }
75 
76  s.p(out, *cur);
77  ++cur;
78  for (; cur != s.end; ++cur) {
79  out << s.sep;
80  s.p(out, *cur);
81  }
82  return out;
83  }
84 };
