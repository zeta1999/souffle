75 /**
76  * An implementation of a meet operation between sets of types computing
77  * the set of pair-wise greatest common subtypes.
78  */
79 struct sub_type {
80  bool operator()(TypeSet& a, const TypeSet& b) const {
81  // compute result set
82  TypeSet greatestCommonSubtypes = getGreatestCommonSubtypes(a, b);
83 
84  // check whether a should change
85  if (greatestCommonSubtypes == a) {
86  return false;
87  }
88 
89  // update a
90  a = greatestCommonSubtypes;
91  return true;
92  }
93 };

126 /**
127  * A constraint factory ensuring that all the types associated to the variable
128  * a are subtypes of type b.
129  */
130 TypeConstraint isSubtypeOf(const TypeVar& variable, const Type& type) {
131  struct C : public Constraint<TypeVar> {
132  TypeVar variable;
133  const Type& type;
134 
135  C(TypeVar variable, const Type& type) : variable(std::move(variable)), type(type) {}
136 
137  bool update(Assignment<TypeVar>& assignments) const override {
138  TypeSet& assignment = assignments[variable];
139 
140  if (assignment.isAll()) {
141  assignment = TypeSet(type);
142  return true;
143  }
144 
145  TypeSet newAssignment;
146  for (const Type& t : assignment) {
147  newAssignment.insert(getGreatestCommonSubtypes(t, type));
148  }
149 
150  // check whether there was a change
151  if (assignment == newAssignment) {
152  return false;
153  }
154  assignment = newAssignment;
155  return true;
156  }
157 
158  void print(std::ostream& out) const override {
159  out << variable << " <: " << type.getName();
160  }
161  };
162 
163  return std::make_shared<C>(variable, type);
164 }

172 TypeConstraint hasSuperTypeInSet(const TypeVar& var, TypeSet values) {
173  struct C : public Constraint<TypeVar> {
174  TypeVar var;
175  TypeSet values;
176 
177  C(TypeVar var, TypeSet values) : var(std::move(var)), values(std::move(values)) {}
178 
179  bool update(Assignment<TypeVar>& assigment) const override {
180  // get current value of variable a
181  TypeSet& assigments = assigment[var];
182 
183  // remove all types that are not sub-types of b
184  if (assigments.isAll()) {
185  assigments = values;
186  return true;
187  }
188 
189  TypeSet newAssigments;
190  for (const Type& type : assigments) {
191  bool existsSuperTypeInValues =
192  any_of(values, [&type](const Type& value) { return isSubtypeOf(type, value); });
193  if (existsSuperTypeInValues) {
194  newAssigments.insert(type);
195  }
196  }
197  // check whether there was a change
198  if (newAssigments == assigments) {
199  return false;
200  }
201  assigments = newAssigments;
202  return true;
203  }
204 
205  void print(std::ostream& out) const override {
206  out << "∃ t ∈ " << values << ": " << var << " <: t";
207  }
208  };
209 
210  return std::make_shared<C>(var, values);
211 }

222 /**
223  * Ensure that types of left and right have the same base types.
224  */
225 TypeConstraint subtypesOfTheSameBaseType(const TypeVar& left, const TypeVar& right) {
226  struct C : public Constraint<TypeVar> {
227  TypeVar left;
228  TypeVar right;
229 
230  C(TypeVar left, TypeVar right) : left(std::move(left)), right(std::move(right)) {}
231 
232  bool update(Assignment<TypeVar>& assigment) const override {
233  // get current value of variable a
234  TypeSet& assigmentsLeft = assigment[left];
235  TypeSet& assigmentsRight = assigment[right];
236 
237  // Base types common to left and right variables.
238  TypeSet baseTypes;
239 
240  // Base types present in left/right variable.
241  TypeSet baseTypesLeft;
242  TypeSet baseTypesRight;
243 
244  // Iterate over possible types extracting base types.
245  // Left
246  if (!assigmentsLeft.isAll()) {
247  for (const Type& type : assigmentsLeft) {
248  if (isA<SubsetType>(type) || isA<ConstantType>(type)) {
249  baseTypesLeft.insert(getBaseType(&type));
250  }
251  }
252  }
253  // Right
254  if (!assigmentsRight.isAll()) {
255  for (const Type& type : assigmentsRight) {
256  if (isA<SubsetType>(type) || isA<ConstantType>(type)) {
257  baseTypesRight.insert(getBaseType(&type));
258  }
259  }
260  }
261 
262  TypeSet resultLeft;
263  TypeSet resultRight;
264 
265  // Handle all
266  if (assigmentsLeft.isAll() && assigmentsRight.isAll()) {
267  return false;
268  }
269 
270  // If left xor right is all, assign base types of the other side as possible values.
271  if (assigmentsLeft.isAll()) {
272  assigmentsLeft = baseTypesRight;
273  return true;
274  }
275  if (assigmentsRight.isAll()) {
276  assigmentsRight = baseTypesLeft;
277  return true;
278  }
279 
280  baseTypes = TypeSet::intersection(baseTypesLeft, baseTypesRight);
281 
282  // Allow types if they are subtypes of any of the common base types.
283  for (const Type& type : assigmentsLeft) {
284  bool isSubtypeOfCommonBaseType = any_of(baseTypes.begin(), baseTypes.end(),
285  [&type](const Type& baseType) { return isSubtypeOf(type, baseType); });
286  if (isSubtypeOfCommonBaseType) {
287  resultLeft.insert(type);
288  }
289  }
290 
291  for (const Type& type : assigmentsRight) {
292  bool isSubtypeOfCommonBaseType = any_of(baseTypes.begin(), baseTypes.end(),
293  [&type](const Type& baseType) { return isSubtypeOf(type, baseType); });
294  if (isSubtypeOfCommonBaseType) {
295  resultRight.insert(type);
296  }
297  }
298 
299  // check whether there was a change
300  if (resultLeft == assigmentsLeft && resultRight == assigmentsRight) {
301  return false;
302  }
303  assigmentsLeft = resultLeft;
304  assigmentsRight = resultRight;
305  return true;
306  }
307  //
308  void print(std::ostream& out) const override {
309  out << "∃ t : (" << left << " <: t)"
310  << " ∧ "
311  << "(" << right << " <: t)"
312  << " where t is a base type";
313  }
314  };
315 
316  return std::make_shared<C>(left, right);
317 }

319 /**
320  * Given a set of overloads, wait the list of candidates to reduce to one and then apply its constraints.
321  * NOTE: `subtypeResult` implies that `func <: overload-return-type`, rather than
322  * `func = overload-return-type`. This is required for old type semantics.
323  * See #1296 and tests/semantic/type_system4
324  */
325 TypeConstraint satisfiesOverload(const TypeEnvironment& typeEnv, IntrinsicFunctors overloads, TypeVar result,
326  std::vector<TypeVar> args, bool subtypeResult) {
327  struct C : public Constraint<TypeVar> {
328  // Check if there already was a non-monotonic update
329  mutable bool nonMonotonicUpdate = false;
330 
331  const TypeEnvironment& typeEnv;
332  mutable IntrinsicFunctors overloads;
333  TypeVar result;
334  std::vector<TypeVar> args;
335  bool subtypeResult;
336 
337  C(const TypeEnvironment& typeEnv, IntrinsicFunctors overloads, TypeVar result,
338  std::vector<TypeVar> args, bool subtypeResult)
339  : typeEnv(typeEnv), overloads(std::move(overloads)), result(std::move(result)),
340  args(std::move(args)), subtypeResult(subtypeResult) {}
341 
342  bool update(Assignment<TypeVar>& assigment) const override {
343  auto subtypesOf = [&](const TypeSet& src, TypeAttribute tyAttr) {
344  auto& ty = typeEnv.getConstantType(tyAttr);
345  return src.filter(TypeSet(true), [&](auto&& x) { return isSubtypeOf(x, ty); });
346  };
347 
348  auto possible = [&](TypeAttribute ty, const TypeVar& var) {
349  auto& curr = assigment[var];
350  return curr.isAll() || any_of(curr, [&](auto&& t) { return getTypeAttribute(t) == ty; });
351  };
352 
353  overloads = filterNot(std::move(overloads), [&](const IntrinsicFunctor& x) -> bool {
354  if (!x.variadic && args.size() != x.params.size()) return true; // arity mismatch?
355 
356  for (size_t i = 0; i < args.size(); ++i)
357  if (!possible(x.params[x.variadic ? 0 : i], args[i])) return true;
358 
359  return !possible(x.result, result);
360  });
361 
362  bool changed = false;
363  auto newResult = [&]() -> std::optional<TypeSet> {
364  if (0 == overloads.size()) return TypeSet();
365  if (1 < overloads.size()) return {};
366 
367  auto& overload = overloads.front().get();
368  // `ord` is freakin' magical: it has the signature `a -> Int`.
369  // As a consequence, we might be given non-primitive arguments (i.e. types for which
370  // `TypeEnv::getConstantType` is undefined).
371  // Handle this by not imposing constraints on the arguments.
372  if (overload.op != FunctorOp::ORD) {
373  for (size_t i = 0; i < args.size(); ++i) {
374  auto argTy = overload.params[overload.variadic ? 0 : i];
375  auto& currArg = assigment[args[i]];
376  auto newArg = subtypesOf(currArg, argTy);
377  changed |= currArg != newArg;
378  // 2020-05-09: CI linter says to remove `std::move`, but clang-tidy-10 is happy.
379  currArg = std::move(newArg); // NOLINT
380  }
381  }
382 
383  if (nonMonotonicUpdate || subtypeResult) {
384  return subtypesOf(assigment[result], overload.result);
385  } else {
386  nonMonotonicUpdate = true;
387  return TypeSet{typeEnv.getConstantType(overload.result)};
388  }
389  }();
390 
391  if (newResult) {
392  auto& curr = assigment[result];
393  changed |= curr != *newResult;
394  // 2020-05-09: CI linter says to remove `std::move`, but clang-tidy-10 is happy.
395  curr = std::move(*newResult); // NOLINT
396  }
397 
398  return changed;
399  }
400 
401  void print(std::ostream& out) const override {
402  // TODO (darth_tytus): is this description correct?
403  out << "∃ t : " << result << " <: t where t is a base type";
404  }
405  };
406 
407  return std::make_shared<C>(
408  typeEnv, std::move(overloads), std::move(result), std::move(args), subtypeResult);
409 }

411 /**
412  * Constraint on record type and its elements.
413  */
414 TypeConstraint isSubtypeOfComponent(
415  const TypeVar& elementVariable, const TypeVar& recordVariable, size_t index) {
416  struct C : public Constraint<TypeVar> {
417  TypeVar elementVariable;
418  TypeVar recordVariable;
419  unsigned index;
420 
421  C(TypeVar elementVariable, TypeVar recordVariable, int index)
422  : elementVariable(std::move(elementVariable)), recordVariable(std::move(recordVariable)),
423  index(index) {}
424 
425  bool update(Assignment<TypeVar>& assignment) const override {
426  // get list of types for b
427  const TypeSet& recordTypes = assignment[recordVariable];
428 
429  // if it is (not yet) constrainted => skip
430  if (recordTypes.isAll()) {
431  return false;
432  }
433 
434  // compute new types for element and record
435  TypeSet newElementTypes;
436  TypeSet newRecordTypes;
437 
438  for (const Type& type : recordTypes) {
439  // A type must be either a record type or a subset of a record type
440  if (!isOfKind(type, TypeAttribute::Record)) {
441  continue;
442  }
443 
444  const auto& typeAsRecord = *as<RecordType>(type);
445 
446  // Wrong size => skip.
447  if (typeAsRecord.getFields().size() <= index) {
448  continue;
449  }
450 
451  // Valid type for record.
452  newRecordTypes.insert(type);
453 
454  // and its corresponding field.
455  newElementTypes.insert(*typeAsRecord.getFields()[index]);
456  }
457 
458  // combine with current types assigned to element
459  newElementTypes = getGreatestCommonSubtypes(assignment[elementVariable], newElementTypes);
460 
461  // update values
462  bool changed = false;
463  if (newRecordTypes != recordTypes) {
464  assignment[recordVariable] = newRecordTypes;
465  changed = true;
466  }
467 
468  if (assignment[elementVariable] != newElementTypes) {
469  assignment[elementVariable] = newElementTypes;
470  changed = true;
471  }
472 
473  return changed;
474  }
475 
476  void print(std::ostream& out) const override {
477  out << elementVariable << " <: " << recordVariable << "::" << index;
478  }
479  };
480 
481  return std::make_shared<C>(elementVariable, recordVariable, index);
482 }

485 /* Return a new clause with type-annotated variables */
486 std::unique_ptr<AstClause> createAnnotatedClause(
487  const AstClause* clause, const std::map<const AstArgument*, TypeSet> argumentTypes) {
488  // Annotates each variable with its type based on a given type analysis result
489  struct TypeAnnotator : public AstNodeMapper {
490  const std::map<const AstArgument*, TypeSet>& types;
491 
492  TypeAnnotator(const std::map<const AstArgument*, TypeSet>& types) : types(types) {}
493 
494  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
495  if (auto* var = dynamic_cast<AstVariable*>(node.get())) {
496  std::stringstream newVarName;
497  newVarName << var->getName() << "&isin;" << types.find(var)->second;
498  return std::make_unique<AstVariable>(newVarName.str());
499  } else if (auto* var = dynamic_cast<AstUnnamedVariable*>(node.get())) {
500  std::stringstream newVarName;
501  newVarName << "_"
502  << "&isin;" << types.find(var)->second;
503  return std::make_unique<AstVariable>(newVarName.str());
504  }
505  node->apply(*this);
506  return node;
507  }
508  };
509 
510  /* Note:
511  * Because the type of each argument is stored in the form [address -> type-set],
512  * the type-analysis result does not immediately apply to the clone due to differing
513  * addresses.
514  * Two ways around this:
515  * (1) Perform the type-analysis again for the cloned clause
516  * (2) Keep track of the addresses of equivalent arguments in the cloned clause
517  * Method (2) was chosen to avoid having to recompute the analysis each time.
518  */
519  auto annotatedClause = souffle::clone(clause);
520 
521  // Maps x -> y, where x is the address of an argument in the original clause, and y
522  // is the address of the equivalent argument in the clone.
523  std::map<const AstArgument*, const AstArgument*> memoryMap;
524 
525  std::vector<const AstArgument*> originalAddresses;
526  visitDepthFirst(*clause, [&](const AstArgument& arg) { originalAddresses.push_back(&arg); });
527 
528  std::vector<const AstArgument*> cloneAddresses;
529  visitDepthFirst(*annotatedClause, [&](const AstArgument& arg) { cloneAddresses.push_back(&arg); });
530 
531  assert(cloneAddresses.size() == originalAddresses.size());
532 
533  for (size_t i = 0; i < originalAddresses.size(); i++) {
534  memoryMap[originalAddresses[i]] = cloneAddresses[i];
535  }
536 
537  // Map the types to the clause clone
538  std::map<const AstArgument*, TypeSet> cloneArgumentTypes;
539  for (auto& pair : argumentTypes) {
540  cloneArgumentTypes[memoryMap[pair.first]] = pair.second;
541  }
542 
543  // Create the type-annotated clause
544  TypeAnnotator annotator(cloneArgumentTypes);
545  annotatedClause->apply(annotator);
546  return annotatedClause;
547 }

549 /**
550  * Constraint analysis framework for types.
551  *
552  * The analysis operates on the concept of sinks and sources.
553  * If the atom is negated or is a head then it's a sink,
554  * and we can only extract the kind constraint from it
555  * Otherwise it is a source, and the type of the element must
556  * be a subtype of source attribute.
557  */
558 class TypeConstraintsAnalysis : public AstConstraintAnalysis<TypeVar> {
559 public:
560  TypeConstraintsAnalysis(const TypeEnvironment& typeEnv, const AstProgram& program)
561  : typeEnv(typeEnv), program(program) {}
562 
563 private:
564  const TypeEnvironment& typeEnv;
565  const AstProgram& program;
566 
567  // Sinks = {head} ∪ {negated atoms}
568  std::set<const AstAtom*> sinks;
569 
570  void collectConstraints(const AstClause& clause) override {
571  sinks.insert(clause.getHead());
572  visitDepthFirstPreOrder(clause, *this);
573  }
574 
575  void visitSink(const AstAtom& atom) {
576  iterateOverAtom(atom, [&](const AstArgument& argument, const Type& attributeType) {
577  if (isA<RecordType>(attributeType)) {
578  addConstraint(isSubtypeOf(getVar(argument), getBaseType(&attributeType)));
579  return;
580  }
581  for (auto& constantType : typeEnv.getConstantTypes()) {
582  if (isSubtypeOf(attributeType, constantType)) {
583  addConstraint(isSubtypeOf(getVar(argument), constantType));
584  }
585  }
586  });
587  }
588 
589  void visitAtom(const AstAtom& atom) override {
590  if (contains(sinks, &atom)) {
591  visitSink(atom);
592  return;
593  }
594 
595  iterateOverAtom(atom, [&](const AstArgument& argument, const Type& attributeType) {
596  addConstraint(isSubtypeOf(getVar(argument), attributeType));
597  });
598  }
599 
600  void visitNegation(const AstNegation& cur) override {
601  sinks.insert(cur.getAtom());
602  }
603 
604  void visitStringConstant(const AstStringConstant& cnst) override {
605  addConstraint(isSubtypeOf(getVar(cnst), typeEnv.getConstantType(TypeAttribute::Symbol)));
606  }

608  void visitNumericConstant(const AstNumericConstant& constant) override {
609  TypeSet possibleTypes;
610 
611  // Check if the type is given.
612  if (constant.getType().has_value()) {
613  switch (*constant.getType()) {
614  // Insert a type, but only after checking that parsing is possible.
615  case AstNumericConstant::Type::Int:
616  if (canBeParsedAsRamSigned(constant.getConstant())) {
617  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Signed));
618  }
619  break;
620  case AstNumericConstant::Type::Uint:
621  if (canBeParsedAsRamUnsigned(constant.getConstant())) {
622  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Unsigned));
623  }
624  break;
625  case AstNumericConstant::Type::Float:
626  if (canBeParsedAsRamFloat(constant.getConstant())) {
627  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Float));
628  }
629  break;
630  }
631  // Else: all numeric types that can be parsed are valid.
632  } else {
633  if (canBeParsedAsRamSigned(constant.getConstant())) {
634  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Signed));
635  }
636 
637  if (canBeParsedAsRamUnsigned(constant.getConstant())) {
638  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Unsigned));
639  }
640 
641  if (canBeParsedAsRamFloat(constant.getConstant())) {
642  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Float));
643  }
644  }
645 
646  addConstraint(hasSuperTypeInSet(getVar(constant), possibleTypes));
647  }

651  auto rhs = getVar(rel.getRHS());

656  void visitFunctor(const AstFunctor& fun) override {
657  auto functorVar = getVar(fun);
658 
659  auto intrFun = as<AstIntrinsicFunctor>(fun);
660  if (intrFun) {
661  auto argVars = map(intrFun->getArguments(), [&](auto&& x) { return getVar(x); });
662  // The type of the user-defined function might not be set at this stage.
663  // If so then add overloads as alternatives
664  if (!intrFun->getFunctionInfo())
665  addConstraint(satisfiesOverload(typeEnv, functorBuiltIn(intrFun->getFunction()), functorVar,
666  argVars, isInfixFunctorOp(intrFun->getFunction())));
667 
668  // In polymorphic case
669  // We only require arguments to share a base type with a return type.
670  // (instead of, for example, requiring them to be of the same type)
671  // This approach is related to old type semantics
672  // See #1296 and tests/semantic/type_system4
673  if (isInfixFunctorOp(intrFun->getFunction())) {
674  for (auto&& var : argVars)
675  addConstraint(subtypesOfTheSameBaseType(var, functorVar));
676 
677  return;
678  }
679 
680  if (!intrFun->getFunctionInfo()) return;
681  }
682 
683  // add a constraint for the return type of the functor
684  addConstraint(isSubtypeOf(functorVar, typeEnv.getConstantType(fun.getReturnType())));
685 
686  // Special case. Ord returns the ram representation of any object.
687  if (intrFun && intrFun->getFunctionInfo()->op == FunctorOp::ORD) return;
688 
689  auto arguments = fun.getArguments();
690  for (size_t i = 0; i < arguments.size(); ++i) {
691  addConstraint(isSubtypeOf(getVar(arguments[i]), typeEnv.getConstantType(fun.getArgType(i))));
692  }
693  }

695  void visitCounter(const AstCounter& counter) override {
696  addConstraint(isSubtypeOf(getVar(counter), typeEnv.getConstantType(TypeAttribute::Signed)));
697  }

699  void visitTypeCast(const AstTypeCast& typeCast) override {
700  auto& typeName = typeCast.getType();
701  if (!typeEnv.isType(typeName)) {
702  return;
703  }
704 
705  addConstraint(isSubtypeOf(getVar(typeCast), typeEnv.getType(typeName)));
706 
707  // If we are dealing with a constant then its type must be deduced from the cast
708  // Otherwise, expression like: to_string(as(2, float)) couldn't be typed.
709  auto* value = typeCast.getValue();
710 
711  if (isA<AstConstant>(value)) {
712  addConstraint(isSubtypeOf(getVar(*value), typeEnv.getType(typeName)));
713  }
714  }

716  void visitRecordInit(const AstRecordInit& record) override {
717  auto arguments = record.getArguments();
718  for (size_t i = 0; i < arguments.size(); ++i) {
719  addConstraint(isSubtypeOfComponent(getVar(arguments[i]), getVar(record), i));
720  }
721  }

723  void visitAggregator(const AstAggregator& agg) override {
724  if (agg.getOperator() == AggregateOp::COUNT) {
725  addConstraint(isSubtypeOf(getVar(agg), typeEnv.getConstantType(TypeAttribute::Signed)));
726  } else if (agg.getOperator() == AggregateOp::MEAN) {
727  addConstraint(isSubtypeOf(getVar(agg), typeEnv.getConstantType(TypeAttribute::Float)));
728  } else {
729  addConstraint(hasSuperTypeInSet(getVar(agg), typeEnv.getConstantNumericTypes()));
730  }
731 
732  // If there is a target expression - it should be of the same type as the aggregator.
733  if (auto expr = agg.getTargetExpression()) {
734  addConstraint(isSubtypeOf(getVar(expr), getVar(agg)));
735  addConstraint(isSubtypeOf(getVar(agg), getVar(expr)));
736  }
737  }

739  /**
740  * Utility function.
741  * Iterate over atoms valid pairs of (argument, type-attribute) and apply procedure `map` for its
742  * side-effects.
743  */
744  void iterateOverAtom(const AstAtom& atom, std::function<void(const AstArgument&, const Type&)> map) {
745  // get relation
746  auto rel = getAtomRelation(&atom, &program);
747  if (rel == nullptr) {
748  return; // error in input program
749  }
750 
751  auto atts = rel->getAttributes();
752  auto args = atom.getArguments();
753  if (atts.size() != args.size()) {
754  return; // error in input program
755  }
756 
757  for (size_t i = 0; i < atts.size(); i++) {
758  const auto& typeName = atts[i]->getTypeName();
759  if (typeEnv.isType(typeName)) {
760  map(*args[i], typeEnv.getType(typeName));
761  }
762  }
763  }
764 };
