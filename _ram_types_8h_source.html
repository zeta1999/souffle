44 /**
45  * Types of elements in a tuple.

47  * Default domain has size of 32 bits; may be overridden by user
48  * defining RAM_DOMAIN_SIZE.

55 #if RAM_DOMAIN_SIZE == 64
56 using RamDomain = int64_t;
57 using RamSigned = RamDomain;
58 using RamUnsigned = uint64_t;

60 using RamFloat = double;
61 #else

69 // Compile time sanity checks
70 static_assert(std::is_integral<RamSigned>::value && std::is_signed<RamSigned>::value,
71  "RamSigned must be represented by a signed type.");
72 static_assert(std::is_integral<RamUnsigned>::value && !std::is_signed<RamUnsigned>::value,
73  "RamUnsigned must be represented by an unsigned type.");
74 static_assert(std::is_floating_point<RamFloat>::value && sizeof(RamFloat) * 8 == RAM_DOMAIN_SIZE,
75  "RamFloat must be represented by a floating point and have the same size as other types.");

77 template <typename T>
78 constexpr bool isRamType = (std::is_same<T, RamDomain>::value || std::is_same<T, RamSigned>::value ||
79  std::is_same<T, RamUnsigned>::value || std::is_same<T, RamFloat>::value);

86 /** Cast a type by reinterpreting its bits. Domain is restricted to Ram Types only.
87  * Template takes two types (second type is never necessary because it can be deduced from the argument)
88  * The following always holds
89  * For type T and a : T
90  * ramBitCast<T>(ramBitCast<RamDomain>(a)) == a
91  **/
92 template <typename To = RamDomain, typename From>
93 To ramBitCast(From source) {
94  static_assert(isRamType<From> && isRamType<To>, "Bit casting should only be used on Ram Types.");
95  static_assert(sizeof(To) == sizeof(From), "Can't bit cast types with different size.");
96  To destination;
97  memcpy(&destination, &source, sizeof(destination));
98  return destination;
99 }
