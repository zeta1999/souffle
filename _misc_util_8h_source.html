40 /**
41  * On windows, the following gcc builtins are missing.
42  *
43  * In the case of popcountll, __popcnt64 is the windows equivalent.
44  *
45  * For ctz and ctzll, BitScanForward and BitScanForward64 are the respective
46  * windows equivalents. However ctz is used in a constexpr context, and we can't
47  * use BitScanForward, so we implement it ourselves.
48  */
49 #define __builtin_popcountll __popcnt64
50 
51 constexpr unsigned long __builtin_ctz(unsigned long value) {
52  unsigned long trailing_zeroes = 0;
53  while ((value = value >> 1) ^ 1) {
54  ++trailing_zeroes;
55  }
56  return trailing_zeroes;
57 }
58 
59 inline unsigned long __builtin_ctzll(unsigned long long value) {
60  unsigned long trailing_zero = 0;
61 
62  if (_BitScanForward64(&trailing_zero, value)) {
63  return trailing_zero;
64  } else {
65  return 64;
66  }
