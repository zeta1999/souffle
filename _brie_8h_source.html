10  *
11  * @file Brie.h
12  *
13  * This header file contains the implementation for a generic, fixed
14  * length integer trie.
15  *
16  * Tries trie is utilized to store n-ary tuples of integers. Each level
17  * is implemented via a sparse array (also covered by this header file),
18  * referencing the following nested level. The leaf level is realized
19  * by a sparse bit-map to minimize the memory footprint.
20  *
21  * Multiple insert operations can be be conducted concurrently on trie
22  * structures. So can read-only operations. However, inserts and read
23  * operations may not be conducted at the same time.
24  *
25  ***********************************************************************/

99 /**
100  * A sparse array simulates an array associating to every element
101  * of uint32_t an element of a generic type T. Any non-defined element
102  * will be default-initialized utilizing the detail::default_factory
103  * functor.
104  *
105  * Internally the array is organized as a balanced tree. The leaf
106  * level of the tree corresponds to the elements of the represented
107  * array. Inner nodes utilize individual bits of the indices to reference
108  * sub-trees. For efficiency reasons, only the minimal sub-tree required
109  * to cover all non-null / non-default values stored in the array is
110  * maintained. Furthermore, several levels of nodes are aggreated in a
111  * B-tree like fashion to inprove cache utilization and reduce the number
112  * of steps required for lookup and insert operations.
113  *
114  * @tparam T the type of the stored elements
115  * @tparam BITS the number of bits consumed per node-level
116  * e.g. if it is set to 3, the resulting tree will be of a degree of
117  * 2^3=8, and thus 8 child-pointers will be stored in each inner node
118  * and as many values will be stored in each leaf node.
119  * @tparam merge_op the functor to be utilized when merging the content of two
120  * instances of this type.
121  * @tparam copy_op a functor to be applied to each stored value when copying an
122  * instance of this array. For instance, this is utilized by the
123  * trie implementation to create a clone of each sub-tree instead
124  * of preserving the original pointer.
125  */
126 template <typename T, unsigned BITS = 6, typename merge_op = detail::default_merge<T>,
127  typename copy_op = detail::identity<T>>

394  /**
395  * Obtains the current version of the root.

398  // here it is assumed that the load of a 64-bit word is atomic
399  return (uint64_t)synced.root;

402  /**
403  * Obtains a snapshot of the current root information.
404  */
405  RootInfoSnapshot getRootInfo() const {
406  RootInfoSnapshot res{};
407  do {
408  // first take the mod counter
409  do {
410  // if res.mod % 2 == 1 .. there is an update in progress
411  res.version = getRootVersion();
412  } while (res.version % 2);
413 
414  // then the rest
415  res.root = synced.root;
416  res.levels = synced.levels;
417  res.offset = synced.offset;
418 
419  // check consistency of obtained data (optimistic locking)
420  } while (res.version != getRootVersion());
421 
422  // got a consistent snapshot
423  return res;
424  }
425 
426  /**
427  * Updates the current root information based on the handed in modified
428  * snapshot instance if the version number of the snapshot still corresponds
429  * to the current version. Otherwise a concurrent update took place and the
430  * operation is aborted.
431  *
432  * @param info the updated information to be assigned to the active root-info data
433  * @return true if successfully updated, false if aborted
434  */
435  bool tryUpdateRootInfo(const RootInfoSnapshot& info) {
436  // check mod counter
437  uintptr_t version = info.version;
438 
439  // update root to invalid pointer (ending with 1)
440  if (!__sync_bool_compare_and_swap(&synced.root, (Node*)version, (Node*)(version + 1))) {
441  return false;
442  }
443 
444  // conduct update
445  synced.levels = info.levels;
446  synced.offset = info.offset;
447 
448  // update root (and thus the version to enable future retrievals)
449  __sync_synchronize();
450  synced.root = info.root;
451 
452  // done
453  return true;
454  }

499  * concurrent setting utilizing a optimistic locking approach.

503  // check mod counter
504  uintptr_t version = info.version;
505 
506  // temporary update first pointer to point to uneven value (lock-out)

513

517

567 private:
568  /**
569  * An internal function capable of navigating to a given leaf node entry.
570  * If the cell does not exist yet it will be created as a side-effect.
571  *
572  * @param i the index of the requested cell
573  * @param ctxt a operation context to exploit state-less temporal locality
574  * @return a reference to the requested cell
575  */
576  inline Cell& getLeaf(index_type i, op_context& ctxt) {
577  // check context
578  if (ctxt.lastNode && (ctxt.lastIndex == (i & ~INDEX_MASK))) {
579  // return reference to referenced
580  return ctxt.lastNode->cell[i & INDEX_MASK];
581  }
582 
583  // get snapshot of root
584  auto info = getRootInfo();
585 
586  // check for emptiness
587  if (info.root == nullptr) {
588  // build new root node
589  info.root = newNode();
590 
591  // initialize the new node
592  info.root->parent = nullptr;
593  info.offset = i & ~(INDEX_MASK);
594 
595  // try updating root information atomically
596  if (tryUpdateRootInfo(info)) {
597  // success -- finish get call
598 
599  // update first
600  auto firstInfo = getFirstInfo();
601  while (info.offset < firstInfo.offset) {
602  firstInfo.node = info.root;
603  firstInfo.offset = info.offset;
604  if (!tryUpdateFirstInfo(firstInfo)) {
605  // there was some concurrent update => check again
606  firstInfo = getFirstInfo();
607  }
608  }
609 
610  // return reference to proper cell
611  return info.root->cell[i & INDEX_MASK];
612  }
613 
614  // somebody else was faster => use standard insertion procedure
615  delete info.root;
616 
617  // retrieve new root info
618  info = getRootInfo();
619 
620  // make sure there is a root
621  assert(info.root);
622  }
623 
624  // for all other inserts
625  // - check boundary
626  // - navigate to node
627  // - insert value
628 
629  // check boundaries
630  while (!inBoundaries(i, info.levels, info.offset)) {
631  // boundaries need to be expanded by growing upwards
632  raiseLevel(info); // try raising level unless someone else did already
633  // update root info
634  info = getRootInfo();
635  }
636 
637  // navigate to node
638  Node* node = info.root;
639  unsigned level = info.levels;
640  while (level != 0) {
641  // get X coordinate
642  auto x = getIndex(i, level);
643 
644  // decrease level counter
645  --level;
646 
647  // check next node
648  std::atomic<Node*>& aNext = node->cell[x].aptr;
649  Node* next = aNext;
650  if (!next) {
651  // create new sub-tree
652  Node* newNext = newNode();
653  newNext->parent = node;
654 
655  // try to update next
656  if (!aNext.compare_exchange_strong(next, newNext)) {
657  // some other thread was faster => use updated next
658  delete newNext;
659  } else {
660  // the locally created next is the new next
661  next = newNext;
662 
663  // update first
664  if (level == 0) {
665  // compute offset of this node
666  auto off = i & ~INDEX_MASK;
667 
668  // fast over-approximation of whether a update is necessary
669  if (off < unsynced.firstOffset) {
670  // update first reference if this one is the smallest
671  auto first_info = getFirstInfo();
672  while (off < first_info.offset) {
673  first_info.node = next;
674  first_info.offset = off;
675  if (!tryUpdateFirstInfo(first_info)) {
676  // there was some concurrent update => check again
677  first_info = getFirstInfo();
678  }
679  }
680  }
681  }
682  }
683 
684  // now next should be defined
685  assert(next);
686  }
687 
688  // continue one level below
689  node = next;
690  }
691 
692  // update context
693  ctxt.lastIndex = (i & ~INDEX_MASK);
694  ctxt.lastNode = node;
695 
696  // return reference to cell
697  return node->cell[i & INDEX_MASK];
698  }

739  * defined previously. A operation context can be provided for
740  * exploiting temporal locality.
741  */
742  value_type lookup(index_type i, op_context& ctxt) const {
743  // check whether it is empty
744  if (!unsynced.root) return souffle::detail::default_factory<value_type>()();
745 
746  // check boundaries
747  if (!inBoundaries(i)) return souffle::detail::default_factory<value_type>()();
748 
749  // check context
750  if (ctxt.lastNode && ctxt.lastIndex == (i & ~INDEX_MASK)) {
751  return ctxt.lastNode->cell[i & INDEX_MASK].value;
752  }
753 
754  // navigate to value
755  Node* node = unsynced.root;
756  unsigned level = unsynced.levels;
757  while (level != 0) {
758  // get X coordinate
759  auto x = getIndex(i, level);
760 
761  // decrease level counter
762  --level;

765  Node* next = node->cell[x].ptr;
766

768  if (!next) return souffle::detail::default_factory<value_type>()();

1521 #if __GNUC__ >= 7

1537 
1538  // it worked, new bit added
1539  return true;
1540  }
1541 
1542 #endif
1543 #endif
1544 
1545  value_t old = val.fetch_or(bit, std::memory_order::memory_order_relaxed);
