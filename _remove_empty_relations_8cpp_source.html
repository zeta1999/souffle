35 bool RemoveEmptyRelationsTransformer::removeEmptyRelations(AstTranslationUnit& translationUnit) {
36  AstProgram& program = *translationUnit.getProgram();
37  auto* ioTypes = translationUnit.getAnalysis<IOType>();
38  bool changed = false;
39  for (auto rel : program.getRelations()) {
40  if (!getClauses(program, *rel).empty() || ioTypes->isInput(rel)) {
41  continue;
42  }
43  changed |= removeEmptyRelationUses(translationUnit, rel);
44 
45  bool usedInAggregate = false;
46  visitDepthFirst(program, [&](const AstAggregator& agg) {
47  for (const auto lit : agg.getBodyLiterals()) {
48  visitDepthFirst(*lit, [&](const AstAtom& atom) {
49  if (getAtomRelation(&atom, &program) == rel) {
50  usedInAggregate = true;
51  }
52  });
53  }
54  });
55 
56  if (!usedInAggregate && !ioTypes->isOutput(rel)) {
57  program.removeRelation(rel->getQualifiedName());
58  changed = true;
59  }
60  }
61  return changed;
62 }

64 bool RemoveEmptyRelationsTransformer::removeEmptyRelationUses(
65  AstTranslationUnit& translationUnit, AstRelation* emptyRelation) {
66  AstProgram& program = *translationUnit.getProgram();
67  bool changed = false;
68 
69  //
70  // (1) drop rules from the program that have empty relations in their bodies.
71  // (2) drop negations of empty relations
72  //
73  // get all clauses
74  std::vector<const AstClause*> clauses;
75  visitDepthFirst(program, [&](const AstClause& cur) { clauses.push_back(&cur); });
76 
77  // clean all clauses
78  for (const AstClause* cl : clauses) {
79  // check for an atom whose relation is the empty relation
80 
81  bool removed = false;
82  for (AstLiteral* lit : cl->getBodyLiterals()) {
83  if (auto* arg = dynamic_cast<AstAtom*>(lit)) {
84  if (getAtomRelation(arg, &program) == emptyRelation) {
85  program.removeClause(cl);
86  removed = true;
87  changed = true;
88  break;
89  }
90  }
91  }
92 
93  if (!removed) {
94  // check whether a negation with empty relations exists
95 
96  bool rewrite = false;
97  for (AstLiteral* lit : cl->getBodyLiterals()) {
98  if (auto* neg = dynamic_cast<AstNegation*>(lit)) {
99  if (getAtomRelation(neg->getAtom(), &program) == emptyRelation) {
100  rewrite = true;
101  break;
102  }
103  }
104  }
105 
106  if (rewrite) {
107  // clone clause without negation for empty relations
108 
109  auto res = std::unique_ptr<AstClause>(cloneHead(cl));
110 
111  for (AstLiteral* lit : cl->getBodyLiterals()) {
112  if (auto* neg = dynamic_cast<AstNegation*>(lit)) {
113  if (getAtomRelation(neg->getAtom(), &program) != emptyRelation) {
114  res->addToBody(souffle::clone(lit));
115  }
116  } else {
117  res->addToBody(souffle::clone(lit));
118  }
119  }
120 
121  program.removeClause(cl);
122  program.addClause(std::move(res));
123  changed = true;
124  }
125  }
126  }
127 
128  return changed;
129 }
