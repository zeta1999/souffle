70 /**
71  * Returns literals of a particular type in the body of a clause.
72  *
73  * @param the clause
74  * @return vector of body literals of the specified type
75  */
76 template <typename T, typename C>
77 std::vector<T*> getBodyLiterals(const C& clause) {
78  std::vector<T*> res;
79  for (auto& lit : clause.getBodyLiterals()) {
80  if (T* t = dynamic_cast<T*>(lit)) {
81  res.push_back(t);
82  }
83  }
84  return res;
85 }
