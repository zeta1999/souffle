36 /**
37  * A record type aggregates a list of fields into a new type.
38  * Each record type has a name making it unique. Two record
39  * types are unrelated to all other types (they do not have
40  * any super or sub types).
41  */
42 class AstRecordType : public AstType {
43 public:
44  AstRecordType(AstQualifiedName name, VecOwn<AstAttribute> fields, SrcLocation loc = {})
45  : AstType(std::move(name), std::move(loc)), fields(std::move(fields)) {}
46 
47  /** add field to record type */
48  void add(std::string name, AstQualifiedName type) {
49  fields.push_back(mk<AstAttribute>(std::move(name), std::move(type)));
50  }
51 
52  /** get fields of record */
53  std::vector<AstAttribute*> getFields() const {
54  return toPtrVector(fields);
55  }
56 
57  /** set field type */
58  void setFieldType(size_t idx, AstQualifiedName type) {
59  fields.at(idx)->setTypeName(std::move(type));
60  }
61 
62  AstRecordType* clone() const override {
63  return new AstRecordType(getQualifiedName(), souffle::clone(fields), getSrcLoc());
64  }
65 
66 protected:
67  void print(std::ostream& os) const override {
68  os << ".type " << getQualifiedName() << "= [" << join(fields, ", ") << "]";
69  }
70 
71  bool equal(const AstNode& node) const override {
72  const auto& other = dynamic_cast<const AstRecordType&>(node);
73  return getQualifiedName() == other.getQualifiedName() && equal_targets(fields, other.fields);
74  }
75 
76 private:
77  /** record fields */
