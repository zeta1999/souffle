108  /**
109  * inserts all nodes from the other relation into this one
110  * @param other the binary relation from which to add elements from
111  */
112  void insertAll(const EquivalenceRelation<TupleType>& other) {
113  other.genAllDisjointSetLists();
114 
115  // iterate over partitions at a time
116  for (typename StatesMap::chunk it : other.equivalencePartition.getChunks(MAX_THREADS)) {
117  for (auto& p : it) {
118  value_type rep = p.first;
119  StatesList& pl = *p.second;
120  const size_t ksize = pl.size();
121  for (size_t i = 0; i < ksize; ++i) {
122  this->sds.unionNodes(rep, pl.get(i));
123  }
124  }
125  }
126  // invalidate iterators unconditionally
127  this->statesMapStale.store(true, std::memory_order_relaxed);
128  }

130  /**
131  * Extend this relation with another relation, expanding this equivalence relation
132  * The supplied relation is the old knowledge, whilst this relation only contains
133  * explicitly new knowledge. After this operation the "implicitly new tuples" are now
134  * explicitly inserted this relation.
135  */
136  void extend(const EquivalenceRelation<TupleType>& other) {
137  // nothing to extend if there's no new/original knowledge
138  if (other.size() == 0 || this->size() == 0) return;
139 
140  this->genAllDisjointSetLists();
141  other.genAllDisjointSetLists();
142 
143  std::set<value_type> repsCovered;
144 
145  // find all the disjoint sets that need to be added to this relation
146  // that exist in other (and exist in this)
147  {
148  auto it = this->sds.sparseToDenseMap.begin();
149  auto end = this->sds.sparseToDenseMap.end();
150  value_type el;
151  for (; it != end; ++it) {
152  std::tie(el, std::ignore) = *it;
153  if (other.containsElement(el)) {
154  value_type rep = other.sds.findNode(el);
155  if (repsCovered.count(rep) == 0) {
156  repsCovered.emplace(rep);
157  }
158  }
159  }
160  }
161 
162  // add the intersecting dj sets into this one
163  {
164  value_type el;
165  value_type rep;
166  auto it = other.sds.sparseToDenseMap.begin();
167  auto end = other.sds.sparseToDenseMap.end();
168  for (; it != end; ++it) {
169  std::tie(el, std::ignore) = *it;
170  rep = other.sds.findNode(el);
171  if (repsCovered.count(rep) != 0) {
172  this->insert(el, rep);
173  }
174  }
175  }
176  }

218  /**
219  * Size of relation
220  * @return the sum of the number of pairs per disjoint set
221  */
222  size_t size() const {
223  genAllDisjointSetLists();
224 
225  statesLock.lock_shared();

247  explicit iterator(const EquivalenceRelation* br)

252  isEndVal = true;
253  return;
254  }
255  // grab the pointer to the list, and make it our current list
256  djSetList = (*djSetMapListIt).second;
257  assert(djSetList->size() != 0);
258

263  // WITHIN: iterator for everything within the same DJset (used for EquivalenceRelation.partition())
264  explicit iterator(const EquivalenceRelation* br, const StatesBucket within)
265  : br(br), ityp(IterType::WITHIN), djSetList(within) {
266  // empty dj set
267  if (djSetList->size() == 0) {
268  isEndVal = true;
269  }
270 
271  updateAnterior();
272  updatePosterior();
273  }
274 
275  // ANTERIOR: iterator that yields all (former, _) \in djset(former) (djset(former) === within)
276  explicit iterator(const EquivalenceRelation* br, const value_type former, const StatesBucket within)

286  // ANTPOST: iterator that yields all (former, latter) \in djset(former), (djset(former) ==

343  /* pre-increment */
344  iterator& operator++() {
345  if (isEndVal) {
346  throw std::out_of_range("error: incrementing an out of range iterator");
347  }
348 
349  switch (ityp) {
350  case IterType::ALL:
351  // move posterior along one
352  // see if we can't move the posterior along
353  if (++cPosteriorIndex == djSetList->size()) {
354  // move anterior along one
355  // see if we can't move the anterior along one
356  if (++cAnteriorIndex == djSetList->size()) {
357  // move the djset it along one
358  // see if we can't move it along one (we're at the end)
359  if (++djSetMapListIt == djSetMapListEnd) {
360  isEndVal = true;
361  return *this;
362  }
363 
364  // we can't iterate along this djset if it is empty
365  djSetList = (*djSetMapListIt).second;
366  if (djSetList->size() == 0) {
367  throw std::out_of_range("error: encountered a zero size djset");
368  }
369 
370  // update our cAnterior and cPosterior
371  cAnteriorIndex = 0;
372  cPosteriorIndex = 0;
373  updateAnterior();
374  updatePosterior();
375  }
376 
377  // we moved our anterior along one
378  updateAnterior();
379 
380  cPosteriorIndex = 0;
381  updatePosterior();
382  }
383  // we just moved our posterior along one
384  updatePosterior();
385 
386  break;
387  case IterType::ANTERIOR:
388  // step posterior along one, and if we can't, then we're done.
389  if (++cPosteriorIndex == djSetList->size()) {
390  isEndVal = true;
391  return *this;
392  }
393  updatePosterior();
394 
395  break;
396  case IterType::ANTPOST:
397  // fixed anterior and posterior literally only points to one, so if we increment, its the
398  // end
399  isEndVal = true;
400  break;
401  case IterType::WITHIN:
402  // move posterior along one
403  // see if we can't move the posterior along
404  if (++cPosteriorIndex == djSetList->size()) {
405  // move anterior along one
406  // see if we can't move the anterior along one
407  if (++cAnteriorIndex == djSetList->size()) {
408  isEndVal = true;
409  return *this;
410  }
411 
412  // we moved our anterior along one
413  updateAnterior();
414 
415  cPosteriorIndex = 0;
416  updatePosterior();
417  }
418  // we just moved our posterior along one
419  updatePosterior();
420  break;
421  }
422 
423  return *this;
424  }

428  // special tombstone value to notify that this iter represents the end

432  enum IterType { ALL, ANTERIOR, ANTPOST, WITHIN };

480  /**
481  * Obtains a range of elements matching the prefix of the given entry up to
482  * levels elements. A operation context may be provided to exploit temporal
483  * locality.
484  *
485  * @tparam levels the length of the requested matching prefix
486  * @param entry the entry to be looking for
487  * @param ctxt the operation context to be utilized
488  * @return the corresponding range of matching elements
489  */
490  template <unsigned levels>
491  range<iterator> getBoundaries(const TupleType& entry, operation_hints&) const {
492  // if nothing is bound => just use begin and end
493  if (levels == 0) return make_range(begin(), end());
494 
495  // as disjoint set is exactly two args (equiv relation)
496  // we only need to handle these cases
497 
498  if (levels == 1) {
499  // need to test if the entry actually exists
500  if (!sds.nodeExists(entry[0])) return make_range(end(), end());
501 
502  // return an iterator over all (entry[0], _)
503  return make_range(anteriorIt(entry[0]), end());
504  }
505 
506  if (levels == 2) {
507  // need to test if the entry actually exists
508  if (!sds.contains(entry[0], entry[1])) return make_range(end(), end());
509 
510  // if so return an iterator containing exactly that node
511  return make_range(antpostit(entry[0], entry[1]), end());
512  }
513 
514  std::cerr << "invalid state, cannot search for >2 arg start point in getBoundaries, in 2 arg tuple "
515  "store\n";
516  throw "invalid state, cannot search for >2 arg start point in getBoundaries, in 2 arg tuple store";
517 
518  return make_range(end(), end());
519  }

592  /**

594  * for a given A and B. If A and B don't exist, or aren't in the same set,

597  * @param posteriorVal: the B value of the tuple

626  /**
627  * Generate an approximate number of iterators for parallel iteration
628  * The iterators returned are not necessarily equal in size, but in practise are approximately similarly
629  * sized
630  * Depending on the structure of the data, there can be more or less partitions returned than requested.
631  * @param chunks the number of requested partitions
632  * @return a list of the iterators as ranges
633  */
634  std::vector<souffle::range<iterator>> partition(size_t chunks) const {
635  // generate all reps
636  genAllDisjointSetLists();
637 
638  size_t numPairs = this->size();
639  if (numPairs == 0) return {};
640  if (numPairs == 1 || chunks <= 1) return {souffle::make_range(begin(), end())};
641 
642  // if there's more dj sets than requested chunks, then just return an iter per dj set
643  std::vector<souffle::range<iterator>> ret;
644  if (chunks <= equivalencePartition.size()) {
645  for (auto& p : equivalencePartition) {
646  ret.push_back(souffle::make_range(closure(p.first), end()));
647  }
648  return ret;
649  }
650 
651  // keep it simple stupid
652  // just go through and if the size of the binrel is > numpairs/chunks, then generate an anteriorIt for
653  // each
654  const size_t perchunk = numPairs / chunks;
655  for (const auto& itp : equivalencePartition) {
656  const size_t s = itp.second->size();
657  if (s * s > perchunk) {
658  for (const auto& i : *itp.second) {
659  ret.push_back(souffle::make_range(anteriorIt(i), end()));
660  }
661  } else {
662  ret.push_back(souffle::make_range(closure(itp.first), end()));
663  }
664  }
665 
666  return ret;
667  }

696  /**
697  * Generate a cache of the sets such that they can be iterated over efficiently.
698  * Each set is partitioned into a PiggyList.
699  */
700  void genAllDisjointSetLists() const {
701  statesLock.lock();
702 
703  // no need to generate again, already done.
704  if (!this->statesMapStale.load(std::memory_order_acquire)) {
705  statesLock.unlock();
706  return;
707  }
708 
709  // btree version
710  emptyPartition();
711 
712  size_t dSetSize = this->sds.ds.a_blocks.size();
713  for (size_t i = 0; i < dSetSize; ++i) {
