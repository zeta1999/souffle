80 /**
81  * @class RamConjunction
82  * @brief A conjunction of conditions
83  *
84  * Condition of the form "LHS and RHS", where LHS
85  * and RHS are conditions
86  *
87  * For example:
88  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
89  * C1 AND C2 AND C3
90  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
91  * Is a RamConjunction, which may have LHS "C1"
92  * and RHS "C2 AND C3"
93  */
94 class RamConjunction : public RamCondition {
95 public:
96  RamConjunction(std::unique_ptr<RamCondition> l, std::unique_ptr<RamCondition> r)
97  : lhs(std::move(l)), rhs(std::move(r)) {
98  assert(lhs != nullptr && "left-hand side of conjunction is a nullptr");
99  assert(rhs != nullptr && "right-hand side of conjunction is a nullptr");
100  }
101 
102  /** @brief Get left-hand side of conjunction */
103  const RamCondition& getLHS() const {
104  return *lhs;
105  }
106 
107  /** @brief Get right-hand side of conjunction */
108  const RamCondition& getRHS() const {
109  return *rhs;
110  }
111 
112  std::vector<const RamNode*> getChildNodes() const override {
113  return {lhs.get(), rhs.get()};
114  }
115 
116  RamConjunction* clone() const override {
117  return new RamConjunction(souffle::clone(lhs), souffle::clone(rhs));
118  }
119 
120  void apply(const RamNodeMapper& map) override {
121  lhs = map(std::move(lhs));
122  rhs = map(std::move(rhs));
123  }
124 
125 protected:
126  void print(std::ostream& os) const override {
127  os << "(" << *lhs << " AND " << *rhs << ")";
128  }
129 
130  bool equal(const RamNode& node) const override {
131  const auto& other = static_cast<const RamConjunction&>(node);
132  return equal_ptr(lhs, other.lhs) && equal_ptr(rhs, other.rhs);
133  }
134 
135  /** Left-hand side of conjunction */
136  std::unique_ptr<RamCondition> lhs;
137 
138  /** Right-hand side of conjunction */
139  std::unique_ptr<RamCondition> rhs;

188 /**
189  * @class RamConstraint
190  * @brief Evaluates a binary constraint with respect to two RamExpressions
191  *
192  * Condition is true if the constraint (a logical operator
193  * such as "<") holds between the two operands
194  *
195  * The following example checks the equality of
196  * the two given tuple elements:
197  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
198  * t0.1 = t1.0
199  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
200  */
201 class RamConstraint : public RamCondition {
202 public:
203  RamConstraint(BinaryConstraintOp op, std::unique_ptr<RamExpression> l, std::unique_ptr<RamExpression> r)
204  : op(op), lhs(std::move(l)), rhs(std::move(r)) {
205  assert(lhs != nullptr && "left-hand side of constraint is a null-pointer");
206  assert(rhs != nullptr && "right-hand side of constraint is a null-pointer");
207  }
208 
209  /** @brief Get left-hand side */
210  const RamExpression& getLHS() const {
211  return *lhs;
212  }
213 
214  /** @brief Get right-hand side */
215  const RamExpression& getRHS() const {
216  return *rhs;
217  }
218 
219  /** @brief Get operator symbol */
220  BinaryConstraintOp getOperator() const {
221  return op;
222  }
223 
224  std::vector<const RamNode*> getChildNodes() const override {
225  return {lhs.get(), rhs.get()};
226  }
227 
228  RamConstraint* clone() const override {
229  return new RamConstraint(op, souffle::clone(lhs), souffle::clone(rhs));
230  }
231 
232  void apply(const RamNodeMapper& map) override {
233  lhs = map(std::move(lhs));
234  rhs = map(std::move(rhs));

237 protected:

260  * @class RamAbstractExistenceCheck
261  * @brief Abstract existence check for a tuple in a relation
262  */
263 class RamAbstractExistenceCheck : public RamCondition {
264 public:
265  RamAbstractExistenceCheck(
266  std::unique_ptr<RamRelationReference> relRef, std::vector<std::unique_ptr<RamExpression>> vals)
267  : relationRef(std::move(relRef)), values(std::move(vals)) {
268  assert(relationRef != nullptr && "Relation reference is a nullptr");
269  for (const auto& v : values) {
270  assert(v != nullptr && "value is a nullptr");
271  }
272  }
273 
274  /** @brief Get relation */
275  const RamRelation& getRelation() const {
276  return *relationRef->get();
277  }
278 
279  /**
280  * @brief Get arguments of the tuple/pattern
281  * A null pointer element in the vector denotes an unspecified
282  * pattern for a tuple element.
283  */
284  const std::vector<RamExpression*> getValues() const {
285  return toPtrVector(values);
286  }
287 
288  std::vector<const RamNode*> getChildNodes() const override {
289  std::vector<const RamNode*> res = {relationRef.get()};
290  for (const auto& cur : values) {
291  res.push_back(cur.get());
292  }
293  return res;
294  }
295 
296  void apply(const RamNodeMapper& map) override {
297  relationRef = map(std::move(relationRef));
298  for (auto& val : values) {
299  val = map(std::move(val));
300  }
301  }
302 
303 protected:
304  void print(std::ostream& os) const override {
305  os << "("
306  << join(values, ",",
307  [](std::ostream& out, const std::unique_ptr<RamExpression>& value) {
308  if (!value) {
309  out << "_";
310  } else {
311  out << *value;
312  }
313  })
314  << ") " << getRelation().getName();
315  }
316 
317  bool equal(const RamNode& node) const override {
318  const auto& other = static_cast<const RamAbstractExistenceCheck&>(node);
319  return equal_ptr(relationRef, other.relationRef) && equal_targets(values, other.values);
320  }
321 
322  /** Relation */
323  std::unique_ptr<RamRelationReference> relationRef;
324 
325  /** Pattern -- nullptr if undefined */
326  std::vector<std::unique_ptr<RamExpression>> values;
327 };
