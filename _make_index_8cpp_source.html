43 ExpressionPair MakeIndexTransformer::getLowerUpperExpression(
44  RamCondition* c, size_t& element, int identifier) {
45  if (auto* binRelOp = dynamic_cast<RamConstraint*>(c)) {
46  // TODO: FIXME: how does this interact w/ `FEQ`?
47 
48  if (isEqConstraint(binRelOp->getOperator())) {
49  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
50  const RamExpression* rhs = &binRelOp->getRHS();
51  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
52  element = lhs->getElement();
53  return {clone(rhs), clone(rhs)};
54  }
55  }
56  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
57  const RamExpression* lhs = &binRelOp->getLHS();
58  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
59  element = rhs->getElement();
60  return {clone(lhs), clone(lhs)};
61  }
62  }
63  }
64 
65  if (isLessEqualConstraint(binRelOp->getOperator())) {
66  // Tuple[level, element] <= <expr>
67  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
68  const RamExpression* rhs = &binRelOp->getRHS();
69  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
70  element = lhs->getElement();
71  return {std::make_unique<RamUndefValue>(), clone(rhs)};
72  }
73  }
74  // <expr> <= Tuple[level, element]
75  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
76  const RamExpression* lhs = &binRelOp->getLHS();
77  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
78  element = rhs->getElement();
79  return {clone(lhs), std::make_unique<RamUndefValue>()};
80  }
81  }
82  }
83 
84  if (isGreaterEqualConstraint(binRelOp->getOperator())) {
85  // Tuple[level, element] >= <expr>
86  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
87  const RamExpression* rhs = &binRelOp->getRHS();
88  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
89  element = lhs->getElement();
90  return {clone(rhs), std::make_unique<RamUndefValue>()};
91  }
92  }
93  // <expr> >= Tuple[level, element]
94  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
95  const RamExpression* lhs = &binRelOp->getLHS();
96  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
97  element = rhs->getElement();
98  return {std::make_unique<RamUndefValue>(), clone(lhs)};
99  }
100  }
101  }
102  if (isLessThanConstraint(binRelOp->getOperator())) {
103  // Tuple[level, element] < <expr>
104  // is equivalent to
105  // Tuple[level, element] <= <expr> - 1
106  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
107  const RamExpression* rhs = &binRelOp->getRHS();
108  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
109  element = lhs->getElement();
110  std::vector<std::unique_ptr<RamExpression>> expressions;
111  expressions.push_back(souffle::clone(rhs));
112  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
113 
114  return {std::make_unique<RamUndefValue>(),
115  std::make_unique<RamIntrinsicOperator>(FunctorOp::SUB, std::move(expressions))};
116  }
117  }
118  // <expr> < Tuple[level, element]
119  // is equivalent to
120  // <expr> + 1 <= Tuple[level, element]
121  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
122  const RamExpression* lhs = &binRelOp->getLHS();
123  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
124  element = rhs->getElement();
125  std::vector<std::unique_ptr<RamExpression>> expressions;
126  expressions.push_back(souffle::clone(lhs));
127  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
128 
129  return {std::make_unique<RamIntrinsicOperator>(FunctorOp::ADD, std::move(expressions)),
130  std::make_unique<RamUndefValue>()};
131  }
132  }
133  }
134 
135  if (isGreaterThanConstraint(binRelOp->getOperator())) {
136  // Tuple[level, element] > <expr>
137  // is equivalent to
138  // Tuple[level, element] >= <expr> + 1
139  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
140  const RamExpression* rhs = &binRelOp->getRHS();
141  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
142  element = lhs->getElement();
143  std::vector<std::unique_ptr<RamExpression>> expressions;
144  expressions.push_back(souffle::clone(rhs));
145  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
146 
147  return {std::make_unique<RamIntrinsicOperator>(FunctorOp::ADD, std::move(expressions)),
148  std::make_unique<RamUndefValue>()};
149  }
150  }
151  // <expr> > Tuple[level, element]
152  // is equivalent to
153  // <expr> - 1 >= Tuple[level, element]
154  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
155  const RamExpression* lhs = &binRelOp->getLHS();
156  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
157  element = rhs->getElement();
158  std::vector<std::unique_ptr<RamExpression>> expressions;
159  expressions.push_back(souffle::clone(lhs));
160  expressions.push_back(std::make_unique<RamSignedConstant>(RamDomain(1)));
161 
162  return {std::make_unique<RamUndefValue>(),
163  std::make_unique<RamIntrinsicOperator>(FunctorOp::SUB, std::move(expressions))};
164  }
165  }
166  }
167  }
168  return {std::make_unique<RamUndefValue>(), std::make_unique<RamUndefValue>()};

171 std::unique_ptr<RamCondition> MakeIndexTransformer::constructPattern(RamPattern& queryPattern,
172  bool& indexable, std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
173  // Remaining conditions which cannot be handled by an index
174  std::unique_ptr<RamCondition> condition;
175  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
176  if (condition != nullptr) {
177  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
178  } else {
179  condition = std::move(c);
180  }
181  };
182 
183  // Build query pattern and remaining condition
184  for (auto& cond : conditionList) {
185  size_t element = 0;
186  std::unique_ptr<RamExpression> lowerExpression;
187  std::unique_ptr<RamExpression> upperExpression;
188  std::tie(lowerExpression, upperExpression) = getLowerUpperExpression(cond.get(), element, identifier);
189 
190  // we have new bounds if both are not nullptr
191  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
192  // if no previous bounds are set then just assign them, consider both bounds to be set (but not
193  // necessarily defined) in all remaining cases
194  indexable = true;
195  if (isRamUndefValue(queryPattern.first[element].get()) &&
196  isRamUndefValue(queryPattern.second[element].get())) {
197  queryPattern.first[element] = std::move(lowerExpression);
198  queryPattern.second[element] = std::move(upperExpression);
199  // if lower bound is undefined and we have a new lower bound then assign it
200  } else if (isRamUndefValue(queryPattern.first[element].get()) &&
201  !isRamUndefValue(lowerExpression.get()) && isRamUndefValue(upperExpression.get())) {
202  queryPattern.first[element] = std::move(lowerExpression);
203  // if upper bound is undefined and we have a new upper bound then assign it
204  } else if (isRamUndefValue(queryPattern.second[element].get()) &&
205  isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
206  queryPattern.second[element] = std::move(upperExpression);
207  // if both bounds are defined ...
208  // and equal then we have a previous equality constraint i.e. Tuple[level, element] = <expr1>
209  } else if (!isRamUndefValue(queryPattern.first[element].get()) &&
210  !isRamUndefValue(queryPattern.second[element].get()) &&
211  (*(queryPattern.first[element]) == *(queryPattern.second[element]))) {
212  // new equality constraint i.e. Tuple[level, element] = <expr2>
213  // simply hoist <expr1> = <expr2> to the outer loop
214  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
215  // FIXME: `FEQ` handling; need to know if the expr is a float exp or not
216  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ,
217  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
218  }
219  // new lower bound i.e. Tuple[level, element] >= <expr2>
220  // we need to hoist <expr1> >= <expr2> to the outer loop
221  else if (!isRamUndefValue(lowerExpression.get()) && isRamUndefValue(upperExpression.get())) {
222  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::GE,
223  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
224  }
225  // new upper bound i.e. Tuple[level, element] <= <expr2>
226  // we need to hoist <expr1> <= <expr2> to the outer loop
227  else if (isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
228  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::LE,
229  souffle::clone(queryPattern.first[element]), std::move(upperExpression)));
230  }
231  // if either bound is defined but they aren't equal we must consider the cases for updating
232  // them note that at this point we know that if we have a lower/upper bound it can't be the
233  // first one
234  } else if (!isRamUndefValue(queryPattern.first[element].get()) ||
235  !isRamUndefValue(queryPattern.second[element].get())) {
236  // if we have a new equality constraint and previous inequality constraints
237  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get()) &&
238  *lowerExpression == *upperExpression) {
239  // if Tuple[level, element] >= <expr1> and we see Tuple[level, element] = <expr2>
240  // need to hoist <expr2> >= <expr1> to the outer loop
241  if (!isRamUndefValue(queryPattern.first[element].get())) {
242  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::GE,
243  souffle::clone(lowerExpression), std::move(queryPattern.first[element])));
244  }
245  // if Tuple[level, element] <= <expr1> and we see Tuple[level, element] = <expr2>
246  // need to hoist <expr2> <= <expr1> to the outer loop
247  if (!isRamUndefValue(queryPattern.second[element].get())) {
248  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::LE,
249  souffle::clone(upperExpression), std::move(queryPattern.second[element])));
250  }
251  // finally replace bounds with equality constraint
252  queryPattern.first[element] = std::move(lowerExpression);
253  queryPattern.second[element] = std::move(upperExpression);
254  // if we have a new lower bound
255  } else if (!isRamUndefValue(lowerExpression.get())) {
256  // we want the tightest lower bound so we take the max
257  std::vector<std::unique_ptr<RamExpression>> maxArguments;
258  maxArguments.push_back(std::move(queryPattern.first[element]));
259  maxArguments.push_back(std::move(lowerExpression));
260 
261  queryPattern.first[element] =
262  std::make_unique<RamIntrinsicOperator>(FunctorOp::MAX, std::move(maxArguments));
263  // if we have a new upper bound
264  } else if (!isRamUndefValue(upperExpression.get())) {
265  // we want the tightest upper bound so we take the min
266  std::vector<std::unique_ptr<RamExpression>> minArguments;
267  minArguments.push_back(std::move(queryPattern.second[element]));
268  minArguments.push_back(std::move(upperExpression));
269 
270  queryPattern.second[element] =
271  std::make_unique<RamIntrinsicOperator>(FunctorOp::MIN, std::move(minArguments));
272  }
273  }
274  } else {
275  addCondition(std::move(cond));
276  }
277  }
278 
279  // Avoid null-pointers for condition and query pattern
280  if (condition == nullptr) {
281  condition = std::make_unique<RamTrue>();
282  }
283  return condition;
284 }

286 std::unique_ptr<RamOperation> MakeIndexTransformer::rewriteAggregate(const RamAggregate* agg) {
287  if (dynamic_cast<const RamTrue*>(&agg->getCondition()) == nullptr) {
288  const RamRelation& rel = agg->getRelation();
289  int identifier = agg->getTupleId();
290  RamPattern queryPattern;
291  for (unsigned int i = 0; i < rel.getArity(); ++i) {
292  queryPattern.first.push_back(std::make_unique<RamUndefValue>());
293  queryPattern.second.push_back(std::make_unique<RamUndefValue>());
294  }
295 
296  bool indexable = false;
297  std::unique_ptr<RamCondition> condition = constructPattern(
298  queryPattern, indexable, toConjunctionList(&agg->getCondition()), identifier);
299  if (indexable) {
300  return std::make_unique<RamIndexAggregate>(souffle::clone(&agg->getOperation()),
301  agg->getFunction(), std::make_unique<RamRelationReference>(&rel),
302  souffle::clone(&agg->getExpression()), std::move(condition), std::move(queryPattern),
303  agg->getTupleId());
304  }
305  }
306  return nullptr;
307 }

309 std::unique_ptr<RamOperation> MakeIndexTransformer::rewriteScan(const RamScan* scan) {
310  if (const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation())) {
311  const RamRelation& rel = scan->getRelation();
312  const int identifier = scan->getTupleId();
313  RamPattern queryPattern;
314  for (unsigned int i = 0; i < rel.getArity(); ++i) {
315  queryPattern.first.push_back(std::make_unique<RamUndefValue>());
316  queryPattern.second.push_back(std::make_unique<RamUndefValue>());
317  }
318 
319  bool indexable = false;
320  std::unique_ptr<RamCondition> condition = constructPattern(
321  queryPattern, indexable, toConjunctionList(&filter->getCondition()), identifier);
322  if (indexable) {
323  std::unique_ptr<RamOperation> op = souffle::clone(&filter->getOperation());
324  if (!isRamTrue(condition.get())) {
325  op = std::make_unique<RamFilter>(std::move(condition), std::move(op));
326  }
327  return std::make_unique<RamIndexScan>(std::make_unique<RamRelationReference>(&rel), identifier,
328  std::move(queryPattern), std::move(op), scan->getProfileText());
329  }
330  }
331  return nullptr;

362 bool MakeIndexTransformer::makeIndex(RamProgram& program) {
363  bool changed = false;
364  visitDepthFirst(program, [&](const RamQuery& query) {
365  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> scanRewriter =
366  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
367  if (const RamScan* scan = dynamic_cast<RamScan*>(node.get())) {
368  if (scan->getRelation().getRepresentation() != RelationRepresentation::INFO) {
369  if (std::unique_ptr<RamOperation> op = rewriteScan(scan)) {
370  changed = true;
371  node = std::move(op);
372  }
373  }
374  } else if (const RamIndexScan* iscan = dynamic_cast<RamIndexScan*>(node.get())) {
375  if (std::unique_ptr<RamOperation> op = rewriteIndexScan(iscan)) {
376  changed = true;
377  node = std::move(op);
378  }
379  } else if (const RamAggregate* agg = dynamic_cast<RamAggregate*>(node.get())) {
380  if (std::unique_ptr<RamOperation> op = rewriteAggregate(agg)) {
381  changed = true;
382  node = std::move(op);
383  }
384  }
385  node->apply(makeLambdaRamMapper(scanRewriter));
386  return node;
387  };
388  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(scanRewriter));
389  });
390  return changed;
391 }
