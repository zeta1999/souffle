30  * This means we can't append, as we don't know the next available element.
31  * insertAt is dangerous. You must be careful not to call it for the same index twice!
32  */

43  this->numElements.store(other.numElements.load());
44 
45  // copy blocks from the old lookup table to this one
46  for (size_t i = 0; i < maxContainers; ++i) {

87  void insertAt(size_t index, T value) {
88  // starting with an initial blocksize requires some shifting to transform into a nice powers of two
89  // series
90  size_t blockNum = (63 - __builtin_clzll(index + INITIALBLOCKSIZE)) - BLOCKBITS;
91 
92  // allocate the block if not allocated
93  if (blockLookupTable[blockNum].load() == nullptr) {
94  slock.lock();
95  if (blockLookupTable[blockNum].load() == nullptr) {
96  blockLookupTable[blockNum].store(new T[INITIALBLOCKSIZE << blockNum]);
97  }
98  slock.unlock();
99  }
100 
101  this->get(index) = value;
102  // we ALWAYS increment size, even if there was something there before (its impossible to tell!)
103  // the onus is up to the user to not call this for an index twice
104  ++numElements;
105  }

109  numElements.store(0);
110  }
