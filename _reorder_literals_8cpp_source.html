50 /**
51  * Counts the number of bound arguments in a given atom.
52  */
53 unsigned int numBoundArguments(const AstAtom* atom, const std::set<std::string>& boundVariables) {
54  int count = 0;
55 
56  for (const AstArgument* arg : atom->getArguments()) {
57  // argument is bound iff all contained variables are bound
58  bool isBound = true;
59 
60  visitDepthFirst(*arg, [&](const AstVariable& var) {
61  if (boundVariables.find(var.getName()) == boundVariables.end()) {
62  // found an unbound variable, so argument is unbound
63  isBound = false;
64  }
65  });
66 
67  if (isBound) {
68  count++;
69  }
70  }
71 
72  return count;
73 }

75 /**
76  * Returns a SIPS function based on the SIPS option provided.
77  * The SIPS function will return the index of the appropriate atom in a clause
78  * given a goal.
79  *
80  * E.g. the 'max-bound' SIPS function will return the atom in the clause with
81  * the maximum number of bound arguments.
82  */
83 sips_t getSipsFunction(const std::string& sipsChosen) {
84  // --- Create the appropriate SIPS function ---
85 
86  // Each SIPS function has a priority metric (e.g. max-bound atoms).
87  // Arguments:
88  // - a vector of atoms to choose from (nullpointers in the vector will be ignored)
89  // - the set of variables bound so far
90  // Returns: the index of the atom maximising the priority metric
91  sips_t getNextAtomSips;
92 
93  if (sipsChosen == "naive") {
94  // Goal: choose the first atom with at least one bound argument, or with no arguments
95  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
96  for (unsigned int i = 0; i < atoms.size(); i++) {
97  const AstAtom* currAtom = atoms[i];
98 
99  if (currAtom == nullptr) {
100  // already processed - move on
101  continue;
102  }
103 
104  if (isProposition(currAtom)) {
105  // no arguments
106  return i;
107  }
108 
109  if (numBoundArguments(currAtom, boundVariables) >= 1) {
110  // at least one bound argument
111  return i;
112  }
113  }
114 
115  // none found; choose the first non-null
116  for (unsigned int i = 0; i < atoms.size(); i++) {
117  if (atoms[i] != nullptr) {
118  return i;
119  }
120  }
121 
122  // fall back to the first
123  return 0U;
124  };
125  } else if (sipsChosen == "all-bound") {
126  // Goal: prioritise atoms with all arguments bound
127  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
128  for (unsigned int i = 0; i < atoms.size(); i++) {
129  const AstAtom* currAtom = atoms[i];
130 
131  if (currAtom == nullptr) {
132  // already processed - move on
133  continue;
134  }
135 
136  if (isProposition(currAtom)) {
137  // no arguments, so all are trivially bound
138  return i;
139  }
140 
141  int arity = currAtom->getArity();
142  int numBound = numBoundArguments(currAtom, boundVariables);
143  if (numBound == arity) {
144  // all arguments are bound!
145  return i;
146  }
147  }
148 
149  // none found; choose the first non-null
150  for (unsigned int i = 0; i < atoms.size(); i++) {
151  if (atoms[i] != nullptr) {
152  return i;
153  }
154  }
155 
156  // fall back to the first
157  return 0U;
158  };
159  } else if (sipsChosen == "max-bound") {
160  // Goal: choose the atom with the maximum number of bound variables
161  // - exception: propositions should be prioritised
162  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
163  int currMaxBound = -1;
164  unsigned int currMaxIdx = 0U;
165 
166  for (unsigned int i = 0; i < atoms.size(); i++) {
167  const AstAtom* currAtom = atoms[i];
168 
169  if (currAtom == nullptr) {
170  // already processed - move on
171  continue;
172  }
173 
174  if (isProposition(currAtom)) {
175  // propositions should be prioritised
176  return i;
177  }
178 
179  int numBound = numBoundArguments(currAtom, boundVariables);
180  if (numBound > currMaxBound) {
181  currMaxBound = numBound;
182  currMaxIdx = i;
183  }
184  }
185 
186  return currMaxIdx;
187  };
188  } else if (sipsChosen == "max-ratio") {
189  // Goal: choose the atom with the maximum ratio of bound to unbound
190  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
191  auto isLargerRatio = [&](std::pair<int, int> lhs, std::pair<int, int> rhs) {
192  return (lhs.first * rhs.second > lhs.second * rhs.first);
193  };
194 
195  auto currMaxRatio = std::pair<int, int>(-1, 1); // set to -1
196  unsigned int currMaxIdx = 0U;
197 
198  for (unsigned int i = 0; i < atoms.size(); i++) {
199  const AstAtom* currAtom = atoms[i];
200 
201  if (currAtom == nullptr) {
202  // already processed - move on
203  continue;
204  }
205 
206  if (isProposition(currAtom)) {
207  // propositions are as bound as possible
208  return i;
209  }
210 
211  int numBound = numBoundArguments(currAtom, boundVariables);
212  int numArgs = currAtom->getArity();
213  auto currRatio = std::pair<int, int>(numBound, numArgs);
214  if (isLargerRatio(currRatio, currMaxRatio)) {
215  currMaxRatio = currRatio;
216  currMaxIdx = i;
217  }
218  }
219 
220  return currMaxIdx;
221  };
222  } else if (sipsChosen == "least-free") {
223  // Goal: choose the atom with the least number of unbound arguments
224  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
225  int currLeastFree = -1;
226  unsigned int currLeastIdx = 0U;
227 
228  for (unsigned int i = 0; i < atoms.size(); i++) {
229  const AstAtom* currAtom = atoms[i];
230 
231  if (currAtom == nullptr) {
232  // already processed - move on
233  continue;
234  }
235 
236  if (isProposition(currAtom)) {
237  // propositions have 0 unbound arguments, which is minimal
238  return i;
239  }
240 
241  int numBound = numBoundArguments(currAtom, boundVariables);
242  int numFree = currAtom->getArity() - numBound;
243  if (currLeastFree == -1 || numFree < currLeastFree) {
244  currLeastFree = numFree;
245  currLeastIdx = i;
246  }
247  }
248 
249  return currLeastIdx;
250  };
251  } else if (sipsChosen == "least-free-vars") {
252  // Goal: choose the atom with the least amount of unbound variables
253  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
254  int currLeastFree = -1;
255  unsigned int currLeastIdx = 0U;
256 
257  for (unsigned int i = 0; i < atoms.size(); i++) {
258  const AstAtom* currAtom = atoms[i];
259 
260  if (currAtom == nullptr) {
261  // already processed - move on
262  continue;
263  }
264 
265  if (isProposition(currAtom)) {
266  // propositions have 0 unbound variables, which is minimal
267  return i;
268  }
269 
270  // use a set to hold all free variables to avoid double-counting
271  std::set<std::string> freeVars;
272  visitDepthFirst(*currAtom, [&](const AstVariable& var) {
273  if (boundVariables.find(var.getName()) == boundVariables.end()) {
274  freeVars.insert(var.getName());
275  }
276  });
277 
278  int numFreeVars = freeVars.size();
279  if (currLeastFree == -1 || numFreeVars < currLeastFree) {
280  currLeastFree = numFreeVars;
281  currLeastIdx = i;
282  }
283  }
284 
285  return currLeastIdx;
286  };
287  } else {
288  // chosen SIPS is not implemented, so keep the same order
289  // Goal: leftmost atom first
290  getNextAtomSips = [&](std::vector<AstAtom*> atoms,
291  const std::set<std::string>& /* boundVariables */) {
292  for (unsigned int i = 0; i < atoms.size(); i++) {
293  if (atoms[i] == nullptr) {
294  // already processed - move on
295  continue;
296  }
297 
298  return i;
299  }
300 
301  // fall back to the first
302  return 0U;
303  };
304  }
305 
306  return getNextAtomSips;
307 }

309 /**
310  * Finds the new ordering of a vector of atoms after the given SIPS is applied.
311  */
312 std::vector<unsigned int> applySips(sips_t sipsFunction, std::vector<AstAtom*> atoms) {
313  std::set<std::string> boundVariables;
314  std::vector<unsigned int> newOrder(atoms.size());
315 
316  unsigned int numAdded = 0;
317  while (numAdded < atoms.size()) {
318  // grab the next atom, based on the SIPS function
319  unsigned int nextIdx = sipsFunction(atoms, boundVariables);
320  AstAtom* nextAtom = atoms[nextIdx];
321 
322  // set all arguments that are variables as bound
323  // note: arguments that are functors, etc., do not newly bind anything
324  for (AstArgument* arg : nextAtom->getArguments()) {
325  if (auto* var = dynamic_cast<AstVariable*>(arg)) {
326  boundVariables.insert(var->getName());
327  }
328  }
329 
330  newOrder[numAdded] = nextIdx; // add to the ordering
331  atoms[nextIdx] = nullptr; // mark as done
332  numAdded++; // move on
333  }
334 
335  return newOrder;
336 }

338 AstClause* reorderClauseWithSips(sips_t sipsFunction, AstClause* clause) {
339  // ignore clauses with fixed execution plans
340  if (clause->getExecutionPlan() != nullptr) {
341  return nullptr;
342  }
343 
344  // get the ordering corresponding to the SIPS
345  std::vector<unsigned int> newOrdering = applySips(sipsFunction, getBodyLiterals<AstAtom>(*clause));
346 
347  // check if we need a change
348  bool changeNeeded = false;
349  for (unsigned int i = 0; i < newOrdering.size(); i++) {
350  if (newOrdering[i] != i) {
351  changeNeeded = true;
352  }
353  }
354 
355  // reorder if needed
356  return changeNeeded ? reorderAtoms(clause, newOrdering) : nullptr;
357 }

359 bool ReorderLiteralsTransformer::transform(AstTranslationUnit& translationUnit) {
360  bool changed = false;
361  AstProgram& program = *translationUnit.getProgram();
362 
363  // --- SIPS-based static reordering ---
364  // ordering is based on the given SIPS
365 
366  // default SIPS to choose is 'all-bound'
367  std::string sipsChosen = "all-bound";
368  if (Global::config().has("SIPS")) {
369  sipsChosen = Global::config().get("SIPS");
370  }
371  auto sipsFunction = getSipsFunction(sipsChosen);
372 
373  // literal reordering is a rule-local transformation
374  std::vector<AstClause*> clausesToRemove;
375 
376  for (const AstRelation* rel : program.getRelations()) {
377  for (AstClause* clause : getClauses(program, *rel)) {
378  AstClause* newClause = reorderClauseWithSips(sipsFunction, clause);
379  if (newClause != nullptr) {
380  // reordering needed - swap around
381  clausesToRemove.push_back(clause);
382  program.addClause(std::unique_ptr<AstClause>(newClause));
383  }
384  }
385  }
386 
387  changed |= !clausesToRemove.empty();
388  for (auto* clause : clausesToRemove) {
389  program.removeClause(clause);
390  }
391 
392  // --- profile-guided reordering ---
393  if (Global::config().has("profile-use")) {
394  // parse supplied profile information
395  auto* profileUse = translationUnit.getAnalysis<AstProfileUseAnalysis>();
396 
397  auto profilerSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
398  // Goal: reorder based on the given profiling information
399  // Metric: cost(atom_R) = log(|atom_R|) * #free/#args
400  // - exception: propositions are prioritised
401 
402  double currOptimalVal = -1;
403  unsigned int currOptimalIdx = 0U;
404  bool set = false;
405 
406  for (unsigned int i = 0; i < atoms.size(); i++) {
407  const AstAtom* currAtom = atoms[i];
408 
409  if (currAtom == nullptr) {
410  // already processed - move on
411  continue;
412  }
413 
414  if (isProposition(currAtom)) {
415  // prioritise propositions
416  return i;
417  }
418 
419  // calculate log(|R|) * #free/#args
420  int numBound = numBoundArguments(currAtom, boundVariables);
421  int numArgs = currAtom->getArity();
422  int numFree = numArgs - numBound;
423  double value = log(profileUse->getRelationSize(currAtom->getQualifiedName()));
424  value *= (numFree * 1.0) / numArgs;
425 
426  if (!set || value < currOptimalVal) {
427  set = true;
428  currOptimalVal = value;
429  currOptimalIdx = i;
430  }
431  }
432 
433  return currOptimalIdx;
434  };
435 
436  // change the ordering of literals within clauses
437  std::vector<AstClause*> clausesToRemove;
438 
439  for (const AstRelation* rel : program.getRelations()) {
440  for (AstClause* clause : getClauses(program, *rel)) {
441  AstClause* newClause = reorderClauseWithSips(profilerSips, clause);
442  if (newClause != nullptr) {
443  // reordering needed - swap around
444  clausesToRemove.push_back(clause);
445  program.addClause(std::unique_ptr<AstClause>(newClause));
446  }
447  }
448  }
449 
450  changed |= !clausesToRemove.empty();
451  for (auto* clause : clausesToRemove) {
452  program.removeClause(clause);
453  }
454  }
455 
456  return changed;
457 }
