48 /** search signature of a RAM operation; each bit represents an attribute of a relation.
49  * A one represents that the attribute has an assigned value; a zero represents that
50  * no value exists (i.e. attribute is unbounded) in the search. */
51 class SearchSignature {
52 public:
53  explicit SearchSignature(size_t arity) : constraints(arity, AttributeConstraint::None) {}
54 
55  inline size_t arity() const {
56  return constraints.size();
57  }
58 
59  // array subscript operator
60  inline AttributeConstraint operator[](std::size_t pos) const {
61  assert(pos < constraints.size());
62  return constraints[pos];
63  }
64 
65  // comparison operators
66  inline bool operator<(const SearchSignature& other) const {
67  assert(constraints.size() == other.constraints.size());
68  size_t len = constraints.size();
69  for (size_t i = 0; i < len; ++i) {
70  size_t index = len - i - 1; // get right to left index order
71 
72  // if ours has a constraint and other's has a constraint then it is smaller
73  if (constraints[index] < other.constraints[index]) {
74  return true;
75  }
76  // if ours has a constraint and other's has no constraint then it is larger
77  else if (constraints[index] > other.constraints[index]) {
78  return false;
79  }
80  }
81  return false;
82  }
83 
84  // needed for asserts
85  inline bool operator==(const SearchSignature& other) const {
86  assert(constraints.size() == other.constraints.size());
87  return constraints == other.constraints;
88  }
89 
90  inline bool empty() const {
91  size_t len = constraints.size();
92  for (size_t i = 0; i < len; ++i) {
93  if (constraints[i] != AttributeConstraint::None) {
94  return false;
95  }
96  }
97  return true;
98  }
99 
100  inline bool isStrictSubset(const SearchSignature& other) const {
101  assert(constraints.size() == other.constraints.size());
102  size_t len = constraints.size();
103  for (size_t i = 0; i < len; ++i) {
104  if ((constraints[i] != AttributeConstraint::None) &&
105  (other.constraints[i] == AttributeConstraint::None)) {
106  return false;
107  }
108  }
109  return constraints != other.constraints;
110  }
111 
112  static SearchSignature getDelta(const SearchSignature& lhs, const SearchSignature& rhs) {
113  assert(lhs.arity() == rhs.arity());
114  SearchSignature delta(lhs.arity());
115  for (size_t i = 0; i < lhs.arity(); ++i) {
116  // if constraints are the same then delta is nothing
117  if (lhs.constraints[i] == rhs.constraints[i]) {
118  delta.constraints[i] = AttributeConstraint::None;
119  continue;
120  }
121 
122  // if rhs has no constraint then delta has lhs constraint
123  if (rhs.constraints[i] == AttributeConstraint::None) {
124  delta.constraints[i] = lhs.constraints[i];
125  continue;
126  }
127 
128  assert(false && "lhs and rhs should not have inequal/equal pair constraints");
129  }
130  return delta;
131  }
132 
133  static SearchSignature getFullSearchSignature(size_t arity) {
134  SearchSignature res(arity);
135  for (size_t i = 0; i < arity; ++i) {
136  res.constraints[i] = AttributeConstraint::Equal;
137  }
138  return res;
139  }
140 
141  // set a constraint
142  inline SearchSignature& set(size_t pos, AttributeConstraint constraint) {
143  assert(pos < constraints.size());
144  constraints[pos] = constraint;
145  return *this;
146  }
147 
148  friend std::ostream& operator<<(std::ostream& out, const SearchSignature& signature);
149 
150 private:
151  std::vector<AttributeConstraint> constraints;
152 };

154 inline std::ostream& operator<<(std::ostream& out, const SearchSignature& signature) {
155  size_t len = signature.constraints.size();
156  for (size_t i = 0; i < len; ++i) {
157  switch (signature.constraints[len - 1 - i]) {
158  case AttributeConstraint::None: out << 0; break;
159  case AttributeConstraint::Equal: out << 1; break;
160  case AttributeConstraint::Inequal: out << 2; break;
161  }
162  }
163  return out;
164 }

167  * @class MaxMatching
168  * @Brief Computes a maximum matching with Hopcroft-Karp algorithm
169  *
170  * This class is a helper class for RamIndexAnalysis.
171  *
172  * This implements a standard maximum matching algorithm for a bi-partite graph
173  * also known as a marriage problem. Given a set of edges in a bi-partite graph
174  * select a subset of edges that each node in the bi-partite graph has at most
175  * one adjacent edge associated with.
176  *
177  * The nodes of the bi-partite graph represent index-signatures stemming from
178  * RAM operations and RAM existence checks for a relation. A relation between
179  * two nodes represent whether an index operation subsumes another operation.
180  *
181  * Source: http://en.wikipedia.org/wiki/Hopcroft%E2%80%93Karp_algorithm#Pseudocode
182  */
183 class MaxMatching {
184 public:
185  using Node = uint32_t;
186  /* The nodes of the bi-partite graph are index signatures of RAM operation */
187  using Nodes = std::set<Node, std::greater<Node>>;
188  /* Distance between nodes */
189  using Distance = int;
190  /**
191  * Matching represent a solution of the matching, i.e., which node in the bi-partite
192  * graph maps to another node. If no map exist for a node, there is no adjacent edge
193  * exists for that node.
194  */
195  using Matchings = std::map<Node, Node, std::greater<Node>>;
196 
197  /* Node constant representing no match */
198  const Node NullVertex = 0;
199 
200  /* Infinite distance */
201  const Distance InfiniteDistance = -1;
202 
203  /**
204  * @Brief solve the maximum matching problem
205  * @result returns the matching
206  */
207  const Matchings& solve();
208 
209  /**
210  * @Brief get number of matches in the solution
211  * @return number of matches
212  */
213  int getNumMatchings() const {
214  return match.size() / 2;
215  }

342  SearchSignature fullIndexKey = SearchSignature::getFullSearchSignature(arity);
343  chain.insert(fullIndexKey);

351  /** Return the attribute position for each inequality that should be discharged.

353  * If an inequality is not in the last position of a lex-order only then is it discharged
