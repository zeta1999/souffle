352  template <typename... Stmts>
353  RamListStatement(std::unique_ptr<Stmts>&&... stmts) {
354  std::unique_ptr<RamStatement> tmp[] = {std::move(stmts)...};
355  for (auto& cur : tmp) {
356  assert(cur.get() != nullptr && "statement is a null-pointer");
357  statements.emplace_back(std::move(cur));
358  }
359  }
360 
361  /** @brief Get statements */
362  std::vector<RamStatement*> getStatements() const {
363  return toPtrVector(statements);
364  }
365 
366  std::vector<const RamNode*> getChildNodes() const override {
367  std::vector<const RamNode*> res;
368  for (const auto& cur : statements) {
369  res.push_back(cur.get());
370  }
371  return res;
372  }
373 
374  void apply(const RamNodeMapper& map) override {
375  for (auto& stmt : statements) {
376  stmt = map(std::move(stmt));
377  }
378  }
379 
380 protected:
381  bool equal(const RamNode& node) const override {
382  const auto& other = static_cast<const RamListStatement&>(node);
383  return equal_targets(statements, other.statements);
384  }
385 
386 protected:
387  /** ordered list of RAM statements */
388  std::vector<std::unique_ptr<RamStatement>> statements;
389 };
