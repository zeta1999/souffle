70 /**
71  * The generic base type of all AstVisitors realizing the dispatching of
72  * visitor calls. Each visitor may define a return type R and a list of
73  * extra parameters to be passed along with the visited AstNodes to the
74  * corresponding visitor function.
75  *
76  * @tparam R the result type produced by a visit call
77  * @tparam Params extra parameters to be passed to the visit call
78  */
79 template <typename R = void, typename... Params>
80 struct AstVisitor : public ast_visitor_tag {
81  /** A virtual destructor */
82  virtual ~AstVisitor() = default;
83 
84  /** The main entry for the user allowing visitors to be utilized as functions */
85  R operator()(const AstNode& node, Params... args) {
86  return visit(node, args...);
87  }
88 
89  /**
90  * The main entry for a visit process conducting the dispatching of
91  * a visit to the various sub-types of AstNodes. Sub-classes may override
92  * this implementation to conduct pre-visit operations.
93  *
94  * @param node the node to be visited
95  * @param args a list of extra parameters to be forwarded
96  */
97  virtual R visit(const AstNode& node, Params... args) {
98  // dispatch node processing based on dynamic type
99 
100 #define FORWARD(Kind) \
101  if (const auto* n = dynamic_cast<const Ast##Kind*>(&node)) return visit##Kind(*n, args...);
102 
103  // types
104  FORWARD(SubsetType);
105  FORWARD(UnionType);
106  FORWARD(RecordType);
107 
108  // arguments
109  FORWARD(Variable)
110  FORWARD(UnnamedVariable)
111  FORWARD(IntrinsicFunctor)
112  FORWARD(UserDefinedFunctor)
113  FORWARD(Counter)
114  FORWARD(NumericConstant)
115  FORWARD(StringConstant)
116  FORWARD(NilConstant)
117  FORWARD(TypeCast)
118  FORWARD(RecordInit)
119  FORWARD(Aggregator)
120  FORWARD(SubroutineArgument)
121 
122  // literals
123  FORWARD(Atom)
124  FORWARD(ProvenanceNegation)
125  FORWARD(Negation)
126  FORWARD(BooleanConstraint)
127  FORWARD(BinaryConstraint)
128 
129  // components
130  FORWARD(ComponentType);
131  FORWARD(ComponentInit);
132  FORWARD(Component);
133 
134  // rest
135  FORWARD(Attribute);
136  FORWARD(Clause);
137  FORWARD(Relation);
138  FORWARD(Program);
139  FORWARD(Pragma);
140 
141 #undef FORWARD
142 
143  // did not work ...
144 
145  fatal("unsupported type: %s", typeid(node).name());
146  }

149 #define LINK(Node, Parent) \
150  virtual R visit##Node(const Ast##Node& n, Params... args) { \

214 /**
215  * A utility function visiting all nodes within the ast rooted by the given node
216  * recursively in a depth-first pre-order fashion applying the given visitor to each
217  * encountered node.
218  *
219  * @param root the root of the AST to be visited
220  * @param visitor the visitor to be applied on each node
221  * @param args a list of extra parameters to be forwarded to the visitor
222  */
223 template <typename R, typename... Ps, typename... Args>
224 void visitDepthFirstPreOrder(const AstNode& root, AstVisitor<R, Ps...>& visitor, Args&... args) {
225  visitor(root, args...);
226  for (const AstNode* cur : root.getChildNodes()) {
227  if (cur != nullptr) {
228  visitDepthFirstPreOrder(*cur, visitor, args...);
229  }
230  }
231 }

268 /**
269  * A specialized visitor wrapping a lambda function -- an auxiliary type required
270  * for visitor convenience functions.
271  */
272 template <typename R, typename N>
273 struct LambdaAstVisitor : public AstVisitor<void> {
274  std::function<R(const N&)> lambda;
275  LambdaAstVisitor(std::function<R(const N&)> lambda) : lambda(std::move(lambda)) {}
276  void visit(const AstNode& node) override {
277  if (const auto* n = dynamic_cast<const N*>(&node)) {
278  lambda(*n);
279  }
280  }
281 };
