52  void setup() override {
53  // for each clause, store a mapping from the head relation name to body relation names
54  for (auto& rel : prog.getAllRelations()) {
55  std::string name = rel->getName();
56 
57  // only process info relations
58  if (name.find("@info") == std::string::npos) {
59  continue;
60  }
61 
62  // find all the info tuples
63  for (auto& tuple : *rel) {
64  std::vector<std::string> bodyLiterals;
65 
66  RamDomain ruleNum;
67  tuple >> ruleNum;
68 
69  for (size_t i = 1; i + 1 < rel->getArity(); i++) {
70  std::string bodyLit;
71  tuple >> bodyLit;
72  bodyLiterals.push_back(bodyLit);
73  }
74 
75  std::string rule;
76  tuple >> rule;
77 
78  info.insert({std::make_pair(name.substr(0, name.find(".@info")), ruleNum), bodyLiterals});
79  rules.insert({std::make_pair(name.substr(0, name.find(".@info")), ruleNum), rule});
80  }
81  }

84  std::unique_ptr<TreeNode> explain(
85  std::string relName, std::vector<RamDomain> tuple, int ruleNum, int levelNum, size_t depthLimit) {
86  std::stringstream joinedArgs;
87  joinedArgs << join(decodeArguments(relName, tuple), ", ");
88  auto joinedArgsStr = joinedArgs.str();
89 
90  // if fact
91  if (levelNum == 0) {
92  return std::make_unique<LeafNode>(relName + "(" + joinedArgsStr + ")");
93  }
94 
95  assert(info.find(std::make_pair(relName, ruleNum)) != info.end() && "invalid rule for tuple");
96 
97  // if depth limit exceeded
98  if (depthLimit <= 1) {
99  tuple.push_back(ruleNum);
100  tuple.push_back(levelNum);
101 
102  // find if subproof exists already
103  size_t idx = 0;
104  auto it = std::find(subproofs.begin(), subproofs.end(), tuple);
105  if (it != subproofs.end()) {
106  idx = it - subproofs.begin();
107  } else {
108  subproofs.push_back(tuple);
109  idx = subproofs.size() - 1;
110  }
111 
112  return std::make_unique<LeafNode>("subproof " + relName + "(" + std::to_string(idx) + ")");
113  }
114 
115  tuple.push_back(levelNum);
116 
117  auto internalNode = std::make_unique<InnerNode>(
118  relName + "(" + joinedArgsStr + ")", "(R" + std::to_string(ruleNum) + ")");
119 
120  // make return vector pointer
121  std::vector<RamDomain> ret;
122 
123  // execute subroutine to get subproofs
124  prog.executeSubroutine(relName + "_" + std::to_string(ruleNum) + "_subproof", tuple, ret);
125 
126  // recursively get nodes for subproofs
127  size_t tupleCurInd = 0;
128  auto bodyRelations = info[std::make_pair(relName, ruleNum)];
129 
130  // start from begin + 1 because the first element represents the head atom
131  for (auto it = bodyRelations.begin() + 1; it < bodyRelations.end(); it++) {
132  std::string bodyLiteral = *it;
133  // split bodyLiteral since it contains relation name plus arguments
134  std::string bodyRel = splitString(bodyLiteral, ',')[0];
135 
136  // check whether the current atom is a constraint
137  assert(bodyRel.size() > 0 && "body of a relation should have positive length");
138  bool isConstraint = contains(constraintList, bodyRel);
139 
140  // handle negated atom names
141  auto bodyRelAtomName = bodyRel;
142  if (bodyRel[0] == '!' && bodyRel != "!=") {
143  bodyRelAtomName = bodyRel.substr(1);
144  }
145 
146  // traverse subroutine return
147  size_t arity;
148  size_t auxiliaryArity;
149  if (isConstraint) {
150  // we only handle binary constraints, and assume arity is 4 to account for hidden provenance
151  // annotations
152  arity = 4;
153  auxiliaryArity = 2;
154  } else {
155  arity = prog.getRelation(bodyRelAtomName)->getArity();
156  auxiliaryArity = prog.getRelation(bodyRelAtomName)->getAuxiliaryArity();
157  }
158  auto tupleEnd = tupleCurInd + arity;
159 
160  // store current tuple
161  std::vector<RamDomain> subproofTuple;
162 
163  for (; tupleCurInd < tupleEnd - auxiliaryArity; tupleCurInd++) {
164  subproofTuple.push_back(ret[tupleCurInd]);
165  }
166 
167  int subproofRuleNum = ret[tupleCurInd];
168  int subproofLevelNum = ret[tupleCurInd + 1];
169 
170  tupleCurInd += 2;
171 
172  // for a negation, display the corresponding tuple and do not recurse
173  if (bodyRel[0] == '!' && bodyRel != "!=") {
174  std::stringstream joinedTuple;
175  joinedTuple << join(decodeArguments(bodyRelAtomName, subproofTuple), ", ");
176  auto joinedTupleStr = joinedTuple.str();
177  internalNode->add_child(std::make_unique<LeafNode>(bodyRel + "(" + joinedTupleStr + ")"));
178  internalNode->setSize(internalNode->getSize() + 1);
179  // for a binary constraint, display the corresponding values and do not recurse
180  } else if (isConstraint) {
181  std::stringstream joinedConstraint;
182 
183  // FIXME: We need type info in order to figure out how to print arguments.
184  BinaryConstraintOp rawBinOp = toBinaryConstraintOp(bodyRel);
185  if (isOrderedBinaryConstraintOp(rawBinOp)) {
186  joinedConstraint << subproofTuple[0] << " " << bodyRel << " " << subproofTuple[1];
187  } else {
188  joinedConstraint << bodyRel << "(\"" << symTable.resolve(subproofTuple[0]) << "\", \""
189  << symTable.resolve(subproofTuple[1]) << "\")";
190  }
191 
192  internalNode->add_child(std::make_unique<LeafNode>(joinedConstraint.str()));
193  internalNode->setSize(internalNode->getSize() + 1);
194  // otherwise, for a normal tuple, recurse
195  } else {
196  auto child =
197  explain(bodyRel, subproofTuple, subproofRuleNum, subproofLevelNum, depthLimit - 1);
198  internalNode->setSize(internalNode->getSize() + child->getSize());
199  internalNode->add_child(std::move(child));
200  }
201 
202  tupleCurInd = tupleEnd;
203  }
204 
205  return internalNode;
206  }

208  std::unique_ptr<TreeNode> explain(
209  std::string relName, std::vector<std::string> args, size_t depthLimit) override {
210  auto tuple = argsToNums(relName, args);
211  if (tuple.empty()) {
212  return std::make_unique<LeafNode>("Relation not found");
213  }
214 
215  std::tuple<int, int> tupleInfo = findTuple(relName, tuple);
216 
217  int ruleNum = std::get<0>(tupleInfo);
218  int levelNum = std::get<1>(tupleInfo);
219 
220  if (ruleNum < 0 || levelNum == -1) {
221  return std::make_unique<LeafNode>("Tuple not found");
222  }
223 
224  return explain(relName, tuple, ruleNum, levelNum, depthLimit);

248  std::vector<std::string> explainNegationGetVariables(
249  std::string relName, std::vector<std::string> args, size_t ruleNum) override {
250  std::vector<std::string> variables;
251 
252  // check that the tuple actually doesn't exist
253  std::tuple<int, int> foundTuple = findTuple(relName, argsToNums(relName, args));
254  if (std::get<0>(foundTuple) != -1 || std::get<1>(foundTuple) != -1) {
255  // return a sentinel value
256  return std::vector<std::string>({"@"});
257  }
258 
259  // atom meta information stored for the current rule
260  auto atoms = info[std::make_pair(relName, ruleNum)];
261 
262  // the info stores the set of atoms, if there is only 1 atom, then it must be the head, so it must be
263  // a fact
264  if (atoms.size() <= 1) {
265  return std::vector<std::string>({"@fact"});
266  }
267 
268  // atoms[0] represents variables in the head atom
269  auto headVariables = splitString(atoms[0], ',');
270 
271  auto isVariable = [&](std::string arg) {
272  return !(isNumber(arg.c_str()) || arg[0] == '\"' || arg == "_");
273  };
274 
275  // check that head variable bindings make sense, i.e. for a head like a(x, x), make sure both x are
276  // the same value
277  std::map<std::string, std::string> headVariableMapping;
278  for (size_t i = 0; i < headVariables.size(); i++) {
279  if (!isVariable(headVariables[i])) {
280  continue;
281  }
282 
283  if (headVariableMapping.find(headVariables[i]) == headVariableMapping.end()) {
284  headVariableMapping[headVariables[i]] = args[i];
285  } else {
286  if (headVariableMapping[headVariables[i]] != args[i]) {
287  return std::vector<std::string>({"@non_matching"});
288  }
289  }
290  }
291 
292  // get body variables
293  std::vector<std::string> uniqueBodyVariables;
294  for (auto it = atoms.begin() + 1; it < atoms.end(); it++) {
295  auto atomRepresentation = splitString(*it, ',');
296 
297  // atomRepresentation.begin() + 1 because the first element is the relation name of the atom
298  // which is not relevant for finding variables
299  for (auto atomIt = atomRepresentation.begin() + 1; atomIt < atomRepresentation.end(); atomIt++) {
300  if (!isVariable(*atomIt)) {
301  continue;
302  }
303 
304  if (!contains(uniqueBodyVariables, *atomIt) && !contains(headVariables, *atomIt)) {
305  uniqueBodyVariables.push_back(*atomIt);
306  }
307  }
308  }
309 
310  return uniqueBodyVariables;
311  }

313  std::unique_ptr<TreeNode> explainNegation(std::string relName, size_t ruleNum,
314  const std::vector<std::string>& tuple,
315  std::map<std::string, std::string>& bodyVariables) override {
316  // construct a vector of unique variables that occur in the rule
317  std::vector<std::string> uniqueVariables;
318 
319  // we also need to know the type of each variable
320  std::map<std::string, char> variableTypes;
321 
322  // atom meta information stored for the current rule
323  auto atoms = info[std::make_pair(relName, ruleNum)];
324 
325  // atoms[0] represents variables in the head atom
326  auto headVariables = splitString(atoms[0], ',');
327 
328  uniqueVariables.insert(uniqueVariables.end(), headVariables.begin(), headVariables.end());
329 
330  auto isVariable = [&](std::string arg) {
331  return !(isNumber(arg.c_str()) || arg[0] == '\"' || arg == "_");
332  };
333 
334  // get body variables
335  for (auto it = atoms.begin() + 1; it < atoms.end(); it++) {
336  auto atomRepresentation = splitString(*it, ',');
337 
338  // atomRepresentation.begin() + 1 because the first element is the relation name of the atom
339  // which is not relevant for finding variables
340  for (auto atomIt = atomRepresentation.begin() + 1; atomIt < atomRepresentation.end(); atomIt++) {
341  if (!contains(uniqueVariables, *atomIt) && !contains(headVariables, *atomIt)) {
342  // ignore non-variables
343  if (!isVariable(*atomIt)) {
344  continue;
345  }
346 
347  uniqueVariables.push_back(*atomIt);
348 
349  if (!contains(constraintList, atomRepresentation[0])) {
350  // store type of variable
351  auto currentRel = prog.getRelation(atomRepresentation[0]);
352  assert(currentRel != nullptr &&
353  ("relation " + atomRepresentation[0] + " doesn't exist").c_str());
354  variableTypes[*atomIt] =
355  *currentRel->getAttrType(atomIt - atomRepresentation.begin() - 1);
356  } else if (atomIt->find("agg_") != std::string::npos) {
357  variableTypes[*atomIt] = 'i';
358  }
359  }
360  }
361  }
362 
363  std::vector<RamDomain> args;
364 
365  size_t varCounter = 0;
366 
367  // construct arguments to pass in to the subroutine
368  // - this contains the variable bindings selected by the user
369 
370  // add number representation of tuple
371  auto tupleNums = argsToNums(relName, tuple);
372  args.insert(args.end(), tupleNums.begin(), tupleNums.end());
373  varCounter += tuple.size();
374 
375  while (varCounter < uniqueVariables.size()) {
376  auto var = uniqueVariables[varCounter];
377  auto varValue = bodyVariables[var];
378  if (variableTypes[var] == 's') {
379  if (varValue.size() >= 2 && varValue[0] == '"' && varValue[varValue.size() - 1] == '"') {
380  auto originalStr = varValue.substr(1, varValue.size() - 2);
381  args.push_back(symTable.lookup(originalStr));
382  } else {
383  // assume no quotation marks
384  args.push_back(symTable.lookup(varValue));
385  }
386  } else {
387  args.push_back(std::stoi(varValue));
388  }
389 
390  varCounter++;
391  }
392 
393  // set up return and error vectors for subroutine calling
394  std::vector<RamDomain> ret;
395 
396  // execute subroutine to get subproofs
397  prog.executeSubroutine(relName + "_" + std::to_string(ruleNum) + "_negation_subproof", args, ret);
398 
399  // ensure the subroutine returns the correct number of results
400  assert(ret.size() == atoms.size() - 1);
401 
402  // construct tree nodes
403  std::stringstream joinedArgsStr;
404  joinedArgsStr << join(tuple, ",");
405  auto internalNode = std::make_unique<InnerNode>(
406  relName + "(" + joinedArgsStr.str() + ")", "(R" + std::to_string(ruleNum) + ")");
407 
408  // store the head tuple in bodyVariables so we can print
409  for (size_t i = 0; i < headVariables.size(); i++) {
410  bodyVariables[headVariables[i]] = tuple[i];
411  }
412 
413  // traverse return vector and construct child nodes
414  // making sure we display existent and non-existent tuples correctly
415  int literalCounter = 1;
416  for (RamDomain returnCounter : ret) {
417  // check what the next contained atom is
418  bool atomExists = true;
419  if (returnCounter == 0) {
420  atomExists = false;
421  }
422 
423  // get the relation of the current atom
424  auto atomRepresentation = splitString(atoms[literalCounter], ',');
425  std::string bodyRel = atomRepresentation[0];
426 
427  // check whether the current atom is a constraint
428  bool isConstraint = contains(constraintList, bodyRel);
429 
430  // handle negated atom names
431  auto bodyRelAtomName = bodyRel;
432  if (bodyRel[0] == '!' && bodyRel != "!=") {
433  bodyRelAtomName = bodyRel.substr(1);
434  }
435 
436  // construct a label for a node containing a literal (either constraint or atom)
437  std::stringstream childLabel;
438  if (isConstraint) {
439  // for a binary constraint, display the corresponding values and do not recurse
440  assert(atomRepresentation.size() == 3 && "not a binary constraint");
441 
442  childLabel << bodyVariables[atomRepresentation[1]] << " " << bodyRel << " "
443  << bodyVariables[atomRepresentation[2]];
444  } else {
445  childLabel << bodyRel << "(";
446  for (size_t i = 1; i < atomRepresentation.size(); i++) {
447  // if it's a non-variable, print either _ for unnamed, or constant value
448  if (!isVariable(atomRepresentation[i])) {
449  childLabel << atomRepresentation[i];
450  } else {
451  childLabel << bodyVariables[atomRepresentation[i]];
452  }
453  if (i < atomRepresentation.size() - 1) {
454  childLabel << ", ";
455  }
456  }
457  childLabel << ")";
458  }
459 
460  // build a marker for existence of body atoms
461  if (atomExists) {
462  childLabel << " ";
463  } else {
464  childLabel << " x";
465  }
466 
467  internalNode->add_child(std::make_unique<LeafNode>(childLabel.str()));
468  internalNode->setSize(internalNode->getSize() + 1);
469 
470  literalCounter++;
471  }
472 
473  return internalNode;
474  }

499  std::string measureRelation(std::string relName) override {
500  auto rel = prog.getRelation(relName);
501 
502  if (rel == nullptr) {
503  return "No relation found\n";
504  }
505 
506  auto size = rel->size();
507  int skip = size / 10;
508 
509  if (skip == 0) {
510  skip = 1;
511  }
512 
513  std::stringstream ss;
514 
515  auto before_time = std::chrono::high_resolution_clock::now();
516 
517  int numTuples = 0;
518  int proc = 0;
519  for (auto& tuple : *rel) {
520  auto tupleStart = std::chrono::high_resolution_clock::now();
521 
522  if (numTuples % skip != 0) {
523  numTuples++;
524  continue;
525  }
526 
527  std::vector<RamDomain> currentTuple;
528  for (size_t i = 0; i < rel->getArity() - rel->getAuxiliaryArity(); i++) {
529  RamDomain n;
530  if (*rel->getAttrType(i) == 's') {
531  std::string s;
532  tuple >> s;
533  n = symTable.lookupExisting(s);
534  } else if (*rel->getAttrType(i) == 'f') {
535  RamFloat element;
536  tuple >> element;
537  n = ramBitCast(element);
538  } else if (*rel->getAttrType(i) == 'u') {
539  RamUnsigned element;
540  tuple >> element;
541  n = ramBitCast(element);
542  } else {
543  tuple >> n;
544  }
545 
546  currentTuple.push_back(n);
547  }
548 
549  RamDomain ruleNum;
550  tuple >> ruleNum;
551 
552  RamDomain levelNum;
553  tuple >> levelNum;
554 
555  std::cout << "Tuples expanded: "
556  << explain(relName, currentTuple, ruleNum, levelNum, 10000)->getSize();
557 
558  numTuples++;
559  proc++;
560 
561  auto tupleEnd = std::chrono::high_resolution_clock::now();
562  auto tupleDuration =
563  std::chrono::duration_cast<std::chrono::duration<double>>(tupleEnd - tupleStart);
564 
565  std::cout << ", Time: " << tupleDuration.count() << "\n";
566  }
567 
568  auto after_time = std::chrono::high_resolution_clock::now();
569  auto duration = std::chrono::duration_cast<std::chrono::duration<double>>(after_time - before_time);
570 
571  ss << "total: " << proc << " ";
572  ss << duration.count() << std::endl;
573 
574  return ss.str();
575  }

592  void queryProcess(const std::vector<std::pair<std::string, std::vector<std::string>>>& rels) override {
593  std::regex varRegex("[a-zA-Z_][a-zA-Z_0-9]*", std::regex_constants::extended);
594  std::regex symbolRegex("\"([^\"]*)\"", std::regex_constants::extended);
595  std::regex numberRegex("[0-9]+", std::regex_constants::extended);
596 
597  std::smatch argsMatcher;
598 
599  // map for variable name and corresponding equivalence class
600  std::map<std::string, Equivalence> nameToEquivalence;
601 
602  // const constraints that solution must satisfy
603  ConstConstraint constConstraints;
604 
605  // relations of tuples containing variables
606  std::vector<Relation*> varRels;
607 
608  // counter for adding element to varRels
609  size_t idx = 0;
610 
611  // parse arguments in each relation Tuple
612  for (const auto& rel : rels) {
613  Relation* relation = prog.getRelation(rel.first);
614  // number/symbol index for constant arguments in tuple
615  std::vector<RamDomain> constTuple;
616  // relation does not exist
617  if (relation == nullptr) {
618  std::cout << "Relation <" << rel.first << "> does not exist" << std::endl;
619  return;
620  }
621  // arity error
622  if (relation->getArity() - relation->getAuxiliaryArity() != rel.second.size()) {
623  std::cout << "<" + rel.first << "> has arity of "
624  << std::to_string(relation->getArity() - relation->getAuxiliaryArity())
625  << std::endl;
626  return;
627  }
628 
629  // check if args contain variable
630  bool containVar = false;
631  for (size_t j = 0; j < rel.second.size(); ++j) {
632  // arg is a variable
633  if (std::regex_match(rel.second[j], argsMatcher, varRegex)) {
634  containVar = true;
635  auto nameToEquivalenceIter = nameToEquivalence.find(argsMatcher[0]);
636  // if variable has not shown up before, create an equivalence class for add it to
637  // nameToEquivalence map, otherwise add its indices to corresponding equivalence class
638  if (nameToEquivalenceIter == nameToEquivalence.end()) {
639  nameToEquivalence.insert(
640  {argsMatcher[0], Equivalence(*(relation->getAttrType(j)), argsMatcher[0],
641  std::make_pair(idx, j))});
642  } else {
643  nameToEquivalenceIter->second.push_back(std::make_pair(idx, j));
644  }
645  continue;
646  }
647 
648  RamDomain rd;
649  switch (*(relation->getAttrType(j))) {
650  case 's':
651  if (!std::regex_match(rel.second[j], argsMatcher, symbolRegex)) {
652  std::cout << argsMatcher.str(0) << " does not match type defined in relation"
653  << std::endl;
654  return;
655  }
656  rd = prog.getSymbolTable().lookup(argsMatcher[1]);
657  break;
658  case 'f':
659  if (!canBeParsedAsRamFloat(rel.second[j])) {
660  std::cout << rel.second[j] << " does not match type defined in relation"
661  << std::endl;
662  return;
663  }
664  rd = ramBitCast(RamFloatFromString(rel.second[j]));
665  break;
666  case 'i':
667  if (!canBeParsedAsRamSigned(rel.second[j])) {

698  relation->getArity() +

700  constConstraints.getConstraints().end());

702  } else {

709  return;
710  }
711  } else {

713  ++idx;
714  }
715  }
716

718  // resolve, Output true and return
719  if (varRels.size() == 0) {
720  std::cout << "true." << std::endl;

788  /*
789  * Find solution for parameterised query satisfying constant constraints and equivalence constraints
790  * @param varRels, reference to vector of relation of tuple contains at least one variable in its
791  * arguments
792  * @param nameToEquivalence, reference to variable name and corresponding equivalence class
793  * @param constConstraints, reference to const constraints must be satisfied
794  * */
795  void findQuerySolution(const std::vector<Relation*>& varRels,
796  const std::map<std::string, Equivalence>& nameToEquivalence,
797  const ConstConstraint& constConstraints) {
798  // vector of iterators for relations in varRels
799  std::vector<Relation::iterator> varRelationIterators;
800  for (auto relation : varRels) {
801  varRelationIterators.push_back(relation->begin());
802  }
803 
804  size_t solutionCount = 0;
805  std::stringstream solution;
806 
807  // iterate through the vector of iterators to find solution
808  while (true) {
809  bool isSolution = true;
810 
811  // vector contains the tuples the iterators currently points to
812  std::vector<tuple> element;
813  for (auto it : varRelationIterators) {
814  element.push_back(*it);
815  }
816  // check if tuple satisfies variable equivalence
817  for (auto var : nameToEquivalence) {
818  if (!var.second.verify(element)) {
819  isSolution = false;
820  break;
821  }
822  }
823  if (isSolution) {
824  // check if tuple satisfies constant constraints
825  isSolution = constConstraints.verify(element);
826  }
827 
828  if (isSolution) {
829  std::cout << solution.str(); // print previous solution (if any)
830  solution.str(std::string()); // reset solution and process
831 
832  size_t c = 0;
833  for (auto&& var : nameToEquivalence) {
834  auto idx = var.second.getFirstIdx();
835  auto raw = element[idx.first][idx.second];
836 
837  solution << var.second.getSymbol() << " = ";
838  switch (var.second.getType()) {
839  case 'i': solution << ramBitCast<RamSigned>(raw); break;
840  case 'f': solution << ramBitCast<RamFloat>(raw); break;
841  case 'u': solution << ramBitCast<RamUnsigned>(raw); break;
842  case 's': solution << prog.getSymbolTable().resolve(raw); break;
843  default: fatal("invalid type: `%c`", var.second.getType());
844  }
845 
846  auto sep = ++c < nameToEquivalence.size() ? ", " : " ";
847  solution << sep;
848  }
849 
850  solutionCount++;
851  // query has more than one solution; query whether to find next solution or stop
852  if (1 < solutionCount) {
853  for (std::string input; getline(std::cin, input);) {
854  if (input == ";") break; // print next solution?
855  if (input == ".") return; // break from query?
856 
857  std::cout << "use ; to find next solution, use . to break from current query\n";
858  }
859  }
860  }
861 
862  // increment the iterators
863  size_t i = varRels.size() - 1;
864  bool terminate = true;
865  for (auto it = varRelationIterators.rbegin(); it != varRelationIterators.rend(); ++it) {
866  if ((++(*it)) != varRels[i]->end()) {
867  terminate = false;
868  break;
869  } else {
870  (*it) = varRels[i]->begin();
871  --i;
872  }
873  }
874 
875  if (terminate) {
876  // if there is no solution, output false
877  if (solutionCount == 0) {
878  std::cout << "false." << std::endl;
879  // otherwise print the last solution
880  } else {
881  std::cout << solution.str() << "." << std::endl;
882  }
883  break;
884  }
885  }
886  }
887 
888  // check if constTuple exists in relation
889  bool containsTuple(Relation* relation, const std::vector<RamDomain>& constTuple) {
890  bool tupleExist = false;
891  for (auto it = relation->begin(); it != relation->end(); ++it) {
892  bool eq = true;
893  for (size_t j = 0; j < constTuple.size(); ++j) {
894  if (constTuple[j] != (*it)[j]) {
895  eq = false;
896  break;
897  }
898  }
899  if (eq) {
900  tupleExist = true;
901  break;
902  }
903  }
904  return tupleExist;
905  }
