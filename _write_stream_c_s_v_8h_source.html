48  void writeNextTupleCSV(std::ostream& destination, const RamDomain* tuple) {
49  writeNextTupleElement(destination, typeAttributes.at(0), tuple[0]);
50 
51  for (size_t col = 1; col < arity; ++col) {
52  destination << delimiter;
53  writeNextTupleElement(destination, typeAttributes.at(col), tuple[col]);
54  }
55 
56  destination << "\n";
57  }
58 
59  void writeNextTupleElement(std::ostream& destination, const std::string& type, RamDomain value) {
60  switch (type[0]) {
61  case 's': destination << symbolTable.unsafeResolve(value); break;
62  case 'i': destination << value; break;
63  case 'u': destination << ramBitCast<RamUnsigned>(value); break;
64  case 'f': destination << ramBitCast<RamFloat>(value); break;
65  case 'r': outputRecord(destination, value, type); break;
66  default: fatal("unsupported type attribute: `%c`", type[0]);
67  }
68  }
69 };
