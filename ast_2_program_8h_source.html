86  /** remove relation */
87  bool removeRelation(const AstQualifiedName& name) {
88  for (auto it = relations.begin(); it != relations.end(); it++) {
89  const auto& rel = *it;
90  if (rel->getQualifiedName() == name) {
91  removeRelationClauses(*this, name);
92  removeRelationIOs(*this, name);
93  relations.erase(it);
94  return true;
95  }
96  }
97  return false;
98  }

111  /** remove a clause */
112  bool removeClause(const AstClause* clause) {
113  for (auto it = clauses.begin(); it != clauses.end(); it++) {
