30 /**
31  * @class ChoiceConversionTransformer
32  * @brief Convert (Scan/If)/(IndexScan/If) operaitons to
33  * (Choice)/(IndexChoice) operations
34 
35  * If there exists Scan/IndexScan operations in the RAM, and the
36  * variables are used in a subsequent Filter operation but no
37  * subsequent operation in the tree (up until and including
38  * the Project), the operations are rewritten to Choice/IndexChoice
39  * operations.
40  *
41  * For example,
42  *
43  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
44  * QUERY
45  * ...
46  * FOR t1 IN A ON INDEX t1.x=10 AND t1.y = 20
47  * IF (t1.x, t1.y) NOT IN A
48  * ... // no occurrence of t1
49  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
50  *
51  * will be rewritten to
52  *
53  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
54  * QUERY
55  * ...
56  * CHOICE A AS t1 ON INDEX t1.x=10 AND t1.y = 20
57  * WHERE (t1.x, t1.y) NOT IN A
58  * ...
59  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
60  *
61  */
62 class ChoiceConversionTransformer : public RamTransformer {
63 public:
64  std::string getName() const override {
65  return "ChoiceConversionTransformer";
66  }

73  * Rewrites Scan/If pair to a Choice operation if value

75  */

95  bool convertScans(RamProgram& program);
