84  /** create utilisation event */
85  void makeUtilisationEvent(const std::string& txt) {
86  /* current time */
87  microseconds time = std::chrono::duration_cast<microseconds>(now().time_since_epoch());
88  /* system CPU time used */
89  struct rusage ru {};
90  getrusage(RUSAGE_SELF, &ru);
91  /* system CPU time used */
92  uint64_t systemTime = ru.ru_stime.tv_sec * 1000000 + ru.ru_stime.tv_usec;
93  /* user CPU time used */
94  uint64_t userTime = ru.ru_utime.tv_sec * 1000000 + ru.ru_utime.tv_usec;
95  /* Maximum resident set size (kb) */
96  size_t maxRSS = ru.ru_maxrss;

140  class ProfileTimer {
141  private:
142  /** time interval between per utilisation read */
143  uint32_t t;
144 
145  /** timer is running */
146  std::atomic<bool> running{false};
147 
148  /** thread timer runs on */
149  std::thread th;

151  std::condition_variable conditionVariable;
152  std::mutex timerMutex;

154  /** number of utilisation events */
155  std::atomic<uint32_t> runCount{0};
156 
157  /** run method for thread th */
158  void run() {
159  ProfileEventSingleton::instance().makeUtilisationEvent("@utilisation");
160  ++runCount;
161  if (runCount % 128 == 0) {
162  increaseInterval();
163  }
164  }

168  }
169 
170  /** Increase value of time interval by factor of 2 */
171  void increaseInterval() {
172  // Don't increase time interval past 60 seconds
173  if (t < 60000) {
174  t = t * 2;
175  }
176  }
177 
178  public:
179  /*
180  * @param interval the size of the timing interval in milliseconds
181  */
182  ProfileTimer(uint32_t interval = 10) : t(interval) {}
183 
184  /** start timer on the thread th */
185  void start() {
186  if (running) {
187  return;
188  }
189  running = true;
190 
191  th = std::thread([this]() {
192  while (running) {
193  run();
194  std::unique_lock<std::mutex> lock(timerMutex);
195  conditionVariable.wait_for(lock, std::chrono::milliseconds(getInterval()));
196  }
197  });
198  }
199 
200  /** stop timer on the thread th */
201  void stop() {
202  running = false;
203  conditionVariable.notify_all();
204  if (th.joinable()) {
205  th.join();
206  }
207  }
208 
209  /** Reset timer interval.
210  *
211  * The timer interval increases as the program executes. Resetting the interval is useful to
212  * ensure that detailed usage information is gathered even in long running programs, if desired.
213  *
214  * @param interval the size of the timing interval in milliseconds
215  */
216  void resetTimerInterval(uint32_t interval = 10) {
217  t = interval;
218  runCount = 0;

220  }
221  };
