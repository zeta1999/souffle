343  /** @brief Helper method for printing */
344  void printIndex(std::ostream& os) const {
345  // const auto& attrib = getRelation().getAttributeNames();
346  bool first = true;
347  for (unsigned int i = 0; i < getRelation().getArity(); ++i) {
348  // TODO: print proper upper lower/bound
349 
350  // early exit if no upper/lower bounds are defined
351  if (isRamUndefValue(queryPattern.first[i].get()) &&
352  isRamUndefValue(queryPattern.second[i].get())) {
353  continue;
354  }
355 
356  if (first) {
357  os << " ON INDEX ";
358  first = false;
359  } else {
360  os << " AND ";
361  }
362 
363  // both bounds defined and equal => equality
364  if (!isRamUndefValue(queryPattern.first[i].get()) &&
365  !isRamUndefValue(queryPattern.second[i].get())) {
366  // print equality when lower bound = upper bound
367  if (*(queryPattern.first[i]) == *(queryPattern.second[i])) {
368  os << "t" << getTupleId() << ".";
369  os << i << " = ";
370  os << *(queryPattern.first[i]);
371  continue;
372  }
373  }
374  // at least one bound defined => inequality
375  if (!isRamUndefValue(queryPattern.first[i].get()) ||
376  !isRamUndefValue(queryPattern.second[i].get())) {
377  if (!isRamUndefValue(queryPattern.first[i].get())) {
378  os << *(queryPattern.first[i]) << " <= ";
379  }
380 
381  os << "t" << getTupleId() << ".";
382  os << i;
383 
384  if (!isRamUndefValue(queryPattern.second[i].get())) {
385  os << " <= " << *(queryPattern.second[i]);
386  }
387 
388  continue;
389  }
390  }
391  }
