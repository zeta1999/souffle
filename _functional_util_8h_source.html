28 /**
29  * A functor comparing the dereferenced value of a pointer type utilizing a
30  * given comparator. Its main use case are sets of non-null pointers which should
31  * be ordered according to the value addressed by the pointer.
32  */

34 struct deref_less {
35  bool operator()(const T* a, const T* b) const {
36  return C()(*a, *b);

92 /**
93  * A generic test checking whether all elements within a container satisfy a
94  * certain predicate.
95  *
96  * @param c the container
97  * @param p the predicate
98  * @return true if for all elements x in c the predicate p(x) is true, false
99  * otherwise; for empty containers the result is always true
100  */
101 template <typename Container, typename UnaryPredicate>
102 bool all_of(const Container& c, UnaryPredicate p) {

134 /**
135  * Filter a vector to exclude certain elements.
136  */
137 template <typename A, typename F>
138 std::vector<A> filterNot(std::vector<A> xs, F&& f) {
139  xs.erase(std::remove_if(xs.begin(), xs.end(), std::forward<F>(f)), xs.end());
