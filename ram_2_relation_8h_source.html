35 class RamRelation : public RamNode {
36 public:
37  RamRelation(std::string name, size_t arity, size_t auxiliaryArity,
38  std::vector<std::string> attributeNames, std::vector<std::string> attributeTypes,
39  RelationRepresentation representation)
40  : representation(representation), name(std::move(name)), arity(arity),
41  auxiliaryArity(auxiliaryArity), attributeNames(std::move(attributeNames)),
42  attributeTypes(std::move(attributeTypes)) {
43  assert(this->attributeNames.size() == arity && "arity mismatch for attributes");
44  assert(this->attributeTypes.size() == arity && "arity mismatch for types");
45  for (std::size_t i = 0; i < arity; i++) {
46  assert(!this->attributeNames[i].empty() && "no attribute name specified");
47  assert(!this->attributeTypes[i].empty() && "no attribute type specified");
48  }
49  }
