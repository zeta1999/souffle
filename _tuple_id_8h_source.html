25 /**
26  * @class TupleIdTransformer
27  * @brief Ordering tupleIds in RamTupleOperation operations correctly
28  *
29  * Transformations, like MakeIndex and IfConversion do not
30  * ensure that RamTupleOperations maintain an appropriate order
31  * with respect to their tupleId's
32  *
33  * For example:
34  * SEARCH ... (tupleId = 2)
35  * ...
36  * SEARCH ... (tupleId = 1)
37  * ...
38  *
39  * Will be converted to
40  * SEARCH ... (tupleId = 0)
41  * ...
42  * SEARCH ... (tupleId = 1)
43  * ...
44  *
45  */
46 class TupleIdTransformer : public RamTransformer {
47 public:
48  std::string getName() const override {
49  return "TupleIdTransformer";
50  }
51 
52  /**
53  * @brief Apply tupleId reordering to the whole program
54  * @param RAM program
55  * @result A flag indicating whether the RAM program has been changed.
56  *
57  * Search for RamTupleOperations and RamTupleElements and rewrite their tupleIds
58  */
59  bool reorderOperations(RamProgram& program);
60 
61 protected:
62  bool transform(RamTranslationUnit& translationUnit) override {
63  return reorderOperations(translationUnit.getProgram());
64  }
