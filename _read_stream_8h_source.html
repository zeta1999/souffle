53  /**
54  * Read a record from a string.
55  *
56  * @param source - string containing a record
57  * @param recordTypeName - record type.
58  * @parem pos - start parsing from this position.
59  * @param consumed - if not nullptr: number of characters read.
60  *
61  */
62  RamDomain readRecord(const std::string& source, const std::string& recordTypeName, size_t pos = 0,
63  size_t* charactersRead = nullptr) {
64  const size_t initial_position = pos;
65 
66  // Check if record type information are present
67  auto&& recordInfo = types["records"][recordTypeName];
68  if (recordInfo.is_null()) {
69  throw std::invalid_argument("Missing record type information: " + recordTypeName);
70  }
71 
72  // Handle nil case
73  consumeWhiteSpace(source, pos);
74  if (source.substr(pos, 3) == "nil") {
75  if (charactersRead != nullptr) {
76  *charactersRead = 3;
77  }
78  return 0;
79  }
80 
81  auto&& recordTypes = recordInfo["types"];
82  const size_t recordArity = recordInfo["arity"].long_value();
83 
84  std::vector<RamDomain> recordValues(recordArity);
85 
86  consumeChar(source, '[', pos);
87 
88  for (size_t i = 0; i < recordArity; ++i) {
89  const std::string& recordType = recordTypes[i].string_value();
90  size_t consumed = 0;
91 
92  if (i > 0) {
93  consumeChar(source, ',', pos);
94  }
95  consumeWhiteSpace(source, pos);
96  switch (recordType[0]) {
97  case 's': {
98  recordValues[i] = readStringInRecord(source, pos, &consumed);
99  break;
100  }
101  case 'i': {
102  recordValues[i] = RamSignedFromString(source.substr(pos), &consumed);
103  break;
104  }
105  case 'u': {
106  recordValues[i] = ramBitCast(RamUnsignedFromString(source.substr(pos), &consumed));
107  break;
108  }
109  case 'f': {
110  recordValues[i] = ramBitCast(RamFloatFromString(source.substr(pos), &consumed));
111  break;
112  }
113  case 'r': {
114  recordValues[i] = readRecord(source, recordType, pos, &consumed);
115  break;
116  }
117  default: fatal("Invalid type attribute");
118  }
119  pos += consumed;
120  }
121  consumeChar(source, ']', pos);
122 
123  if (charactersRead != nullptr) {
124  *charactersRead = pos - initial_position;
125  }
126 
127  return recordTable.pack(recordValues.data(), recordValues.size());
128  }

130  RamDomain readStringInRecord(const std::string& source, const size_t pos, size_t* charactersRead) {
131  size_t endOfSymbol = source.find_first_of(",]", pos);
132 
133  if (endOfSymbol == std::string::npos) {
134  throw std::invalid_argument("Unexpected end of input in record");
135  }
136 
137  *charactersRead = endOfSymbol - pos;
138  std::string str = source.substr(pos, *charactersRead);
139 
140  return symbolTable.unsafeLookup(str);
141  }

143  /**

145  */
146  void consumeChar(const std::string& str, char c, size_t& pos) {
147  consumeWhiteSpace(str, pos);
148  if (pos >= str.length()) {
149  throw std::invalid_argument("Unexpected end of input in record");
150  }
151  if (str[pos] != c) {
152  std::stringstream error;
153  error << "Expected: \'" << c << "\', got: " << str[pos];
154  throw std::invalid_argument(error.str());
155  }
156  ++pos;
157  }

163  while (pos < str.length() && std::isspace(static_cast<unsigned char>(str[pos]))) {
