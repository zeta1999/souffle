80 template <typename R>
81 struct TypeVisitor {
82  virtual ~TypeVisitor() = default;
83 
84  R operator()(const Type& type) const {
85  return visit(type);
86  }
87 
88 #define FORWARD(TYPE) \
89  if (auto* t = dynamic_cast<const TYPE##Type*>(&type)) return visit##TYPE##Type(*t);
90 
91  virtual R visit(const Type& type) const {
92  FORWARD(Constant);
93  FORWARD(Subset);
94  FORWARD(Union);
95  FORWARD(Record);
96 
97  fatal("Unsupported type encountered!");
98  }
99 #undef FORWARD
100 
101 #define VISIT(TYPE) \
102  virtual R visit##TYPE##Type(const TYPE##Type& type) const { \
103  return visitType(type); \
104  }
105 
106  VISIT(Constant)
107  VISIT(Subset)
108  VISIT(Union)
109  VISIT(Record)
110 
111  virtual R visitType(const Type& /*type*/) const {
112  return R();
113  }
114 
115 #undef VISIT
116 };

118 /**
119  * A visitor for types visiting each type only once (effectively breaking
120  * recursive cycles).
121  */
122 template <typename R>
123 class VisitOnceTypeVisitor : public TypeVisitor<R> {
124 protected:
125  mutable std::map<const Type*, R> seen;
126 
127 public:
128  R visit(const Type& type) const override {
129  auto pos = seen.find(&type);
130  if (pos != seen.end()) {
131  return pos->second;
132  }
133  auto& res = seen[&type]; // mark as seen
134  return res = TypeVisitor<R>::visit(type);
135  }
136 };

138 /**
139  * Determines whether the given type is a sub-type of the given root type.
140  */
141 bool isOfRootType(const Type& type, const Type& root) {
142  struct visitor : public VisitOnceTypeVisitor<bool> {
143  const Type& root;
144 
145  explicit visitor(const Type& root) : root(root) {}
146 
147  bool visitConstantType(const ConstantType& type) const override {
148  return type == root;
149  }
150  bool visitSubsetType(const SubsetType& type) const override {
151  return type == root || isOfRootType(type.getBaseType(), root);
152  }
153  bool visitUnionType(const UnionType& type) const override {
154  return type == root ||
155  all_of(type.getElementTypes(), [&](const Type* cur) { return this->visit(*cur); });
156  }
157 
158  bool visitRecordType(const RecordType& type) const override {
159  return type == root;
160  }
161 
162  bool visitType(const Type& /*unused*/) const override {
163  return false;
164  }
165  };
166 
167  return visitor(root).visit(type);
168 }

182 std::string getTypeQualifier(const Type& type) {
183  struct visitor : public VisitOnceTypeVisitor<std::string> {
184  std::string visitUnionType(const UnionType& type) const override {
185  return tfm::format("%s[%s]", visitType(type),
186  join(type.getElementTypes(), ", ",
187  [&](std::ostream& out, const auto* elementType) { out << visit(*elementType); }));
188  }
189 
190  std::string visitRecordType(const RecordType& type) const override {
191  return tfm::format("%s{%s}", visitType(type),
192  join(type.getFields(), ", ",
193  [&](std::ostream& out, const auto* field) { out << visit(*field); }));
194  }
195 
196  std::string visitType(const Type& type) const override {
197  std::string str;
198 
199  if (isOfKind(type, TypeAttribute::Signed)) {
200  str.append("i");
201  } else if (isOfKind(type, TypeAttribute::Unsigned)) {
202  str.append("u");
203  } else if (isOfKind(type, TypeAttribute::Float)) {
204  str.append("f");
205  } else if (isOfKind(type, TypeAttribute::Symbol)) {
206  str.append("s");
207  } else if (isOfKind(type, TypeAttribute::Record)) {
208  str.append("r");
209  } else {
210  fatal("Unsupported kind");
211  }
212 
213  str.append(":");
214  str.append(toString(type.getName()));
215  seen[&type] = str;
216  return str;
217  }
218  };
219 
220  return visitor().visit(type);
221 }

223 bool isSubtypeOf(const Type& a, const Type& b) {
224  assert(&a.getTypeEnvironment() == &b.getTypeEnvironment() &&
225  "Types must be in the same type environment");
226 
227  if (isOfRootType(a, b)) {
228  return true;
229  }
230 
231  if (isA<UnionType>(a)) {
232  return all_of(static_cast<const UnionType&>(a).getElementTypes(),
233  [&b](const Type* type) { return isSubtypeOf(*type, b); });
234  }
235 
236  if (isA<UnionType>(b)) {
237  return any_of(static_cast<const UnionType&>(b).getElementTypes(),
238  [&a](const Type* type) { return isSubtypeOf(a, *type); });
239  }
240 
241  return false;
242 }

251 TypeSet getGreatestCommonSubtypes(const Type& a, const Type& b) {
252  assert(&a.getTypeEnvironment() == &b.getTypeEnvironment() &&
253  "Types must be in the same type environment");
254 
255  if (isSubtypeOf(a, b)) {
256  return TypeSet(a);
257  }
258  if (isSubtypeOf(b, a)) {
259  return TypeSet(b);
260  }
261 
262  TypeSet res;
263  if (isA<UnionType>(a) && isA<UnionType>(b)) {
264  // collect common sub-types of union types
265  struct collector : public TypeVisitor<void> {
266  const Type& b;
267  TypeSet& res;
268  collector(const Type& b, TypeSet& res) : b(b), res(res) {}
269 
270  void visit(const Type& type) const override {
271  if (isSubtypeOf(type, b)) {
272  res.insert(type);
273  } else {
274  TypeVisitor<void>::visit(type);
275  }
276  }
277  void visitUnionType(const UnionType& type) const override {
278  for (const auto& cur : type.getElementTypes()) {
279  visit(*cur);
280  }
281  }
282  };
283 
284  // collect all common sub-types
285  collector(b, res).visit(a);
286  }
287 
288  // otherwise there is no common super type
289  return res;
290 }

292 TypeSet getGreatestCommonSubtypes(const TypeSet& set) {
293  // Edge cases.
294  if (set.empty() || set.isAll()) {
295  return TypeSet();
296  }
297 
298  TypeSet greatestCommonSubtypes;
299  greatestCommonSubtypes.insert(*set.begin());
300 
301  for (auto& type : set) {
302  greatestCommonSubtypes = getGreatestCommonSubtypes(TypeSet(type), greatestCommonSubtypes);
303  }

308 TypeSet getGreatestCommonSubtypes(const TypeSet& a, const TypeSet& b) {
309  // special cases
310  if (a.empty()) {
311  return a;
312  }
313  if (b.empty()) {
314  return b;
315  }
316 
317  if (a.isAll()) {
318  return b;
319  }
320  if (b.isAll()) {
321  return a;
322  }
323 
324  // compute pairwise greatest common sub types
325  TypeSet res;
326  for (const Type& x : a) {
327  for (const Type& y : b) {
328  res.insert(getGreatestCommonSubtypes(x, y));
329  }
330  }

334 bool haveCommonSupertype(const Type& a, const Type& b) {
335  assert(&a.getTypeEnvironment() == &b.getTypeEnvironment() &&
336  "Types must be in the same type environment");
337 
338  if (a == b) {
339  return true;
340  }
341 
342  if (isSubtypeOf(a, b) || isSubtypeOf(b, a)) {
343  return true;
344  }
345 
346  return any_of(a.getTypeEnvironment().getTypes(),
347  [&](const Type& type) { return isSubtypeOf(a, type) && isSubtypeOf(b, type); });
348 }
