48 const AstComponent* AstComponentChecker::checkComponentNameReference(ErrorReport& report,
49  const AstComponent* enclosingComponent, const ComponentLookup& componentLookup,
50  const std::string& name, const SrcLocation& loc, const TypeBinding& binding) {
51  const AstQualifiedName& forwarded = binding.find(name);
52  if (!forwarded.empty()) {
53  // for forwarded types we do not check anything, because we do not know
54  // what the actual type will be
55  return nullptr;
56  }
57 
58  const AstComponent* c = componentLookup.getComponent(enclosingComponent, name, binding);
59  if (c == nullptr) {
60  report.addError("Referencing undefined component " + name, loc);
61  return nullptr;
62  }
63 
64  return c;
65 }

99 void AstComponentChecker::checkComponent(ErrorReport& report, const AstComponent* enclosingComponent,
100  const ComponentLookup& componentLookup, const AstComponent& component, const TypeBinding& binding) {
101  // -- inheritance --
102 
103  // Update type binding:
104  // Since we are not compiling, i.e. creating concrete instance of the
105  // components with type parameters, we are only interested in whether
106  // component references refer to existing components or some type parameter.
107  // Type parameter for us here is unknown type that will be bound at the template
108  // instantiation time.
109  auto parentTypeParameters = component.getComponentType()->getTypeParameters();
110  std::vector<AstQualifiedName> actualParams(parentTypeParameters.size(), "<type parameter>");
111  TypeBinding activeBinding = binding.extend(parentTypeParameters, actualParams);
112 
113  // check parents of component
114  for (const auto& cur : component.getBaseComponents()) {
115  checkComponentReference(
116  report, enclosingComponent, componentLookup, *cur, component.getSrcLoc(), activeBinding);
117 
118  // Note: type parameters can also be atomic types like number, or anything defined through .type
119  // The original semantic check permitted any identifier (existing or non-existing) to be actual
120  // parameter
121  // In order to maintain the compatibility, we do not check the actual parameters
122 
123  // for (const std::string& param : cur.getTypeParameters()) {
124  // checkComponentNameReference(report, scope, param, component.getSrcLoc(), activeBinding);
125  //}
126  }
127 
128  // get all parents
129  std::set<const AstComponent*> parents;
130  std::function<void(const AstComponent&)> collectParents = [&](const AstComponent& cur) {
131  for (const auto& base : cur.getBaseComponents()) {
132  auto c = componentLookup.getComponent(enclosingComponent, base->getName(), binding);
133  if (c == nullptr) {
134  continue;
135  }
136  if (parents.insert(c).second) {
137  collectParents(*c);
138  }
139  }
140  };
141  collectParents(component);
142 
143  // check overrides
144  for (const AstRelation* relation : component.getRelations()) {
145  if (component.getOverridden().count(relation->getQualifiedName().getQualifiers()[0]) != 0u) {
146  report.addError("Override of non-inherited relation " +
147  relation->getQualifiedName().getQualifiers()[0] + " in component " +
148  component.getComponentType()->getName(),
149  component.getSrcLoc());
150  }
151  }
152  for (const AstComponent* parent : parents) {
153  for (const AstRelation* relation : parent->getRelations()) {
154  if ((component.getOverridden().count(relation->getQualifiedName().getQualifiers()[0]) != 0u) &&
155  !relation->hasQualifier(RelationQualifier::OVERRIDABLE)) {
156  report.addError("Override of non-overridable relation " +
157  relation->getQualifiedName().getQualifiers()[0] + " in component " +
158  component.getComponentType()->getName(),
159  component.getSrcLoc());
160  }
161  }
162  }
163 
164  // check for a cycle
165  if (parents.find(&component) != parents.end()) {
166  report.addError(
167  "Invalid cycle in inheritance for component " + component.getComponentType()->getName(),
168  component.getSrcLoc());
169  }
170 
171  // -- nested components --
172 
173  // check nested components
174  for (const auto& cur : component.getComponents()) {
175  checkComponent(report, &component, componentLookup, *cur, activeBinding);
176  }
177 
178  // check nested instantiations
179  for (const auto& cur : component.getInstantiations()) {
180  checkComponentInit(report, &component, componentLookup, *cur, activeBinding);
181  }
182 }

195 // Check that component names are disjoint from type and relation names.
196 void AstComponentChecker::checkComponentNamespaces(ErrorReport& report, const AstProgram& program) {
197  std::map<std::string, SrcLocation> names;
198 
199  // Type and relation name error reporting performed by the AstSemanticChecker instead
200 
201  // Find all names and report redeclarations as we go.
202  for (const auto& type : program.getTypes()) {
203  const std::string name = toString(type->getQualifiedName());
204  if (names.count(name) == 0u) {
205  names[name] = type->getSrcLoc();
206  }
207  }
208 
209  for (const auto& rel : program.getRelations()) {
210  const std::string name = toString(rel->getQualifiedName());
211  if (names.count(name) == 0u) {
212  names[name] = rel->getSrcLoc();
213  }
214  }
215 
216  // Note: Nested component and instance names are not obtained.
217  for (const auto& comp : program.getComponents()) {
218  const std::string name = toString(comp->getComponentType()->getName());
219  if (names.count(name) != 0u) {
220  report.addError("Name clash on component " + name, comp->getSrcLoc());
221  } else {
222  names[name] = comp->getSrcLoc();
223  }
224  }
225 
226  for (const auto& inst : program.getComponentInstantiations()) {
227  const std::string name = toString(inst->getInstanceName());
228  if (names.count(name) != 0u) {
229  report.addError("Name clash on instantiation " + name, inst->getSrcLoc());
230  } else {
231  names[name] = inst->getSrcLoc();
232  }
233  }
234 }
