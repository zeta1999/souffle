31 bool EliminateDuplicatesTransformer::eliminateDuplicates(RamProgram& program) {
32  bool changed = false;
33  visitDepthFirst(program, [&](const RamQuery& query) {
34  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
35  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
36  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
37  const RamCondition* condition = &filter->getCondition();
38  std::vector<std::unique_ptr<RamCondition>> conds = toConjunctionList(condition);
39  bool eliminatedDuplicate = false;
40  for (std::size_t i = 0; i < conds.size(); i++) {
41  for (std::size_t j = i + 1; j < conds.size(); j++) {
42  if (*conds[i] == *conds[j]) {
43  conds.erase(conds.begin() + j);
44  i = -1;
45  eliminatedDuplicate = true;
46  break;
47  }
48  }
49  }
50  if (eliminatedDuplicate) {
51  changed = true;
52  node = std::make_unique<RamFilter>(std::unique_ptr<RamCondition>(toCondition(conds)),
53  souffle::clone(&filter->getOperation()));
54  }
55  }
56  node->apply(makeLambdaRamMapper(filterRewriter));
57  return node;
58  };
59  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
60  });
61  return changed;
62 }
