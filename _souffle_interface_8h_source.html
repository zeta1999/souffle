393 /**
394  * Defines a tuple for the OO interface such that
395  * relations with varying columns can be accessed.
396  *
397  * Tuples are stored in relations.
398  * In Souffle, one row of data to be stored into a relation is represented as a tuple.
399  * For example if we have a relation called dog with attributes name, colour and age which are string, string
400  * and interger type respectively. One row of data a relation to be stored can be (mydog, black, 3). However,
401  * this is not directly stored as a tuple. There will be a symbol table storing the actual content and
402  * associate them with numbers (For example, |1|mydog| |2|black| |3|3|). And when this row of data is stored
403  * as a tuple, (1, 2, 3) will be stored.
404  */
405 class tuple {
406  /**
407  * The relation to which the tuple belongs.
408  */

411  /**
412  * Dynamic array used to store the elements in a tuple.
413  */

415 
416  /**
417  * pos shows what the current position of a tuple is.
418  * Initially, pos is 0 meaning we are at the head of the tuple.
419  * If we have an empty tuple and try to insert things, pos lets us know where to insert the element. After
420  * the element is inserted, pos will be incremented by 1. If we have a tuple with content, pos lets us
421  * know where to read the element. After we have read one element, pos will be incremented by 1. pos also
422  * helps to make sure we access an insert a tuple within the bound by making sure pos never exceeds the
423  * arity of the relation.
424  */

427 public:

434  * element is of integer type, it will be stored directly within the vector. Otherwise, if the element is
435  * of a string type, the index of that string within the associated symbol table will be stored instead.
436  * The tuple also stores the index of some "current" element, referred to as its position. The constructor
437  * initially sets this position to the first (zeroth) element of the tuple, while subsequent methods of

679  /**

682  std::vector<Relation*> allRelations;
683  /**

686  std::size_t numThreads = 1;

688 protected:
689  /**
690  * Add the relation to relationMap (with its name) and allRelations,
691  * depends on the properties of the relation, if the relation is an input relation, it will be added to
692  * inputRelations, else if the relation is an output relation, it will be added to outputRelations,
693  * otherwise will add to internalRelations. (a relation could be both input and output at the same time.)
694  *
695  * @param name the name of the relation (std::string)
696  * @param rel a pointer to the relation (std::string)
697  * @param isInput a bool argument, true if the relation is a input relation, else false (bool)
698  * @param isOnput a bool argument, true if the relation is a ouput relation, else false (bool)
699  */
700  void addRelation(const std::string& name, Relation* rel, bool isInput, bool isOutput) {
701  relationMap[name] = rel;
702  allRelations.push_back(rel);
703  if (isInput) {
704  inputRelations.push_back(rel);
705  }
706  if (isOutput) {
707  outputRelations.push_back(rel);
708  }
709  if (!isInput && !isOutput) {
710  internalRelations.push_back(rel);
711  }
712  }

954  * Abstract program factory class.
955  */
956 class ProgramFactory {
957 protected:
958  /**
959  * Singly linked-list to store all program factories
960  * Note that STL data-structures are not possible due
961  * to "static initialization order fiasco (problem)".
962  * (The problem of the order static objects get initialized, causing effect
963  * such as program access static variables before they initialized.)
964  * The static container needs to be a primitive type such as pointer
965  * set to NULL.
966  * Link to next factory.
967  */
968  ProgramFactory* link = nullptr;
969 
970  /**
971  * The name of factory.
972  */
973  std::string name;
974 
975 protected:
976  /**
977  * Constructor.
978  *
979  * Constructor adds factory to static singly-linked list
980  * for registration.
981  */
982  ProgramFactory(std::string name) : name(std::move(name)) {
983  registerFactory(this);
984  }
985 
986 private:
987  /**
988  * Helper method for creating a factory map, which map key is the name of the program factory, map value
989  * is the pointer of the ProgramFactory.
990  *
991  * TODO (NubKel) : Improve documentation of use and interaction between inline and static, here and for
992  * the whole class.
993  *
994  * @return The factory registration map (std::map)
995  */
996  static inline std::map<std::string, ProgramFactory*>& getFactoryRegistry() {
997  static std::map<std::string, ProgramFactory*> factoryReg;
998  return factoryReg;
999  }
1000 
1001 protected:
1002  /**
1003  * Create and insert a factory into the factoryReg map.
1004  *
1005  * @param factory Pointer of the program factory (ProgramFactory*)
1006  */
1007  static inline void registerFactory(ProgramFactory* factory) {
1008  auto& entry = getFactoryRegistry()[factory->name];
1009  assert(!entry && "double-linked/defined souffle analyis");
1010  entry = factory;
1011  }
1012 
1013  /**
1014  * Find a factory by its name, return the fatory if found, return nullptr if the
1015  * factory not found.
1016  *
1017  * @param factoryName The factory name (const std::string)
1018  * @return The pointer of the target program factory, or null pointer if the program factory not found
1019  * (ProgramFactory*)
1020  */
1021  static inline ProgramFactory* find(const std::string& factoryName) {
1022  const auto& reg = getFactoryRegistry();
1023  auto pos = reg.find(factoryName);
1024  return (pos == reg.end()) ? nullptr : pos->second;
1025  }

1027  /**
1028  * Create new instance (abstract).
1029  */
1030  virtual SouffleProgram* newInstance() = 0;
1031 
1032 public:
1033  /**
1034  * Destructor.
1035  *
1036  * Destructor of ProgramFactory.
1037  */
1038  virtual ~ProgramFactory() = default;
1039 
1040  /**
1041  * Create an instance by finding the name of the program factory, return nullptr if the instance not
1042  * found.
1043  *
1044  * @param name Instance name (const std::string)
1045  * @return The new instance(SouffleProgram*), or null pointer if the instance not found
1046  */
1047  static SouffleProgram* newInstance(const std::string& name) {
1048  ProgramFactory* factory = find(name);
1049  if (factory != nullptr) {
1050  return factory->newInstance();
1051  } else {
1052  return nullptr;
1053  }
1054  }
