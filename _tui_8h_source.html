108  void runCommand(std::vector<std::string> c) {
109  if (linereader.hasReceivedInput() && c.empty()) {
110  return;
111  }
112  if (!loaded) {
113  std::cout << "Error: File cannot be loaded\n";
114  return;
115  }
116 
117  if (alive) {
118  updateDB();
119  // remake tables to get new data
120  ruleTable = out.getRulTable();
121  relationTable = out.getRelTable();
122 
123  setupTabCompletion();
124  }
125 
126  // If we have not received any input yet in live mode then run top.
127  if ((!linereader.hasReceivedInput() && c.empty())) {
128  // Move up n lines and overwrite the previous top output.
129  std::cout << "\x1b[3D";
130  std::cout << "\x1b[27A";
131  top();
132  std::cout << "\x1b[B> ";
133  } else if (c[0] == "top") {
134  top();
135  } else if (c[0] == "rel") {
136  if (c.size() == 2) {
137  relRul(c[1]);
138  } else if (c.size() == 1) {
139  rel(resultLimit);
140  } else {
141  std::cout << "Invalid parameters to rel command.\n";
142  }
143  } else if (c[0] == "rul") {
144  if (c.size() > 1) {
145  if (c.size() == 3 && c[1] == "id") {
146  std::printf("%7s%2s%s\n\n", "ID", "", "NAME");
147  id(c[2]);
148  } else if (c.size() == 2 && c[1] == "id") {
149  id("0");
150  } else if (c.size() == 2) {
151  verRul(c[1]);
152  } else {
153  std::cout << "Invalid parameters to rul command.\n";
154  }
155  } else {
156  rul(resultLimit);
157  }
158  } else if (c[0] == "graph") {
159  if (c.size() == 3 && c[1].find(".") == std::string::npos) {
160  iterRel(c[1], c[2]);
161  } else if (c.size() == 3 && c[1].at(0) == 'C') {
162  iterRul(c[1], c[2]);
163  } else if (c.size() == 4 && c[1] == "ver" && c[2].at(0) == 'C') {
164  verGraph(c[2], c[3]);
165  } else {
166  std::cout << "Invalid parameters to graph command.\n";
167  }
168  } else if (c[0] == "memory") {
169  memoryUsage();
170  } else if (c[0] == "usage") {
171  if (c.size() > 1) {
172  if (c[1][0] == 'R') {
173  usageRelation(c[1]);
174  } else {
175  usageRule(c[1]);
176  }
177  } else {
178  usage();
179  }
180  } else if (c[0] == "help") {
181  help();
182  } else if (c[0] == "limit") {
183  if (c.size() == 1) {
184  setResultLimit(20000);
185  } else {
186  try {
187  setResultLimit(std::stoul(c[1]));
188  } catch (...) {
189  std::cout << "Invalid parameters to limit command.\n";
190  }
191  }
192  } else if (c[0] == "configuration") {
193  configuration();
194  } else {
195  std::cout << "Unknown command. Use \"help\" for a list of commands.\n";
196  }
197  }

751  std::set<Usage> getUsageStats(size_t width = size_t(-1)) {
752  std::set<Usage> usages;
753  DirectoryEntry* usageStats = dynamic_cast<DirectoryEntry*>(
754  ProfileEventSingleton::instance().getDB().lookupEntry({"program", "usage", "timepoint"}));
755  if (usageStats == nullptr || usageStats->getKeys().size() < 2) {
756  return usages;
757  }
758  std::chrono::microseconds endTime{};
759  std::chrono::microseconds startTime{};
760  std::chrono::microseconds timeStep{};
761  // Translate the string ordered text usage stats to a time ordered binary form.
762  std::set<Usage> allUsages;
763  for (auto& currentKey : usageStats->getKeys()) {
764  Usage currentUsage{};
765  uint64_t cur = std::stoul(currentKey);
766  currentUsage.time = std::chrono::duration<uint64_t, std::micro>(cur);
767  cur = dynamic_cast<SizeEntry*>(
768  usageStats->readDirectoryEntry(currentKey)->readEntry("systemtime"))
769  ->getSize();
770  currentUsage.systemtime = std::chrono::duration<uint64_t, std::micro>(cur);
771  cur = dynamic_cast<SizeEntry*>(usageStats->readDirectoryEntry(currentKey)->readEntry("usertime"))
772  ->getSize();
773  currentUsage.usertime = std::chrono::duration<uint64_t, std::micro>(cur);
774  currentUsage.maxRSS =
775  dynamic_cast<SizeEntry*>(usageStats->readDirectoryEntry(currentKey)->readEntry("maxRSS"))
776  ->getSize();
777 
778  // Duplicate times are possible
779  if (allUsages.find(currentUsage) != allUsages.end()) {
780  auto& existing = *allUsages.find(currentUsage);
781  currentUsage.systemtime = std::max(existing.systemtime, currentUsage.systemtime);
782  currentUsage.usertime = std::max(existing.usertime, currentUsage.usertime);
783  currentUsage.maxRSS = std::max(existing.maxRSS, currentUsage.maxRSS);
784  allUsages.erase(currentUsage);
785  }
786  allUsages.insert(currentUsage);
787  }
788 
789  // cpu times aren't quite recorded in a monotonic way, so skip the invalid ones.
790  for (auto it = ++allUsages.begin(); it != allUsages.end(); ++it) {
791  auto previous = std::prev(it);
792  if (it->usertime < previous->usertime || it->systemtime < previous->systemtime ||
793  it->time == previous->time) {
794  it = allUsages.erase(it);
795  --it;
796  }
797  }
798 
799  // Extract our overall stats
800  startTime = allUsages.begin()->time;
801  endTime = allUsages.rbegin()->time;
802 
803  // If we don't have enough records, just return what we can
804  if (allUsages.size() < width) {
805  return allUsages;
806  }
807 
808  timeStep = (endTime - startTime) / width;
809 
810  // Store the timepoints we need for the graph
811  for (uint32_t i = 1; i <= width; ++i) {
812  auto it = allUsages.upper_bound(Usage{startTime + timeStep * i, 0, {}, {}});
813  if (it != allUsages.begin()) {
814  --it;
815  }
816  usages.insert(*it);
817  }
818 
819  return usages;
820  }
