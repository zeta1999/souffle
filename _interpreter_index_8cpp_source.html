36 // -------- generic tuple comparator ----------

38 template <unsigned... Columns>
39 struct comparator;
40 
41 template <unsigned First, unsigned... Rest>
42 struct comparator<First, Rest...> {
43  template <typename T>
44  int operator()(const T& a, const T& b) const {
45  return (a[First] < b[First]) ? -1 : ((a[First] > b[First]) ? 1 : comparator<Rest...>()(a, b));
46  }
47  template <typename T>
48  bool less(const T& a, const T& b) const {
49  return a[First] < b[First] || (a[First] == b[First] && comparator<Rest...>().less(a, b));
50  }
51  template <typename T>
52  bool equal(const T& a, const T& b) const {
53  return a[First] == b[First] && comparator<Rest...>().equal(a, b);
54  }
55 };

142 bool Order::valid() const {
143  // Check that all indices are in range.
144  for (int i : order) {
145  if (i < 0 || i >= int(order.size())) {
146  return false;
147  }
148  }
149  // Check that there are no duplicates.
150  for (std::size_t i = 0; i < order.size(); i++) {
151  for (std::size_t j = i + 1; j < order.size(); j++) {
152  if (order[i] == order[j]) {
153  return false;
154  }
155  }
156  }
157  return true;
158 }

308 protected:
309  using Entry = typename Structure::element_type;

321  // a source adapter for streaming through data
322  class Source : public Stream::Source {
323  const Order& order;
324 
325  // the begin and end of the stream
326  iter cur;
327  iter end;
328 
329  // an internal buffer for re-ordered elements
330  std::array<Entry, Stream::BUFFER_SIZE> buffer;
331 
332  public:
333  Source(const Order& order, iter begin, iter end)
334  : order(order), cur(std::move(begin)), end(std::move(end)) {}

336  int load(TupleRef* out, int max) override {
337  int c = 0;
338  while (cur != end && c < max) {
339  buffer[c] = order.decode(*cur);
340  out[c] = buffer[c];
341  ++cur;
342  ++c;

348  int c = 0;
349  max = std::min(max, Stream::BUFFER_SIZE);
350  while (c < max) {

352  ++c;
353  }
354  return c;
355  }

358  auto source = std::make_unique<Source>(order, cur, end);
359  source->buffer = this->buffer;
360  return source;
361  }

468 /* B-Tree Indirect indexes */
469 class IndirectIndex : public InterpreterIndex {
470 public:
471  using AttributeIndex = uint32_t;
472  using AttributeOrder = std::vector<AttributeIndex>;
473  /* lexicographical comparison operation on two tuple pointers */
474  struct comparator {
475  const AttributeOrder order;
476 
477  /* constructor to initialize state */
478  comparator(AttributeOrder order) : order(std::move(order)) {}
479 
480  /* comparison function */
481  int operator()(const TupleRef& x, const TupleRef& y) const {
482  for (auto& i : order) {
483  if (x[i] < y[i]) {
484  return -1;
485  }
486  if (x[i] > y[i]) {
487  return 1;
488  }
489  }
490  return 0;
491  }
492 
493  /* less comparison */
494  bool less(const TupleRef& x, const TupleRef& y) const {
495  return operator()(x, y) < 0;
496  }
497 
498  /* equal comparison */
499  bool equal(const TupleRef& x, const TupleRef& y) const {
500  for (auto& i : order) {
501  if (x[i] != y[i]) {
502  return false;
503  }
504  }
505  return true;
506  }
507  };
508 
509  /* btree for storing tuple pointers with a given lexicographical order */
510  using index_set = btree_multiset<TupleRef, comparator, std::allocator<TupleRef>, 512>;
511  using Hints = typename index_set::operation_hints;
512 
513  class Source : public Stream::Source {
514  // the begin and end of the stream
515  using iter = btree_multiset<TupleRef, comparator, std::allocator<TupleRef>, 512>::iterator;
516  iter cur;
517  iter end;
518 
519  // an internal buffer for re-ordered elements
520  std::array<TupleRef, Stream::BUFFER_SIZE> buffer;
521 
522  public:
523  Source(iter begin, iter end) : cur(begin), end(end) {}
524 
525  int load(TupleRef* out, int max) override {
526  int c = 0;
527  while (cur != end && c < max) {
528  buffer[c] = *cur;
529  out[c] = buffer[c];
530  ++cur;

532  }
533  return c;
534  }
535 
536  int reload(TupleRef* out, int max) override {
537  int c = 0;
538  max = std::min(max, Stream::BUFFER_SIZE);
539  while (c < max) {
540  out[c] = buffer[c];
