40 Graph<AstQualifiedName> createTypeDependencyGraph(const std::vector<AstType*>& programTypes) {
41  Graph<AstQualifiedName> typeDependencyGraph;
42  for (const auto* astType : programTypes) {
43  if (auto type = as<AstSubsetType>(astType)) {
44  typeDependencyGraph.insert(type->getQualifiedName(), type->getBaseType());
45  } else if (isA<AstRecordType>(astType)) {
46  // do nothing
47  } else if (auto type = as<AstUnionType>(astType)) {
48  for (const auto& subtype : type->getTypes()) {
49  typeDependencyGraph.insert(type->getQualifiedName(), subtype);
50  }
51  } else {
52  fatal("unsupported type construct: %s", typeid(astType).name());
53  }
54  }
55  return typeDependencyGraph;
56 }

59  * Find all the type with a cyclic definition (in terms of being a subtype)
60  */
61 std::set<AstQualifiedName> analyseCyclicTypes(
62  const Graph<AstQualifiedName>& dependencyGraph, const std::vector<AstType*>& programTypes) {
63  std::set<AstQualifiedName> cyclicTypes;
64  for (const auto& astType : programTypes) {
65  AstQualifiedName typeName = astType->getQualifiedName();
66 
67  if (dependencyGraph.reaches(typeName, typeName)) {
68  cyclicTypes.insert(std::move(typeName));
69  }
70  }
71  return cyclicTypes;
72 }

74 /**
75  * Find all the primitive types that are the subtypes of the union types.
76  */
77 std::map<AstQualifiedName, std::set<AstQualifiedName>> analysePrimitiveTypesInUnion(
78  const Graph<AstQualifiedName>& dependencyGraph, const std::vector<AstType*>& programTypes,
79  const TypeEnvironment& env) {
80  std::map<AstQualifiedName, std::set<AstQualifiedName>> primitiveTypesInUnions;
81 
82  for (const auto& astType : programTypes) {
83  auto* unionType = as<AstUnionType>(astType);
84  if (unionType == nullptr) {
85  continue;
86  }
87  AstQualifiedName unionName = unionType->getQualifiedName();
88 
89  auto iteratorToUnion = primitiveTypesInUnions.find(unionName);
90 
91  // Initialize with the empty set
92  if (iteratorToUnion == primitiveTypesInUnions.end()) {
93  iteratorToUnion = primitiveTypesInUnions.insert({unionName, {}}).first;
94  }
95 
96  auto& associatedTypes = iteratorToUnion->second;
97 
98  // Insert any reachable primitive type
99  for (auto& type : env.getPrimitiveTypes()) {
100  if (dependencyGraph.reaches(unionName, type.getName())) {
101  associatedTypes.insert(type.getName());
102  }
103  }
104  }
105  return primitiveTypesInUnions;
106 }

110 void TypeEnvironmentAnalysis::run(const AstTranslationUnit& translationUnit) {
111  const AstProgram& program = *translationUnit.getProgram();
112 
113  auto rawProgramTypes = program.getTypes();
114  Graph<AstQualifiedName> typeDependencyGraph{createTypeDependencyGraph(rawProgramTypes)};
115 
116  cyclicTypes = analyseCyclicTypes(typeDependencyGraph, rawProgramTypes);
117  primitiveTypesInUnions = analysePrimitiveTypesInUnion(typeDependencyGraph, rawProgramTypes, env);
118 
119  std::map<AstQualifiedName, const AstType*> nameToAstType;
120 
121  // Filter redefined primitive types and cyclic types.
122  std::vector<AstType*> programTypes;
123  for (auto* type : program.getTypes()) {
124  if (env.isType(type->getQualifiedName()) || isCyclic(type->getQualifiedName())) {
125  continue;
126  }
127  programTypes.push_back(type);
128  nameToAstType.insert({type->getQualifiedName(), type});
129  }
130 
131  for (const auto* astType : programTypes) {
132  createType(astType->getQualifiedName(), nameToAstType);
133  }
134 }

136 const Type* TypeEnvironmentAnalysis::createType(
137  const AstQualifiedName& typeName, const std::map<AstQualifiedName, const AstType*>& nameToAstType) {
138  // base case
139  if (env.isType(typeName)) {
140  return &env.getType(typeName);
141  }
142 
143  // Handle undeclared type in the definition of another type.
144  auto iterToType = nameToAstType.find(typeName);
145  if (iterToType == nameToAstType.end()) {
146  return nullptr;
147  }
148  const auto& astType = iterToType->second;
149 
150  if (isA<AstSubsetType>(astType)) {
151  // First create a base type.
152  auto* baseType = createType(as<AstSubsetType>(astType)->getBaseType(), nameToAstType);
153 
154  if (baseType == nullptr) {
155  return nullptr;
156  }
157 
158  // Subset of a record is a special case.
159  if (isA<RecordType>(baseType)) {
160  return &env.createType<SubsetRecordType>(typeName, *as<RecordType>(baseType));
161  }
162 
163  return &env.createType<SubsetType>(typeName, *baseType);
164 
165  } else if (isA<AstUnionType>(astType)) {
166  // Create all elements and then the type itself
167  std::vector<const Type*> elements;
168  for (const auto& element : as<AstUnionType>(astType)->getTypes()) {
169  auto* elementType = createType(element, nameToAstType);
170  if (elementType == nullptr) {
171  return nullptr;
172  }
173  elements.push_back(elementType);
174  }
175  return &env.createType<UnionType>(typeName, elements);
176 
177  } else if (isA<AstRecordType>(astType)) {
178  // Create anonymous base type first.
179  // The types need to be initialized upfront,
180  // because we may have mutually recursive record definitions.
181  auto& recordBase = env.createType<RecordType>(tfm::format("__%sConstant", typeName));
182  auto& recordType = env.createType<SubsetRecordType>(typeName, recordBase);
183 
184  std::vector<const Type*> elements;
185  for (const auto* field : as<AstRecordType>(astType)->getFields()) {
186  if (field->getTypeName() == typeName) {
187  elements.push_back(&recordBase);
188  continue;
189  }
190 
191  auto* elementType = createType(field->getTypeName(), nameToAstType);
192  if (elementType == nullptr) {
193  return nullptr;
194  }
195  elements.push_back(elementType);
196  }
197 
198  recordBase.setFields(elements);
199  std::replace(elements.begin(), elements.end(), dynamic_cast<Type*>(&recordBase),
200  dynamic_cast<Type*>(&recordType));
201  recordType.setFields(std::move(elements));
202 
203  return &recordType;
204 
205  } else {
206  fatal("unsupported type construct: %s", typeid(*astType).name());
207  }
208 }
